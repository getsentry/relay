31/// Kafka producer errors.
32#[derive(Error, Debug)]
33pub enum ClientError {
34    /// Failed to send a kafka message.
35    #[error("failed to send kafka message")]
36    SendFailed(#[source] rdkafka::error::KafkaError),
37
38    /// Failed to find configured producer for the requested kafka topic.
39    #[error("failed to find producer for the requested kafka topic")]
40    InvalidTopicName,
41
42    /// Failed to create a kafka producer because of the invalid configuration.
43    #[error("failed to create kafka producer: invalid kafka config")]
44    InvalidConfig(#[source] rdkafka::error::KafkaError),
45
46    /// Failed to serialize the message.
47    #[error("failed to serialize kafka message")]
48    InvalidMsgPack(#[source] rmp_serde::encode::Error),
49
50    /// Failed to serialize the json message using serde.
51    #[error("failed to serialize json message")]
52    InvalidJson(#[source] serde_json::Error),
53
54    /// Failed to run schema validation on message.
55    #[cfg(feature = "schemas")]
56    #[error("failed to run schema validation on message")]
57    SchemaValidationFailed(#[source] schemas::SchemaError),
58
59    /// Configuration is wrong and it cannot be used to identify the number of a shard.
60    #[error("invalid kafka shard")]
61    InvalidShard,
62
63    /// Failed to fetch the metadata of Kafka.
64    #[error("failed to fetch the metadata of Kafka")]
65    MetadataFetchError(rdkafka::error::KafkaError),
66
67    /// Failed to validate the topic.
68    #[error("failed to validate the topic with name {0}: {1:?}")]
69    TopicError(String, rdkafka_sys::rd_kafka_resp_err_t),
70
71    /// Failed to encode the protobuf into the buffer
72    /// because the buffer is too small.
73    #[error("failed to encode protobuf because the buffer is too small")]
74    ProtobufEncodingFailed,
75}

77/// Describes the type which can be sent using kafka producer provided by this crate.
78pub trait Message {
79    /// Returns the partitioning key for this kafka message determining.
80    fn key(&self) -> [u8; 16];
81
82    /// Returns the type of the message.
83    fn variant(&self) -> &'static str;
84
85    /// Return the list of headers to be provided when payload is sent to Kafka.
86    fn headers(&self) -> Option<&BTreeMap<String, String>>;
87
88    /// Serializes the message into its binary format.
89    ///
90    /// # Errors
91    /// Returns the [`ClientError::InvalidMsgPack`] or [`ClientError::InvalidJson`] if the
92    /// serialization failed.
93    fn serialize(&self) -> Result<Cow<'_, [u8]>, ClientError>;
94}

96/// Single kafka producer config with assigned topic.
97struct Producer {
98    /// Kafka topic name.
99    topic_name: String,
100    /// Real kafka producer.
101    producer: Arc<ThreadedProducer>,
102    /// Debouncer for metrics.
103    metrics: Debounced,
104    /// Optional rate limits to apply.
105    rate_limiter: Option<KafkaRateLimits>,
106}
107
108impl Producer {
109    fn new(
110        topic_name: String,
111        producer: Arc<ThreadedProducer>,
112        rate_limiter: Option<KafkaRateLimits>,
113    ) -> Self {
114        Self {
115            topic_name,
116            producer,
117            metrics: Debounced::new(REPORT_FREQUENCY_SECS),
118            rate_limiter,
119        }
120    }
121
122    /// Validates the topic by fetching the metadata of the topic directly from Kafka.
123    fn validate_topic(&self) -> Result<(), ClientError> {
124        let client = self.producer.client();
125        let metadata = client
126            .fetch_metadata(Some(&self.topic_name), KAFKA_FETCH_METADATA_TIMEOUT)
127            .map_err(ClientError::MetadataFetchError)?;
128
129        for topic in metadata.topics() {
130            if let Some(error) = topic.error() {
131                return Err(ClientError::TopicError(topic.name().to_string(), error));
132            }
133        }
134
135        Ok(())
136    }
137}

139impl fmt::Debug for Producer {
140    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
141        f.debug_struct("Producer")
142            .field("topic_name", &self.topic_name)
143            .field("producer", &"<ThreadedProducer>")
144            .finish_non_exhaustive()
145    }

148/// Keeps all the configured kafka producers and responsible for the routing of the messages.
149#[derive(Debug)]
150pub struct KafkaClient {
151    producers: HashMap<KafkaTopic, Producer>,
152    #[cfg(feature = "schemas")]
153    schema_validator: schemas::Validator,
154}
155
156impl KafkaClient {
157    /// Returns the [`KafkaClientBuilder`]
158    pub fn builder() -> KafkaClientBuilder {
159        KafkaClientBuilder::default()
160    }
161
162    /// Sends message to the provided kafka topic.
163    ///
164    /// Returns the name of the kafka topic to which the message was produced.
165    pub fn send_message(
166        &self,
167        topic: KafkaTopic,
168        message: &impl Message,
169    ) -> Result<&str, ClientError> {
170        let serialized = message.serialize()?;
171        #[cfg(feature = "schemas")]
172        self.schema_validator
173            .validate_message_schema(topic, &serialized)
174            .map_err(ClientError::SchemaValidationFailed)?;
175
176        self.send(
177            topic,
178            message.key(),
179            message.headers(),
180            message.variant(),
181            &serialized,
182        )
183    }
184
185    /// Sends the payload to the correct producer for the current topic.
186    ///
187    /// Returns the name of the kafka topic to which the message was produced.
188    pub fn send(
189        &self,
190        topic: KafkaTopic,
191        key: [u8; 16],
192        headers: Option<&BTreeMap<String, String>>,
193        variant: &str,
194        payload: &[u8],
195    ) -> Result<&str, ClientError> {
196        let producer = self.producers.get(&topic).ok_or_else(|| {
197            relay_log::error!(
198                "attempted to send message to {topic:?} using an unconfigured kafka producer",
199            );
200            ClientError::InvalidTopicName
201        })?;
202
203        producer.send(key, headers, variant, payload)
204    }
205}

207/// Helper structure responsible for building the actual [`KafkaClient`].
208#[derive(Default)]
209pub struct KafkaClientBuilder {
210    reused_producers: BTreeMap<Option<String>, Arc<ThreadedProducer>>,
211    producers: HashMap<KafkaTopic, Producer>,
212}
213
214impl KafkaClientBuilder {
215    /// Creates an empty KafkaClientBuilder.
216    pub fn new() -> Self {
217        Self::default()
218    }
219
220    /// Adds topic configuration to the current [`KafkaClientBuilder`], which in return assigns
221    /// dedicates producer to the topic which can will be used to send the messages.
222    ///
223    /// # Errors
224    /// Returns [`ClientError::InvalidConfig`] error if the provided configuration is wrong and
225    /// the producer could not be created.
226    pub fn add_kafka_topic_config(
227        mut self,
228        topic: KafkaTopic,
229        params: &KafkaParams,
230        validate_topic: bool,
231    ) -> Result<Self, ClientError> {
232        let mut client_config = ClientConfig::new();
233
234        let KafkaParams {
235            topic_name,
236            config_name,
237            params,
238            key_rate_limit,
239        } = params;
240
241        let rate_limiter = key_rate_limit.map(|limit| {
242            KafkaRateLimits::new(
243                limit.limit_per_window,
244                Duration::from_secs(limit.window_secs),
245            )
246        });
247
248        let config_name = config_name.map(str::to_string);
249
250        if let Some(producer) = self.reused_producers.get(&config_name) {
251            let producer = Producer::new(
252                (*topic_name).to_string(),
253                Arc::clone(producer),
254                rate_limiter,
255            );
256            if validate_topic {
257                producer.validate_topic()?;
258            }
259            self.producers.insert(topic, producer);
260            return Ok(self);
261        }
262
263        for config_p in *params {
264            client_config.set(config_p.name.as_str(), config_p.value.as_str());
265        }
266
267        let producer = Arc::new(
268            client_config
269                .create_with_context(Context)
270                .map_err(ClientError::InvalidConfig)?,
271        );
272
273        self.reused_producers
274            .insert(config_name, Arc::clone(&producer));
275
276        let producer = Producer::new((*topic_name).to_string(), producer, rate_limiter);
277        if validate_topic {
278            producer.validate_topic()?;
279        }
280        self.producers.insert(topic, producer);
281
282        Ok(self)
283    }
284
285    /// Consumes self and returns the built [`KafkaClient`].
286    pub fn build(self) -> KafkaClient {
287        KafkaClient {
288            producers: self.producers,
289            #[cfg(feature = "schemas")]
290            schema_validator: schemas::Validator::default(),
291        }
292    }
293}

304impl Producer {
305    /// Sends the payload to the correct producer for the current topic.
306    fn send(
307        &self,
308        key: [u8; 16],
309        headers: Option<&BTreeMap<String, String>>,
310        variant: &str,
311        payload: &[u8],
312    ) -> Result<&str, ClientError> {
313        let now = Instant::now();
314        let topic_name = self.topic_name.as_str();
315
316        metric!(
317            histogram(KafkaHistograms::KafkaMessageSize) = payload.len() as u64,
318            variant = variant,
319            topic = topic_name,
320        );
321
322        let mut headers = headers
323            .unwrap_or(&BTreeMap::new())
324            .iter()
325            .map(|(key, value)| Header {
326                key,
327                value: Some(value),
328            })
329            .collect::<KafkaHeaders>();
330
331        let mut key = key;
332        if let Some(ref limiter) = self.rate_limiter {
333            if limiter.try_increment(now, key, 1) < 1 {
334                metric!(
335                    counter(KafkaCounters::ProducerPartitionKeyRateLimit) += 1,
336                    variant = variant,
337                    topic = topic_name,
338                );
339
340                key = Uuid::new_v4().into_bytes();
341                headers.insert(Header {
342                    key: "sentry-reshuffled",
343                    value: Some("1"),
344                });
345            }
346        }
347
348        let mut record = BaseRecord::to(topic_name).key(&key).payload(payload);
349
350        if let Some(headers) = headers.into_inner() {
351            record = record.headers(headers);
352        }
353
354        self.metrics.debounce(now, || {
355            metric!(
356                gauge(KafkaGauges::InFlightCount) = self.producer.in_flight_count() as u64,
357                variant = variant,
358                topic = topic_name
359            );
360        });
361
362        self.producer.send(record).map_err(|(error, _message)| {
363            relay_log::error!(
364                error = &error as &dyn std::error::Error,
365                tags.variant = variant,
366                tags.topic = topic_name,
367                "error sending kafka message",
368            );
369            metric!(
370                counter(KafkaCounters::ProducerEnqueueError) += 1,
371                variant = variant,
372                topic = topic_name
373            );
374            ClientError::SendFailed(error)
375        })?;
376
377        Ok(topic_name)
378    }
