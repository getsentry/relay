5use sentry_kafka_schemas::{Schema as SentrySchema, SchemaError as SentrySchemaError};
6use thiserror::Error;
7
8use crate::config::KafkaTopic;

10#[derive(Debug, Error)]
11pub enum SchemaError {
12    /// The "logical topic" is a concept in sentry-kafka-schemas and Snuba that identifies topics
13    /// irrespective of how they are structured in production. Snuba has a mapping from "logical
14    /// topic" to "physical topic" that is very similar to our `TopicAssignments`.
15    ///
16    /// The name of a logical topic is almost always equal to the Kafka topic's name in local
17    /// development. So, in order to determine a logical topic name from a `KafkaTopic`, we get
18    /// `TopicAssignments::default()` and try to resolve the name through that.
19    ///
20    /// When doing that we assume that Relay's default topic assignments don't use slicing/sharding
21    /// and no custom clusters.
22    ///
23    /// If somebody changes `impl Default for TopicAssignments` to have more complex defaults, this
24    /// error will start occurring. But it should not happen in prod.
25    #[error("failed to determine logical topic")]
26    LogicalTopic,
27
28    /// Failed to compile schema
29    #[error("failed to compile schema")]
30    SchemaCompiled(#[source] SentrySchemaError),
31
32    /// Failed to validate message JSON against schema
33    #[error("message violates schema")]
34    Validation(#[source] SentrySchemaError),
35}
36
37/// Validates payloads for their given topic's schema.
38#[derive(Debug, Default)]
39pub struct Validator {
40    /// Caches the schema for given topics.
41    schemas: Mutex<BTreeMap<KafkaTopic, Option<Arc<SentrySchema>>>>,
42}
43
44impl Validator {
45    /// Validate a message for a given topic's schema.
46    pub fn validate_message_schema(
47        &self,
48        topic: KafkaTopic,
49        message: &[u8],
50    ) -> Result<(), SchemaError> {
51        let Some(schema) = self.get_schema(topic)? else {
52            return Ok(());
53        };
54
55        schema
56            .validate_json(message)
57            .map_err(SchemaError::Validation)
58            .map(drop)
59    }
60
61    fn get_schema(&self, topic: KafkaTopic) -> Result<Option<Arc<SentrySchema>>, SchemaError> {
62        let mut schemas = self.schemas.lock().unwrap_or_else(|e| e.into_inner());
63
64        Ok(match schemas.entry(topic) {
65            Entry::Vacant(entry) => entry.insert({
66                let logical_topic_name = topic.logical_topic_name();
67
68                match sentry_kafka_schemas::get_schema(logical_topic_name, None) {
69                    Ok(schema) => Some(Arc::new(schema)),
70                    Err(SentrySchemaError::TopicNotFound) => None,
71                    Err(err) => return Err(SchemaError::SchemaCompiled(err)),
72                }
73            }),
74            Entry::Occupied(entry) => entry.into_mut(),
75        }
76        .as_ref()
77        .map(Arc::clone))
78    }
79}
