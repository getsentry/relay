3pub enum KafkaCounters {
4    /// Number of producer errors occurred after an envelope was already enqueued for sending to
5    /// Kafka.
6    ///
7    /// These errors include, for example, _"MessageTooLarge"_ errors when the broker does not
8    /// accept the requests over a certain size, which is usually due to invalid or inconsistent
9    /// broker/producer configurations.
10    ///
11    /// This metric is tagged with:
12    ///  - `topic`: The Kafka topic being produced to.
13    ProcessingProduceError,
14
15    /// Number of messages that failed to be enqueued in the Kafka producer's memory buffer.
16    ///
17    /// These errors include, for example, _"UnknownTopic"_ errors when attempting to send a
18    /// message a topic that does not exist.
19    ///
20    /// This metric is tagged with:
21    /// - `topic`: The Kafka topic being produced to.
22    /// - `variant`: The Kafka message variant.
23    ProducerEnqueueError,
24}
25
26impl CounterMetric for KafkaCounters {
27    fn name(&self) -> &'static str {
28        match self {
29            Self::ProcessingProduceError => "processing.produce.error",
30            Self::ProducerEnqueueError => "producer.enqueue.error",
31        }
32    }
33}

47/// Gauge metrics for the Kafka producer.
48///
49/// Most of these metrics are taken from the [`rdkafka::statistics`] module.
50pub enum KafkaGauges {
51    /// The number of messages waiting to be sent to, or acknowledged by, the broker.
52    ///
53    /// See <https://docs.confluent.io/platform/7.5/clients/librdkafka/html/rdkafka_8h.html#ad4b3b7659cf9a79d3353810d6b625bb7>.
54    ///
55    /// This metric is tagged with:
56    /// - `topic`: The Kafka topic being produced to.
57    /// - `variant`: The Kafka message variant.
58    InFlightCount,
59
60    /// The current number of messages in producer queues.
61    MessageCount,
62
63    /// The maximum number of messages allowed in the producer queues.
64    MessageCountMax,
65
66    /// The current total size of messages in producer queues.
67    MessageSize,
68
69    /// The maximum total size of messages allowed in the producer queues.
70    MessageSizeMax,
71
72    /// The number of requests awaiting transmission to the broker.
73    ///
74    /// This metric is tagged with:
75    /// - `broker_name`: The broker hostname, port, and ID, in the form HOSTNAME:PORT/ID.
76    OutboundBufferRequests,
77
78    /// The number of messages awaiting transmission to the broker.
79    ///
80    /// This metric is tagged with:
81    /// - `broker_name`: The broker hostname, port, and ID, in the form HOSTNAME:PORT/ID.
82    OutboundBufferMessages,
83
84    /// The number of connection attempts, including successful and failed attempts, and name resolution failures.
85    ///
86    /// This metric is tagged with:
87    /// - `broker_name`: The broker hostname, port, and ID, in the form HOSTNAME:PORT/ID.
88    Connects,
89
90    /// The number of disconnections, whether triggered by the broker, the network, the load balancer, or something else.
91    ///
92    /// This metric is tagged with:
93    /// - `broker_name`: The broker hostname, port, and ID, in the form HOSTNAME:PORT/ID.
94    Disconnects,
95
96    /// Maximum internal producer queue latency, in microseconds.
97    ///
98    /// This metric is tagged with:
99    /// - `broker_name`: The broker hostname, port, and ID, in the form HOSTNAME:PORT/ID.
100    ProducerQueueLatency,
101
102    /// Maximum internal request queue latency, in microseconds.
103    ///
104    /// This metric is tagged with:
105    /// - `broker_name`: The broker hostname, port, and ID, in the form HOSTNAME:PORT/ID.
106    RequestQueueLatency,
107}
108
109impl GaugeMetric for KafkaGauges {
110    fn name(&self) -> &'static str {
111        match self {
112            KafkaGauges::InFlightCount => "kafka.in_flight_count",
113            KafkaGauges::MessageCount => "kafka.stats.message_count",
114            KafkaGauges::MessageCountMax => "kafka.stats.message_count_max",
115            KafkaGauges::MessageSize => "kafka.stats.message_size",
116            KafkaGauges::MessageSizeMax => "kafka.stats.message_size_max",
117            KafkaGauges::OutboundBufferRequests => "kafka.stats.broker.outbuf.requests",
118            KafkaGauges::OutboundBufferMessages => "kafka.stats.broker.outbuf.messages",
119            KafkaGauges::Connects => "kafka.stats.broker.connects",
120            KafkaGauges::Disconnects => "kafka.stats.broker.disconnects",
121            KafkaGauges::ProducerQueueLatency => "kafka.stats.broker.int_latency",
122            KafkaGauges::RequestQueueLatency => "kafka.stats.broker.outbuf_latency",
123        }
124    }
125}
