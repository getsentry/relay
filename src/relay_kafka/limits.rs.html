7use parking_lot::RwLock;
8

10
11type Key = [u8; 16];
12
13/// A really basic rate limiter to protect downstream systems from partition imbalance on ingestion
14/// topics.
15///
16/// The rate limiter is not used to drop messages, but rather to have them be randomly partitioned
17/// (instead of by trace ID) to reduce the partition imbalance again. If the topic is semantically
18/// partitioned this almost certainly will have product impact.
19///
20/// A partition key's message rate is measured in a single window. If the span rate ever exceeds this rate
21/// (`limit_per_window / window_size`), the partition key is limited for up to a duration of
22/// `window_size`.
23pub struct KafkaRateLimits {
24    limit_per_window: u64,
25    map: RwLock<HashMap<Key, AtomicU64>>,
26    wipe_debouncer: Debounced,
27    window_size: Duration,
28}
29
30impl KafkaRateLimits {
31    pub fn new(limit_per_window: u64, window_size: Duration) -> Self {
32        KafkaRateLimits {
33            map: Default::default(),
34            wipe_debouncer: Debounced::new(window_size.as_secs()),
35            limit_per_window,
36            window_size,
37        }
38    }
39
40    #[must_use]
41    pub fn try_increment(&self, now: Instant, key: Key, amount: u64) -> u64 {
42        self.wipe_debouncer.debounce(now, || {
43            self.map.write().clear();
44        });
45
46        let increment = |value: &AtomicU64| {
47            let actual_count = value.fetch_add(amount, Ordering::Relaxed);
48            let headroom = self.limit_per_window.saturating_sub(actual_count);
49            cmp::min(amount, headroom)
50        };
51
52        // Fast path, check if we already have a stored bucket.
53        if let Some(value) = self.map.read().get(&key) {
54            return increment(value);
55        }
56
57        // Slow path, get an exclusive lock and create the bucket if it wasn't already created in
58        // the meantime.
59        let mut map_w = self.map.write();
60        increment(map_w.entry(key).or_default())
61    }
62}
63
64impl fmt::Debug for KafkaRateLimits {
65    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
66        f.debug_struct("KafkaRateLimits")
67            .field("limit_per_window", &self.limit_per_window)
68            .field("window_size", &self.window_size)
69            .finish_non_exhaustive()
70    }
71}

73#[cfg(test)]
74mod tests {
75    use super::*;
76
77    #[test]
78    fn test_limits() {
79        let limiter = KafkaRateLimits::new(10, Duration::from_secs(10));
80        let now = Instant::now();
81        let key = [0; 16];
82
83        for i in 1..=5 {
84            assert_eq!(
85                limiter.try_increment(now + Duration::from_secs(i), key, 2),
86                2
87            );
88        }
89
90        for i in 6..=10 {
91            assert_eq!(
92                limiter.try_increment(now + Duration::from_secs(i), key, 2),
93                0
94            );
95        }
96
97        assert_eq!(
98            limiter.try_increment(now + Duration::from_secs(11), key, 2),
99            2
100        );
101    }
