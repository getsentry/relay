7/// A ratio is converted to a divisor to perform integer arithmetic, instead of floating point.
8///
9/// This is done with the configured precision here.
10const RATIO_PRECISION: u64 = 10;
11
12/// A quota to be checked with the [`OpportunisticQuotaCache`].
13#[derive(Debug, Clone, Copy)]
14pub struct Quota<T> {
15    /// The quota limit.
16    ///
17    /// A negative limit is treated as infinite/unlimited.
18    pub limit: i64,
19    /// The quota window size in seconds.
20    pub window: u64,
21    /// A unique identifier for the quota bucket.
22    pub key: T,
23    /// The expiry of the bucket.
24    pub expiry: UnixTimestamp,
25}
26
27/// An opportunistic cache for quotas.
28#[derive(Debug)]
29pub struct OpportunisticQuotaCache<T>
30where
31    T: std::hash::Hash + Eq,
32{
33    /// A cache for all currently consumed quotas.
34    ///
35    /// The cache is keyed with the individual quota, the value is the last known, currently
36    /// consumed amount of the quota.
37    cache: papaya::HashMap<T, CachedQuota, ahash::RandomState>,
38
39    /// The amount the cache is allowed to opportunistically over-accept based on the remaining
40    /// quota.
41    ///
42    /// For example: Setting this to `10 * RATIO_PRECISION` means, if there is 100 quota remaining,
43    /// the cache will opportunistically accept the next 10 items, if there is a quota of 90 remaining,
44    /// the cache will accept the next 9 items.
45    max_over_spend_divisor: NonZeroU64,
46
47    /// The maximum amount of quota the cache considers for activation.
48    ///
49    /// See also: [`Self::with_max`].
50    limit_max_divisor: Option<NonZeroU64>,
51
52    /// Minimum interval between vacuum of the cache.
53    vacuum_interval: Duration,
54    /// Unix timestamp of the next time the vacuum should be run.
55    ///
56    /// This is a [`UnixTimestamp`] stored in an atomic.
57    next_vacuum: AtomicU64,
58}
59
60impl<T> OpportunisticQuotaCache<T>
61where
62    T: std::hash::Hash + Eq,
63{
64    /// Creates a new [`Self`], with the configured maximum amount the cache accepts per quota
65    /// until it requires synchronization.
66    ///
67    /// The configured ratio must be in range `[0, 1]`.
68    pub fn new(max_over_spend: f32) -> Self {
69        let max_over_spend_divisor = 1.0f32 / max_over_spend * RATIO_PRECISION as f32;
70        let max_over_spend_divisor =
71            NonZeroU64::new(max_over_spend_divisor as u64).unwrap_or(NonZeroU64::MIN);
72
73        Self {
74            cache: papaya::HashMap::builder()
75                // `ahash` for a faster hasher
76                .hasher(ahash::RandomState::new())
77                // Blocking resizes help with iteration, which we need for cleanups,
78                // as well as our workload seems to match the description of `Blocking`
79                // very well.
80                .resize_mode(papaya::ResizeMode::Blocking)
81                .build(),
82            max_over_spend_divisor,
83            limit_max_divisor: None,
84            vacuum_interval: Duration::from_secs(30),
85            // Initialize to 0, this means a vacuum run immediately, but it is going to be fast
86            // (empty cache) and it requires us to be time/environment independent, time is purely
87            // driven by the user of the cache.
88            next_vacuum: AtomicU64::new(0),
89        }
90    }
91
92    /// Relative amount of the total quota limit to which caching is applied.
93    ///
94    /// If exceeded the cache will no longer cache values for the quota.
95    /// Lowering this value reduces the probability of incorrectly over-accepting.
96    ///
97    /// For example: A quota with limit `100` and a configured limit threshold of `0.7` (70%),
98    /// will no longer be considered for caching if 70% (70) of the quota is consumed.
99    ///
100    /// By default, no maximum is configured and the entire quota is cached.
101    pub fn with_max(mut self, max: Option<f32>) -> Self {
102        self.limit_max_divisor = max.map(|v| {
103            // Inverting the threshold here simplifies the checking code, but also retains more
104            // precision for the integer division, since we can expect this value to be large.
105            //
106            // 1.0 / 0.95 * 10 = 10
107            // 1.0 / (1.0 - 0.95) * 10 = 200
108            //
109            // 100 * 10 / 10 = 100
110            // 100 - (100 * 10 / 200) = 95
111            let max_div = 1.0f32 / (1.0f32 - v.clamp(0.0, 1.0)) * RATIO_PRECISION as f32;
112            NonZeroU64::new(max_div as u64).unwrap_or(NonZeroU64::MAX)
113        });
114
115        self
116    }
117
118    /// Checks a quota with quantity against the cache.
119    ///
120    /// The cache may return [`Action::Accept`] indicating the quantity should be accepted.
121    /// If the cache can not make a decision it returns [`Action::Check`] indicating the returned
122    /// quantity needs to be synchronized with a centralized store.
123    ///
124    /// Whenever the cache returns [`Action::Check`], the cache requires a call to [`Self::set_quota`],
125    /// with a synchronized 'consumed' amount.
126    pub fn check_quota(&self, quota: Quota<T>, quantity: u64) -> Action {
127        let cache = self.cache.pin();
128
129        let Ok(limit) = u64::try_from(quota.limit) else {
130            // Negative limits are infinite.
131            return Action::Accept;
132        };
133
134        // We can potentially short circuit here with a simple read, the cases:
135        // 1. `NeedsSync`
136        // 2. Active with `consumed >= limit`
137        // 3. No entry.
138        //
139        // This may be faster because the map does not require a mutation.
140        // Needs more investigation if the additional lookup is worth it.
141
142        let value = cache.update(quota.key, |q| {
143            let (consumed, local_use) = match q {
144                // There is already a sync requested which we need to wait for to get consistent
145                // data, we cannot make a decision here at this time.
146                CachedQuota::NeedsSync => return CachedQuota::NeedsSync,
147                // This variant only exists as a way of returning a quantity on the first
148                // `NeedsSync` decision. We can safely erase it into `NeedsSync` without a value.
149                CachedQuota::NeedsSyncWithQuantity(_) => return CachedQuota::NeedsSync,
150                CachedQuota::Active {
151                    consumed,
152                    local_use,
153                    expiry: _,
154                } => (*consumed, *local_use),
155            };
156
157            let total_local_use = local_use + quantity;
158
159            let threshold = match self.limit_max_divisor.map(NonZeroU64::get) {
160                Some(div) => limit * RATIO_PRECISION / div,
161                None => 0,
162            };
163
164            // Can short circuit here already if consumed is already above or equal to the limit.
165            //
166            // We could also propagate this out to the caller as a definitive negative in the
167            // future. This does require some additional consideration how this would interact with
168            // refunds, which can reduce the consumed.
169            if consumed + total_local_use >= limit.saturating_sub(threshold) {
170                return CachedQuota::new_needs_sync(total_local_use);
171            }
172
173            let remaining = limit.saturating_sub(consumed);
174            let max_allowed_spend = remaining
175                // Normalize the remaining quota with the window size, to apply the ratio/divisor to the
176                // per second rate.
177                //
178                // This means we get a consistent behaviour for short (10s) quotas (e.g. abuse) as well
179                // as long (1h) quotas (e.g. spike protection) with a more predictable error.
180                / quota.window.max(1)
181                // Apply ratio precision, which is already pre-multiplied into `max_over_spend_divisor`.
182                * RATIO_PRECISION
183                // Apply the actual ratio with the pre-computed divisor.
184                / self.max_over_spend_divisor.get();
185
186            match total_local_use > max_allowed_spend {
187                true => CachedQuota::new_needs_sync(total_local_use),
188                false => CachedQuota::Active {
189                    consumed,
190                    local_use: total_local_use,
191                    expiry: quota.expiry,
192                },
193            }
194        });
195
196        match value {
197            Some(CachedQuota::NeedsSyncWithQuantity(q)) => Action::Check(q.get()),
198            Some(CachedQuota::NeedsSync) | None => Action::Check(quantity),
199            Some(CachedQuota::Active { .. }) => Action::Accept,
200        }
201    }
202
203    /// Sets the quota state in the cache for the specified quota.
204    ///
205    /// `consumed` is the absolute value reported by the synchronized store (Redis) not a relative
206    /// change. The value may also be negative, this may happen with refunded quotas.
207    ///
208    /// The cache will use the synchronized `consumed` value to derive future decisions whether it
209    /// can accept quota requests.
210    pub fn set_quota(&self, quota: Quota<T>, consumed: i64) {
211        let cache = self.cache.pin();
212
213        // Consumed quota can be negative due to refunds, we choose to deal with negative quotas
214        // like they are simply unused.
215        //
216        // This only makes the cache stricter and less likely to over accept.
217        let consumed = u64::try_from(consumed).unwrap_or(0);
218
219        cache.update_or_insert(
220            quota.key,
221            |q| match q {
222                // We got the sync, start fresh!
223                CachedQuota::NeedsSync | CachedQuota::NeedsSyncWithQuantity(_) => {
224                    CachedQuota::Active {
225                        consumed,
226                        local_use: 0,
227                        expiry: quota.expiry,
228                    }
229                }
230                // This is slightly tricky, one could assume quotas are only growing and we should use
231                // the bigger `consumed` count here. The problem is refunds, a quota may decrease again
232                // due to refunds.
233                //
234                // Without a way to differentiate which `consumed` value is actually more recent, we
235                // just have the implicit assumption that the value arriving last is the most recent.
236                //
237                // This is not perfect, but for our best effort opportunistic cache, this is not a big
238                // problem, at worst, we over accept a little bit more for the next synchronization
239                // round.
240                CachedQuota::Active {
241                    consumed: _,
242                    local_use,
243                    expiry: _,
244                } => CachedQuota::Active {
245                    consumed,
246                    local_use: *local_use,
247                    expiry: quota.expiry,
248                },
249            },
250            CachedQuota::Active {
251                consumed,
252                local_use: 0,
253                expiry: quota.expiry,
254            },
255        );
256    }
257
258    /// Attempts to run a vacuum on the internal cache.
259    ///
260    /// The vacuum is internally debounced and may not run. Callers should call attempt to vacuum
261    /// periodically.
262    ///
263    /// Returns `true` if a vacuum was performed.
264    pub fn try_vacuum(&self, now: UnixTimestamp) -> bool {
265        // Check for the lucky winner to run an expiration.
266        let next_vacuum = self.next_vacuum.load(Ordering::Relaxed);
267
268        if next_vacuum > now.as_secs() {
269            // It's not yet time to vacuum.
270            return false;
271        }
272
273        let exchange = self.next_vacuum.compare_exchange(
274            next_vacuum,
275            now.as_secs() + self.vacuum_interval.as_secs(),
276            Ordering::Relaxed,
277            Ordering::Relaxed,
278        );
279
280        if exchange.is_err() {
281            // Somebody else took the vacuum, that's fine.
282            return false;
283        }
284
285        let mut cache = self.cache.pin();
286        cache.retain(|_, v| v.is_still_needed(now));
287
288        true
289    }
290}

292/// Action to take when accessing cached quotas via [`OpportunisticQuotaCache::check_quota`].
293#[derive(Debug, PartialEq, Eq)]
294pub enum Action {
295    /// Accept the quota request.
296    Accept,
297    /// Synchronize the quota with the returned quantity.
298    Check(u64),
299}
300
301/// State of a cached quota.
302#[derive(Debug)]
303enum CachedQuota {
304    /// The cache currently cannot make a decision and needs an updated consumption value from the
305    /// synchronized store.
306    NeedsSync,
307    /// Like [`Self::NeedsSync`], but also carries a total quantity which needs to be synchronized
308    /// with the store.
309    NeedsSyncWithQuantity(NonZeroU64),
310    /// The cache is active and can still make decisions without a synchronization.
311    Active {
312        consumed: u64,
313        local_use: u64,
314        expiry: UnixTimestamp,
315    },
316}
317
318impl CachedQuota {
319    /// Creates [`Self::NeedsSync`] for a quantity of `0`, [`Self::NeedsSyncWithQuantity`] otherwise.
320    pub fn new_needs_sync(quantity: u64) -> Self {
321        NonZeroU64::new(quantity)
322            .map(Self::NeedsSyncWithQuantity)
323            .unwrap_or(Self::NeedsSync)
324    }
325
326    /// Returns `true` when the entry is no longer needed in the cache.
327    ///
328    /// An item is no longer needed when it is expired or a sync is required.
329    fn is_still_needed(&self, now: UnixTimestamp) -> bool {
330        match self {
331            Self::NeedsSync => false,
332            Self::NeedsSyncWithQuantity(_) => false,
333            Self::Active { expiry, .. } => now <= *expiry,
334        }
335    }
336}

352    #[test]
353    fn test_opp_quota() {
354        let cache = OpportunisticQuotaCache::new(0.1);
355
356        let q1 = Quota {
357            limit: 100,
358            window: 1,
359            key: "k1",
360            expiry: UnixTimestamp::from_secs(300),
361        };
362        let q2 = Quota {
363            limit: 50,
364            window: 1,
365            key: "k2",
366            expiry: UnixTimestamp::from_secs(300),
367        };
368
369        // First access must always go to synchronized store.
370        assert_eq!(cache.check_quota(q1, 10), Action::Check(10));
371        // So does any following access until there was a sync.
372        assert_eq!(cache.check_quota(q1, 9), Action::Check(9));
373
374        // Same for other keys.
375        assert_eq!(cache.check_quota(q2, 12), Action::Check(12));
376
377        // Sync internal state to 30, for this test, there will be 70 remaining, meaning the cache
378        // is expected to accept 7 more items without requiring another sync.
379        cache.set_quota(q1, 30);
380
381        // A different key still needs a sync.
382        assert_eq!(cache.check_quota(q2, 12), Action::Check(12));
383
384        // 70 remaining, 10% -> 7 accepted.
385        assert_eq!(cache.check_quota(q1, 5), Action::Accept);
386        assert_eq!(cache.check_quota(q1, 2), Action::Accept);
387        // This one goes over the limit, we need to now check a total of 8.
388        assert_eq!(cache.check_quota(q1, 1), Action::Check(8));
389
390        // A new sync is required
391        assert_eq!(cache.check_quota(q1, 1), Action::Check(1));
392        assert_eq!(cache.check_quota(q1, 2), Action::Check(2));
393        // A different key is still a different key.
394        assert_eq!(cache.check_quota(q2, 3), Action::Check(3));
395
396        // Consumed state is absolute not relative.
397        cache.set_quota(q1, 50);
398        // This is too much.
399        assert_eq!(cache.check_quota(q2, 6), Action::Check(6));
400        // Need another sync again.
401        assert_eq!(cache.check_quota(q2, 1), Action::Check(1));
402
403        // Negative state can exist due to refunds.
404        cache.set_quota(q1, -123);
405        // The cache considers a negative quota like `0`, `100` remaining quota -> 10 (= 10%).
406        assert_eq!(cache.check_quota(q1, 10), Action::Accept);
407        // Too much, check the entire local usage.
408        assert_eq!(cache.check_quota(q1, 1), Action::Check(11));
409    }

591    fn test_opp_quota_vacuum() {
592        let cache = OpportunisticQuotaCache::new(0.1);
593
594        let q1 = Quota {
595            limit: 100,
596            window: 1,
597            key: "k1",
598            expiry: UnixTimestamp::from_secs(100),
599        };
600        let q2 = Quota {
601            limit: 100,
602            window: 1,
603            key: "k2",
604            expiry: UnixTimestamp::from_secs(200),
605        };
606
607        // Initialize the cache.
608        cache.set_quota(q1, 0);
609        cache.set_quota(q2, 0);
610
611        // Make sure some elements are stored in the cache.
612        assert_eq!(cache.check_quota(q1, 9), Action::Accept);
613        assert_eq!(cache.check_quota(q2, 9), Action::Accept);
614
615        // This vacuum should succeed, no concurrent vacuums.
616        assert!(cache.try_vacuum(UnixTimestamp::from_secs(150)));
617        // A vacuum was just ran, there is no need to run it again.
618        assert!(!cache.try_vacuum(UnixTimestamp::from_secs(150)));
619
620        // This entry was purged from the cache, now with the same quota, we should no longer have
621        // an old entry.
622        assert_eq!(cache.check_quota(q1, 6), Action::Check(6));
623        // This entry was not purged (different expiry) and should therefor return the full amount.
624        assert_eq!(cache.check_quota(q2, 6), Action::Check(15));
625    }
