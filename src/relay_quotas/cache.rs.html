7/// A ratio is converted to a divisor to perform integer arithmetic, instead of floating point.
8///
9/// This is done with the configured precision here.
10const RATIO_PRECISION: usize = 10;
11
12/// A quota to be checked with the [`OpportunisticQuotaCache`].
13#[derive(Debug, Clone, Copy)]
14pub struct Quota<T> {
15    /// The quota limit.
16    pub limit: i64,
17    /// The quota window size in seconds.
18    pub window: u64,
19    /// A unique identifier for the quota bucket.
20    pub key: T,
21    /// The expiry of the bucket.
22    pub expiry: UnixTimestamp,
23}
24
25/// An opportunistic cache for quotas.
26#[derive(Debug)]
27pub struct OpportunisticQuotaCache<T>
28where
29    T: std::hash::Hash + Eq,
30{
31    /// A cache for all currently consumed quotas.
32    ///
33    /// The cache is keyed with the individual quota, the value is the last known, currently
34    /// consumed amount of the quota.
35    cache: papaya::HashMap<T, CachedQuota>,
36
37    /// The amount the cache is allowed to opportunistically over-accept based on the remaining
38    /// quota.
39    ///
40    /// For example: Setting this to `10 * RATIO_PRECISION` means, if there is 100 quota remaining,
41    /// the cache will opportunistically accept the next 10 items, if there is a quota of 90 remaining,
42    /// the cache will accept the next 9 items.
43    max_over_spend_divisor: NonZeroUsize,
44
45    /// Minimum interval between vacuum of the cache.
46    vacuum_interval: Duration,
47    /// Unix timestamp of the next time the vacuum should be run.
48    ///
49    /// This is a [`UnixTimestamp`] stored in an atomic.
50    next_vacuum: AtomicU64,
51}
52
53impl<T> OpportunisticQuotaCache<T>
54where
55    T: std::hash::Hash + Eq,
56{
57    /// Creates a new [`Self`], with the configured maximum amount the cache accepts per quota
58    /// until it requires synchronization.
59    ///
60    /// The configured ratio must be in range `[0, 1]`.
61    pub fn new(max_over_spend_ratio: f32) -> Self {
62        let max_over_spend_divisor = 1.0f32 / max_over_spend_ratio * RATIO_PRECISION as f32;
63        let max_over_spend_divisor =
64            NonZeroUsize::new(max_over_spend_divisor as usize).unwrap_or(NonZeroUsize::MIN);
65
66        Self {
67            cache: Default::default(),
68            max_over_spend_divisor,
69            vacuum_interval: Duration::from_secs(30),
70            // Initialize to 0, this means a vacuum run immediately, but it is going to be fast
71            // (empty cache) and it requires us to be time/environment independent, time is purely
72            // driven by the user of the cache.
73            next_vacuum: AtomicU64::new(0),
74        }
75    }
76
77    /// Checks a quota with quantity against the cache.
78    ///
79    /// The cache may return [`Action::Accept`] indicating the quantity should be accepted.
80    /// If the cache can not make a decision it returns [`Action::Check`] indicating the returned
81    /// quantity needs to be synchronized with a centralized store.
82    ///
83    /// Whenever the cache returns [`Action::Check`], the cache requires a call to [`Self::update_quota`],
84    /// with a synchronized 'consumed' amount.
85    pub fn check_quota(&self, quota: Quota<T>, quantity: usize) -> Action {
86        let cache = self.cache.pin();
87
88        // We can potentially short circuit here with a simple read, the cases:
89        // 1. `NeedsSync`
90        // 2. Active with `consumed >= limit`
91        // 3. No entry.
92        //
93        // This may be faster because the map does not require a mutation.
94        // Needs more investigation if the additional lookup is worth it.
95
96        let value = cache.update(quota.key, |q| {
97            let (consumed, local_use) = match q {
98                // There is already a sync requested which we need to wait for to get consistent
99                // data, we cannot make a decision here at this time.
100                CachedQuota::NeedsSync => return CachedQuota::NeedsSync,
101                // This variant only exists as a way of returning a quantity on the first
102                // `NeedsSync` decision. We can safely erase it into `NeedsSync` without a value.
103                CachedQuota::NeedsSyncWithQuantity(_) => return CachedQuota::NeedsSync,
104                CachedQuota::Active {
105                    consumed,
106                    local_use,
107                    expiry: _,
108                } => (*consumed, *local_use),
109            };
110
111            let total_local_use = local_use + quantity;
112
113            // Can short circuit here already if consumed is already above or equal to the limit.
114            //
115            // We could also propagate this out to the caller as a definitive negative in the
116            // future. This does require some additional consideration how this would interact with
117            // refunds, which can reduce the consumed.
118            if consumed >= quota.limit {
119                return CachedQuota::new_needs_sync(total_local_use);
120            }
121
122            let remaining = usize::try_from(quota.limit - consumed).unwrap_or(usize::MAX);
123            let max_allowed_spend = remaining
124                // Normalize the remaining quota with the window size, to apply the ratio/divisor to the
125                // per second rate.
126                //
127                // This means we get a consistent behaviour for short (10s) quotas (e.g. abuse) as well
128                // as long (1h) quotas (e.g. spike protection) with a more predictable error.
129                / usize::try_from(quota.window).unwrap_or(usize::MAX).max(1)
130                // Apply ratio precision, which is already pre-multiplied into `max_over_spend_divisor`.
131                * RATIO_PRECISION
132                // Apply the actual ratio with the pre-computed divisor.
133                / self.max_over_spend_divisor.get();
134
135            match total_local_use > max_allowed_spend {
136                true => CachedQuota::new_needs_sync(total_local_use),
137                false => CachedQuota::Active {
138                    consumed,
139                    local_use: total_local_use,
140                    expiry: quota.expiry,
141                },
142            }
143        });
144
145        match value {
146            Some(CachedQuota::NeedsSyncWithQuantity(q)) => Action::Check(q.get()),
147            Some(CachedQuota::NeedsSync) | None => Action::Check(quantity),
148            Some(CachedQuota::Active { .. }) => Action::Accept,
149        }
150    }
151
152    /// Updates the quota state in the cache for the specified quota.
153    ///
154    /// The cache will use the synchronized `consumed` value to derive future decisions whether it
155    /// can accept quota requests.
156    pub fn update_quota(&self, quota: Quota<T>, consumed: i64) {
157        let cache = self.cache.pin();
158
159        cache.update_or_insert(
160            quota.key,
161            |q| match q {
162                // We got the sync, start fresh!
163                CachedQuota::NeedsSync | CachedQuota::NeedsSyncWithQuantity(_) => {
164                    CachedQuota::Active {
165                        consumed,
166                        local_use: 0,
167                        expiry: quota.expiry,
168                    }
169                }
170                // This is slightly tricky, one could assume quotas are only growing and we should use
171                // the bigger `consumed` count here. The problem is refunds, a quota may decrease again
172                // due to refunds.
173                //
174                // Without a way to differentiate which `consumed` value is actually more recent, we
175                // just have the implicit assumption that the value arriving last is the most recent.
176                //
177                // This is not perfect, but for our best effort opportunistic cache, this is not a big
178                // problem, at worst, we over accept a little bit more for the next synchronization
179                // round.
180                CachedQuota::Active {
181                    consumed: _,
182                    local_use,
183                    expiry: _,
184                } => CachedQuota::Active {
185                    consumed,
186                    local_use: *local_use,
187                    expiry: quota.expiry,
188                },
189            },
190            CachedQuota::Active {
191                consumed,
192                local_use: 0,
193                expiry: quota.expiry,
194            },
195        );
196    }
197
198    /// Attempts to run a vacuum on the internal cache.
199    ///
200    /// The vacuum is internally debounced and may not run. Callers should call attempt to vacuum
201    /// periodically.
202    ///
203    /// Returns `true` if a vacuum was performed.
204    pub fn try_vacuum(&self, now: UnixTimestamp) -> bool {
205        // Check for the lucky winner to run an expiration.
206        let next_vacuum = self.next_vacuum.load(Ordering::Relaxed);
207
208        if next_vacuum > now.as_secs() {
209            // It's not yet time to vacuum.
210            return false;
211        }
212
213        let exchange = self.next_vacuum.compare_exchange(
214            next_vacuum,
215            now.as_secs() + self.vacuum_interval.as_secs(),
216            Ordering::Relaxed,
217            Ordering::Relaxed,
218        );
219
220        if exchange.is_err() {
221            // Somebody else took the vacuum, that's fine.
222            return false;
223        }
224
225        let mut cache = self.cache.pin();
226        cache.retain(|_, v| v.is_still_needed(now));
227
228        true
229    }
230}

232/// Action to take when accessing cached quotas via [`OpportunisticQuotaCache::check_quota`].
233#[derive(Debug, PartialEq, Eq)]
234pub enum Action {
235    /// Accept the quota request.
236    Accept,
237    /// Synchronize the quota with the returned quantity.
238    Check(usize),
239}
240
241/// State of a cached quota.
242#[derive(Debug)]
243enum CachedQuota {
244    /// The cache currently cannot make a decision and needs an updated consumption value from the
245    /// synchronized store.
246    NeedsSync,
247    /// Like [`Self::NeedsSync`], but also carries a total quantity which needs to be synchronized
248    /// with the store.
249    NeedsSyncWithQuantity(NonZeroUsize),
250    /// The cache is active and can still make decisions without a synchronization.
251    Active {
252        consumed: i64,
253        local_use: usize,
254        expiry: UnixTimestamp,
255    },
256}
257
258impl CachedQuota {
259    /// Creates [`Self::NeedsSync`] for a quantity of `0`, [`Self::NeedsSyncWithQuantity`] otherwise.
260    pub fn new_needs_sync(quantity: usize) -> Self {
261        NonZeroUsize::new(quantity)
262            .map(Self::NeedsSyncWithQuantity)
263            .unwrap_or(Self::NeedsSync)
264    }
265
266    /// Returns `true` when the entry is no longer needed in the cache.
267    ///
268    /// An item is no longer needed when it is expired or a sync is required.
269    fn is_still_needed(&self, now: UnixTimestamp) -> bool {
270        match self {
271            Self::NeedsSync => false,
272            Self::NeedsSyncWithQuantity(_) => false,
273            Self::Active { expiry, .. } => now <= *expiry,
274        }
275    }
276}

282    #[test]
283    fn test_opp_quota() {
284        let cache = OpportunisticQuotaCache::new(0.1);
285
286        let q1 = Quota {
287            limit: 100,
288            window: 1,
289            key: "k1",
290            expiry: UnixTimestamp::from_secs(300),
291        };
292        let q2 = Quota {
293            limit: 50,
294            window: 1,
295            key: "k2",
296            expiry: UnixTimestamp::from_secs(300),
297        };
298
299        // First access must always go to synchronized store.
300        assert_eq!(cache.check_quota(q1, 10), Action::Check(10));
301        // So does any following access until there was a sync.
302        assert_eq!(cache.check_quota(q1, 9), Action::Check(9));
303
304        // Same for other keys.
305        assert_eq!(cache.check_quota(q2, 12), Action::Check(12));
306
307        // Sync internal state to 30, for this test, there will be 70 remaining, meaning the cache
308        // is expected to accept 7 more items without requiring another sync.
309        cache.update_quota(q1, 30);
310
311        // A different key still needs a sync.
312        assert_eq!(cache.check_quota(q2, 12), Action::Check(12));
313
314        // 70 remaining, 10% -> 7 accepted.
315        assert_eq!(cache.check_quota(q1, 5), Action::Accept);
316        assert_eq!(cache.check_quota(q1, 2), Action::Accept);
317        // This one goes over the limit, we need to now check a total of 8.
318        assert_eq!(cache.check_quota(q1, 1), Action::Check(8));
319
320        // A new sync is required
321        assert_eq!(cache.check_quota(q1, 1), Action::Check(1));
322        assert_eq!(cache.check_quota(q1, 2), Action::Check(2));
323        // A different key is still a different key.
324        assert_eq!(cache.check_quota(q2, 3), Action::Check(3));
325
326        // Consumed state is absolute not relative.
327        cache.update_quota(q1, 50);
328        // This is too much.
329        assert_eq!(cache.check_quota(q2, 6), Action::Check(6));
330        // Need another sync again.
331        assert_eq!(cache.check_quota(q2, 1), Action::Check(1));
332
333        // Negative state can exist due to refunds.
334        cache.update_quota(q1, -100);
335        // We now have `200` remaining quota -> 20 (= 10%).
336        assert_eq!(cache.check_quota(q1, 20), Action::Accept);
337        // Too much, check the entire local usage.
338        assert_eq!(cache.check_quota(q1, 1), Action::Check(21));
339    }

342    fn test_opp_quota_normalized_to_window() {
343        let cache = OpportunisticQuotaCache::new(0.1);
344
345        let q1 = Quota {
346            limit: 1000,
347            window: 10,
348            key: "k1",
349            expiry: UnixTimestamp::from_secs(300),
350        };
351
352        // First access always needs synchronization.
353        assert_eq!(cache.check_quota(q1, 1), Action::Check(1));
354
355        // 700 remaining -> 70 per second -> 7 (10%).
356        cache.update_quota(q1, 300);
357
358        // 7 is the exact synchronization breakpoint.
359        assert_eq!(cache.check_quota(q1, 8), Action::Check(8));
360        // Under 7, but already over consumed before.
361        assert_eq!(cache.check_quota(q1, 6), Action::Check(6));
362
363        // Reset.
364        cache.update_quota(q1, 300);
365        // Under 7 -> that's fine.
366        assert_eq!(cache.check_quota(q1, 7), Action::Accept);
367        // Way over the limit now.
368        assert_eq!(cache.check_quota(q1, 90), Action::Check(97));
369
370        // 100 remaining -> 10 per second -> 1 (10%).
371        cache.update_quota(q1, 900);
372        assert_eq!(cache.check_quota(q1, 1), Action::Accept);
373        assert_eq!(cache.check_quota(q1, 1), Action::Check(2));
374
375        // 90 remaining -> 9 per second -> 0 (10% floored).
376        cache.update_quota(q1, 910);
377        assert_eq!(cache.check_quota(q1, 1), Action::Check(1));
378
379        // Same for even less remaining.
380        cache.update_quota(q1, 999);
381        assert_eq!(cache.check_quota(q1, 1), Action::Check(1));
382
383        // Same for nothing remaining.
384        cache.update_quota(q1, 1000);
385        assert_eq!(cache.check_quota(q1, 1), Action::Check(1));
386
387        // Same for less than nothing remaining.
388        cache.update_quota(q1, 1001);
389        assert_eq!(cache.check_quota(q1, 1), Action::Check(1));
390    }

394    fn test_opp_quota_limit_change() {
395        let cache = OpportunisticQuotaCache::new(0.1);
396        let window = 3;
397
398        let limit_100 = Quota {
399            limit: 100 * window,
400            window: window as u64,
401            key: "k1",
402            expiry: UnixTimestamp::from_secs(300),
403        };
404        let limit_50 = Quota {
405            // Same quota, but a different limit.
406            limit: 50 * window,
407            ..limit_100
408        };
409
410        // Sync internal state to an initial value.
411        cache.update_quota(limit_100, 50 * window);
412
413        // With limit 100 there is enough (5) in the cache remaining.
414        assert_eq!(cache.check_quota(limit_100, 3), Action::Accept);
415        // With limit 50 there is not enough in the cache remaining,  the over accepted amount needs
416        // to be checked though.
417        assert_eq!(cache.check_quota(limit_50, 1), Action::Check(4));

442    #[test]
443    fn test_opp_quota_vacuum() {
444        let cache = OpportunisticQuotaCache::new(0.1);
445
446        let q1 = Quota {
447            limit: 100,
448            window: 1,
449            key: "k1",
450            expiry: UnixTimestamp::from_secs(100),
451        };
452        let q2 = Quota {
453            limit: 100,
454            window: 1,
455            key: "k2",
456            expiry: UnixTimestamp::from_secs(200),
457        };
458
459        // Initialize the cache.
460        cache.update_quota(q1, 0);
461        cache.update_quota(q2, 0);
462
463        // Make sure some elements are stored in the cache.
464        assert_eq!(cache.check_quota(q1, 9), Action::Accept);
465        assert_eq!(cache.check_quota(q2, 9), Action::Accept);
466
467        // This vacuum should succeed, no concurrent vacuums.
468        assert!(cache.try_vacuum(UnixTimestamp::from_secs(150)));
469        // A vacuum was just ran, there is no need to run it again.
470        assert!(!cache.try_vacuum(UnixTimestamp::from_secs(150)));
471
472        // This entry was purged from the cache, now with the same quota, we should no longer have
473        // an old entry.
474        assert_eq!(cache.check_quota(q1, 6), Action::Check(6));
475        // This entry was not purged (different expiry) and should therefor return the full amount.
476        assert_eq!(cache.check_quota(q2, 6), Action::Check(15));
477    }
