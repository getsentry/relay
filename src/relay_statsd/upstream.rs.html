10pub enum Remote {
11    Udp(UdpSocket),
12    #[cfg(unix)]
13    UnixDatagram(UnixDatagram),
14}
15
16impl Remote {
17    fn connect(addr: &str) -> io::Result<Self> {
18        // Try treating the address as a fully-qualified URL, where the scheme is the transport identifier.
19        if let Some((scheme, path)) = addr.split_once("://") {
20            return match scheme {
21                "udp" => {
22                    let socket = UdpSocket::bind((Ipv4Addr::UNSPECIFIED, 0))?;
23                    socket.connect(path)?;
24                    socket.set_nonblocking(true)?;
25
26                    Ok(Self::Udp(socket))
27                }
28                #[cfg(unix)]
29                "unixgram" => {
30                    let socket = UnixDatagram::unbound()?;
31                    socket.connect(path)?;
32                    socket.set_nonblocking(true)?;
33
34                    Ok(Self::UnixDatagram(socket))
35                }
36                _ => Err(io::Error::other(format!(
37                    "invalid scheme '{scheme}', expected one of 'udp', 'unixgram'"
38                ))),
39            };
40        }
41
42        // If there is no scheme, fall back to a UDP socket
43        let socket = UdpSocket::bind((Ipv4Addr::UNSPECIFIED, 0))?;
44        socket.connect(addr)?;
45        socket.set_nonblocking(true)?;
46
47        Ok(Self::Udp(socket))
48    }
49
50    fn bufsize(&self) -> usize {
51        match self {
52            // The original statsdproxy uses.
53            Self::Udp(_) => 512,
54            #[cfg(unix)]
55            Self::UnixDatagram(_) => 1024,
56        }
57    }
58
59    fn send(&mut self, buf: &[u8]) {
60        let result = match self {
61            Self::Udp(socket) => socket.send(buf).map(drop),
62            #[cfg(unix)]
63            Self::UnixDatagram(socket) => socket.send(buf).map(drop),
64        };
65
66        if let Err(err) = result {
67            relay_log::warn!("failed to send metrics to upstream: {err}");
68        }
69    }
70}

72pub struct Upstream {
73    remote: Remote,
74    buffer: Vec<u8>,
75    buf_used: usize,
76    last_sent_at: SystemTime,
77}
78
79impl Upstream {
80    pub fn connect(upstream: &str, bufsize: Option<usize>) -> io::Result<Self> {
81        let remote = Remote::connect(upstream)?;
82
83        Ok(Upstream {
84            buffer: vec![0; bufsize.unwrap_or_else(|| remote.bufsize())],
85            remote,
86            buf_used: 0,
87            last_sent_at: UNIX_EPOCH,
88        })
89    }
90
91    fn flush(&mut self) {
92        if self.buf_used > 0 {
93            self.remote.send(&self.buffer[..self.buf_used]);
94            self.buf_used = 0;
95        }
96        self.last_sent_at = SystemTime::now(); // Annoyingly superfluous call to now().
97    }
98
99    fn timed_flush(&mut self) {
100        let now = SystemTime::now();
101        if now
102            .duration_since(self.last_sent_at)
103            .map_or(true, |x| x > Duration::from_secs(1))
104        {
105            // We have not sent any metrics in a while. Flush the buffer.
106            self.flush();
107        }
108    }
109}
110
111impl Drop for Upstream {
112    fn drop(&mut self) {
113        self.flush();
114    }
115}
116
117impl Middleware for Upstream {
118    fn submit(&mut self, metric: &mut Metric) {
119        let metric_len = metric.raw.len();
120        if metric_len + 1 > self.buffer.len() - self.buf_used {
121            // Message bigger than space left in buffer. Flush the buffer.
122            self.flush();
123        }
124        if metric_len > self.buffer.len() {
125            // Message too big for the entire buffer, send it and pray.
126            self.remote.send(&metric.raw);
127        } else {
128            // Put the message in the buffer, separating it from the previous message if any.
129            if self.buf_used > 0 {
130                self.buffer[self.buf_used] = b'\n';
131                self.buf_used += 1;
132            }
133            self.buffer[self.buf_used..self.buf_used + metric_len].copy_from_slice(&metric.raw);
134            self.buf_used += metric_len;
135        }
136        // poll gets called before submit, so if the buffer needed to be flushed for time reasons,
137        // it already was.
138    }
139
140    fn poll(&mut self) {
141        self.timed_flush();
142    }
143}

145/// A [`Upstream`] which falls back to noop operations if connecting to the statsd sink fails.
146pub enum TryUpstream {
147    Upstream(Upstream),
148    Error,
149}
150
151impl TryUpstream {
152    pub fn connect(upstream: &str, bufsize: Option<usize>) -> Self {
153        match Upstream::connect(upstream, bufsize) {
154            Ok(upstream) => Self::Upstream(upstream),
155            Err(err) => {
156                relay_log::error!(
157                    error = &err as &dyn std::error::Error,
158                    "failed to connect to statsd sink at {upstream}"
159                );
160                Self::Error
161            }
162        }
163    }
164}
165
166impl Middleware for TryUpstream {
167    fn submit(&mut self, metric: &mut Metric) {
168        match self {
169            Self::Upstream(upstream) => upstream.submit(metric),
170            Self::Error => {}
171        }
172    }
173
174    fn poll(&mut self) {
175        match self {
176            Self::Upstream(upstream) => upstream.poll(),
177            Self::Error => {}
178        }
179    }
