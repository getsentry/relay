14/// Default name of the pool.
15const DEFAULT_POOL_NAME: &str = "unnamed";

17/// [`AsyncPool`] is a thread-based executor that runs asynchronous tasks on dedicated worker threads.
18///
19/// The pool collects tasks through a bounded channel and distributes them among threads, each of which runs its own
20/// Tokio executor. This design enables controlled concurrency and efficient use of system resources.
21#[derive(Debug)]
22pub struct AsyncPool<F> {
23    /// Name of the pool.
24    name: &'static str,
25    /// Transmission containing all tasks.
26    tx: flume::Sender<F>,
27    /// The maximum number of tasks that are expected to run concurrently at any point in time.
28    max_tasks: u64,
29    /// Vector containing all the metrics collected individually in each thread.
30    threads_metrics: Arc<Vec<Arc<ThreadMetrics>>>,
31}
32
33impl<F> AsyncPool<F> {
34    /// Returns the `name` of the [`AsyncPool`].
35    pub fn name(&self) -> &'static str {
36        self.name
37    }
38
39    /// Returns the [`AsyncPoolMetrics`] that are updated by the pool.
40    pub fn metrics(&self) -> AsyncPoolMetrics {
41        AsyncPoolMetrics {
42            max_tasks: self.max_tasks,
43            queue_size: self.tx.len() as u64,
44            threads_metrics: &self.threads_metrics,
45        }
46    }
47}

49impl<F> AsyncPool<F>
50where
51    F: Future<Output = ()> + Send + 'static,
52{
53    /// Creates a new [`AsyncPool`] based on the configuration specified by [`AsyncPoolBuilder`].
54    ///
55    /// This method initializes the dedicated worker threads and configures each executor with the defined
56    /// concurrency limits.
57    pub fn new<S>(mut builder: AsyncPoolBuilder<S>) -> io::Result<Self>
58    where
59        S: ThreadSpawn,
60    {
61        let pool_name = builder.pool_name.unwrap_or(DEFAULT_POOL_NAME);
62        let (tx, rx) = flume::bounded(builder.num_threads * 2);
63        let mut threads_metrics = Vec::with_capacity(builder.num_threads);
64
65        for thread_id in 0..builder.num_threads {
66            let rx = rx.clone();
67            let thread_name: Option<String> = builder.thread_name.as_mut().map(|f| f(thread_id));
68
69            let metrics = Arc::new(ThreadMetrics::default());
70            let thread = Thread {
71                id: thread_id,
72                max_concurrency: builder.max_concurrency,
73                name: thread_name.clone(),
74                runtime: builder.runtime.clone(),
75                panic_handler: builder.thread_panic_handler.clone(),
76                task: Multiplexed::new(
77                    pool_name,
78                    builder.max_concurrency,
79                    rx.into_stream(),
80                    builder.task_panic_handler.clone(),
81                    metrics.clone(),
82                )
83                .boxed(),
84            };
85
86            threads_metrics.push(metrics);
87
88            builder.spawn_handler.spawn(thread)?;
89        }
90
91        Ok(Self {
92            name: pool_name,
93            tx,
94            max_tasks: (builder.num_threads * builder.max_concurrency) as u64,
95            threads_metrics: Arc::new(threads_metrics),
96        })
97    }
98
99    /// Schedules a future for execution within the [`AsyncPool`].
100    ///
101    /// The task is added to the pool's internal queue to be executed by an available worker thread.
102    ///
103    /// # Panics
104    ///
105    /// This method panics if all receivers have been dropped which can happen when all threads of
106    /// the pool panicked.
107    pub fn spawn(&self, future: F) {
108        assert!(
109            self.tx.send(future).is_ok(),
110            "failed to schedule task: all worker threads have terminated (either none were spawned or all have panicked)"
111        );
112    }
113
114    /// Asynchronously enqueues a future for execution within the [`AsyncPool`].
115    ///
116    /// This method awaits until the task is successfully added to the internal queue.
117    ///
118    /// # Panics
119    ///
120    /// This method panics if all receivers have been dropped which can happen when all threads of
121    /// the pool panicked.
122    pub async fn spawn_async(&self, future: F) {
123        assert!(
124            self.tx.send_async(future).await.is_ok(),
125            "failed to schedule task: all worker threads have terminated (either none were spawned or all have panicked)"
126        );
127    }
128}

141/// [`Thread`] represents a dedicated worker thread within an [`AsyncPool`] that executes scheduled tasks.
142pub struct Thread {
143    id: usize,
144    max_concurrency: usize,
145    name: Option<String>,
146    runtime: tokio::runtime::Handle,
147    panic_handler: Option<Arc<PanicHandler>>,
148    task: BoxFuture<'static, ()>,
149}
150
151impl Thread {
152    /// Returns the unique index assigned to this [`Thread`].
153    ///
154    /// The index can help identify the thread during debugging or logging.
155    pub fn id(&self) -> usize {
156        self.id
157    }
158
159    /// Returns the maximum number of concurrent tasks permitted on this [`Thread`].
160    ///
161    /// This reflects the concurrency limit configured via the [`AsyncPoolBuilder`].
162    pub fn max_concurrency(&self) -> usize {
163        self.max_concurrency
164    }
165
166    /// Returns the human-readable name of this [`Thread`], if one was set.
167    ///
168    /// Thread names can assist in monitoring and debugging the execution environment.
169    pub fn name(&self) -> Option<&str> {
170        self.name.as_deref()
171    }
172}
173
174impl Thread {
175    /// Runs the task multiplexer associated with this [`Thread`].
176    ///
177    /// This method drives the execution of tasks on the worker thread.
178    ///
179    /// # Panics
180    ///
181    /// Panics are either handled by the custom handler or propagated if no handler is specified.
182    pub fn run(self) {
183        let result =
184            std::panic::catch_unwind(AssertUnwindSafe(|| self.runtime.block_on(self.task)));
185
186        match (self.panic_handler, result) {
187            // Panic handler and error, we swallow the panic and invoke the callback.
188            (Some(panic_handler), Err(error)) => {
189                panic_handler(error);
190            }
191            // No panic handler and error, we propagate the panic.
192            (None, Err(error)) => {
193                std::panic::resume_unwind(error);
194            }
195            // Otherwise, we do nothing.
196            (_, Ok(())) => {}
197        }
198    }
199}

201/// [`ThreadSpawn`] defines how threads are spawned in an [`AsyncPool`].
202///
203/// This trait allows customization of thread creation (for example, setting names or adjusting stack sizes)
204/// without altering the core functionality of the pool.
205pub trait ThreadSpawn {
206    /// Spawns a new thread using the provided configuration.
207    fn spawn(&mut self, thread: Thread) -> io::Result<()>;
208}
209
210/// [`DefaultSpawn`] is the default implementation of [`ThreadSpawn`] that delegates to the system's
211/// standard thread creation mechanism.
212///
213/// It applies any provided thread name using the standard thread builder.
214#[derive(Clone)]
215pub struct DefaultSpawn;
216
217impl ThreadSpawn for DefaultSpawn {
218    fn spawn(&mut self, thread: Thread) -> io::Result<()> {
219        let mut b = std::thread::Builder::new();
220        if let Some(name) = thread.name() {
221            b = b.name(name.to_owned());
222        }
223        b.spawn(|| thread.run())?;
224
225        Ok(())
226    }
227}

229/// [`CustomSpawn`] is an alternative implementation of [`ThreadSpawn`] that uses a user-supplied closure
230/// for custom thread configuration.
231///
232/// This allows for fine-grained control over thread properties, enabling application-specific setups.
233#[derive(Clone)]
234pub struct CustomSpawn<B>(B);
235
236impl<B> CustomSpawn<B> {
237    /// Creates a new instance of [`CustomSpawn`] with the specified configuration closure.
238    pub fn new(spawn_handler: B) -> Self {
239        CustomSpawn(spawn_handler)
240    }
241}
242
243impl<B> ThreadSpawn for CustomSpawn<B>
244where
245    B: FnMut(Thread) -> io::Result<()>,
246{
247    /// Applies the custom configuration closure when spawning a new thread.
248    fn spawn(&mut self, thread: Thread) -> io::Result<()> {
249        self.0(thread)
250    }
251}

309    async fn test_async_pool_executes_all_tasks() {
310        let pool = AsyncPoolBuilder::new(Handle::current())
311            .num_threads(1)
312            .max_concurrency(2)
313            .build()
314            .unwrap();
315        let counter = Arc::new(AtomicUsize::new(0));
316        let barrier = TestBarrier::new(20).await;
317
318        // Spawn 20 tasks that wait briefly and then update the counter.
319        for _ in 0..20 {
320            let counter_clone = counter.clone();
321            barrier
322                .spawn(&pool, move || async move {
323                    sleep(Duration::from_millis(50)).await;
324                    counter_clone.fetch_add(1, Ordering::SeqCst);
325                })
326                .await;
327        }
328
329        barrier.wait().await;
330        assert_eq!(counter.load(Ordering::SeqCst), 20);
331    }
