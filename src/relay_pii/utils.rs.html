12pub fn process_pairlist<P: Processor, T: ProcessValue + AsPair>(
13    slf: &mut P,
14    value: &mut PairList<T>,
15    state: &ProcessingState,
16) -> ProcessingResult {
17    // View pairlists as objects just for the purpose of PII stripping (e.g. `event.tags.mykey`
18    // instead of `event.tags.42.0`). For other purposes such as trimming we would run into
19    // problems:
20    //
21    // * tag keys need to be trimmed too and therefore need to have a path
22
23    for (idx, annotated) in value.iter_mut().enumerate() {
24        if let Some(pair) = annotated.value_mut() {
25            let (key, value) = pair.as_pair_mut();
26            let value_type = ValueType::for_field(value);
27
28            if let Some(key_name) = key.as_str() {
29                // if the pair has no key name, we skip over it for PII stripping. It is
30                // still processed with index-based path in the invocation of
31                // `process_child_values`.
32                let entered = state.enter_borrowed(key_name, state.inner_attrs(), value_type);
33                processor::process_value(value, slf, &entered)?;
34            } else {
35                let entered = state.enter_index(idx, state.inner_attrs(), value_type);
36                processor::process_value(value, slf, &entered)?;
37            }
38        }
39    }
40
41    Ok(())
42}

44pub fn process_attributes<P: Processor>(
45    value: &mut Attributes,
46    slf: &mut P,
47    state: &ProcessingState,
48) -> ProcessingResult {
49    // Check if we're in default rules (no root ValueType)
50    // This is a workaround to support explicit selectors eg. $log.attributes.KEY.value to work but keep implicit selectors for default rules.
51    let is_advanced_rules = state
52        .iter()
53        .nth(1)
54        .is_some_and(|s| s.value_type().contains(ValueType::OurLog));
55
56    for (key, annotated_attribute) in value.iter_mut() {
57        if let Some(attribute) = annotated_attribute.value_mut() {
58            if is_advanced_rules {
59                // Process normally to allow explicit selectors like $log.attributes.KEY.value to work
60                let field_value_type = ValueType::for_field(annotated_attribute);
61                let key_state = state.enter_borrowed(key, state.inner_attrs(), field_value_type);
62                processor::process_value(annotated_attribute, slf, &key_state)?;
63            } else {
64                // For the default rules, we want Pii::True since we want them to always be scrubbed.
65                let attrs = FieldAttrs::new().pii(Pii::True);
66                let inner_value = &mut attribute.value.value;
67                let inner_value_type = ValueType::for_field(inner_value);
68                let entered =
69                    state.enter_borrowed(key, Some(Cow::Borrowed(&attrs)), inner_value_type);
70                processor::process_value(inner_value, slf, &entered)?;
71
72                let object_value_type = enum_set!(ValueType::Object);
73                for (key, value) in attribute.other.iter_mut() {
74                    let other_state =
75                        state.enter_borrowed(key, Some(Cow::Borrowed(&attrs)), object_value_type);
76                    processor::process_value(value, slf, &other_state)?;
77                }
78            }
79        }
80    }
81    Ok(())
82}

84pub fn hash_value(data: &[u8]) -> String {
85    let mut mac = Hmac::<Sha1>::new_from_slice(&[]).unwrap();
86    mac.update(data);
87    format!("{:X}", mac.finalize().into_bytes())
