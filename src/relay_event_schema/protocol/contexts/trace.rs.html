14/// Represents a W3C Trace Context `trace-id`.
15///
16/// The `trace-id` is a globally unique identifier for a distributed trace,
17/// used to correlate requests across service boundaries.
18///
19/// Format:
20/// - 16-byte array (128 bits), represented as 32-character hexadecimal string
21/// - Example: `"4bf92f3577b34da6a3ce929d0e0e4736"`
22/// - MUST NOT be all zeros (`"00000000000000000000000000000000"`)
23/// - MUST contain only hex digits (`0-9`, `a-f`, `A-F`)
24///
25/// Our implementation allows uppercase hexadecimal characters for backward compatibility, even
26/// though the original spec only allows lowercase hexadecimal characters.
27///
28/// See: <https://www.w3.org/TR/trace-context/#trace-id>
29#[derive(Clone, Copy, PartialEq, Empty, ProcessValue)]
30pub struct TraceId(Uuid);
31
32impl TraceId {
33    /// Creates a new, random, trace id.
34    pub fn random() -> Self {
35        Self(Uuid::new_v4())
36    }
37
38    /// Parses a [`TraceId`] from a value that can be converted into one.
39    ///
40    /// If parsing fails, uses a [`Self::random`] id instead and adds a remark.
41    pub fn try_from_or_random<T>(value: T) -> Annotated<Self>
42    where
43        T: TryInto<Self> + AsRef<[u8]> + Copy,
44    {
45        value.try_into().map(Annotated::new).unwrap_or_else(|_| {
46            let mut meta = Meta::default();
47            let rule_id = match value.as_ref().is_empty() {
48                true => "trace_id.missing",
49                false => "trace_id.invalid",
50            };
51            meta.add_remark(Remark::new(RemarkType::Substituted, rule_id));
52            Annotated(Some(TraceId::random()), meta)
53        })
54    }
55
56    /// Parses a [`TraceId`] from a slice, if it fails uses a [`Self::random`] id instead.
57    pub fn try_from_slice_or_random(value: &[u8]) -> Annotated<Self> {
58        Self::try_from_or_random(value)
59    }
60
61    /// Parses a [`TraceId`] from a string, if it fails uses a [`Self::random`] id instead.
62    pub fn try_from_str_or_random(value: &str) -> Annotated<Self> {
63        Self::try_from_or_random(value)
64    }
65}

167pub struct SpanId([u8; 8]);

169relay_common::impl_str_serde!(SpanId, "a span identifier");
170
171impl FromStr for SpanId {
172    type Err = Error;
173
174    fn from_str(s: &str) -> Result<Self, Self::Err> {
175        match u64::from_str_radix(s, 16) {
176            Ok(id) if s.len() == 16 && id > 0 => Ok(Self(id.to_be_bytes())),
177            _ => Err(Error::invalid("not a valid span id")),
178        }
179    }
180}
181
182impl TryFrom<&[u8]> for SpanId {
183    type Error = Error;
184
185    fn try_from(value: &[u8]) -> Result<Self, Self::Error> {
186        match <[u8; 8]>::try_from(value) {
187            Ok(bytes) if !bytes.iter().all(|&x| x == 0) => Ok(Self(bytes)),
188            _ => Err(Error::invalid("not a valid span id")),
189        }
190    }
191}
192
193impl fmt::Debug for SpanId {
194    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
195        write!(f, "SpanId(\"")?;
196        for b in self.0 {
197            write!(f, "{b:02x}")?;
198        }
199        write!(f, "\")")
200    }
201}

272/// Trace context
273#[derive(Clone, Debug, Default, PartialEq, Empty, FromValue, IntoValue, ProcessValue)]
274#[metastructure(process_func = "process_trace_context")]
275pub struct TraceContext {
276    /// The trace ID.
277    #[metastructure(required = true)]
278    pub trace_id: Annotated<TraceId>,
279
280    /// The ID of the span.
281    #[metastructure(required = true)]
282    pub span_id: Annotated<SpanId>,
283
284    /// The ID of the span enclosing this span.
285    pub parent_span_id: Annotated<SpanId>,
286
287    /// Span type (see `OperationType` docs).
288    #[metastructure(max_chars = 128)]
289    pub op: Annotated<OperationType>,
290
291    /// Whether the trace failed or succeeded. Currently only used to indicate status of individual
292    /// transactions.
293    pub status: Annotated<SpanStatus>,
294
295    /// The amount of time in milliseconds spent in this transaction span,
296    /// excluding its immediate child spans.
297    pub exclusive_time: Annotated<f64>,
298
299    /// The client-side sample rate as reported in the envelope's `trace.sample_rate` header.
300    ///
301    /// The server takes this field from envelope headers and writes it back into the event. Clients
302    /// should not ever send this value.
303    pub client_sample_rate: Annotated<f64>,
304
305    /// The origin of the trace indicates what created the trace (see [OriginType] docs).
306    #[metastructure(max_chars = 128, allow_chars = "a-zA-Z0-9_.")]
307    pub origin: Annotated<OriginType>,
308
309    /// Track whether the trace connected to this event has been sampled entirely.
310    ///
311    /// This flag only applies to events with [`Error`] type that have an associated dynamic sampling context.
312    pub sampled: Annotated<bool>,
313
314    /// Data of the trace's root span.
315    #[metastructure(pii = "maybe", skip_serialization = "null")]
316    pub data: Annotated<SpanData>,
317
318    /// Links to other spans from the trace's root span.
319    #[metastructure(pii = "maybe", skip_serialization = "null")]
320    pub links: Annotated<Array<SpanLink>>,
321
322    /// Additional arbitrary fields for forwards compatibility.
323    #[metastructure(additional_properties, retain = true, pii = "maybe")]
324    pub other: Object<Value>,
325}
326

328    fn default_key() -> &'static str {
329        "trace"
330    }
331
332    fn from_context(context: super::Context) -> Option<Self> {
333        match context {
334            super::Context::Trace(c) => Some(*c),
335            _ => None,
336        }
337    }
338
339    fn cast(context: &super::Context) -> Option<&Self> {
340        match context {
341            super::Context::Trace(c) => Some(c),
342            _ => None,
343        }
344    }
345
346    fn cast_mut(context: &mut super::Context) -> Option<&mut Self> {
347        match context {
348            super::Context::Trace(c) => Some(c),
349            _ => None,
350        }
351    }
352
353    fn into_context(self) -> super::Context {
354        super::Context::Trace(Box::new(self))
355    }

577    #[test]
578    fn test_try_from_or_random() {
579        // Test valid string
580        let valid_str = "4c79f60c11214eb38604f4ae0781bfb2";
581        let annotated = TraceId::try_from_str_or_random(valid_str);
582        assert_eq!(
583            annotated.value().unwrap().as_u128(),
584            0x4c79f60c11214eb38604f4ae0781bfb2
585        );
586        assert!(annotated.meta().is_empty());
587
588        // Test invalid string (should return random + remark)
589        let invalid_str = "invalid";
590        let annotated = TraceId::try_from_str_or_random(invalid_str);
591        assert!(annotated.value().is_some()); // Random trace ID
592        assert_ne!(annotated.value().unwrap().as_u128(), 0);
593        assert_eq!(annotated.meta().iter_remarks().count(), 1);
594        let remark = annotated.meta().iter_remarks().next().unwrap();
595        assert_eq!(remark.rule_id(), "trace_id.invalid");
596
597        // Test empty string (should return random + remark)
598        let empty_str = "";
599        let annotated = TraceId::try_from_str_or_random(empty_str);
600        assert!(annotated.value().is_some());
601        let remark = annotated.meta().iter_remarks().next().unwrap();
602        assert_eq!(remark.rule_id(), "trace_id.missing");
603
604        // Test valid slice
605        let valid_bytes = b"\x4c\x79\xf6\x0c\x11\x21\x4e\xb3\x86\x04\xf4\xae\x07\x81\xbf\xb2";
606        let annotated = TraceId::try_from_slice_or_random(valid_bytes.as_slice());
607        assert_eq!(
608            annotated.value().unwrap().as_u128(),
609            0x4c79f60c11214eb38604f4ae0781bfb2
610        );
611
612        // Test invalid slice length
613        let invalid_bytes = b"\x00";
614        let annotated = TraceId::try_from_slice_or_random(invalid_bytes.as_slice());
615        let remark = annotated.meta().iter_remarks().next().unwrap();
616        assert_eq!(remark.rule_id(), "trace_id.invalid");
617    }
