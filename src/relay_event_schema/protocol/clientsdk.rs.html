15/// The SDK Interface describes the Sentry SDK and its configuration used to capture and transmit an event.
16#[derive(Clone, Debug, Default, PartialEq, Empty, FromValue, IntoValue, ProcessValue)]
17#[metastructure(process_func = "process_client_sdk_info", value_type = "ClientSdkInfo")]
18pub struct ClientSdkInfo {
19    /// Unique SDK name. _Required._
20    ///
21    /// The name of the SDK. The format is `entity.ecosystem[.flavor]` where entity identifies the
22    /// developer of the SDK, ecosystem refers to the programming language or platform where the
23    /// SDK is to be used and the optional flavor is used to identify standalone SDKs that are part
24    /// of a major ecosystem.
25    ///
26    /// Official Sentry SDKs use the entity `sentry`, as in `sentry.python` or
27    /// `sentry.javascript.react-native`. Please use a different entity for your own SDKs.
28    #[metastructure(required = true, max_chars = 256, max_chars_allowance = 20)]
29    pub name: Annotated<String>,
30
31    /// The version of the SDK. _Required._
32    ///
33    /// It should have the [Semantic Versioning](https://semver.org/) format `MAJOR.MINOR.PATCH`,
34    /// without any prefix (no `v` or anything else in front of the major version number).
35    ///
36    /// Examples: `0.1.0`, `1.0.0`, `4.3.12`
37    #[metastructure(required = true, max_chars = 256, max_chars_allowance = 20)]
38    pub version: Annotated<String>,
39
40    /// List of integrations that are enabled in the SDK. _Optional._
41    ///
42    /// The list should have all enabled integrations, including default integrations. Default
43    /// integrations are included because different SDK releases may contain different default
44    /// integrations.
45    #[metastructure(skip_serialization = "empty_deep")]
46    pub integrations: Annotated<Array<String>>,
47
48    /// List of features that are enabled in the SDK. _Optional._
49    ///
50    /// A list of feature names identifying enabled SDK features. This list
51    /// should contain all enabled SDK features. On some SDKs, enabling a feature in the
52    /// options also adds an integration. We encourage tracking such features with either
53    /// integrations or features but not both to reduce the payload size.
54    #[metastructure(skip_serialization = "empty_deep")]
55    pub features: Annotated<Array<String>>,
56
57    /// List of installed and loaded SDK packages. _Optional._
58    ///
59    /// A list of packages that were installed as part of this SDK or the activated integrations.
60    /// Each package consists of a name in the format `source:identifier` and `version`. If the
61    /// source is a Git repository, the `source` should be `git`, the identifier should be a
62    /// checkout link and the version should be a Git reference (branch, tag or SHA).
63    #[metastructure(skip_serialization = "empty_deep")]
64    pub packages: Annotated<Array<ClientSdkPackage>>,
65
66    /// IP Address of sender??? Seems unused. Do not send, this only leads to surprises wrt PII, as
67    /// the value appears nowhere in the UI.
68    #[metastructure(pii = "true", skip_serialization = "empty", omit_from_schema)]
69    pub client_ip: Annotated<IpAddr>,
70
71    /// Additional arbitrary fields for forwards compatibility.
72    #[metastructure(additional_properties)]
73    pub other: Object<Value>,
74}
75
76impl ClientSdkInfo {
77    pub fn has_integration<T: AsRef<str>>(&self, integration: T) -> bool {
78        if let Some(integrations) = self.integrations.value() {
79            for x in integrations {
80                if x.as_str().unwrap_or_default() == integration.as_ref() {
81                    return true;
82                }
83            }
84        };
85        false
86    }
87}

96    #[test]
97    fn test_client_sdk_roundtrip() {
98        let json = r#"{
99  "name": "sentry.rust",
100  "version": "1.0.0",
101  "integrations": [
102    "actix"
103  ],
104  "features": [
105    "feature1"
106  ],
107  "packages": [
108    {
109      "name": "cargo:sentry",
110      "version": "0.10.0"
111    },
112    {
113      "name": "cargo:sentry-actix",
114      "version": "0.10.0"
115    }
116  ],
117  "client_ip": "127.0.0.1",
118  "other": "value"
119}"#;
120        let sdk = Annotated::new(ClientSdkInfo {
121            name: Annotated::new("sentry.rust".to_string()),
122            version: Annotated::new("1.0.0".to_string()),
123            integrations: Annotated::new(vec![Annotated::new("actix".to_string())]),
124            features: Annotated::new(vec![Annotated::new("feature1".to_string())]),
125            packages: Annotated::new(vec![
126                Annotated::new(ClientSdkPackage {
127                    name: Annotated::new("cargo:sentry".to_string()),
128                    version: Annotated::new("0.10.0".to_string()),
129                }),
130                Annotated::new(ClientSdkPackage {
131                    name: Annotated::new("cargo:sentry-actix".to_string()),
132                    version: Annotated::new("0.10.0".to_string()),
133                }),
134            ]),
135            client_ip: Annotated::new(IpAddr("127.0.0.1".to_owned())),
136            other: {
137                let mut map = Map::new();
138                map.insert(
139                    "other".to_string(),
140                    Annotated::new(Value::String("value".to_string())),
141                );
142                map
143            },
144        });
145
146        assert_eq!(sdk, Annotated::from_json(json).unwrap());
147        assert_eq!(json, sdk.to_json_pretty().unwrap());
148    }
