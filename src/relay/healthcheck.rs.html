1use std::net::SocketAddr;
2use std::time::Duration;
3
4use anyhow::{Result, format_err};
5use clap::ArgMatches;
6use relay_config::Config;
7use reqwest::blocking::Client;
8
9pub fn healthcheck(config: &Config, matches: &ArgMatches) -> Result<()> {
10    let mode = matches
11        .get_one::<String>("mode")
12        .expect("`mode` is required");
13
14    let timeout = matches
15        .get_one::<u64>("timeout")
16        .expect("`timeout` is required");
17
18    let addr = matches
19        .get_one::<SocketAddr>("addr")
20        .copied()
21        .or_else(|| config.listen_addr_internal())
22        .unwrap_or(config.listen_addr());
23
24    let client = Client::builder()
25        .timeout(Some(Duration::from_secs(*timeout)))
26        .build()
27        .unwrap_or_default();
28
29    let response = client
30        .get(format!("http://{addr}/api/relay/healthcheck/{mode}/"))
31        .send();
32
33    match response {
34        Ok(response) => {
35            if response.status().is_success() {
36                Ok(())
37            } else {
38                relay_log::error!("Relay is unhealthy. Status code: {}", response.status());
39                Err(format_err!(
40                    "Relay is unhealthy. Status code: {}",
41                    response.status()
42                ))
43            }
44        }
45        Err(err) => {
46            relay_log::error!("Relay is unhealthy. Error: {err}");
47            Err(err.into())
48        }
49    }
50}
