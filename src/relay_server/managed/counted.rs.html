11/// A list of data categories and amounts.
12pub type Quantities = SmallVec<[(DataCategory, usize); 1]>;
13
14/// A counted item.
15///
16/// An item may represent multiple categories with different counts at once.
17pub trait Counted {
18    /// Returns the contained item quantities.
19    ///
20    /// Implementation are expected to be pure.
21    fn quantities(&self) -> Quantities;
22}
23
24impl Counted for () {
25    fn quantities(&self) -> Quantities {
26        Quantities::new()
27    }
28}
29
30impl Counted for Item {
31    fn quantities(&self) -> Quantities {
32        self.quantities()
33    }
34}
35
36impl Counted for Box<Envelope> {
37    fn quantities(&self) -> Quantities {
38        let mut quantities = Quantities::new();
39
40        // This matches the implementation of `ManagedEnvelope::reject`.
41        let summary = EnvelopeSummary::compute(self);
42        if let Some(category) = summary.event_category {
43            quantities.push((category, 1));
44            if let Some(category) = category.index_category() {
45                quantities.push((category, 1));
46            }
47        }
48
49        let data = [
50            (DataCategory::Attachment, summary.attachment_quantity),
51            (DataCategory::Profile, summary.profile_quantity),
52            (DataCategory::ProfileIndexed, summary.profile_quantity),
53            (DataCategory::Span, summary.span_quantity),
54            (DataCategory::SpanIndexed, summary.span_quantity),
55            (
56                DataCategory::Transaction,
57                summary.secondary_transaction_quantity,
58            ),
59            (DataCategory::Span, summary.secondary_span_quantity),
60            (DataCategory::Replay, summary.replay_quantity),
61            (DataCategory::ProfileChunk, summary.profile_chunk_quantity),
62            (
63                DataCategory::ProfileChunkUi,
64                summary.profile_chunk_ui_quantity,
65            ),
66            (DataCategory::LogItem, summary.log_item_quantity),
67            (DataCategory::LogByte, summary.log_byte_quantity),
68            (DataCategory::Monitor, summary.monitor_quantity),
69        ];
70
71        for (category, quantity) in data {
72            if quantity > 0 {
73                quantities.push((category, quantity));
74            }
75        }
76
77        quantities
78    }
79}

105impl Counted for ExtractedMetrics {
106    fn quantities(&self) -> Quantities {
107        // We only consider project metrics, sampling project metrics should never carry outcomes,
108        // as they would be for a *different* project.
109        let SourceQuantities {
110            transactions,
111            spans,
112            profiles,
113            buckets,
114        } = metrics::extract_quantities(&self.project_metrics);
115
116        [
117            (DataCategory::Transaction, transactions),
118            (DataCategory::Span, spans),
119            (DataCategory::Profile, profiles),
120            (DataCategory::MetricBucket, buckets),
121        ]
122        .into_iter()
123        .filter(|(_, q)| *q > 0)
124        .collect()
125    }
126}
