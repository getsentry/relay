15/// Type used for encoding string lengths.
16type Len = u32;
17
18/// Serializes a Pascal-style string with a 4 byte little-endian length prefix.
19fn write_string<W>(mut writer: W, string: &str) -> io::Result<()>
20where
21    W: io::Write,
22{
23    writer.write_all(&(string.len() as Len).to_le_bytes())?;
24    writer.write_all(string.as_bytes())?;
25
26    Ok(())
27}

29/// Safely consumes a slice of the given length.
30fn split_front<'a>(data: &mut &'a [u8], len: usize) -> Option<&'a [u8]> {
31    if data.len() < len {
32        *data = &[];
33        return None;
34    }
35
36    let (slice, rest) = data.split_at(len);
37    *data = rest;
38    Some(slice)
39}
40
41/// Consumes the 4-byte length prefix of a string.
42fn consume_len(data: &mut &[u8]) -> Option<usize> {
43    let len = std::mem::size_of::<Len>();
44    let slice = split_front(data, len)?;
45    let bytes = slice.try_into().ok();
46    bytes.map(|b| Len::from_le_bytes(b) as usize)
47}
48
49/// Consumes a Pascal-style string with a 4 byte little-endian length prefix.
50fn consume_string<'a>(data: &mut &'a [u8]) -> Option<&'a str> {
51    let len = consume_len(data)?;
52    let bytes = split_front(data, len)?;
53    std::str::from_utf8(bytes).ok()
54}

56/// An entry in a serialized form data item.
57#[derive(Clone, Debug, Eq, PartialEq, Deserialize, Serialize)]
58pub struct FormDataEntry<'a>(&'a str, &'a str);
59
60impl<'a> FormDataEntry<'a> {
61    pub fn new(key: &'a str, value: &'a str) -> Self {
62        Self(key, value)
63    }
64
65    pub fn key(&self) -> &'a str {
66        self.0
67    }
68
69    pub fn value(&self) -> &'a str {
70        self.1
71    }
72
73    fn to_writer<W: io::Write>(&self, mut writer: W) {
74        write_string(&mut writer, self.key()).ok();
75        write_string(&mut writer, self.value()).ok();
76    }
77
78    fn read(data: &mut &'a [u8]) -> Option<Self> {
79        let key = consume_string(data)?;
80        let value = consume_string(data)?;
81        Some(Self::new(key, value))
82    }
83}

85/// A writer for serialized form data.
86///
87/// This writer is used to serialize multiple plain fields from a multipart form data request into a
88/// single envelope item. Use `FormDataIter` to iterate all entries.
89struct FormDataWriter {
90    data: Vec<u8>,
91}
92
93impl FormDataWriter {
94    pub fn new() -> Self {
95        Self { data: Vec::new() }
96    }
97
98    pub fn append(&mut self, key: &str, value: &str) {
99        let entry = FormDataEntry::new(key, value);
100        entry.to_writer(&mut self.data);
101    }
102
103    pub fn into_inner(self) -> Vec<u8> {
104        self.data
105    }
106}

108/// Iterates through serialized form data written with `FormDataWriter`.
109pub struct FormDataIter<'a> {
110    data: &'a [u8],
111}
112
113impl<'a> FormDataIter<'a> {
114    pub fn new(data: &'a [u8]) -> Self {
115        Self { data }
116    }
117}
118
119impl<'a> Iterator for FormDataIter<'a> {
120    type Item = FormDataEntry<'a>;
121
122    fn next(&mut self) -> Option<Self::Item> {
123        while !self.data.is_empty() {
124            match FormDataEntry::read(&mut self.data) {
125                Some(entry) => return Some(entry),
126                None => relay_log::error!("form data deserialization failed"),
127            }
128        }
129
130        None
131    }
132}

134/// Looks for a multipart boundary at the beginning of the data
135/// and returns it as a `&str` if it is found
136///
137/// A multipart boundary starts at the beginning of the data (possibly
138/// after some blank lines) and it is prefixed by '--' (two dashes)
139///
140/// ```ignore
141/// let boundary = get_multipart_boundary(b"--The boundary\r\n next line");
142/// assert_eq!(Some("The boundary"), boundary);
143///
144/// let invalid_boundary = get_multipart_boundary(b"The boundary\r\n next line");
145/// assert_eq!(None, invalid_boundary);
146/// ```
147pub fn get_multipart_boundary(data: &[u8]) -> Option<&str> {
148    data.split(|&byte| byte == b'\r' || byte == b'\n')
149        // Get the first non-empty line
150        .find(|slice| !slice.is_empty())
151        // Check for the form boundary indicator
152        .filter(|slice| slice.len() > 2 && slice.starts_with(b"--"))
153        // Form boundaries must be valid UTF-8 strings
154        .and_then(|slice| std::str::from_utf8(&slice[2..]).ok())
155}

157/// Strategy for how to infer attachment type and add a multipart attachment to an envelope item.
158///
159/// This enables different endpoints to have different ways of dealing with multipart attachments,
160/// for instance, one endpoint can upload attachments and add a ref to the item, while another
161/// endpoint can add attachments to items directly.
162pub trait AttachmentStrategy {
163    fn infer_type(&self, field: &Field) -> AttachmentType;
164
165    /// Defines how individual multipart items should be handled.
166    ///
167    /// Returns
168    ///  - `Ok(Some(item))` if everything was successful.
169    ///  - `Ok(None)` if there was an error adding the attachment, but the rest of the request
170    ///    should still be handled.
171    ///  - `Err(..)` if there was an unexpected error adding the attachment and the request should
172    ///    be cancelled.
173    fn add_to_item(
174        &self,
175        field: Field<'static>,
176        item: Item,
177        config: &Config,
178    ) -> impl Future<Output = Result<Option<Item>, multer::Error>> + Send;
179}

181pub async fn read_attachment_bytes_into_item(
182    field: Field<'static>,
183    mut item: Item,
184    config: &Config,
185    ignore_size_exceeded: bool,
186) -> Result<Option<Item>, multer::Error> {
187    let content_type = field.content_type().cloned();
188    let field = LimitedField::new(field, config.max_attachment_size());
189    match field.bytes().await {
190        Ok(bytes) => {
191            if let Some(content_type) = content_type {
192                let ct = content_type
193                    .as_ref()
194                    .parse()
195                    .unwrap_or(ContentType::OctetStream);
196                item.set_payload(ct, bytes);
197            } else {
198                item.set_payload_without_content_type(bytes);
199            }
200            Ok(Some(item))
201        }
202        Err(multer::Error::FieldSizeExceeded { .. }) if ignore_size_exceeded => Ok(None),
203        Err(err) => Err(err),
204    }
205}

207pub async fn multipart_items(
208    mut multipart: Multipart<'static>,
209    config: &Config,
210    attachment_strategy: impl AttachmentStrategy,
211) -> Result<Items, multer::Error> {
212    let mut items = Items::new();
213    let mut form_data = FormDataWriter::new();
214    let mut attachments_size = 0;
215
216    while let Some(field) = multipart.next_field().await? {
217        if let Some(file_name) = field.file_name() {
218            let mut item = Item::new(ItemType::Attachment);
219            let attachment_type = attachment_strategy.infer_type(&field);
220            item.set_attachment_type(attachment_type);
221            item.set_filename(file_name);
222            let item = attachment_strategy.add_to_item(field, item, config).await?;
223            if let Some(item) = item {
224                // This increases the attachments byte count even if the item is an attachment ref.
225                // This is by design as the total number of bytes read into memory should be
226                // constrained.
227                attachments_size += item.len();
228                if attachments_size > config.max_attachments_size() {
229                    return Err(multer::Error::StreamSizeExceeded {
230                        limit: config.max_attachments_size() as u64,
231                    });
232                }
233                items.push(item);
234            }
235        } else if let Some(field_name) = field.name().map(str::to_owned) {
236            // Ensure to decode this SAFELY to match Django's POST data behavior. This allows us to
237            // process sentry event payloads even if they contain invalid encoding.
238            let string = field.text().await?;
239            form_data.append(&field_name, &string);
240        } else {
241            relay_log::trace!("multipart content without name or file_name");
242        }
243    }
244
245    let form_data = form_data.into_inner();
246    if !form_data.is_empty() {
247        let mut item = Item::new(ItemType::FormData);
248        // Content type is `Text` (since it is not a json object but multiple
249        // json arrays serialized one after the other).
250        item.set_payload(ContentType::Text, form_data);
251        items.push(item);
252    }
253
254    Ok(items)
255}

257/// Wrapper around `multer::Field` which consumes the entire underlying stream even when the
258/// size limit is exceeded.
259///
260/// The idea being that you can process fields in a multi-part form even if one fields is too large.
261struct LimitedField<'a> {
262    field: Field<'a>,
263    consumed_size: usize,
264    size_limit: usize,
265    inner_finished: bool,
266}
267
268impl<'a> LimitedField<'a> {
269    fn new(field: Field<'a>, limit: usize) -> Self {
270        LimitedField {
271            field,
272            consumed_size: 0,
273            size_limit: limit,
274            inner_finished: false,
275        }
276    }
277
278    async fn bytes(self) -> Result<Bytes, multer::Error> {
279        self.try_fold(BytesMut::new(), |mut acc, x| async move {
280            acc.extend_from_slice(&x);
281            Ok(acc)
282        })
283        .await
284        .map(|x| x.freeze())
285    }
286}
287
288impl futures::Stream for LimitedField<'_> {
289    type Item = Result<Bytes, multer::Error>;
290
291    fn poll_next(
292        mut self: std::pin::Pin<&mut Self>,
293        cx: &mut std::task::Context<'_>,
294    ) -> std::task::Poll<Option<Self::Item>> {
295        if self.inner_finished {
296            return Poll::Ready(None);
297        }
298
299        match self.field.poll_next_unpin(cx) {
300            err @ Poll::Ready(Some(Err(_))) => err,
301            Poll::Ready(Some(Ok(t))) => {
302                self.consumed_size += t.len();
303                match self.consumed_size <= self.size_limit {
304                    true => Poll::Ready(Some(Ok(t))),
305                    false => {
306                        cx.waker().wake_by_ref();
307                        Poll::Pending
308                    }
309                }
310            }
311            Poll::Ready(None) if self.consumed_size > self.size_limit => {
312                self.inner_finished = true;
313                Poll::Ready(Some(Err(multer::Error::FieldSizeExceeded {
314                    limit: self.consumed_size as u64,
315                    field_name: self.field.name().map(Into::into),
316                })))
317            }
318            Poll::Ready(None) => {
319                self.inner_finished = true;
320                Poll::Ready(None)
321            }
322            Poll::Pending => Poll::Pending,
323        }
324    }
325}

327pub fn multipart_from_request(
328    request: Request,
329    stream_size_limit: usize,
330) -> Result<Multipart<'static>, BadStoreRequest> {
331    let content_type = request
332        .headers()
333        .get("content-type")
334        .and_then(|v| v.to_str().ok())
335        .unwrap_or("");
336    let boundary =
337        multer::parse_boundary(content_type).map_err(BadStoreRequest::InvalidMultipart)?;
338
339    // Limits the overall stream size, preventing overly long processing times which can cause
340    // incidents like the one described in [#4836](https://github.com/getsentry/relay/pull/4836).
341    let stream_size_limit = SizeLimit::new().whole_stream(stream_size_limit as u64);
342    Ok(Multipart::with_constraints(
343        request.into_body().into_data_stream(),
344        boundary,
345        Constraints::new().size_limit(stream_size_limit),
346    ))
347}
