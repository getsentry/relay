6/// A shared item that can be consumed exactly once.
7pub struct TakeOnce<T>(Arc<Mutex<Option<T>>>);
8
9impl<S> Clone for TakeOnce<S> {
10    fn clone(&self) -> Self {
11        Self(Arc::clone(&self.0))
12    }
13}
14
15impl<T> TakeOnce<T> {
16    /// Creates a new shared item.
17    pub fn new(inner: T) -> Self {
18        Self(Arc::new(Mutex::new(Some(inner))))
19    }
20
21    /// Checks whether the item was already taken.
22    pub fn is_taken(&self) -> bool {
23        self.lock().is_none()
24    }
25
26    /// Takes the item, making it inaccessible for other owners.
27    ///
28    /// Returns `None` if the item was already taken by somebody else.
29    pub fn take(&self) -> Option<T> {
30        self.lock().take()
31    }
32
33    /// Gets locked access to the underlying option.
34    fn lock(&self) -> MutexGuard<'_, Option<T>> {
35        self.0.lock().unwrap_or_else(PoisonError::into_inner)
36    }
37}

39/// A stream that can be retried as long as it hasn't been polled.
40pub enum RetryableStream<S> {
41    /// The stream has not been polled.
42    /// Other owners of `S` can recover it by calling [`TakeOnce::take`].
43    New(TakeOnce<S>),
44    /// The stream has been polled at least once and is no longer retryable.
45    ///
46    /// This state is an optimization such that the stream does not have to lock a mutex
47    /// on every poll.
48    Used(S),
49}
50
51impl<S> RetryableStream<S> {
52    /// Creates a new retryable stream.
53    ///
54    /// Returns `None` if the underlying item has already been consumed.
55    pub fn new(inner: TakeOnce<S>) -> Option<Self> {
56        if inner.is_taken() {
57            None
58        } else {
59            Some(Self::New(inner))
60        }
61    }
62
63    /// Marks the underlying stream as used and returns a mutable reference to it.
64    ///
65    /// After calling this function, other owners cannot recover the stream.
66    /// It is no longer retryable.
67    fn make_used(&mut self) -> Option<&mut S> {
68        match self {
69            Self::New(s) => {
70                let inner = s.take()?;
71                *self = Self::Used(inner);
72                self.make_used() // recurse once
73            }
74            Self::Used(s) => Some(s),
75        }
76    }
77}
78
79impl<S: Stream + Unpin> Stream for RetryableStream<S> {
80    type Item = S::Item;
81
82    fn poll_next(
83        self: std::pin::Pin<&mut Self>,
84        cx: &mut std::task::Context<'_>,
85    ) -> std::task::Poll<Option<Self::Item>> {
86        match self.get_mut().make_used() {
87            Some(s) => s.poll_next_unpin(cx),
88            None => {
89                debug_assert!(false, "stream was taken while streaming");
90                Poll::Ready(None)
91            }
92        }
93    }
94
95    fn size_hint(&self) -> (usize, Option<usize>) {
96        match self {
97            Self::New(s) => s.lock().as_ref().map_or((0, None), |s| s.size_hint()),
98            Self::Used(s) => s.size_hint(),
99        }
100    }
101}

135    async fn test_retryable_before_use() {
136        let inner = TakeOnce::new(stream::iter(vec![1, 2]));
137
138        // First stream is created but never polled — inner is still available.
139        let _s1 = RetryableStream::new(inner.clone()).unwrap();
140
141        // A second stream can be created because the first hasn't taken the inner yet.
142        assert!(RetryableStream::new(inner).is_some());

145    #[tokio::test]
146    async fn test_first_poll_consumes_inner() {
147        let inner = TakeOnce::new(stream::iter(vec![1, 2]));
148
149        let mut s1 = RetryableStream::new(inner.clone()).unwrap();
150        // Polling the first item transitions s1 from Retryable -> Used.
151        assert_eq!(s1.next().await, Some(1));
152
153        // Now the inner is consumed; no new stream can be created.
154        assert!(RetryableStream::new(inner).is_none());
155
156        // But s1 still works.
157        assert_eq!(s1.next().await, Some(2));
158        assert_eq!(s1.next().await, None);
159    }
