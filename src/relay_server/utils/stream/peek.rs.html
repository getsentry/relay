1use bytes::{Bytes, BytesMut};
2use futures::{Stream, StreamExt, future, stream};

4/// Peeks `n` bytes into a `stream`, returning the peeked bytes together with the stream containing
5/// all the bytes of the original `stream`.
6pub async fn peek_n<S, E>(
7    stream: S,
8    n: usize,
9) -> Result<(Bytes, impl Stream<Item = Result<Bytes, E>> + Send), E>
10where
11    S: Stream<Item = Result<Bytes, E>> + Send,
12    E: Send,
13{
14    let mut stream = Box::pin(stream);
15    let mut head = BytesMut::with_capacity(n);
16    while head.len() < n {
17        match stream.next().await {
18            Some(Ok(chunk)) => head.extend_from_slice(&chunk),
19            Some(Err(e)) => return Err(e),
20            None => break,
21        }
22    }
23    let head = head.freeze();
24    let replay = stream::once(future::ready(Ok(head.clone()))).chain(stream);
25    Ok((head, replay))
26}
