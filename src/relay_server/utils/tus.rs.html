1//! TUS protocol types and helpers.
2//!
3//! Contains constants, error types, and parsing utilities for the TUS resumable upload protocol
4//! v1.0.0.
5//!
6//! Reference: <https://tus.io/protocols/resumable-upload>
7

16#[derive(Debug, thiserror::Error)]
17pub enum Error {
18    /// The TUS version is missing or does not match the server's version.
19    #[error("expected Tus-Resumable: 1.0.0, got: {}", .0.as_deref().unwrap_or("(missing)"))]
20    Version(Option<String>),
21    /// The `Upload-Length` and `Upload-Defer-Length` headers are both missing, incorrect, or cannot be parsed.
22    #[error(
23        "expected Upload-Length or Upload-Defer-Length=1, got Upload-Length={upload_length:?}, Upload-Defer-Length={upload_defer_length:?}"
24    )]
25    UploadLength {
26        upload_length: Option<usize>,
27        upload_defer_length: Option<usize>,
28    },
29    /// The `Upload-Offset` header is missing or invalid
30    #[error("expected Upload-Offset: 0, got: {0:?}")]
31    UploadOffset(Option<usize>),
32    /// The `Upload-Defer-Length` header is not allowed for external/untrusted requests.
33    #[error("Upload-Defer-Length not allowed")]
34    DeferLengthNotAllowed,
35    /// The `Content-Type` header is not what TUS expects.
36    #[error("expected Content-Type: application/offset+octet-stream, got: {}", .0.as_deref().unwrap_or("(missing)"))]
37    ContentType(Option<String>),
38}

40/// TUS protocol header for the protocol version.
41///
42/// See <https://tus.io/protocols/resumable-upload#tus-version>.
43pub const TUS_RESUMABLE: &str = "Tus-Resumable";

50const SUPPORTED_EXTENSIONS: HeaderValue = HeaderValue::from_static("creation,creation-with-upload");

65/// TUS protocol header for the current upload offset.

67/// See <https://tus.io/protocols/resumable-upload#upload-offset>.
68pub const UPLOAD_OFFSET: &str = "Upload-Offset";

70/// Expected value of the content-type header.
71pub const EXPECTED_CONTENT_TYPE: HeaderValue =
72    HeaderValue::from_static("application/offset+octet-stream");

74/// Parsed and validated header values.
75#[derive(Debug, PartialEq, Eq)]
76pub struct ParsedHeaders {
77    /// Whether the request has bytes to be uploaded. Defered from `Content-Type` header.
78    pub has_upload: bool,
79    /// Value of the `Upload-Length` header, if given.
80    pub upload_length: Option<usize>,
81}
82
83/// Validates TUS protocol headers and returns a subset of parsed values.
84pub fn validate_post_headers(
85    headers: &HeaderMap,
86    allow_defer_length: bool,
87) -> Result<ParsedHeaders, Error> {
88    let tus_version = headers.get(TUS_RESUMABLE);
89    if tus_version != Some(&TUS_VERSION) {
90        return Err(Error::Version(
91            tus_version.and_then(|v| v.to_str().ok()).map(str::to_owned),
92        ));
93    }
94
95    let has_upload = match headers.get(http::header::CONTENT_TYPE) {
96        Some(ct) if ct == EXPECTED_CONTENT_TYPE => true,
97        None => false,
98        Some(ct) => return Err(Error::ContentType(ct.to_str().ok().map(str::to_owned))),
99    };
100
101    let upload_length: Option<usize> = parse_header(headers, UPLOAD_LENGTH);
102    let upload_defer_length: Option<usize> = parse_header(headers, UPLOAD_DEFER_LENGTH);
103
104    // Exactly one of Upload-Length and Upload-Defer-Length must be present.
105    // Upload-Defer-Length is only accepted if its value is 1 (as demanded by the TUS protocol)
106    // and `allow_defer_length` is true (i.e. the sender is trusted/internal).
107    let upload_length = match (upload_length, upload_defer_length, allow_defer_length) {
108        (Some(u), None, _) => Ok(Some(u)),
109        (None, Some(1), true) => Ok(None),
110        (None, Some(1), false) => Err(Error::DeferLengthNotAllowed),
111        _ => Err(Error::UploadLength {
112            upload_length,
113            upload_defer_length,
114        }),
115    }?;
116
117    Ok(ParsedHeaders {
118        has_upload,
119        upload_length,
120    })
121}

123/// Validates TUS protocol headers and returns the expected upload length.
124pub fn validate_patch_headers(headers: &HeaderMap) -> Result<(), Error> {
125    let tus_version = headers.get(TUS_RESUMABLE);
126    if tus_version != Some(&TUS_VERSION) {
127        return Err(Error::Version(
128            tus_version.and_then(|v| v.to_str().ok()).map(str::to_owned),
129        ));
130    }
131
132    let content_type = headers.get(http::header::CONTENT_TYPE);
133    if content_type != Some(&EXPECTED_CONTENT_TYPE) {
134        return Err(Error::ContentType(
135            content_type
136                .and_then(|v| v.to_str().ok())
137                .map(str::to_owned),
138        ));
139    }
140
141    let upload_offset: usize =
142        parse_header(headers, UPLOAD_OFFSET).ok_or(Error::UploadOffset(None))?;
143    if upload_offset != 0 {
144        // Only allow full uploads for now.
145        return Err(Error::UploadOffset(Some(upload_offset)));
146    }
147
148    Ok(())
149}

151/// Prepares the required TUS request headers for upstream requests.
152pub fn add_creation_headers(upload_length: Option<usize>, builder: &mut RequestBuilder) {
153    builder.header(TUS_RESUMABLE, TUS_VERSION);
154    if let Some(upload_length) = upload_length {
155        builder.header(UPLOAD_LENGTH, HeaderValue::from(upload_length));
156    } else {
157        builder.header(UPLOAD_DEFER_LENGTH, "1");
158    }
159}
160
161/// Prepares the required TUS request headers for upstream requests.
162pub fn add_upload_headers(builder: &mut RequestBuilder) {
163    builder.header(TUS_RESUMABLE, TUS_VERSION);
164    builder.header(http::header::CONTENT_TYPE, EXPECTED_CONTENT_TYPE);
165    builder.header(UPLOAD_OFFSET, "0"); // always zero until we implement retries / chunking
166}
167
168/// Prepares the required TUS response headers.
169pub fn response_headers() -> HeaderMap {
170    let mut headers = HeaderMap::new();
171    headers.insert(TUS_RESUMABLE, TUS_VERSION);
172    headers.insert(TUS_EXTENSION, SUPPORTED_EXTENSIONS);
173    headers
174}
175
176fn parse_header<K: AsHeaderName, V: FromStr>(headers: &HeaderMap, header_name: K) -> Option<V> {
177    headers.get(header_name)?.to_str().ok()?.parse().ok()
178}
