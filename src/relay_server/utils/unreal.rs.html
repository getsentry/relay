29fn get_event_item(data: &[u8]) -> Result<Option<Item>, Unreal4Error> {
30    let mut context = Unreal4Context::parse(data)?;
31    let json = match context.game_data.remove(SENTRY_PAYLOAD_KEY) {
32        Some(json) if !json.is_empty() => json,
33        _ => return Ok(None),
34    };
35
36    relay_log::trace!("adding event payload from unreal context");
37    let mut item = Item::new(ItemType::Event);
38    item.set_payload(ContentType::Json, json);
39    Ok(Some(item))
40}

42/// Expands Unreal 4 items inside an envelope.
43///
44/// If the envelope does NOT contain an `UnrealReport` item, it doesn't do anything. If the envelope
45/// contains an `UnrealReport` item, it removes it from the envelope and inserts new items for each
46/// of its contents.
47///
48/// After this, the `EnvelopeProcessor` should be able to process the envelope the same way it
49/// processes any other envelopes.
50pub fn expand_unreal_envelope(
51    unreal_item: Item,
52    envelope: &mut Envelope,
53    config: &Config,
54) -> Result<(), ProcessingError> {
55    let payload = unreal_item.payload();
56    let crash = Unreal4Crash::parse_with_limit(&payload, config.max_envelope_size())?;
57
58    let mut has_event = envelope
59        .get_item_by(|item| item.ty() == &ItemType::Event)
60        .is_some();
61
62    for file in crash.files() {
63        let (content_type, attachment_type) = match file.ty() {
64            Unreal4FileType::Minidump => (ContentType::Minidump, AttachmentType::Minidump),
65            Unreal4FileType::AppleCrashReport => {
66                (ContentType::Text, AttachmentType::AppleCrashReport)
67            }
68            Unreal4FileType::Log => (ContentType::Text, AttachmentType::UnrealLogs),
69            Unreal4FileType::Config => (ContentType::OctetStream, AttachmentType::Attachment),
70            Unreal4FileType::Context => (ContentType::Xml, AttachmentType::UnrealContext),
71            _ => match file.name() {
72                self::ITEM_NAME_EVENT => (ContentType::MsgPack, AttachmentType::EventPayload),
73                self::ITEM_NAME_BREADCRUMBS1 => (ContentType::MsgPack, AttachmentType::Breadcrumbs),
74                self::ITEM_NAME_BREADCRUMBS2 => (ContentType::MsgPack, AttachmentType::Breadcrumbs),
75                _ => (ContentType::OctetStream, AttachmentType::Attachment),
76            },
77        };
78
79        if !has_event
80            && attachment_type == AttachmentType::UnrealContext
81            && let Some(event_item) = get_event_item(file.data())?
82        {
83            envelope.add_item(event_item);
84            has_event = true;
85        }
86
87        let mut item = Item::new(ItemType::Attachment);
88        item.set_filename(file.name());
89        // TODO: This clones data. Update symbolic to allow moving the bytes out.
90        item.set_payload(content_type, file.data().to_owned());
91        item.set_attachment_type(attachment_type);
92        envelope.add_item(item);
93    }
94
95    if let Err(offender) = super::check_envelope_size_limits(config, envelope) {
96        return Err(ProcessingError::PayloadTooLarge(offender));
97    }
98
99    Ok(())
100}

102fn merge_unreal_user_info(event: &mut Event, user_info: &str) {
103    let mut parts = user_info.split('|');
104
105    if let Some(user_id) = parts.next().filter(|id| !id.is_empty()) {
106        let user = event.user.value_mut().get_or_insert_with(User::default);
107        user.id = Annotated::new(LenientString(user_id.to_owned()));
108    }
109
110    if let Some(epic_account_id) = parts.next().filter(|id| !id.is_empty()) {
111        let tags = event.tags.value_mut().get_or_insert_with(Tags::default);
112        tags.push(Annotated::new(TagEntry(
113            Annotated::new("epic_account_id".to_owned()),
114            Annotated::new(epic_account_id.to_owned()),
115        )));
116    }
117
118    if let Some(machine_id) = parts.next().filter(|id| !id.is_empty()) {
119        let tags = event.tags.value_mut().get_or_insert_with(Tags::default);
120        tags.push(Annotated::new(TagEntry::from_pair((
121            Annotated::new("machine_id".to_owned()),
122            Annotated::new(machine_id.to_owned()),
123        ))));
124    }
125}

127/// Merge an unreal logs object into event breadcrumbs.
128fn merge_unreal_logs(
129    event: &mut Event,
130    logs: impl IntoIterator<Item = Bytes>,
131) -> Result<(), Unreal4Error> {
132    let breadcrumbs = event
133        .breadcrumbs
134        .value_mut()
135        .get_or_insert_with(Values::default)
136        .values
137        .value_mut()
138        .get_or_insert_with(Array::default);
139
140    for breadcrumb in logs
141        .into_iter()
142        .flat_map(parse_unreal_logs)
143        .take(MAX_NUM_UNREAL_LOGS)
144    {
145        breadcrumbs.push(Annotated::new(breadcrumb?));
146    }
147
148    Ok(())
149}

151/// Parses unreal logs into breadcrumbs.
152fn parse_unreal_logs(data: Bytes) -> impl Iterator<Item = Result<Breadcrumb, Unreal4Error>> {
153    let (logs, err) = match Unreal4LogEntry::parse(&data, MAX_NUM_UNREAL_LOGS) {
154        Ok(logs) => (logs, None),
155        Err(err) => (Vec::new(), Some(Err(err))),
156    };
157
158    let logs = logs.into_iter().map(|log| {
159        let timestamp = log
160            .timestamp
161            .and_then(|ts| {
162                Utc.timestamp_opt(ts.unix_timestamp(), ts.nanosecond())
163                    .latest()
164            })
165            .map(Timestamp);
166
167        Ok(Breadcrumb {
168            timestamp: Annotated::from(timestamp),
169            category: Annotated::from(log.component),
170            message: Annotated::new(log.message),
171            ..Breadcrumb::default()
172        })
173    });
174
175    err.into_iter().chain(logs)
176}

178/// Parses `UserReport` from unreal user information.
179fn get_unreal_user_report(event_id: EventId, context: &mut Unreal4Context) -> Option<Item> {
180    let runtime_props = context.runtime_properties.as_mut()?;
181    let user_description = runtime_props.user_description.take()?;
182
183    let user_name = runtime_props
184        .username
185        .clone()
186        .unwrap_or_else(|| "unknown".to_owned());
187
188    let user_report = UserReport {
189        email: "".to_owned(),
190        comments: user_description,
191        event_id,
192        name: user_name,
193    };
194
195    let json = serde_json::to_string(&user_report).ok()?;
196
197    let mut item = Item::new(ItemType::UserReport);
198    item.set_payload(ContentType::Json, json);
199    Some(item)
200}

202/// Merges an unreal context object into an event
203fn merge_unreal_context(event: &mut Event, context: Unreal4Context) {
204    let mut runtime_props = match context.runtime_properties {
205        Some(runtime_props) => runtime_props,
206        None => return,
207    };
208
209    if let Some(msg) = runtime_props.error_message.take() {
210        let logentry = event.logentry.get_or_insert_with(LogEntry::default);
211        if logentry.formatted.is_empty() {
212            logentry.formatted = Annotated::new(Message::from(msg));
213        }
214    }
215
216    if let Some(username) = runtime_props.username.take() {
217        let user = event.user.get_or_insert_with(User::default);
218        if user.username.is_empty() {
219            user.username.set_value(Some(username.into()));
220        }
221    }
222
223    if let Some(login_id) = &runtime_props.login_id {
224        let id = event.user.get_or_insert_with(User::default).id.value_mut();
225
226        if id.as_ref().is_none_or(|s| s.is_empty()) {
227            *id = Some(login_id.clone().into());
228        }
229    }
230
231    let contexts = event.contexts.get_or_insert_with(Contexts::default);
232
233    if let Some(memory_physical) = runtime_props.memory_stats_total_physical.take() {
234        let device_context = contexts.get_or_default::<DeviceContext>();
235        if device_context.memory_size.is_empty() {
236            device_context.memory_size = Annotated::new(memory_physical);
237        }
238    }
239
240    // OS information is likely overwritten by Minidump processing later.
241    if let Some(os_major) = runtime_props.misc_os_version_major.take() {
242        let os_context = contexts.get_or_default::<OsContext>();
243        if os_context.raw_description.is_empty() {
244            os_context.raw_description = Annotated::new(os_major);
245        }
246    }
247
248    // See https://github.com/EpicGames/UnrealEngine/blob/5.3.2-release/Engine/Source/Runtime/RHI/Private/DynamicRHI.cpp#L368-L376
249    if let Some(adapter_name) = context.engine_data.get("RHI.AdapterName") {
250        let gpu_context = contexts.get_or_default::<GpuContext>();
251        if gpu_context.name.is_empty() {
252            gpu_context.name = Annotated::new(adapter_name.into());
253        }
254    } else if let Some(gpu_brand) = runtime_props.misc_primary_gpu_brand.take() {
255        let gpu_context = contexts.get_or_default::<GpuContext>();
256        if gpu_context.name.is_empty() {
257            gpu_context.name = Annotated::new(gpu_brand);
258        }
259    }
260
261    if let Some(device_id) = context.engine_data.get("RHI.DeviceId") {
262        let gpu_context = contexts.get_or_default::<GpuContext>();
263        if gpu_context.id.is_empty() {
264            gpu_context.id = Annotated::new(Value::String(device_id.into()));
265        }
266    }
267
268    if let Some(feature_level) = context.engine_data.get("RHI.FeatureLevel") {
269        let gpu_context = contexts.get_or_default::<GpuContext>();
270        if gpu_context.graphics_shader_level.is_empty() {
271            gpu_context.graphics_shader_level = Annotated::new(feature_level.into());
272        }
273    }
274
275    if let Some(vendor_name) = context.engine_data.get("RHI.GPUVendor") {
276        let gpu_context = contexts.get_or_default::<GpuContext>();
277        if gpu_context.vendor_name.is_empty() {
278            gpu_context.vendor_name = Annotated::new(vendor_name.into());
279        }
280    }
281
282    if let Some(driver_version) = context.engine_data.get("RHI.UserDriverVersion") {
283        let gpu_context = contexts.get_or_default::<GpuContext>();
284        if gpu_context.version.is_empty() {
285            gpu_context.version = Annotated::new(driver_version.into());
286        }
287    }
288
289    if let Some(rhi_name) = context.engine_data.get("RHI.RHIName") {
290        let gpu_context = contexts.get_or_default::<GpuContext>();
291        if gpu_context.api_type.is_empty() {
292            gpu_context.api_type = Annotated::new(rhi_name.into());
293        }
294    }
295
296    if event.level.is_empty() {
297        if runtime_props.is_assert.unwrap_or(false) {
298            event.level = Annotated::new(Level::Error)
299        } else if runtime_props.is_ensure.unwrap_or(false) {
300            event.level = Annotated::new(Level::Warning)
301        }
302    }
303
304    // Modules are not used and later replaced with Modules from the Minidump or Apple Crash Report.
305    runtime_props.modules.take();
306
307    // Promote all game data (except the special `__sentry` key) into a context.
308    if !context.game_data.is_empty() {
309        let game_context = contexts.get_or_insert_with("game", || Context::Other(Object::new()));
310        if let Context::Other(game_context) = game_context {
311            let filtered_keys = context
312                .game_data
313                .into_iter()
314                .filter(|(key, _)| key != SENTRY_PAYLOAD_KEY)
315                .map(|(key, value)| (key, Annotated::new(Value::String(value))));
316
317            game_context.extend(filtered_keys);
318        }
319    }
320
321    // Add sdk information for analytics.
322    event.client_sdk.get_or_insert_with(|| ClientSdkInfo {
323        name: CLIENT_SDK_NAME.to_owned().into(),
324        version: runtime_props
325            .crash_reporter_client_version
326            .take()
327            .unwrap_or_else(|| "0.0.0".to_owned())
328            .into(),
329        ..ClientSdkInfo::default()
330    });
331
332    if let Ok(Some(Value::Object(props))) = relay_protocol::to_value(&runtime_props) {
333        let unreal_context =
334            contexts.get_or_insert_with("unreal", || Context::Other(Object::new()));
335
336        if let Context::Other(unreal_context) = unreal_context {
337            unreal_context.extend(props);
338        }
339    }
340}

342/// Processes an unreal envelope.
343///
344/// This function returns either the processing error, or a boolean indicating
345/// whether the envelope contained an unreal item.
346pub fn process_unreal_envelope(
347    event: &mut Annotated<Event>,
348    envelope: &mut Envelope,
349) -> Result<bool, Unreal4Error> {
350    let user_header = envelope
351        .get_header(UNREAL_USER_HEADER)
352        .and_then(Value::as_str);
353    let context_item =
354        envelope.get_item_by(|item| item.attachment_type() == Some(&AttachmentType::UnrealContext));
355    let mut logs_items = envelope
356        .items()
357        .filter(|item| item.attachment_type() == Some(&AttachmentType::UnrealLogs))
358        .map(|item| item.payload())
359        .peekable();
360
361    // Early exit if there is no information.
362    if user_header.is_none() && context_item.is_none() && logs_items.peek().is_none() {
363        return Ok(false);
364    }
365
366    // If we have UE4 info, ensure an event is there to fill. DO NOT fill if there is no unreal
367    // information, or otherwise `EnvelopeProcessorService::process` breaks.
368    let event = event.get_or_insert_with(Event::default);
369
370    if let Some(user_info) = user_header {
371        merge_unreal_user_info(event, user_info);
372    }
373
374    merge_unreal_logs(event, logs_items)?;
375
376    if let Some(context_item) = context_item {
377        let mut context = Unreal4Context::parse(&context_item.payload())?;
378
379        // the `unwrap_or_default` here can produce an invalid user report if the envelope id
380        // is indeed missing. This should not happen under normal circumstances since the EventId is
381        // created statically.
382        let event_id = envelope.event_id().unwrap_or_default();
383        debug_assert!(!event_id.is_nil());
384
385        if let Some(report) = get_unreal_user_report(event_id, &mut context) {
386            envelope.add_item(report);
387        }
388
389        merge_unreal_context(event, context);
390    }
391
392    Ok(true)
393}
