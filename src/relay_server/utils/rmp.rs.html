6/// Maximum nesting depth allowed when deserializing MessagePack payloads.
7///
8/// `rmp_serde` default (1024) too permissive for
9/// Relay's types (e.g. `Annotated<Value>`). Each recursive deserialization frame is
10/// several kilobytes, so a payload can exhaust the thread stack well before the built-in counter
11/// trips.
12/// This limit matches the one in `serde_json`.
13pub const MAX_DEPTH: usize = 128;
14
15/// Returns a Deserializer with a bounded recursion depth.
16///
17/// Caps nesting at
18/// [`MAX_DEPTH`] to protect against stack overflows from maliciously nested payloads.
19pub fn slice_deserializer<'a>(data: &'a [u8]) -> Deserializer<ReadRefReader<'a, [u8]>> {
20    let mut deserializer = Deserializer::from_read_ref(data);
21    deserializer.set_max_depth(MAX_DEPTH);
22    deserializer
23}

25/// Returns a Deserializer with a bounded recursion depth.
26///
27/// Caps nesting at
28/// [`MAX_DEPTH`] to protect against stack overflows from maliciously nested payloads.
29///
30/// Like `slice_deserializer`, but allows consuming multiple items from the slice.
31pub fn stream_deserializer(data: &[u8]) -> Deserializer<ReadReader<&[u8]>> {
