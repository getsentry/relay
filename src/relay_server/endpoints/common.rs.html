127impl BadStoreRequest {
128    pub fn to_outcome(&self) -> Option<Outcome> {
129        let discard_reason = match self {
130            Self::EmptyBody => DiscardReason::EmptyBody,
131            Self::InternalEnvelope => DiscardReason::InternalEnvelope,
132            Self::InvalidBody(_) => DiscardReason::InvalidBody,
133            Self::InvalidJson(_) => DiscardReason::InvalidJson,
134            Self::InvalidMsgpack(_) => DiscardReason::InvalidMsgpack,
135            Self::InvalidEnvelope(_) => DiscardReason::InvalidEnvelope,
136            Self::InvalidMultipart(_) => DiscardReason::InvalidMultipart,
137            Self::InvalidMinidump => DiscardReason::InvalidMinidump,
138            Self::MissingMinidump => DiscardReason::MissingMinidump,
139            Self::InvalidUnrealReport => DiscardReason::InvalidUnrealReport,
140            #[cfg(sentry)]
141            Self::InvalidProsperodump => DiscardReason::InvalidProsperodump,
142            #[cfg(sentry)]
143            Self::MissingProsperodump => DiscardReason::MissingProsperodump,
144            Self::InvalidCompression(_) => DiscardReason::InvalidCompression,
145            Self::InvalidEventId => DiscardReason::InvalidEventId,
146            Self::QueueFailed(_) => DiscardReason::QueueFailed,
147            Self::ItemTooLarge(item_type) => DiscardReason::ItemTooLarge(*item_type),
148            Self::RequestTooLarge => DiscardReason::RequestTooLarge,
149            Self::RateLimited(_) => DiscardReason::RateLimited,
150            Self::EventRejected(discard_reason) => *discard_reason,
151            Self::ProjectUnavailable => DiscardReason::ProjectUnavailable,
152            Self::ObjectstoreUploadFailed => DiscardReason::ObjectstoreUploadFailed,
153        };
154        Some(Outcome::Invalid(discard_reason))
155    }
156}

164impl From<BytesRejection> for BadStoreRequest {
165    fn from(value: BytesRejection) -> Self {
166        match value {
167            BytesRejection::FailedToBufferBody(FailedToBufferBody::LengthLimitError(_)) => {
168                BadStoreRequest::RequestTooLarge
169            }
170            other => BadStoreRequest::InvalidBody(io::Error::other(other)),
171        }

175impl From<multer::Error> for BadStoreRequest {
176    fn from(value: multer::Error) -> Self {
177        match value {
178            multer::Error::StreamSizeExceeded { .. } => BadStoreRequest::RequestTooLarge,
179            multer::Error::StreamReadFailed(error)
180                if find_error_source(error.as_ref(), is_length_limit_error).is_some() =>
181            {
182                // This happens when the stream suddenly stops because `RequestBodyLimit` capped
183                // a request with `Transfer-Encoding: Chunked`.
184                BadStoreRequest::RequestTooLarge
185            }
186            other => BadStoreRequest::InvalidMultipart(other),
187        }
188    }
189}

191impl IntoResponse for BadStoreRequest {
192    fn into_response(self) -> axum::response::Response {
193        let body = ApiErrorResponse::from_error(&self);
194
195        let response = match &self {
196            BadStoreRequest::RateLimited(rate_limits) => {
197                let retry_after_header = rate_limits
198                    .longest()
199                    .map(|limit| limit.retry_after.remaining_seconds().to_string())
200                    .unwrap_or_default();
201
202                let rate_limits_header = utils::format_rate_limits(rate_limits);
203
204                // For rate limits, we return a special status code and indicate the client to hold
205                // off until the rate limit period has expired. Currently, we only support the
206                // delay-seconds variant of the Rate-Limit header.
207                let headers = [
208                    (header::RETRY_AFTER.as_str(), retry_after_header),
209                    (utils::RATE_LIMITS_HEADER, rate_limits_header),
210                ];
211
212                (StatusCode::TOO_MANY_REQUESTS, headers, body).into_response()
213            }
214            BadStoreRequest::QueueFailed(_) | BadStoreRequest::ProjectUnavailable => {
215                // These errors indicate that something's wrong with our service system, most likely
216                // mailbox congestion or a faulty shutdown. Indicate an unavailable service to the
217                // client. It might retry event submission at a later time.
218                (StatusCode::SERVICE_UNAVAILABLE, body).into_response()
219            }
220            BadStoreRequest::EventRejected(_) => {
221                // The event has been discarded, which is generally indicated with a 403 error.
222                // Originally, Sentry also used this status code for event filters, but these are
223                // now executed asynchronously in `EnvelopeProcessor`.
224                (StatusCode::FORBIDDEN, body).into_response()
225            }
226            BadStoreRequest::ItemTooLarge(_) | BadStoreRequest::RequestTooLarge => {
227                (StatusCode::PAYLOAD_TOO_LARGE, body).into_response()
228            }
229            BadStoreRequest::ObjectstoreUploadFailed => {
230                (StatusCode::INTERNAL_SERVER_ERROR, body).into_response()
231            }
232            BadStoreRequest::InvalidMultipart(multer::Error::LockFailure) => {
233                (StatusCode::INTERNAL_SERVER_ERROR, body).into_response()
234            }
235            _ => {
236                // In all other cases, we indicate a generic bad request to the client and render
237                // the cause. This was likely the client's fault.
238                (StatusCode::BAD_REQUEST, body).into_response()
239            }
240        };
241
242        metric!(counter(RelayCounters::EnvelopeRejected) += 1);
243        if response.status() == http::StatusCode::SERVICE_UNAVAILABLE {
244            relay_log::warn!(
245                error = &self as &dyn std::error::Error,
246                "not handling request: service unavailable"
247            );
248        } else if response.status().is_server_error() {
249            relay_log::error!(
250                error = &self as &dyn std::error::Error,
251                "error handling request"
252            );
253        } else if response.status().is_client_error() {
254            relay_log::debug!(
255                error = &self as &dyn std::error::Error,
256                "error handling request"
257            );
258        }
259
260        response
261    }
262}

273/// Parses a minimal subset of the event payload.
274///
275/// This function validates that the provided payload is valid and returns an `Err` on parse errors.
276pub fn minimal_event_from_json(data: &[u8]) -> Result<MinimalEvent, BadStoreRequest> {
277    serde_json::from_slice(data).map_err(BadStoreRequest::InvalidJson)
278}

280/// Extracts the event id from a JSON payload.
281///
282/// If the payload contains no event id, `Ok(None)` is returned. This function also validates that
283/// the provided is valid and returns an `Err` on parse errors. If the event id itself is malformed,
284/// an `Err` is returned.
285pub fn event_id_from_json(data: &[u8]) -> Result<Option<EventId>, BadStoreRequest> {
286    minimal_event_from_json(data).map(|event| event.id)
287}
288
289/// Extracts the event id from a MessagePack payload.
290///
291/// If the payload contains no event id, `Ok(None)` is returned. This function also validates that
292/// the provided is valid and returns an `Err` on parse errors. If the event id itself is malformed,
293/// an `Err` is returned.
294pub fn event_id_from_msgpack(data: &[u8]) -> Result<Option<EventId>, BadStoreRequest> {
295    rmp_serde::from_slice(data)
296        .map(|MinimalEvent { id, .. }| id)
297        .map_err(BadStoreRequest::InvalidMsgpack)
298}
299
300/// Extracts the event id from `sentry` JSON payload or the `sentry[event_id]` formdata key.
301///
302/// If the event id itself is malformed, an `Err` is returned. If there is a `sentry` key containing
303/// malformed JSON, an error is returned.
304pub fn event_id_from_formdata(data: &[u8]) -> Result<Option<EventId>, BadStoreRequest> {
305    for entry in FormDataIter::new(data) {
306        if entry.key() == "sentry" {
307            return event_id_from_json(entry.value().as_bytes());
308        } else if entry.key() == "sentry[event_id]" {
309            return entry
310                .value()
311                .parse()
312                .map(Some)
313                .map_err(|_| BadStoreRequest::InvalidEventId);
314        }
315    }
316
317    Ok(None)
318}

320/// Extracts the event id from multiple items.
321///
322/// Submitting multiple event payloads is undefined behavior. This function will check for an event
323/// id in the following precedence:
324///
325///  1. The `Event` item.
326///  2. The `__sentry-event` event attachment.
327///  3. The `sentry` JSON payload.
328///  4. The `sentry[event_id]` formdata key.
329///
330/// # Limitations
331///
332/// Extracting the event id from chunked formdata fields on the Minidump endpoint (`sentry__1`,
333/// `sentry__2`, ...) is not supported. In this case, `None` is returned.
334pub fn event_id_from_items(items: &Items) -> Result<Option<EventId>, BadStoreRequest> {
335    if let Some(item) = items.iter().find(|item| item.ty() == &ItemType::Event)
336        && let Some(event_id) = event_id_from_json(&item.payload())?
337    {
338        return Ok(Some(event_id));
339    }
340
341    if let Some(item) = items
342        .iter()
343        .find(|item| item.attachment_type() == Some(AttachmentType::EventPayload))
344        && let Some(event_id) = event_id_from_msgpack(&item.payload())?
345    {
346        return Ok(Some(event_id));
347    }
348
349    if let Some(item) = items.iter().find(|item| item.ty() == &ItemType::FormData) {
350        // Swallow all other errors here since it is quite common to receive invalid secondary
351        // payloads. `EnvelopeProcessor` also retains events in such cases.
352        if let Ok(Some(event_id)) = event_id_from_formdata(&item.payload()) {
353            return Ok(Some(event_id));
354        }
355    }
356
357    Ok(None)
358}

360/// Queues an envelope for processing.
361///
362/// Depending on the items in the envelope, there are multiple outcomes:
363///
364/// - Events and event related items, such as attachments, are always queued together. See the
365///   [crate-level documentation](crate) for a full description of how envelopes are
366///   queued and processed.
367/// - Sessions and Session batches are always queued separately. If they occur in the same envelope
368///   as an event, they are split off. Their path is the same as other Envelopes.
369/// - Metrics are directly sent to the [`crate::services::processor::EnvelopeProcessor`], bypassing the manager's queue and
370///   going straight into metrics aggregation. See [`ProcessMetrics`] for a full description.
371///
372/// Queueing can fail if the queue exceeds `envelope_buffer_size`. In this case, `Err` is
373/// returned and the envelope is not queued.
374fn queue_envelope(
375    state: &ServiceState,
376    mut envelope: Managed<Box<Envelope>>,
377) -> Result<(), Rejected<BadStoreRequest>> {
378    if state.config().relay_mode() != RelayMode::Proxy {
379        // Remove metrics from the envelope and queue them directly on the project's `Aggregator`.
380        // In proxy mode, we cannot aggregate metrics because we may not have a project ID.
381        let is_metric = |i: &Item| matches!(i.ty(), ItemType::Statsd | ItemType::MetricBuckets);
382
383        let metrics;
384        (envelope, metrics) = envelope.split_once(|mut envelope, _| {
385            let metrics = envelope.take_items_by(is_metric).into_vec();
386            (envelope, metrics)
387        });
388
389        metrics.accept(|metrics| {
390            if metrics.is_empty() {
391                return;
392            }
393
394            state.processor().send(ProcessMetrics {
395                data: MetricData::Raw(metrics),
396                received_at: envelope.received_at(),
397                sent_at: envelope.sent_at(),
398                project_key: envelope.meta().public_key(),
399                source: BucketSource::from_meta(envelope.meta()),
400            });
401        })
402    }
403
404    let pkp = ProjectKeyPair::from_envelope(&envelope);
405    state
406        .envelope_buffer(pkp)
407        .try_push(envelope)
408        .map_err(|e| e.map(BadStoreRequest::QueueFailed))
409}

411/// Convert `envelope` to a managed envelope and call [`handle_managed_envelope`].
412///
413/// See [`handle_managed_envelope`] for full details.
414pub async fn handle_envelope(
415    state: &ServiceState,
416    envelope: Box<Envelope>,
417) -> Result<HandledEnvelope, Rejected<BadStoreRequest>> {
418    let envelope = Managed::from_envelope(envelope, state.outcome_aggregator().clone());
419    handle_managed_envelope(state, envelope).await
420}

422/// Handles an envelope store request.
423///
424/// Sentry envelopes may come either directly from an HTTP request (the envelope endpoint calls this
425/// method directly) or are generated inside Relay from requests to other endpoints (e.g. the
426/// security endpoint).
427///
428/// This returns `Some(EventId)` if the envelope contains an event, either explicitly as payload or
429/// implicitly through an item that will create an event during ingestion.
430pub async fn handle_managed_envelope(
431    state: &ServiceState,
432    mut envelope: Managed<Box<Envelope>>,
433) -> Result<HandledEnvelope, Rejected<BadStoreRequest>> {
434    emit_envelope_metrics(&envelope);
435
436    if state.memory_checker().check_memory().is_exceeded() {
437        return Err(envelope.reject_err((
438            Outcome::Invalid(DiscardReason::Internal),
439            BadStoreRequest::QueueFailed(PushError::OutOfMemory),
440        )));
441    };
442
443    // If configured, remove unknown items at the very beginning. If the envelope is
444    // empty, we fail the request with a special control flow error to skip checks and
445    // queueing, that still results in a `200 OK` response.
446    utils::remove_unknown_items(state.config(), &mut envelope);
447
448    let event_id = envelope.event_id();
449    if envelope.is_empty() {
450        return Ok(HandledEnvelope {
451            event_id,
452            rate_limits: Default::default(),
453        });
454    }
455
456    let project_key = envelope.meta().public_key();
457
458    let rate_limits = state
459        .project_cache_handle()
460        .get(project_key)
461        .check_envelope(&mut envelope)
462        .await
463        .map_err(|err| err.map(BadStoreRequest::EventRejected))?;
464
465    if envelope.is_empty() {
466        return Ok(HandledEnvelope {
467            event_id,
468            rate_limits,
469        });
470    }
471
472    if let Err(offender) = utils::check_envelope_size_limits(state.config(), &envelope) {
473        return Err(envelope.reject_err((
474            Outcome::Invalid(DiscardReason::ItemTooLarge(offender)),
475            BadStoreRequest::ItemTooLarge(offender),
476        )));
477    }
478
479    queue_envelope(state, envelope)?;
480
481    Ok(HandledEnvelope {
482        event_id,
483        // Even if some envelope items have been queued, there might be active rate limits on
484        // other items. Communicate these rate limits to the downstream (Relay or SDK).
485        rate_limits,
486    })
487}

489/// A successfully handled envelope, returned by [`handle_envelope`].
490#[derive(Debug)]
491#[must_use = "rate limits of a handled envelope must be used"]
492pub struct HandledEnvelope {
493    /// The event id of the envelope.
494    pub event_id: Option<EventId>,
495    /// All active, but not necessarily enforced, rate limits.
496    ///
497    /// These rate limits should always be communicated to the client on envelope endpoints.
498    ///
499    /// See also: [`BadStoreRequest::RateLimited`].
500    pub rate_limits: RateLimits,
501}
502
503impl HandledEnvelope {
504    /// Check if any rate limits were enforced (i.e. led to one or more items being dropped) and
505    /// return an error if so.
506    pub fn check_rate_limits(self) -> Result<Option<EventId>, BadStoreRequest> {
507        if self.rate_limits.is_limited() {
508            return Err(BadStoreRequest::RateLimited(self.rate_limits));
509        }
510        Ok(self.event_id)
511    }
512
513    /// Silence rate limits, even if they caused items to be dropped from the envelope.
514    ///
515    /// Endpoints which choose to not propagate rate limits should use this method to explicitly
516    /// state the fact that they do so.
517    pub fn ignore_rate_limits(self) -> Option<EventId> {
518        self.event_id
519    }
520}

522fn emit_envelope_metrics(envelope: &Envelope) {
523    let client_name = utils::client_name_tag(envelope.meta().client_name());
524    for item in envelope.items() {
525        let item_type = item.ty().name();
526        let is_container = if item.is_container() { "true" } else { "false" };
527
528        metric!(
529            distribution(RelayDistributions::EnvelopeItemSize) = item.payload().len() as u64,
530            item_type = item_type,
531            is_container = is_container,
532        );
533        metric!(
534            counter(RelayCounters::EnvelopeItems) += item.item_count().unwrap_or(1) as u64,
535            item_type = item_type,
536            is_container = is_container,
537            sdk = client_name,
538        );
539        metric!(
540            counter(RelayCounters::EnvelopeItemBytes) += item.payload().len() as u64,
541            item_type = item_type,
542            is_container = is_container,
543            sdk = client_name,
544        );
545    }
546}

548/// Uploads the content of `field` to the objectstore and returns an [Item] with an
549/// [AttachmentPlaceholder] as payload.
550pub async fn upload_to_objectstore<S, E>(
551    stream: S,
552    content_type: Option<String>,
553    mut item: Managed<Item>,
554    config: &Config,
555    project: ProjectContext,
556    upload: &Addr<Upload>,
557    referrer: &'static str,
558) -> Result<Managed<Item>, Rejected<()>>
559where
560    S: futures::Stream<Item = Result<Bytes, E>> + Send + 'static,
561    E: Into<Box<dyn std::error::Error + Send + Sync>> + 'static,
562{
563    let res = upload_to_objectstore_inner(
564        stream,
565        content_type,
566        &mut item,
567        config,
568        project,
569        upload,
570        referrer,
571    )
572    .await;
573    match res {
574        Some(()) => Ok(item),
575        None => Err(Outcome::Invalid(DiscardReason::Internal)).reject(&item),
576    }
577}
578
579async fn upload_to_objectstore_inner<S, E>(
580    stream: S,
581    content_type: Option<String>,
582    item: &mut Managed<Item>,
583    config: &Config,
584    project: ProjectContext,
585    upload: &Addr<Upload>,
586    referrer: &'static str,
587) -> Option<()>
588where
589    S: futures::Stream<Item = Result<Bytes, E>> + Send + 'static,
590    E: Into<Box<dyn std::error::Error + Send + Sync>> + 'static,
591{
592    let stream: BoxStream<'static, io::Result<Bytes>> = Box::pin(stream.map_err(io::Error::other));
593    let stream = MeteredStream::new(stream, referrer);
594    let stream = BoundedStream::new(stream, 1, config.max_upload_size());
595    let byte_counter = stream.byte_counter();
596
597    let location = upload
598        .send(Create {
599            project: project.clone(),
600            length: None,
601            attachment_type: item.attachment_type(),
602        })
603        .await
604        .ok()?
605        .ok()?;
606
607    let scoping = project.scoping;
608
609    let result = upload
610        .send(Stream {
611            received: Utc::now(),
612            project,
613            location,
614            stream,
615        })
616        .await
617        .ok()?;
618
619    let location = result
620        .inspect_err(|e| {
621            relay_log::warn!(
622                error = e as &dyn std::error::Error,
623                referrer = referrer,
624                organization_id = scoping.organization_id.value(),
625                project_id = scoping.project_id.value(),
626                bytes_uploaded = byte_counter.get(),
627                "multipart item upload failed",
628            );
629        })
630        .ok()?;
631    let location = location.into_header_value().ok()?;
632    let location = location.to_str().ok()?;
633    let placeholder = serde_json::to_vec(&AttachmentPlaceholder {
634        location,
635        content_type,
636    })
637    .ok()?;
638
639    item.modify(|inner, records| {
640        inner.set_payload(ContentType::AttachmentRef, placeholder);
641        inner.set_attachment_length(byte_counter.get());
642        records.lenient(DataCategory::Attachment); // item was empty before
643    });
644    Some(())
645}

649
650impl IntoResponse for TextResponse {
651    fn into_response(self) -> axum::response::Response {
652        // Event id is set statically in the ingest path.
653        let EventId(id) = self.0.unwrap_or_default();
654        debug_assert!(!id.is_nil());
655
656        // the minidump client expects the response to contain an event id as a hyphenated UUID
657        // i.e. xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx
658        let text = id.as_hyphenated().to_string();
