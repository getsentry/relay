36#[derive(Debug, thiserror::Error)]
37enum Error {
38    #[error("TUS protocol error: {0}")]
39    Tus(#[from] tus::Error),
40
41    #[error("request error: {0}")]
42    Request(#[from] BadStoreRequest),
43
44    #[error("upload error: {0}")]
45    Upload(#[from] upload::Error),
46}
47
48impl IntoResponse for Error {
49    fn into_response(self) -> Response {
50        match self {
51            Error::Tus(tus::Error::DeferLengthNotAllowed) => StatusCode::FORBIDDEN,
52            Error::Tus(_) => StatusCode::BAD_REQUEST,
53            Error::Request(error) => return error.into_response(),
54            Error::Upload(error) => match error {
55                upload::Error::Send(_) => StatusCode::SERVICE_UNAVAILABLE,
56                upload::Error::UpstreamRequest(e) => match e {
57                    UpstreamRequestError::SendFailed(e)
58                        if find_error_source(&e, is_hyper_user_error).is_some() =>
59                    {
60                        return StatusCode::BAD_REQUEST.into_response();
61                    }
62                    _ => return e.into_response(),
63                },
64                upload::Error::Timeout(_) => StatusCode::GATEWAY_TIMEOUT,
65                upload::Error::Upstream(status) => status,
66                upload::Error::InvalidLocation | upload::Error::SigningFailed => {
67                    StatusCode::INTERNAL_SERVER_ERROR
68                }
69                upload::Error::ServiceUnavailable => StatusCode::SERVICE_UNAVAILABLE,
70                #[cfg(feature = "processing")]
71                upload::Error::UploadService(service_error) => match service_error {
72                    ServiceError::Timeout => StatusCode::GATEWAY_TIMEOUT,
73                    ServiceError::LoadShed => StatusCode::SERVICE_UNAVAILABLE,
74                    ServiceError::UploadFailed(error) => match error {
75                        objectstore::Error::Reqwest(error) => match error.status() {
76                            Some(status) => status,
77                            None => StatusCode::INTERNAL_SERVER_ERROR,
78                        },
79                        _ => StatusCode::INTERNAL_SERVER_ERROR,
80                    },
81                    ServiceError::Uuid(_) => StatusCode::INTERNAL_SERVER_ERROR,
82                },
83                upload::Error::Internal => {
84                    debug_assert!(false);
85                    relay_log::error!(error = &error as &dyn std::error::Error);
86                    StatusCode::INTERNAL_SERVER_ERROR
87                }
88            },
89        }
90        .into_response()
91    }
92}

94impl IntoResponse for SignedLocation {
95    fn into_response(self) -> Response {
96        let mut headers = tus::response_headers();
97        match self.into_header_value() {
98            Ok(uri) => headers.insert(header::LOCATION, uri),
99            Err(_) => return StatusCode::INTERNAL_SERVER_ERROR.into_response(),
100        };
101
102        (StatusCode::CREATED, headers, ()).into_response()
103    }
104}

106/// Handles TUS upload requests (Creation With Upload).
107///
108/// This endpoint accepts POST requests with a body containing the complete upload data.
109/// Unlike the full TUS protocol, this implementation only supports uploading all data
110/// in a single request - partial uploads and resumption are not supported.
111///
112/// The body is processed as a stream to avoid loading the entire upload into memory.
113async fn handle(
114    state: ServiceState,
115    meta: RequestMeta,
116    headers: HeaderMap,
117    body: Body,
118) -> axum::response::Result<impl IntoResponse> {
119    let upload_length =
120        tus::validate_headers(&headers, meta.request_trust().is_trusted()).map_err(Error::from)?;
121    let config = state.config();
122
123    if upload_length.is_some_and(|len| len > config.max_upload_size()) {
124        return Err(StatusCode::PAYLOAD_TOO_LARGE.into());
125    }
126
127    // There is no real "fast path" for streaming uploads. Always wait for the project config
128    // to be loaded:
129    let project = state
130        .project_cache_handle()
131        .ready(meta.public_key(), config.query_timeout()) // uses same timeout as `Upstream`
132        .await
133        .ok_or(StatusCode::SERVICE_UNAVAILABLE)?;
134
135    let scoping = check_request(&state, meta, upload_length, project).await?;
136    let stream = body
137        .into_data_stream()
138        .map(|result| result.map_err(io::Error::other))
139        .boxed();
140    let (lower_bound, upper_bound) = match upload_length {
141        None => (1, config.max_upload_size()),
142        Some(u) => (u, u),
143    };
144    let stream = BoundedStream::new(stream, lower_bound, upper_bound);
145    let byte_counter = stream.byte_counter();
146
147    let location = upload::Sink::new(&state)
148        .upload(config, upload::Stream { scoping, stream })
149        .await
150        .map_err(|e| {
151            relay_log::warn!(error = &e as &dyn std::error::Error, "upload failed");
152            Error::from(e)
153        })?;
154
155    let mut response = location.into_response();
156    response
157        .headers_mut()
158        .insert(tus::UPLOAD_OFFSET, byte_counter.get().into());
159
160    Ok(response)
161}

163/// Check request by converting it into a pseudo-envelope.
164///
165/// This is currently the easiest way to guarantee that the upload gets checked the same way as
166/// the envelope.
167async fn check_request(
168    state: &ServiceState,
169    meta: RequestMeta,
170    upload_length: Option<usize>,
171    project: Project<'_>,
172) -> Result<Scoping, BadStoreRequest> {
173    let mut envelope = Envelope::from_request(None, meta);
174    envelope.require_feature(Feature::UploadEndpoint);
175    let mut item = Item::new(ItemType::Attachment);
176    item.set_payload(ContentType::AttachmentRef, vec![]);
177    item.set_attachment_length(upload_length.unwrap_or(1) as u64);
178    envelope.add_item(item);
179    let mut envelope = Managed::from_envelope(envelope, state.outcome_aggregator().clone());
180    let rate_limits = project
181        .check_envelope(&mut envelope)
182        .await
183        .map_err(|err| err.map(BadStoreRequest::EventRejected).into_inner())?;
184    if envelope.is_empty() {
185        return Err(envelope
186            .reject_err((None, BadStoreRequest::RateLimited(rate_limits)))
187            .into_inner());
188    }
189
190    // We are not really processing an envelope here, only keep the updated scoping:
191    let scoping = envelope.scoping();
192    envelope.accept(|x| x);
193    Ok(scoping)
194}

196pub fn route(config: &Config) -> MethodRouter<ServiceState> {
197    post(handle).route_layer(RequestBodyLimitLayer::new(config.max_upload_size()))
198}

200fn is_hyper_user_error(error: &(dyn std::error::Error + 'static)) -> bool {
201    error
202        .downcast_ref::<hyper::Error>()
203        .is_some_and(hyper::Error::is_user)
