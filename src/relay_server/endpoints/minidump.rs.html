53/// Minidump attachments should have these magic bytes, little- and big-endian.
54const MINIDUMP_MAGIC_HEADER_LE: &[u8] = b"MDMP";
55const MINIDUMP_MAGIC_HEADER_BE: &[u8] = b"PMDM";
56
57/// Magic bytes for gzip compressed minidump containers.
58const GZIP_MAGIC_HEADER: &[u8] = b"\x1F\x8B";
59/// Magic bytes for xz compressed minidump containers.
60const XZ_MAGIC_HEADER: &[u8] = b"\xFD\x37\x7A\x58\x5A\x00";
61/// Magic bytes for bzip2 compressed minidump containers.
62const BZIP2_MAGIC_HEADER: &[u8] = b"\x42\x5A\x68";
63/// Magic bytes for zstd compressed minidump containers.
64const ZSTD_MAGIC_HEADER: &[u8] = b"\x28\xB5\x2F\xFD";
65
66/// Longest magic header we recognize (XZ is 6 bytes).
67const MAGIC_PEEK: usize = 6;
68
69/// Content types by which standalone uploads can be recognized.
70const MINIDUMP_RAW_CONTENT_TYPES: &[&str] = &["application/octet-stream", "application/x-dmp"];

73enum PeekError<E> {
74    #[error("compressed minidump payloads are not supported for streaming upload")]
75    Compressed,
76    #[error(transparent)]
77    Source(#[from] E),
78}

80/// Peek the first bytes of `stream` and reject if they look compressed (gzip/xz/bzip2/zstd).
81/// Returns the original stream contents if not.
82async fn reject_if_compressed<S, E>(
83    stream: S,
84) -> Result<impl Stream<Item = Result<Bytes, E>> + Send, PeekError<E>>
85where
86    S: Stream<Item = Result<Bytes, E>> + Send,
87    E: Send,
88{
89    let (head, stream) = utils::stream::peek_n(stream, MAGIC_PEEK).await?;
90
91    if head.starts_with(GZIP_MAGIC_HEADER)
92        || head.starts_with(XZ_MAGIC_HEADER)
93        || head.starts_with(BZIP2_MAGIC_HEADER)
94        || head.starts_with(ZSTD_MAGIC_HEADER)
95    {
96        Err(PeekError::Compressed)
97    } else {
98        Ok(stream)
99    }
100}

102fn validate_minidump(data: &[u8]) -> Result<(), BadStoreRequest> {
103    if !data.starts_with(MINIDUMP_MAGIC_HEADER_LE) && !data.starts_with(MINIDUMP_MAGIC_HEADER_BE) {
104        relay_log::trace!("invalid minidump file");
105        return Err(BadStoreRequest::InvalidMinidump);
106    }
107
108    Ok(())
109}

111/// Convenience wrapper to let a decoder decode its full input into a buffer.
112///
113/// Stops reading once `max_size` is exceeded and returns an error. This prevents
114/// decompression bombs from exhausting memory.
115fn run_decoder(mut decoder: impl Read) -> std::io::Result<Vec<u8>> {
116    let mut buffer = Vec::new();
117    decoder.read_to_end(&mut buffer)?;
118    Ok(buffer)
119}

121/// Creates a decoder based on the magic bytes the minidump payload
122fn decoder_from(minidump_data: Bytes) -> Option<Box<dyn Read>> {
123    if minidump_data.starts_with(GZIP_MAGIC_HEADER) {
124        return Some(Box::new(GzDecoder::new(Cursor::new(minidump_data))));
125    } else if minidump_data.starts_with(XZ_MAGIC_HEADER) {
126        return Some(Box::new(XzDecoder::new(Cursor::new(minidump_data))));
127    } else if minidump_data.starts_with(BZIP2_MAGIC_HEADER) {
128        return Some(Box::new(BzDecoder::new(Cursor::new(minidump_data))));
129    } else if minidump_data.starts_with(ZSTD_MAGIC_HEADER) {
130        return match ZstdDecoder::new(Cursor::new(minidump_data)) {
131            Ok(decoder) => Some(Box::new(decoder)),
132            Err(ref err) => {
133                relay_log::error!(error = err as &dyn Error, "failed to create ZstdDecoder");
134                None
135            }
136        };
137    }
138
139    None
140}

142/// Tries to decode a minidump using any of the supported compression formats
143/// or returns the provided minidump payload untouched if no format where detected.
144///
145/// Returns an `Overflow` error if the decompressed size exceeds `max_size`.
146fn decode_minidump(minidump_data: Bytes, max_size: usize) -> Result<Bytes, BadStoreRequest> {
147    let Some(decoder) = decoder_from(minidump_data.clone()) else {
148        // this means we haven't detected any compression container
149        // proceed to process the payload untouched (as a plain minidump).
150        return Ok(minidump_data);
151    };
152
153    // Determine if this is a niche use-case of if this happens frequently.
154    relay_statsd::metric!(counter(RelayCounters::CompressedMinidump) += 1);
155
156    let decoder = decoder.take(max_size.saturating_add(1) as u64);
157
158    match run_decoder(decoder) {
159        Ok(decoded) => {
160            if decoded.len() > max_size {
161                let item_type = DiscardItemType::Attachment(DiscardAttachmentType::Minidump);
162                return Err(BadStoreRequest::ItemTooLarge(item_type));
163            }
164            Ok(Bytes::from(decoded))
165        }
166        Err(err) => {
167            // we detected a compression container but failed to decode it
168            relay_log::trace!("invalid compression container");
169            Err(BadStoreRequest::InvalidCompression(err))
170        }
171    }
172}

174/// Removes any compression container file extensions from the minidump
175/// filename so it can be updated in the item. Otherwise, attachments that
176/// have been decoded would still show the extension in the UI, which is misleading.
177fn remove_container_extension(filename: &str) -> &str {
178    [".gz", ".xz", ".bz2", ".zst"]
179        .into_iter()
180        .find_map(|suffix| filename.strip_suffix(suffix))
181        .unwrap_or(filename)
182}

184/// Extract a minidump from a nested multipart form.
185///
186/// This field is not a minidump (i.e. it doesn't start with the minidump magic header). It could be
187/// a multipart field containing a minidump; this happens in old versions of the Linux Electron SDK.
188///
189/// Unfortunately, the embedded multipart field is not recognized by the multipart parser as a
190/// multipart field containing a multipart body. For this case we will look if the field starts with
191/// a '--' and manually extract the boundary (which is what follows '--' up to the end of line) and
192/// manually construct a multipart with the detected boundary. If we can extract a multipart with an
193/// embedded minidump, then use that field.
194async fn extract_embedded_minidump(payload: Bytes) -> Result<Option<Bytes>, BadStoreRequest> {
195    let boundary = match utils::get_multipart_boundary(&payload) {
196        Some(boundary) => boundary,
197        None => return Ok(None),
198    };
199
200    let stream = futures::stream::once(async { Ok::<_, Infallible>(payload.clone()) });
201    let mut multipart = Multipart::new(stream, boundary);
202
203    while let Some(field) = multipart.next_field().await? {
204        if field.name() == Some(MINIDUMP_FIELD_NAME) {
205            return Ok(Some(field.bytes().await?));
206        }
207    }
208
209    Ok(None)
210}

212#[derive(Clone, Debug)]
213enum UploadDecision {
214    /// Put the item into the envelope as-is (default behavior).
215    Inline,
216    /// Upload the item to objectstore and put a placeholder into the envelope.
217    ///
218    /// The behavior for supersized items.
219    Upload,
220    /// Drop the item rather than uploading it or putting it in an envelope.
221    ///
222    /// Since we already check the quota in the endpoint an item can be dropped even before being
223    /// uploaded or put into an envelope.
224    Drop(RateLimits),
225}

234impl UploadContext<'_> {
235    fn upload_decision(&self, attachment_type: Option<AttachmentType>) -> &UploadDecision {
236        match attachment_type {

251impl<'a> AttachmentStrategy for MinidumpAttachmentStrategy<'a> {
252    async fn add_to_item(
253        &self,
254        field: Field<'static>,
255        item: Managed<Item>,
256        config: &Config,
257    ) -> Result<Option<Managed<Item>>, BadStoreRequest> {
258        let read_inline = async |field: Field<'static>, item: Managed<Item>| {
259            let is_minidump = matches!(item.attachment_type(), Some(AttachmentType::Minidump));
260            match read_bytes_into_item(field, item, config).await {
261                // Don't bubble up errors caused by large items unless it is the minidump itself.
262                Err(multer::Error::FieldSizeExceeded { .. }) if !is_minidump => Ok(None),
263                r => Ok(Some(r?)),
264            }
265        };
266
267        // If we have no upload context just fall back to the old behavior.
268        let Some(ref upload_context) = self.upload_context else {
269            return read_inline(field, item).await;
270        };
271
272        match upload_context.upload_decision(item.attachment_type()) {
273            UploadDecision::Upload => {
274                let is_minidump = matches!(item.attachment_type(), Some(AttachmentType::Minidump));
275                let content_type = field.content_type().map(ToString::to_string);
276                match upload_to_objectstore_checked(
277                    field,
278                    content_type,
279                    item,
280                    config,
281                    upload_context.project.clone(),
282                    upload_context.upload,
283                    "minidump",
284                )
285                .await
286                {
287                    Ok(item) => Ok(Some(item)),
288                    // A failed minidump upload should cause the entire request to be rejected.
289                    Err(e) if is_minidump => Err(e),
290                    // A failed attachment upload should not cause the entire request to be rejected.
291                    Err(_) => Ok(None),
292                }
293            }
294            UploadDecision::Inline => read_inline(field, item).await,
295            UploadDecision::Drop(limits) => {
296                // This is best effort, the item here does not yet have its content set hence size
297                // is not correct.
298                let _ = item.reject_err(Outcome::RateLimited(
299                    limits.longest().and_then(|l| l.reason_code.clone()),
300                ));
301                Ok(None)
302            }
303        }
304    }
305
306    fn infer_type(&self, field: &Field) -> AttachmentType {
307        match field.name().unwrap_or("") {
308            MINIDUMP_FIELD_NAME => AttachmentType::Minidump,
309            ITEM_NAME_BREADCRUMBS1 => AttachmentType::Breadcrumbs,
310            ITEM_NAME_BREADCRUMBS2 => AttachmentType::Breadcrumbs,
311            ITEM_NAME_EVENT => AttachmentType::EventPayload,
312            VIEW_HIERARCHY_FIELD_NAME => AttachmentType::ViewHierarchy,
313            _ => AttachmentType::Attachment,
314        }
315    }
316}

318/// Wrapper around [`upload_to_objectstore`] that enforces that minidumps are not compressed.
319pub async fn upload_to_objectstore_checked<S, E>(
320    stream: S,
321    content_type: Option<String>,
322    item: Managed<Item>,
323    config: &Config,
324    project: ProjectContext,
325    upload: &Addr<Upload>,
326    referrer: &'static str,
327) -> Result<Managed<Item>, BadStoreRequest>
328where
329    S: futures::Stream<Item = Result<Bytes, E>> + Send + 'static,
330    E: Into<Box<dyn std::error::Error + Send + Sync>> + Send + 'static,
331{
332    if !matches!(item.attachment_type(), Some(AttachmentType::Minidump)) {
333        return upload_to_objectstore(
334            stream,
335            content_type,
336            item,
337            config,
338            project,
339            upload,
340            referrer,
341        )
342        .await
343        .map_err(|_| BadStoreRequest::ObjectstoreUploadFailed);
344    }
345
346    let stream = match reject_if_compressed(stream).await {
347        Ok(stream) => stream,
348        Err(_) => {
349            let _ = item.reject_err(Outcome::Invalid(DiscardReason::InvalidMinidump));
350            return Err(BadStoreRequest::InvalidMinidump);
351        }
352    };
353
354    upload_to_objectstore(
355        stream,
356        content_type,
357        item,
358        config,
359        project,
360        upload,
361        referrer,
362    )
363    .await
364    .map_err(|_| BadStoreRequest::ObjectstoreUploadFailed)
365}

367async fn multipart_to_items(
368    multipart: Multipart<'static>,
369    meta: &RequestMeta,
370    state: &ServiceState,
371    upload_context: Option<UploadContext<'_>>,
372) -> Result<Managed<Items>, BadStoreRequest> {
373    let minidump_attachment_strategy = MinidumpAttachmentStrategy { upload_context };
374    let config = state.config();
375
376    let mut items = utils::multipart_items(
377        multipart,
378        config,
379        minidump_attachment_strategy,
380        meta,
381        state.outcome_aggregator(),
382    )
383    .await?;
384
385    let minidump_idx = items
386        .iter()
387        .position(|item| item.attachment_type() == Some(AttachmentType::Minidump))
388        .ok_or(BadStoreRequest::MissingMinidump)
389        .reject(&items)?;
390
391    let minidump_item = items
392        .get(minidump_idx)
393        .ok_or(BadStoreRequest::MissingMinidump)
394        .reject(&items)?;
395    // Doing these operations does not make sense if we already streamed the minidump to objectstore.
396    if !minidump_item.is_attachment_ref() {
397        let payload = minidump_item.payload();
398        let payload = extract_embedded_minidump(payload.clone())
399            .await
400            .reject(&items)?
401            .unwrap_or(payload);
402        let payload = decode_minidump(payload, config.max_attachment_size()).reject(&items)?;
403
404        items.try_modify(|items, records| -> Result<(), BadStoreRequest> {
405            let minidump_item = items
406                .get_mut(minidump_idx)
407                .ok_or(BadStoreRequest::MissingMinidump)?;
408            minidump_item.set_payload(ContentType::Minidump, payload);
409            records.lenient(DataCategory::Attachment); // decoding the minidump changes its size
410            if let Some(minidump_filename) = minidump_item.filename() {
411                minidump_item.set_filename(remove_container_extension(minidump_filename).to_owned())
412            }
413            validate_minidump(&minidump_item.payload())?;
414            Ok(())
415        })?;
416    }
417
418    Ok(items)
419}

421/// Creates an [UploadContext].
422async fn upload_context<'a>(
423    meta: &RequestMeta,
424    state: &'a ServiceState,
425) -> Result<Option<UploadContext<'a>>, BadStoreRequest> {
426    if !state
427        .global_config_handle()
428        .current()
429        .unwrap_or_default()
430        .options
431        .endpoint_fetch_config_enabled
432    {
433        return Ok(None);
434    }
435
436    let project = state
437        .project_cache_handle()
438        .ready(meta.public_key(), state.config().query_timeout())
439        .await
440        .ok_or(BadStoreRequest::ProjectUnavailable)?;
441
442    let project_config = match project.state() {
443        ProjectState::Enabled(info) => info.clone(),
444        // Note: In Proxy mode we should never make it here since the endpoint_fetch_config_enabled
445        // check should already fail.
446        ProjectState::Dummy => return Ok(None),
447        ProjectState::Disabled | ProjectState::Pending => {
448            return Err(BadStoreRequest::EventRejected(DiscardReason::ProjectId));
449        }
450    };
451
452    let scoping = project_config
453        .scoping(meta.public_key())
454        .ok_or(BadStoreRequest::EventRejected(DiscardReason::ProjectId))?;
455
456    let rate_limits = project.rate_limits().current_limits().check_with_quotas(
457        project_config.get_quotas(),
458        scoping.item(DataCategory::Error),
459    );
460
461    let attachment_rate_limits = project.rate_limits().current_limits().check_with_quotas(
462        project_config.get_quotas(),
463        scoping.item(DataCategory::Attachment),
464    );
465
466    let upload_minidumps = if !project_config.has_feature(Feature::MinidumpUploads) {
467        UploadDecision::Inline
468    } else if rate_limits.is_limited() {
469        UploadDecision::Drop(rate_limits.clone())
470    } else {
471        UploadDecision::Upload
472    };
473
474    let upload_attachments = if matches!(upload_minidumps, UploadDecision::Drop(_)) {
475        UploadDecision::Drop(rate_limits)
476    } else if !project_config.has_feature(Feature::MinidumpAttachmentUploads) {
477        UploadDecision::Inline
478    } else if attachment_rate_limits.is_limited() {
479        UploadDecision::Drop(attachment_rate_limits)
480    } else {
481        UploadDecision::Upload
482    };
483
484    Ok(Some(UploadContext {
485        upload: state.upload(),
486        project: ProjectContext {
487            scoping,
488            upstream: project_config.upstream.clone(),
489        },
490        upload_attachments,
491        upload_minidumps,
492    }))
493}

495async fn raw_minidump_to_item(
496    request: Request,
497    meta: &RequestMeta,
498    state: &ServiceState,
499    content_type: RawContentType,
500    upload_context: Option<UploadContext<'_>>,
501) -> Result<Managed<Item>, BadStoreRequest> {
502    debug_assert!(!matches!(
503        upload_context.as_ref().map(|c| &c.upload_minidumps),
504        Some(UploadDecision::Drop(_))
505    ));
506
507    let mut item = Item::new(ItemType::Attachment);
508    item.set_filename(MINIDUMP_FILE_NAME);
509    item.set_attachment_type(AttachmentType::Minidump);
510    let mut item = Managed::with_meta_from_request_meta(meta, state.outcome_aggregator(), item);
511    if let Some(upload_context) = upload_context
512        && matches!(upload_context.upload_minidumps, UploadDecision::Upload)
513    {
514        let stream = reject_if_compressed(request.into_body().into_data_stream())
515            .await
516            .map_err(|_| BadStoreRequest::InvalidMinidump)?;
517
518        item = upload_to_objectstore(
519            stream,
520            Some(content_type.to_string()).filter(|s| !s.is_empty()),
521            item,
522            state.config(),
523            upload_context.project,
524            upload_context.upload,
525            "minidump",
526        )
527        .await
528        .map_err(|_| BadStoreRequest::ObjectstoreUploadFailed)?;
529    } else {
530        let minidump_data = request.extract().await?;
531        item.try_modify(|inner, records| -> Result<(), BadStoreRequest> {
532            let payload = decode_minidump(minidump_data, state.config().max_attachment_size())?;
533            inner.set_payload(ContentType::Minidump, payload);
534            records.lenient(DataCategory::Attachment); // decoding the minidump changes its size
535            validate_minidump(&inner.payload())?;
536            Ok(())
537        })?;
538    };
539
540    Ok(item)
541}

543async fn items(
544    upload_context: Option<UploadContext<'_>>,
545    state: &ServiceState,
546    meta: &RequestMeta,
547    content_type: RawContentType,
548    request: Request,
549) -> Result<Managed<Items>, BadStoreRequest> {
550    let items = if MINIDUMP_RAW_CONTENT_TYPES.contains(&content_type.as_ref()) {
551        raw_minidump_to_item(request, meta, state, content_type, upload_context)
552            .await?
553            .map(|item, _| smallvec![item])
554    } else {
555        let multipart = utils::multipart_from_request(request)?;
556        multipart_to_items(multipart, meta, state, upload_context).await?
557    };
558    Ok(items)
559}

561fn envelope(
562    items: Managed<Items>,
563    meta: RequestMeta,
564    managed_err: Managed<(DataCategory, usize)>,
565) -> Result<Managed<Box<Envelope>>, BadStoreRequest> {
566    let event_id = common::event_id_from_items(&items)
567        .reject2(&items, &managed_err)?
568        .unwrap_or_else(EventId::new);
569    let envelope = items.map(|items, records| {
570        managed_err.accept(|_| ()); // There will be an envelope with (DataCategory::Error, 1) now
571        records.modify_by(DataCategory::Error, 1);
572        Box::new(Envelope::from_request(Some(event_id), meta).with_items(items))
573    });
574    Ok(envelope)
575}

577async fn handle(
578    state: ServiceState,
579    meta: RequestMeta,
580    content_type: RawContentType,
581    request: Request,
582) -> axum::response::Result<impl IntoResponse> {
583    // The minidump can either be transmitted as the request body, or as
584    // `upload_file_minidump` in a multipart form-data/ request.
585    // Minidump request payloads do not have the same structure as usual events from other SDKs. The
586    // minidump can either be transmitted as request body, or as `upload_file_minidump` in a
587    // multipart formdata request.
588
589    // If something goes wrong before the envelope is created, this managed error ensures an
590    // outcome is emitted. This is a best effort outcome, in the sense that we also drop some
591    // attachments but since we know neither the amount or size we can't emit an accurate outcome
592    // for them.
593    let err = (DataCategory::Error, 1);
594    let managed_err = Managed::with_meta_from_request_meta(&meta, state.outcome_aggregator(), err);
595
596    let upload_context = upload_context(&meta, &state).await.reject(&managed_err)?;
597
598    if let Some(upload_context) = &upload_context
599        && let UploadDecision::Drop(limits) = &upload_context.upload_minidumps
600    {
601        let _ = managed_err.reject_err(Outcome::RateLimited(
602            limits.longest().and_then(|l| l.reason_code.clone()),
603        ));
604        return Ok(TextResponse(Some(EventId::new())));
605    }
606    let items = items(upload_context, &state, &meta, content_type, request)
607        .await
608        .reject(&managed_err)?;
609    let envelope = envelope(items, meta, managed_err)?;
610
611    let id = envelope.event_id();
612
613    // Never respond with a 429 since clients often retry these
614    common::handle_managed_envelope(&state, envelope)
615        .await?
616        .ignore_rate_limits();
617
618    // The return here is only useful for consistency because the UE4 crash reporter doesn't
619    // care about it.
620    Ok(TextResponse(id))
621}
622
623pub fn route(config: &Config) -> MethodRouter<ServiceState> {
624    post(handle)
625        .route_layer(RequestBodyLimitLayer::new(
626            config.max_upload_size() + config.max_attachments_size(),
627        ))
628        .route_layer(DefaultBodyLimit::disable())
629        .route_layer(axum::middleware::from_fn(middlewares::content_length))
630}

662    fn encode_gzip(be_minidump: &[u8]) -> Result<Bytes, Box<dyn std::error::Error>> {
663        let mut encoder = GzEncoder::new(Vec::new(), GzCompression::default());
664        encoder.write_all(be_minidump)?;
665        let compressed = encoder.finish()?;

667    }

670        encoder.write_all(be_minidump)?;
671        let compressed = encoder.finish()?;
672        Ok(Bytes::from(compressed))
673    }
674    fn encode_xz(be_minidump: &[u8]) -> Result<Bytes, Box<dyn std::error::Error>> {

681        let mut encoder = ZstdEncoder::new(Vec::new(), 0)?;

689        let encoders: Vec<EncodeFunction> = vec![encode_gzip, encode_zst, encode_bzip, encode_xz];
690        for encoder in &encoders {
691            let be_minidump = b"PMDMxxxxxx";
692            let compressed = encoder(be_minidump)?;
693            let decoder = decoder_from(compressed).unwrap();

697            let compressed = encoder(le_minidump)?;

700
701            let garbage = b"xxxxxx";

703            let decoder = decoder_from(compressed).unwrap();
704            let decoded = run_decoder(decoder);
705            assert!(decoded.is_ok());
706            assert!(validate_minidump(&decoded.unwrap()).is_err());
707        }

715        let minidump_data = b"xxxxxxxxxx".repeat(10);
716        let compressed = encode_gzip(&minidump_data)?;
717
718        // With a limit larger than the decompressed size, decoding should succeed
719        let result = decode_minidump(compressed.clone(), 200);
