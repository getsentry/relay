28/// The field name of a minidump in the multipart form-data upload.
29///
30/// Sentry requires
31const MINIDUMP_FIELD_NAME: &str = "upload_file_minidump";
32
33/// The field name of a view hierarchy file in the multipart form-data upload.
34/// It matches the expected file name of the view hierarchy, as outlined in RFC#33
35/// <https://github.com/getsentry/rfcs/blob/main/text/0033-view-hierarchy.md>
36const VIEW_HIERARCHY_FIELD_NAME: &str = "view-hierarchy.json";

38/// File name for a standalone minidump upload.
39///
40/// In contrast to the field name, this is used when a standalone minidump is uploaded not in a
41/// multipart request. The file name is later used to display the event attachment.
42const MINIDUMP_FILE_NAME: &str = "Minidump";
43
44/// Minidump attachments should have these magic bytes, little- and big-endian.
45const MINIDUMP_MAGIC_HEADER_LE: &[u8] = b"MDMP";
46const MINIDUMP_MAGIC_HEADER_BE: &[u8] = b"PMDM";
47
48/// Magic bytes for gzip compressed minidump containers.
49const GZIP_MAGIC_HEADER: &[u8] = b"\x1F\x8B";
50/// Magic bytes for xz compressed minidump containers.
51const XZ_MAGIC_HEADER: &[u8] = b"\xFD\x37\x7A\x58\x5A\x00";
52/// Magic bytes for bzip2 compressed minidump containers.
53const BZIP2_MAGIC_HEADER: &[u8] = b"\x42\x5A\x68";
54/// Magic bytes for zstd compressed minidump containers.
55const ZSTD_MAGIC_HEADER: &[u8] = b"\x28\xB5\x2F\xFD";
56
57/// Content types by which standalone uploads can be recognized.
58const MINIDUMP_RAW_CONTENT_TYPES: &[&str] = &["application/octet-stream", "application/x-dmp"];
59
60fn validate_minidump(data: &[u8]) -> Result<(), BadStoreRequest> {
61    if !data.starts_with(MINIDUMP_MAGIC_HEADER_LE) && !data.starts_with(MINIDUMP_MAGIC_HEADER_BE) {
62        relay_log::trace!("invalid minidump file");
63        return Err(BadStoreRequest::InvalidMinidump);
64    }
65
66    Ok(())
67}

69/// Convenience wrapper to let a decoder decode its full input into a buffer.
70///
71/// Stops reading once `max_size` is exceeded and returns an error. This prevents
72/// decompression bombs from exhausting memory.
73fn run_decoder(mut decoder: impl Read) -> std::io::Result<Vec<u8>> {
74    let mut buffer = Vec::new();
75    decoder.read_to_end(&mut buffer)?;
76    Ok(buffer)
77}
78
79/// Creates a decoder based on the magic bytes the minidump payload
80fn decoder_from(minidump_data: Bytes) -> Option<Box<dyn Read>> {
81    if minidump_data.starts_with(GZIP_MAGIC_HEADER) {
82        return Some(Box::new(GzDecoder::new(Cursor::new(minidump_data))));
83    } else if minidump_data.starts_with(XZ_MAGIC_HEADER) {
84        return Some(Box::new(XzDecoder::new(Cursor::new(minidump_data))));
85    } else if minidump_data.starts_with(BZIP2_MAGIC_HEADER) {
86        return Some(Box::new(BzDecoder::new(Cursor::new(minidump_data))));
87    } else if minidump_data.starts_with(ZSTD_MAGIC_HEADER) {
88        return match ZstdDecoder::new(Cursor::new(minidump_data)) {
89            Ok(decoder) => Some(Box::new(decoder)),
90            Err(ref err) => {
91                relay_log::error!(error = err as &dyn Error, "failed to create ZstdDecoder");
92                None
93            }
94        };
95    }
96
97    None
98}

100/// Tries to decode a minidump using any of the supported compression formats
101/// or returns the provided minidump payload untouched if no format where detected.
102///
103/// Returns an `Overflow` error if the decompressed size exceeds `max_size`.
104fn decode_minidump(minidump_data: Bytes, max_size: usize) -> Result<Bytes, BadStoreRequest> {
105    let Some(decoder) = decoder_from(minidump_data.clone()) else {
106        // this means we haven't detected any compression container
107        // proceed to process the payload untouched (as a plain minidump).
108        return Ok(minidump_data);
109    };
110
111    let decoder = decoder.take(max_size.saturating_add(1) as u64);
112
113    match run_decoder(decoder) {
114        Ok(decoded) => {
115            if decoded.len() > max_size {
116                return Err(BadStoreRequest::Overflow(DiscardItemType::Attachment(
117                    DiscardAttachmentType::Minidump,
118                )));
119            }
120            Ok(Bytes::from(decoded))
121        }
122        Err(err) => {
123            // we detected a compression container but failed to decode it
124            relay_log::trace!("invalid compression container");
125            Err(BadStoreRequest::InvalidCompressionContainer(err))
126        }
127    }
128}

130/// Removes any compression container file extensions from the minidump
131/// filename so it can be updated in the item. Otherwise, attachments that
132/// have been decoded would still show the extension in the UI, which is misleading.
133fn remove_container_extension(filename: &str) -> &str {
134    [".gz", ".xz", ".bz2", ".zst"]
135        .into_iter()
136        .find_map(|suffix| filename.strip_suffix(suffix))
137        .unwrap_or(filename)
138}

140fn infer_attachment_type(field_name: Option<&str>, _file_name: &str) -> AttachmentType {
141    match field_name.unwrap_or("") {
142        MINIDUMP_FIELD_NAME => AttachmentType::Minidump,
143        ITEM_NAME_BREADCRUMBS1 => AttachmentType::Breadcrumbs,
144        ITEM_NAME_BREADCRUMBS2 => AttachmentType::Breadcrumbs,
145        ITEM_NAME_EVENT => AttachmentType::EventPayload,
146        VIEW_HIERARCHY_FIELD_NAME => AttachmentType::ViewHierarchy,
147        _ => AttachmentType::Attachment,
148    }
149}

151/// Extract a minidump from a nested multipart form.
152///
153/// This field is not a minidump (i.e. it doesn't start with the minidump magic header). It could be
154/// a multipart field containing a minidump; this happens in old versions of the Linux Electron SDK.
155///
156/// Unfortunately, the embedded multipart field is not recognized by the multipart parser as a
157/// multipart field containing a multipart body. For this case we will look if the field starts with
158/// a '--' and manually extract the boundary (which is what follows '--' up to the end of line) and
159/// manually construct a multipart with the detected boundary. If we can extract a multipart with an
160/// embedded minidump, then use that field.
161async fn extract_embedded_minidump(payload: Bytes) -> Result<Option<Bytes>, BadStoreRequest> {
162    let boundary = match utils::get_multipart_boundary(&payload) {
163        Some(boundary) => boundary,
164        None => return Ok(None),
165    };
166
167    let stream = futures::stream::once(async { Ok::<_, Infallible>(payload.clone()) });
168    let mut multipart = Multipart::new(stream, boundary);
169
170    while let Some(field) = multipart.next_field().await? {
171        if field.name() == Some(MINIDUMP_FIELD_NAME) {
172            return Ok(Some(field.bytes().await?));
173        }
174    }
175
176    Ok(None)
177}

179async fn extract_multipart(
180    multipart: ConstrainedMultipart,
181    meta: RequestMeta,
182    config: &Config,
183) -> Result<Box<Envelope>, BadStoreRequest> {
184    let mut items = multipart.items(infer_attachment_type, config).await?;
185
186    let minidump_item = items
187        .iter_mut()
188        .find(|item| item.attachment_type() == Some(&AttachmentType::Minidump))
189        .ok_or(BadStoreRequest::MissingMinidump)?;
190
191    let embedded_opt = extract_embedded_minidump(minidump_item.payload()).await?;
192    if let Some(embedded) = embedded_opt {
193        minidump_item.set_payload(Minidump, embedded);
194    }
195
196    minidump_item.set_payload(
197        Minidump,
198        decode_minidump(minidump_item.payload(), config.max_attachment_size())?,
199    );
200
201    validate_minidump(&minidump_item.payload())?;
202
203    if let Some(minidump_filename) = minidump_item.filename() {
204        minidump_item.set_filename(remove_container_extension(minidump_filename).to_owned());
205    }
206
207    let event_id = common::event_id_from_items(&items)?.unwrap_or_else(EventId::new);
208    let mut envelope = Envelope::from_request(Some(event_id), meta);
209
210    for item in items {
211        envelope.add_item(item);
212    }
213
214    Ok(envelope)
215}

217fn extract_raw_minidump(
218    data: Bytes,
219    meta: RequestMeta,
220    max_size: usize,
221) -> Result<Box<Envelope>, BadStoreRequest> {
222    let mut item = Item::new(ItemType::Attachment);
223
224    item.set_payload(Minidump, decode_minidump(data, max_size)?);
225    validate_minidump(&item.payload())?;
226    item.set_filename(MINIDUMP_FILE_NAME);
227    item.set_attachment_type(AttachmentType::Minidump);
228
229    // Create an envelope with a random event id.
230    let mut envelope = Envelope::from_request(Some(EventId::new()), meta);
231    envelope.add_item(item);
232    Ok(envelope)
233}

235async fn handle(
236    state: ServiceState,
237    meta: RequestMeta,
238    content_type: RawContentType,
239    request: Request,
240) -> axum::response::Result<impl IntoResponse> {
241    // The minidump can either be transmitted as the request body, or as
242    // `upload_file_minidump` in a multipart form-data/ request.
243    // Minidump request payloads do not have the same structure as usual events from other SDKs. The
244    // minidump can either be transmitted as request body, or as `upload_file_minidump` in a
245    // multipart formdata request.
246    let config = state.config();
247    let envelope = if MINIDUMP_RAW_CONTENT_TYPES.contains(&content_type.as_ref()) {
248        extract_raw_minidump(request.extract().await?, meta, config.max_attachment_size())?
249    } else {
250        let multipart = request.extract_with_state(&state).await?;
251        extract_multipart(multipart, meta, config).await?
252    };
253
254    let id = envelope.event_id();
255
256    // Never respond with a 429 since clients often retry these
257    match common::handle_envelope(&state, envelope)
258        .await
259        .map_err(|err| err.into_inner())
260    {
261        Ok(_) | Err(BadStoreRequest::RateLimited(_)) => (),
262        Err(error) => return Err(error.into()),
263    };
264
265    // The return here is only useful for consistency because the UE4 crash reporter doesn't
266    // care about it.
267    Ok(TextResponse(id))
268}

270pub fn route(config: &Config) -> MethodRouter<ServiceState> {
271    // Set the single-attachment limit that applies only for raw minidumps. Multipart bypasses the
272    // limited body and applies its own limits.
273    post(handle)
274        .route_layer(DefaultBodyLimit::max(config.max_attachment_size()))
275        .route_layer(axum::middleware::from_fn(middlewares::content_length))
276}

308    fn encode_gzip(be_minidump: &[u8]) -> Result<Bytes, Box<dyn std::error::Error>> {
309        let mut encoder = GzEncoder::new(Vec::new(), GzCompression::default());
310        encoder.write_all(be_minidump)?;
311        let compressed = encoder.finish()?;
312        Ok(Bytes::from(compressed))
313    }

317        let compressed = encoder.finish()?;
318        Ok(Bytes::from(compressed))
319    }

327        let mut encoder = ZstdEncoder::new(Vec::new(), 0)?;
328        encoder.write_all(be_minidump)?;
329        let compressed = encoder.finish()?;

334    fn test_validate_encoded_minidump() -> Result<(), Box<dyn std::error::Error>> {
335        let encoders: Vec<EncodeFunction> = vec![encode_gzip, encode_zst, encode_bzip, encode_xz];
336        for encoder in &encoders {
337            let be_minidump = b"PMDMxxxxxx";
338            let compressed = encoder(be_minidump)?;
339            let decoder = decoder_from(compressed).unwrap();

346
347            let garbage = b"xxxxxx";

350            let decoded = run_decoder(decoder);
351            assert!(decoded.is_ok());
352            assert!(validate_minidump(&decoded.unwrap()).is_err());

359    fn test_decode_minidump_size_limit() -> Result<(), Box<dyn std::error::Error>> {
360        // Create a minidump that will decompress to 100 bytes
361        let minidump_data = b"xxxxxxxxxx".repeat(10);
362        let compressed = encode_gzip(&minidump_data)?;
363
364        // With a limit larger than the decompressed size, decoding should succeed
365        let result = decode_minidump(compressed.clone(), 200);
366        assert!(result.is_ok());
367        assert_eq!(result.unwrap().len(), 100);
368
369        // With a limit smaller than the decompressed size, decoding should fail with Overflow
370        let result = decode_minidump(compressed, 50);
371        assert!(matches!(result, Err(BadStoreRequest::Overflow(_))));
