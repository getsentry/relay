18#[derive(Debug, Deserialize)]
19struct SecurityReportQuery {
20    sentry_release: Option<String>,
21    sentry_environment: Option<String>,
22}
23
24#[derive(Debug, FromRequest)]
25#[from_request(state(ServiceState))]
26struct SecurityReportParams {
27    meta: RequestMeta,
28    #[from_request(via(Query))]
29    query: SecurityReportQuery,
30    body: Bytes,
31}
32
33impl SecurityReportParams {
34    fn create_security_item(query: &SecurityReportQuery, item: Bytes) -> Item {
35        let mut report_item = Item::new(ItemType::RawSecurity);
36        report_item.set_payload(ContentType::Json, item);
37
38        if let Some(sentry_release) = &query.sentry_release {
39            report_item.set_header("sentry_release", sentry_release.clone());
40        }
41
42        if let Some(sentry_environment) = &query.sentry_environment {
43            report_item.set_header("sentry_environment", sentry_environment.clone());
44        }
45
46        report_item
47    }
48
49    fn extract_envelope(self) -> Result<Box<Envelope>, BadStoreRequest> {
50        let Self { meta, query, body } = self;
51
52        if body.is_empty() {
53            return Err(BadStoreRequest::EmptyBody);
54        }
55
56        let mut envelope = Envelope::from_request(Some(EventId::new()), meta);
57        let variant =
58            serde_json::from_slice::<Vec<&RawValue>>(&body).map_err(BadStoreRequest::InvalidJson);
59
60        if let Ok(items) = variant {
61            for item in items {
62                let report_item =
63                    Self::create_security_item(&query, Bytes::from(item.to_owned().to_string()));
64                envelope.add_item(report_item);
65            }
66        } else {
67            let report_item = Self::create_security_item(&query, body);
68            envelope.add_item(report_item);
69        }
70
71        Ok(envelope)
72    }
73}

75fn is_security_mime(mime: Mime) -> bool {
76    let ty = mime.type_().as_str();
77    let subty = mime.subtype().as_str();
78    let suffix = mime.suffix().map(|suffix| suffix.as_str());
79
80    matches!(
81        (ty, subty, suffix),
82        ("application", "json", None)
83            | ("application", "csp-report", None)
84            | ("application", "expect-ct-report", None)
85            | ("application", "expect-ct-report", Some("json"))
86            | ("application", "expect-staple-report", None)
87            | ("application", "reports", Some("json"))
88    )
89}

91/// This handles all messages coming on the Security endpoint.
92///
93/// The security reports will be checked.
94async fn handle(
95    state: ServiceState,
96    mime: Mime,
97    params: SecurityReportParams,
98) -> Result<impl IntoResponse, BadStoreRequest> {
99    if !is_security_mime(mime) {
100        return Ok(StatusCode::UNSUPPORTED_MEDIA_TYPE.into_response());
101    }
102
103    let envelope = params.extract_envelope()?;
104    common::handle_envelope(&state, envelope).await?;
105    Ok(().into_response())
106}
107
108pub fn route(config: &Config) -> MethodRouter<ServiceState> {
109    post(handle).route_layer(DefaultBodyLimit::max(config.max_event_size()))
