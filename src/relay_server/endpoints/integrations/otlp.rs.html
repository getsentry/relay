14/// All routes configured for the OTLP integration.
15///
16/// The integration currently supports the following endpoints:
17///  - V1 Traces
18///  - V1 Logs
19pub fn routes(config: &Config) -> axum::Router<ServiceState> {
20    axum::Router::new()
21        .route("/v1/traces", traces::route(config))
22        .route("/v1/traces/", traces::route(config))
23        .route("/v1/logs", logs::route(config))
24        .route("/v1/logs/", logs::route(config))
25}
26
27mod traces {
28    use super::*;
29
30    async fn handle(
31        content_type: RawContentType,
32        state: ServiceState,
33        builder: IntegrationBuilder,
34    ) -> axum::response::Result<impl IntoResponse> {
35        let format = match ContentType::from(content_type.as_ref()) {
36            ContentType::Json => OtelFormat::Json,
37            ContentType::Protobuf => OtelFormat::Protobuf,
38            _ => return Ok(StatusCode::UNSUPPORTED_MEDIA_TYPE),
39        };
40
41        let envelope = builder
42            .with_type(SpansIntegration::OtelV1 { format })
43            .with_required_feature(Feature::OtelTracesEndpoint)
44            .build();
45
46        common::handle_envelope(&state, envelope).await?;
47
48        Ok(StatusCode::ACCEPTED)
49    }
50
51    pub fn route(config: &Config) -> MethodRouter<ServiceState> {
52        post(handle).route_layer(DefaultBodyLimit::max(config.max_envelope_size()))
53    }
54}
