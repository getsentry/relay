18/// All routes configured for the OTLP integration.
19///
20/// The integration currently supports the following endpoints:
21///  - V1 Traces
22///  - V1 Logs
23pub fn routes(config: &Config) -> axum::Router<ServiceState> {
24    axum::Router::new()
25        .route("/v1/traces", traces::route(config))
26        .route("/v1/traces/", traces::route(config))
27        .route("/v1/logs", logs::route(config))
28        .route("/v1/logs/", logs::route(config))
29}

31mod traces {
32    use super::*;
33
34    async fn handle(
35        content_type: RawContentType,
36        state: ServiceState,
37        builder: Result<IntegrationBuilder, IntegrationBuilderRejection>,
38    ) -> Result<SuccessResponse<ExportTraceServiceResponse>, OtlpError> {
39        let format = match content_type.as_ref().parse::<ContentType>() {
40            Ok(ContentType::Json) => OtelFormat::Json,
41            Ok(ContentType::Protobuf) => OtelFormat::Protobuf,
42            _ => return Err(OtlpError::unsupported_content_type(content_type)),
43        };
44
45        let envelope = builder
46            .map_err(|err| OtlpError::from_err(format, err))?
47            .with_type(SpansIntegration::OtelV1 { format })
48            .build();
49
50        common::handle_envelope(&state, envelope)
51            .await
52            .map_err(Rejected::into_inner)
53            .and_then(|r| r.check_rate_limits())
54            .map_err(|err| OtlpError::from_err(format, err))?;
55
56        Ok(SuccessResponse::new(format))
57    }
58
59    pub fn route(config: &Config) -> MethodRouter<ServiceState> {
60        post(handle).route_layer(DefaultBodyLimit::max(config.max_spans_integration_size()))
61    }
62}

98struct SuccessResponse<T> {
99    format: OtelFormat,
100    _proto: std::marker::PhantomData<T>,
101}
102
103impl<T> SuccessResponse<T> {
104    fn new(format: OtelFormat) -> Self {
105        Self {
106            format,
107            _proto: std::marker::PhantomData,
108        }
109    }
110}

112impl<T> IntoResponse for SuccessResponse<T>
113where
114    T: prost::Message,
115    T: Default,
116{
117    fn into_response(self) -> Response {
118        match self.format {
119            OtelFormat::Json => {
120                let empty = serde_json::Value::Object(Default::default());
121                axum::Json(empty).into_response()
122            }
123            OtelFormat::Protobuf => Protobuf(T::default()).into_response(),
124        }
125    }
126}

128/// Protobuf-compatible subset of `google.rpc.Status`.
129#[derive(Clone, PartialEq, Eq, serde::Serialize, prost::Message)]
130#[serde(rename_all = "camelCase")]
131struct OtlpStatus {
132    #[prost(int32, tag = "1")]
133    code: i32,
134    #[prost(string, tag = "2")]
135    message: String,
136}
137
138#[repr(i32)]
139#[derive(Clone, Copy, Debug)]
140enum GrpcStatusCode {
141    Unknown = 2,
142    InvalidArgument = 3,
143    DeadlineExceeded = 4,
144    NotFound = 5,
145    PermissionDenied = 7,
146    ResourceExhausted = 8,
147    Unimplemented = 12,
148    Internal = 13,
149    Unavailable = 14,
150    Unauthenticated = 16,
151}
152
153impl GrpcStatusCode {
154    fn from_http_status(status: StatusCode) -> Self {
155        match status {
156            StatusCode::BAD_REQUEST | StatusCode::UNSUPPORTED_MEDIA_TYPE => Self::InvalidArgument,
157            StatusCode::UNAUTHORIZED => Self::Unauthenticated,
158            StatusCode::FORBIDDEN => Self::PermissionDenied,
159            StatusCode::NOT_FOUND => Self::NotFound,
160            StatusCode::PAYLOAD_TOO_LARGE | StatusCode::TOO_MANY_REQUESTS => {
161                Self::ResourceExhausted
162            }
163            StatusCode::NOT_IMPLEMENTED => Self::Unimplemented,
164            StatusCode::BAD_GATEWAY | StatusCode::SERVICE_UNAVAILABLE => Self::Unavailable,
165            StatusCode::GATEWAY_TIMEOUT => Self::DeadlineExceeded,
166            status if status.is_server_error() => Self::Internal,
167            _ => Self::Unknown,
168        }
169    }
170}
171struct OtlpError {
172    format: OtelFormat,
173    status: StatusCode,
174    headers: header::HeaderMap,
175    message: String,
176}
177
178impl OtlpError {
179    fn unsupported_content_type(content_type: RawContentType) -> Self {
180        Self {
181            format: OtelFormat::Json,
182            status: StatusCode::UNSUPPORTED_MEDIA_TYPE,
183            headers: Default::default(),
184            message: format!("unsupported content type: {content_type}"),
185        }
186    }
187
188    fn from_err<E>(format: OtelFormat, err: E) -> Self
189    where
190        E: IntoResponse,
191        E: std::fmt::Display,
192    {
193        let message = err.to_string();
194        let (parts, _) = err.into_response().into_parts();
195
196        Self {
197            format,
198            status: parts.status,
199            headers: parts.headers,
200            message,
201        }
202    }
203}
204
205impl IntoResponse for OtlpError {
206    fn into_response(mut self) -> Response {
207        let status = OtlpStatus {
208            code: GrpcStatusCode::from_http_status(self.status) as i32,
209            message: self.message,
210        };
211
212        let mut response = match self.format {
213            OtelFormat::Json => axum::Json(status).into_response(),
214            OtelFormat::Protobuf => Protobuf(status).into_response(),
215        };
216
217        *response.status_mut() = self.status;
218
219        self.headers.remove(header::CONTENT_LENGTH);
220        self.headers.remove(header::CONTENT_TYPE);
221        response.headers_mut().extend(self.headers);
222
223        response
224    }
225}

227struct Protobuf<T>(T);
228
229impl<T> IntoResponse for Protobuf<T>
230where
231    T: prost::Message,
232{
233    fn into_response(self) -> Response {
234        let body = self.0.encode_to_vec();
235        let mut res = axum::body::Body::from(body).into_response();
236        res.headers_mut().insert(
237            header::CONTENT_TYPE,
238            header::HeaderValue::from_static("application/x-protobuf"),
239        );
240        res
241    }
