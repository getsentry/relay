16/// All routes configured for the OTLP integration.
17///
18/// The integration currently supports the following endpoints:
19///  - V1 Traces
20///  - V1 Logs
21pub fn routes(config: &Config) -> axum::Router<ServiceState> {
22    axum::Router::new()
23        .route("/v1/traces", traces::route(config))
24        .route("/v1/traces/", traces::route(config))
25        .route("/v1/logs", logs::route(config))
26        .route("/v1/logs/", logs::route(config))
27}
28
29// Public until we can remove the old, pre integration, routes.
30pub mod traces {
31    use super::*;
32
33    async fn handle(
34        state: ServiceState,
35        content_type: RawContentType,
36        meta: RequestMeta,
37        request: Request,
38    ) -> axum::response::Result<impl IntoResponse> {
39        let content_type @ (ContentType::Json | ContentType::Protobuf) =
40            ContentType::from(content_type.as_ref())
41        else {
42            return Ok(StatusCode::UNSUPPORTED_MEDIA_TYPE);
43        };
44
45        let payload: Bytes = request.extract().await?;
46        let mut envelope = Envelope::from_request(None, meta);
47        envelope.require_feature(Feature::OtelEndpoint);
48
49        envelope.add_item({
50            let mut item = Item::new(ItemType::OtelTracesData);
51            item.set_payload(content_type, payload);
52            item
53        });
54
55        common::handle_envelope(&state, envelope).await?;
56
57        Ok(StatusCode::ACCEPTED)
58    }
59
60    pub fn route(config: &Config) -> MethodRouter<ServiceState> {
61        post(handle).route_layer(DefaultBodyLimit::max(config.max_envelope_size()))
62    }
63}

65mod logs {
66    use super::*;
67
68    async fn handle(
69        content_type: RawContentType,
70        state: ServiceState,
71        builder: IntegrationBuilder,
72    ) -> axum::response::Result<impl IntoResponse> {
73        let format = match ContentType::from(content_type.as_ref()) {
74            ContentType::Json => OtelFormat::Json,
75            ContentType::Protobuf => OtelFormat::Protobuf,
76            _ => return Ok(StatusCode::UNSUPPORTED_MEDIA_TYPE),
77        };
78
79        let envelope = builder
80            .with_type(LogsIntegration::OtelV1 { format })
81            .with_required_feature(Feature::OtelLogsEndpoint)
82            .build();
83
84        common::handle_envelope(&state, envelope).await?;
85
86        Ok(StatusCode::ACCEPTED)
87    }
88
89    pub fn route(config: &Config) -> MethodRouter<ServiceState> {
90        post(handle).route_layer(DefaultBodyLimit::max(config.max_envelope_size()))
91    }
92}
