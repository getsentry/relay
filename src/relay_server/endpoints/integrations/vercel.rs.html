13/// All routes configured for the Vercel integration.
14///
15/// The integration currently supports the following endpoints:
16///  - Vercel Log Drain
17pub fn routes(config: &Config) -> axum::Router<ServiceState> {
18    axum::Router::new()
19        .route("/logs", logs::route(config))
20        .route("/logs/", logs::route(config))
21}
22
23mod logs {
24    use super::*;
25
26    async fn handle(
27        content_type: RawContentType,
28        state: ServiceState,
29        builder: IntegrationBuilder,
30    ) -> axum::response::Result<impl IntoResponse> {
31        let format = match ContentType::from(content_type.as_ref()) {
32            ContentType::Json => VercelLogDrainFormat::Json,
33            ContentType::NdJson => VercelLogDrainFormat::NdJson,
34            _ => return Ok(StatusCode::UNSUPPORTED_MEDIA_TYPE),
35        };
36
37        let envelope = builder
38            .with_type(LogsIntegration::VercelDrainLog { format })
39            .build();
40
41        common::handle_envelope(&state, envelope).await?;
42
43        Ok(StatusCode::ACCEPTED)
44    }
45
46    pub fn route(config: &Config) -> MethodRouter<ServiceState> {
47        post(handle).route_layer(DefaultBodyLimit::max(config.max_logs_integration_size()))
48    }
49}
