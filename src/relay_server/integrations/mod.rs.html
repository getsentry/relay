1//! The Home of Relay's integrations with other protocols and vendors.
2//!
3//! Integrations are protocols and vendor APIs Relay implements. Data received by integrations must
4//! be converted into a Sentry native format within the same Relay instance.
5//!
6//! Relay needs to temporarily store integration payloads in the protocol in an [`Envelope`](`crate::Envelope`)
7//! until it can be converted by a [`Processor`](`crate::processing::Processor`).
8//! Integration *must not* be transported via the [`Envelope`](crate::Envelope) network protocol.
9//! See also: [`ItemType::is_internal`](`crate::envelope::ItemType::is_internal`) and
10//! [`ItemType::Integration`](`crate::envelope::ItemType::Integration`).
11//!
12//! This module contains an exhaustive list of all integrations Relay supports.
13
14macro_rules! define_integrations {
15    (@as_content_type, $($ct:literal => $value:pat,)*) => {
16        impl Integration {
17            pub fn as_content_type(&self) -> &'static str {
18                match self {
19                    $($value => $ct),*
20                }
21            }
22        }
23    };
24    (@from_content_type, $($ct:literal => $value:expr,)*) => {
25        impl Integration {
26            pub fn from_content_type(s: &str) -> Option<Integration> {
27                match s {
28                    $(s if s.eq_ignore_ascii_case($ct) => Some($value),)*
29                    _ => None
30                }
31            }
32        }
33    };
34    ($($tt:tt)*) => {
35        define_integrations!(@as_content_type, $($tt)*);
36        define_integrations!(@from_content_type, $($tt)*);
37    };
38}
39
40define_integrations!(
41    "application/vnd.sentry.integration.otel.logs+json" => Integration::Logs(LogsIntegration::OtelV1 { format: OtelFormat::Json }),
42    "application/vnd.sentry.integration.otel.logs+protobuf" => Integration::Logs(LogsIntegration::OtelV1 { format: OtelFormat::Protobuf }),
43    "application/vnd.sentry.integration.otel.spans+json" => Integration::Spans(SpansIntegration::OtelV1 { format: OtelFormat::Json }),
44    "application/vnd.sentry.integration.otel.spans+protobuf" => Integration::Spans(SpansIntegration::OtelV1 { format: OtelFormat::Protobuf }),
45    "application/vnd.sentry.integration.vercel.logs+json" => Integration::Logs(LogsIntegration::VercelDrainLog { format: VercelLogDrainFormat::Json }),
46    "application/vnd.sentry.integration.vercel.logs+ndjson" => Integration::Logs(LogsIntegration::VercelDrainLog { format: VercelLogDrainFormat::NdJson }),
47);

49/// An exhaustive list of all integrations supported by Relay.
50///
51/// While the list is currently exhaustive more integrations may be added at any time.
52#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
53pub enum Integration {
54    /// All logging integrations.
55    Logs(LogsIntegration),
56    /// All tracing/spans integrations.
57    Spans(SpansIntegration),
58}
59
60impl From<LogsIntegration> for Integration {
61    fn from(value: LogsIntegration) -> Self {
62        Self::Logs(value)
63    }
64}
65
66impl From<SpansIntegration> for Integration {
67    fn from(value: SpansIntegration) -> Self {
68        Self::Spans(value)
69    }
70}

72/// All logging integrations supported by Relay.
73#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
74pub enum LogsIntegration {
75    /// The OTeL logging integration.
76    ///
77    /// Supports OTeL's [`LogsData`](opentelemetry_proto::tonic::logs::v1::LogsData).
78    OtelV1 { format: OtelFormat },
79    /// The Vercel Log Drain integration.
80    ///
81    /// Supports the [`relay_ourlogs::VercelLog`] format.
82    VercelDrainLog { format: VercelLogDrainFormat },
83}
84
85/// All span integrations supported by Relay.
86#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
87pub enum SpansIntegration {
88    /// The OTeL traces integration.
89    ///
90    /// Supports OTeL's [`TracesData`](opentelemetry_proto::tonic::trace::v1::TracesData).
91    OtelV1 { format: OtelFormat },
92}
93
94/// An OTeL wire format.
95#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
96pub enum OtelFormat {
97    /// OTeL data in a Protocol Buffers container.
98    Protobuf,
99    /// OTeL data in a JSON container.
100    Json,
101}
102
103#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
104pub enum VercelLogDrainFormat {
105    // Vercel Log Drain data in a JSON array payload
106    Json,
107    // Vercel Log Drain data in a newline delimited JSON payload
108    NdJson,
109}
