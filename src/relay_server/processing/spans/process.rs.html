19/// Parses all serialized spans.
20///
21/// Individual, invalid spans are discarded.
22pub fn expand(spans: Managed<SerializedSpans>) -> Result<Managed<ExpandedSpans>, Rejected<Error>> {
23    spans.try_map(|spans, records| {
24        let SerializedSpans {
25            headers,
26            items,
27            invalid,
28            attachments,
29        } = spans;
30
31        debug_assert!(
32            invalid.is_empty(),
33            "invalid items should already be rejected"
34        );
35
36        let (settings, spans) = match items {
37            SpanItems::Container(item) => expand_span_container(&item)?,
38            SpanItems::Legacy(items) => expand_legacy_spans(items, records),
39            SpanItems::Integration(item) => spans::integrations::expand(records, &[item]),
40            SpanItems::None => (Default::default(), Vec::new()),
41        };
42
43        let mut span_id_mapping: BTreeMap<_, _> = BTreeMap::new();
44        for span in spans {
45            if let Some(id) = span.value().and_then(|span| span.span_id.value().copied()) {
46                // Although span_ids should be unique it could be that they collied in which case we
47                // want to drop one of the offending spans.
48                if let Some(old_span) = span_id_mapping.insert(id, ExpandedSpan::new(span)) {
49                    relay_log::debug!("span id collision");
50                    records.reject_err(Error::Invalid(DiscardReason::Duplicate), old_span);
51                }
52            } else {
53                relay_log::debug!("failed to extract span id from span");
54                records.reject_err(Error::Invalid(DiscardReason::InvalidSpan), span);
55            }
56        }
57
58        let mut stand_alone_attachments: Vec<ExpandedAttachment> = Vec::new();
59        for attachment in attachments {
60            match parse_and_validate_span_attachment(&attachment) {
61                Ok((None, attachment)) => {
62                    stand_alone_attachments.push(attachment);
63                }
64                Ok((Some(span_id), attachment)) => {
65                    if let Some(entry) = span_id_mapping.get_mut(&span_id) {
66                        entry.attachments.push(attachment);
67                    } else {
68                        relay_log::debug!("span attachment invalid associated span id");
69                        records.reject_err(
70                            Error::Invalid(DiscardReason::InvalidSpanAttachment),
71                            attachment,
72                        );
73                    }
74                }
75                Err(err) => {
76                    records.reject_err(err, attachment);
77                }
78            }
79        }
80
81        Ok::<_, Error>(ExpandedSpans {
82            headers,
83            settings,
84            spans: span_id_mapping.into_values().collect(),
85            server_sample_rate: None,
86            stand_alone_attachments,
87            category: spans::TotalAndIndexed,
88        })
89    })
90}

92fn expand_span_container(item: &Item) -> Result<(Settings, ContainerItems<SpanV2>)> {
93    let (metadata, spans) = ItemContainer::<SpanV2>::parse(item)
94        .map_err(|err| {
95            relay_log::debug!("failed to parse span container: {err}");
96            Error::Invalid(DiscardReason::InvalidJson)
97        })?
98        .into_parts();
99
100    relay_log::trace!("span container metadata: {metadata:?}");
101    let settings = metadata
102        .map(|metadata| {
103            let is = metadata.ingest_settings.as_ref();
104
105            match metadata.version {
106                None => Settings::default(),
107                // Technically invalid.
108                Some(0 | 1) => Settings::default(),
109                Some(2) => Settings {
110                    infer_ip: is
111                        .and_then(|is| is.infer_ip)
112                        .is_some_and(|infer| infer.is_auto()),
113                    infer_user_agent: is
114                        .and_then(|is| is.infer_user_agent)
115                        .is_some_and(|infer| infer.is_auto()),
116                    // We don't want to infer names for V2 spans. If an SDK sent a
117                    // V2 span without a name it's just invalid.
118                    infer_name: false,
119                },
120                // Unsupported, fall back to the safe default.
121                Some(_) => Default::default(),
122            }
123        })
124        .unwrap_or_default();
125
126    Ok((settings, spans))
127}

129fn expand_legacy_spans(
130    items: Vec<Item>,
131    records: &mut RecordKeeper<'_>,
132) -> (Settings, ContainerItems<SpanV2>) {
133    let spans = items
134        .into_iter()
135        .filter_map(|item| match expand_legacy_span(&item) {
136            Ok(span) => Some(span),
137            Err(err) => {
138                records.reject_err(err, item);
139                None
140            }
141        })
142        .collect();
143
144    let settings = Settings {
145        // In the legacy standalone span pipeline we always
146        // inferred both IPs and user agents. If this function
147        // is ever applied to transactions, we may need to rethink this.
148        infer_ip: true,
149        infer_user_agent: true,
150        // We want to infer names during processing for legacy spans.
151        // The inference can't happen during the conversion
152        // because PII scrubbing needs to run first.
153        infer_name: true,
154    };
155
156    (settings, spans)
157}

159fn expand_legacy_span(item: &Item) -> Result<WithHeader<SpanV2>> {
160    let payload = item.payload();
161
162    let span = Annotated::<Span>::from_json_bytes(&payload)
163        .map_err(|err| {
164            relay_log::debug!("failed to parse span: {err}");
165            Error::Invalid(DiscardReason::InvalidJson)
166        })?
167        // We can't enable `infer_name` here:
168        // These spans haven't been PII scrubbed yet, so inferring a name would risk
169        // leaking PII.
170        .map_value(|span| relay_spans::span_v1_to_span_v2(span, false));
171
172    Ok(WithHeader::new(span))
173}

175/// Parses and validates a span attachment, converting it into a structured type.
176fn parse_and_validate_span_attachment(item: &Item) -> Result<(Option<SpanId>, ExpandedAttachment)> {
177    let associated_span_id = match item.parent_id() {
178        Some(ParentId::SpanId(span_id)) => span_id,
179        None => {
180            relay_log::debug!("span attachment missing associated span id");
181            return Err(Error::Invalid(DiscardReason::InvalidSpanAttachment));
182        }
183    };
184
185    let expanded_attachment =
186        trace_attachments::process::parse_and_validate(item).map_err(Error::Invalid)?;
187
188    Ok((associated_span_id, expanded_attachment))
189}

191/// Normalizes individual spans.
192pub fn normalize(spans: &mut Managed<ExpandedSpans>, geo_lookup: &GeoIpLookup, ctx: Context<'_>) {
193    let settings = spans.settings;
194
195    spans.retain_with_context(
196        |spans| (&mut spans.spans, &spans.headers),
197        |span, headers, _| {
198            normalize_span(&mut span.span, settings, headers, geo_lookup, ctx).inspect_err(|err| {
199                relay_log::debug!("failed to normalize span: {err}");
200            })
201        },
202    );
203}

205fn normalize_span(
206    span: &mut Annotated<SpanV2>,
207    settings: Settings,
208    headers: &EnvelopeHeaders,
209    geo_lookup: &GeoIpLookup,
210    ctx: Context<'_>,
211) -> Result<()> {
212    let meta = headers.meta();
213
214    eap::time::normalize(
215        span,
216        utils::normalize::time_config(headers, |f| f.span.as_ref(), ctx),
217    );
218
219    if let Some(span) = span.value_mut() {
220        let duration = span_duration(span);
221        let allowed_hosts = ctx.global_config.options.http_span_allowed_hosts.as_slice();
222        let model_metdata = ctx.global_config.ai_model_metadata();
223        let client_ua_info = settings.infer_user_agent.then(|| ClientUserAgentInfo {
224            user_agent: meta.user_agent(),
225            hints: meta.client_hints(),
226        });
227        let performance_score = ctx.project_info.config().performance_score.as_ref();
228
229        validate_timestamps(span)?;
230
231        eap::normalize_attribute_types(&mut span.attributes);
232        eap::normalize_attribute_names(&mut span.attributes);
233        // normalize_sentry_op must be called before normalize_span_category
234        // because category derivation depends on having the sentry.op attribute
235        // available.
236        eap::normalize_sentry_op(&mut span.attributes);
237        eap::normalize_web_vital_span_segment(span);
238        eap::normalize_span_category(&mut span.attributes);
239        eap::normalize_received(&mut span.attributes, meta.received_at());
240        eap::normalize_client_address(&mut span.attributes, meta.client_addr());
241        if settings.infer_ip {
242            eap::normalize_inject_client_address(&mut span.attributes, meta.client_addr());
243        }
244        eap::normalize_user_agent(&mut span.attributes, client_ua_info);
245        eap::normalize_user_geo(&mut span.attributes, |ip| geo_lookup.lookup(ip));
246        eap::normalize_dsc(&mut span.attributes, &span.is_segment, headers.dsc());
247        if ctx.is_processing() {
248            eap::normalize_ai(&mut span.attributes, duration, model_metdata);
249        }
250        // In the old pipeline, the profile would get saved in the performance score context.
251        // In the new pipeline, we are not storing it for now.
252        let _ = relay_event_normalization::normalize_performance_score(span, performance_score);
253        eap::normalize_mobile_measurements(&mut span.attributes, duration);
254        eap::normalize_attribute_values(&mut span.attributes, allowed_hosts);
255        eap::write_legacy_attributes(&mut span.attributes);
256        eap::normalize_client_sample_rate(&mut span.attributes);
257    };
258
259    if let Annotated(None, meta) = span {
260        relay_log::debug!("empty span: {meta:?}");
261        return Err(Error::Invalid(DiscardReason::NoData));
262    }
263
264    Ok(())
265}

267/// Normalize derived fields and attributes.
268///
269/// This is separate from [`normalize`] because it needs to run
270/// after PII scrubbing; PII might get leaked otherwise.
271///
272/// It also takes care of trimming and schema validation because those procedures need
273/// to also run for derived fields.
274pub fn normalize_derived(spans: &mut Managed<ExpandedSpans>, ctx: Context<'_>) {
275    let settings = spans.settings;
276    spans.retain_with_context(
277        |spans| (&mut spans.spans, &()),
278        |span, _, _| {
279            normalize_span_derived(&mut span.span, settings, ctx).inspect_err(|err| {
280                relay_log::debug!("failed to normalize span: {err}");
281            })
282        },
283    );
284}
285
286fn normalize_span_derived(
287    span: &mut Annotated<SpanV2>,
288    settings: Settings,
289    ctx: Context<'_>,
290) -> Result<()> {
291    if let Some(span) = span.value_mut() {
292        // The order of operations shouldn't matter here—we should never _actually_
293        // need to synthesize both the name and description. If the span was sent as
294        // V2 it should have a name and if it was sent as V1 it should have a description.
295        if settings.infer_name {
296            eap::normalize_span_name(span);
297        }
298        eap::normalize_sentry_description(&mut span.attributes, &span.name);
299    }
300
301    // Set a max_bytes value on the root state if it's defined in the project config.
302    // This causes the whole item to be trimmed down to the limit.
303    let max_bytes = ctx
304        .project_info
305        .config()
306        .trimming
307        .span
308        .map(|cfg| cfg.max_size as usize);
309    let trimming_root = ProcessingState::root_builder().max_bytes(max_bytes).build();
310
311    process_value(
312        span,
313        &mut eap::TrimmingProcessor::new(ctx.config.max_removed_attribute_key_size()),
314        &trimming_root,
315    )?;
316
317    process_value(
318        span,
319        &mut SchemaProcessor::new()
320            .with_required(RequiredMode::DeleteParent)
321            .with_verbose_errors(relay_log::enabled!(relay_log::Level::DEBUG)),
322        ProcessingState::root(),
323    )?;
324
325    if let Annotated(None, meta) = span {
326        relay_log::debug!("empty span: {meta:?}");
327        return Err(Error::Invalid(DiscardReason::NoData));
328    }
329
330    Ok(())
331}

333/// Validates the start and end timestamps of a span.
334///
335/// The start timestamp must not be after the end timestamp.
336fn validate_timestamps(span: &SpanV2) -> Result<()> {
337    match (span.start_timestamp.value(), span.end_timestamp.value()) {
338        (Some(start), Some(end)) if start <= end => Ok(()),
339        _ => Err(Error::Invalid(DiscardReason::Timestamp)),
340    }
341}

343/// Applies PII scrubbing to individual spans.
344pub fn scrub(spans: &mut Managed<ExpandedSpans>, ctx: Context<'_>) {
345    spans.retain(
346        |spans| &mut spans.spans,
347        |span, r| {
348            scrub_span(&mut span.span, ctx)
349                .inspect_err(|err| relay_log::debug!("failed to scrub pii from span: {err}"))?;
350
351            span.attachments.retain_mut(|attachment| {
352                match trace_attachments::process::scrub_attachment(attachment, ctx) {
353                    Ok(()) => true,
354                    Err(err) => {
355                        relay_log::debug!("failed to scrub pii from span attachment: {err}");
356                        r.reject_err(Error::from(err), &*attachment);
357                        false
358                    }
359                }
360            });
361
362            Ok::<(), Error>(())
363        },
364    );
365
366    spans.retain(
367        |spans| &mut spans.stand_alone_attachments,
368        |attachment, _| {
369            trace_attachments::process::scrub_attachment(attachment, ctx)
370                .inspect_err(|err| {
371                    relay_log::debug!("failed to scrub pii from span attachment: {err}")
372                })
373                .map_err(Error::from)
374        },
375    );
376}

378fn scrub_span(span: &mut Annotated<SpanV2>, ctx: Context<'_>) -> Result<()> {
379    let pii_config_from_scrubbing = ctx.project_info.config.datascrubbing_settings.pii_config();
380
381    relay_pii::eap::scrub(
382        ValueType::Span,
383        span,
384        ctx.project_info.config.pii_config.as_ref(),
385        pii_config_from_scrubbing.as_ref(),
386    )?;
387
388    Ok(())
389}

391fn span_duration(span: &SpanV2) -> Option<Duration> {
392    let start_timestamp = *span.start_timestamp.value()?;
393    let end_timestamp = *span.end_timestamp.value()?;
394    (end_timestamp - start_timestamp).to_std().ok()
395}

977    fn prepare_normalize_span_params(
978        string_attributes: &[(&str, &str)],
979        float_attributes: &[(&str, f64)],
980    ) -> (
981        Annotated<SpanV2>,
982        EnvelopeHeaders,
983        GeoIpLookup,
984        Context<'static>,
985    ) {
986        let mut attributes = Attributes::new();
987        string_attributes
988            .iter()
989            .for_each(|(key, value)| attributes.insert(*key, value.to_owned()));
990        float_attributes
991            .iter()
992            .for_each(|(key, value)| attributes.insert(*key, *value));
993        let attrs_json =
994            serde_json::to_string(&SerializableAnnotated(&Annotated::new(attributes))).unwrap();
995        let span_json = format!(
996            r#"{{
997             "trace_id": "5b8efff798038103d269b633813fc60c",
998             "span_id": "eee19b7ec3c1b175",
999             "start_timestamp": 1715000000.0,
1000             "end_timestamp": 1715000001.0,
1001             "name": "test",
1002             "status": "ok",
1003             "attributes": {attrs_json}
1004         }}"#
1005        );
1006        let span = Annotated::<SpanV2>::from_json(&span_json).unwrap();
1007
1008        let dsn = "https://a94ae32be2584e0bbd7a4cbb95971fee:@sentry.io/42"
1009            .parse()
1010            .unwrap();
1011        let meta = RequestMeta::new(dsn);
1012        let envelope = Envelope::from_request(Some(EventId::new()), meta);
1013        let headers = envelope.headers().to_owned();
1014
1015        (span, headers, GeoIpLookup::empty(), Context::for_test())
1016    }

1018    fn assert_attributes_contains(
1019        span: &Annotated<SpanV2>,
1020        string_attributes: &[(&str, &str)],
1021        float_attributes: &[(&str, f64)],
1022    ) {
1023        let attrs = span.value().unwrap().attributes.value().unwrap();
1024        string_attributes.iter().for_each(|(key, value)| {
1025            assert_eq!(
1026                attrs.get_value(*key).and_then(|v| v.as_str()),
1027                Some(*value),
1028                "attribute mismatch for {key}"
1029            )
1030        });
1031        float_attributes.iter().for_each(|(key, value)| {
1032            assert_eq!(
1033                attrs.get_value(*key).and_then(|v| v.as_f64()),
1034                Some(*value),
1035                "attribute mismatch for {key}"
1036            )
1037        });
1038    }
