26/// Validates all sampling configurations.
27///
28/// The function validates the sampling config of the project and sampling project (from the trace
29/// root) are valid and supported.
30/// This check is not required to run, but makes sure eventual mis-configurations and errors in the
31/// project config are caught and not silently ignored.
32pub fn validate_configs(ctx: Context<'_>) {
33    // It is possible there is a new version of the dynamic sampling configuration rolled out
34    // and only the innermost, processing, Relay supports it, in which case dynamic sampling
35    // will be delayed to the processing Relay.
36    //
37    // If the processing Relay does not support it, there is a real problem.
38    if !ctx.is_processing() {
39        return;
40    }
41
42    if !is_sampling_config_supported(ctx.project_info)
43        || !ctx
44            .sampling_project_info
45            .is_none_or(is_sampling_config_supported)
46    {
47        relay_log::error!(
48            project_id = ?ctx.project_info.project_id,
49            sampling_project_id = ?ctx.sampling_project_info.and_then(|c| c.project_id.as_ref()),
50            "found unsupported dynamic sampling rules in a processing relay"
51        );
52    }
53}

55/// Validates the presence of a dynamic sampling context when processing Spans.
56///
57/// Each envelope received by Relay must contain a valid dynamic sampling context.
58/// This is not a technical requirement as a missing dynamic sampling context is treated as having
59/// a sample rate of 100%, but to ensure SDKs implement the protocol correctly it is validated.
60///
61/// An exception exists for our OTeL integration, which currently never sets a dynamic sampling
62/// context. In the future we may want to extract a DSC from the OTeL payload to allow dynamic
63/// sampling if the necessary attributes are present.
64pub fn validate_dsc_presence(spans: &SerializedSpans) -> Result<()> {
65    let dsc = spans.headers.dsc();
66
67    // Envelopes created by Relay may not carry a DSC. This is currently the case for envelopes
68    // created through the OTeL integration.
69    //
70    // This validation is only best effort (not a hard requirement) to get SDKs to implement DSC
71    // correctly, so it is okay to be a bit more lenient here and trust ourselves.
72    let client_is_relay = spans.headers.meta().client_name() == ClientName::Relay;
73
74    if !client_is_relay && dsc.is_none() {
75        return Err(Error::MissingDynamicSamplingContext);
76    }
77
78    Ok(())
79}

81/// Validates the dynamic sampling context against the parsed spans.
82///
83/// The values of the dynamic sampling context must match the values provided in the spans.
84/// Currently this only validates the trace id.
85pub fn validate_dsc(spans: &ExpandedSpans) -> Result<()> {
86    let Some(dsc) = spans.headers.dsc() else {
87        // It's okay if we don't have a DSC here. We may be in a processing path which has spans
88        // from mixed traces without a DSC (=100% sample rate).
89        // For example via the OTeL integration.
90        return Ok(());
91    };
92
93    for span in &spans.spans {
94        let trace_id = get_value!(span.trace_id);
95
96        if trace_id != Some(&dsc.trace_id) {
97            return Err(Error::DynamicSamplingContextMismatch);
98        }
99    }
100
101    Ok(())
102}

104/// Computes the sampling decision for a batch of spans.
105///
106/// All spans are evaluated in one go as they are required by the protocol to share the same
107/// DSC, which contains all the sampling relevant information.
108pub async fn run(
109    spans: Managed<SerializedSpans>,
110    ctx: Context<'_>,
111) -> Result<Managed<SampledSpans>, Managed<ExtractedMetrics>> {
112    let sampling_result = compute(&spans, ctx).await;
113
114    relay_statsd::metric!(
115        counter(RelayCounters::SamplingDecision) += 1,
116        decision = sampling_result.decision().as_str(),
117        item = "span"
118    );
119
120    let sampling_match = match sampling_result {
121        SamplingResult::Match(m) if m.decision().is_drop() => m,
122        sampling_result => {
123            let sample_rate = sampling_result.sample_rate();
124            return Ok(spans.map(|spans, _| spans.sampled(sample_rate)));
125        }
126    };
127
128    // At this point the decision is to drop the spans.
129    let span_count = outcome_count(&spans.spans)
130        + outcome_count(&spans.legacy)
131        + outcome_count(&spans.integrations);
132    let metrics = create_metrics(
133        spans.scoping(),
134        span_count,
135        spans.headers.dsc(),
136        SamplingDecision::Drop,
137    );
138    let (spans, metrics) = spans.split_once(|spans| (UnsampledSpans::from(spans), metrics));
139
140    let outcome = Outcome::FilteredSampling(sampling_match.into_matched_rules().into());
141    let _ = spans.reject_err(outcome);
142
143    Err(metrics)
144}

146/// Type returned by [`create_indexed_metrics`].
147///
148/// Contains the indexed spans and the metrics extracted from the spans.
149type SpansAndMetrics = (Managed<ExpandedSpans<Indexed>>, Managed<ExtractedMetrics>);
150
151/// Creates/extracts metrics for spans which have been determined to be kept by dynamic sampling.
152///
153/// Indexed metrics can only be extracted from the Relay making the final sampling decision,
154/// if the current Relay is not the final Relay, the function returns the original spans unchanged.
155pub fn create_indexed_metrics(
156    spans: Managed<ExpandedSpans>,
157    ctx: Context<'_>,
158) -> Either<Managed<ExpandedSpans>, SpansAndMetrics> {
159    if !ctx.is_processing() {
160        return Either::Left(spans);
161    }
162
163    let scoping = spans.scoping();
164    let (indexed, metrics) = spans.split_once(|spans| {
165        let metrics = create_metrics(
166            scoping,
167            spans.spans.len() as u32,
168            spans.headers.dsc(),
169            SamplingDecision::Keep,
170        );
171
172        (spans.into_indexed(), metrics)
173    });
174
175    Either::Right((indexed, metrics))
176}

178async fn compute(spans: &Managed<SerializedSpans>, ctx: Context<'_>) -> SamplingResult {
179    // The DSC is always required, we need it to evaluate all rules, if it is missing,
180    // no rules can be applied -> we sample the item.
181    let Some(dsc) = spans.headers.dsc() else {
182        return SamplingResult::NoMatch;
183    };
184
185    let project_sampling_config = get_sampling_config(ctx.project_info);
186    let root_sampling_config = ctx
187        .sampling_project_info
188        // Fallback to current project if there is no trace root project, this may happen,
189        // if the trace root is from a different organization.
190        .or(Some(ctx.project_info))
191        .and_then(get_sampling_config);
192
193    // The root sampling config is always required for dynamic sampling. It determines the sample
194    // rate which is applied to the item.
195    let Some(root_sampling_config) = root_sampling_config else {
196        return SamplingResult::NoMatch;
197    };
198
199    // Currently there is no support planned for reservoir sampling.
200    let mut evaluator = SamplingEvaluator::new(Utc::now());
201
202    // Apply project rules before trace rules, to give projects a chance to override the trace root
203    // sample rate.
204    if let Some(sampling_config) = project_sampling_config {
205        let rules = sampling_config.filter_rules(RuleType::Project);
206
207        // We need a segment consistent seed to make sure all spans of the same segment in this
208        // project get a proper minimum sample rate applied.
209        //
210        // The trace id gives us this property and it will also have the upside of consistently
211        // sampling multiple segments of the same trace.
212        evaluator = match evaluator.match_rules(*dsc.trace_id, dsc, rules).await {
213            ControlFlow::Continue(evaluator) => evaluator,
214            ControlFlow::Break(sampling_match) => return SamplingResult::Match(sampling_match),
215        }
216    }
217
218    let rules = root_sampling_config.filter_rules(RuleType::Trace);
219    evaluator
220        .match_rules(*dsc.trace_id, dsc, rules)
221        .await
222        .into()
223}

225fn get_sampling_config(info: &ProjectInfo) -> Option<&SamplingConfig> {
226    let config = info.config.sampling.as_ref()?.as_ref().ok()?;
227    (!config.unsupported()).then_some(config)
228}

230fn is_sampling_config_supported(project_info: &ProjectInfo) -> bool {
231    let Some(config) = &project_info.config.sampling else {
232        return true;
233    };
234    matches!(config, ErrorBoundary::Ok(config) if !config.unsupported())
235}

237fn create_metrics(
238    scoping: Scoping,
239    span_count: u32,
240    dsc: Option<&DynamicSamplingContext>,
241    sampling_decision: SamplingDecision,
242) -> ExtractedMetrics {
243    let mut metrics = ExtractedMetrics::default();
244
245    if span_count == 0 {
246        return metrics;
247    }
248
249    // For extracted metrics, Relay has always used the moment when the metrics were extracted
250    // as the received time, instead of the source item's received time.
251    let timestamp = UnixTimestamp::now();
252    let mut metadata = BucketMetadata::new(timestamp);
253    if sampling_decision.is_keep() {
254        metadata.extracted_from_indexed = true;
255    }
256
257    metrics.sampling_metrics.push(Bucket {
258        timestamp,
259        width: 0,
260        name: "c:spans/count_per_root_project@none".into(),
261        value: BucketValue::counter(span_count.into()),
262        tags: {
263            let mut tags = BTreeMap::new();
264            tags.insert("decision".to_owned(), sampling_decision.to_string());
265            tags.insert(
266                "target_project_id".to_owned(),
267                scoping.project_id.to_string(),
268            );
269            if let Some(tx) = dsc.and_then(|dsc| dsc.transaction.clone()) {
270                tags.insert("transaction".to_owned(), tx);
271            }
272            tags
273        },
274        metadata,
275    });
276    metrics.project_metrics.push(Bucket {
277        timestamp,
278        width: 0,
279        name: "c:spans/usage@none".into(),
280        value: BucketValue::counter(span_count.into()),
281        tags: Default::default(),
282        metadata,
283    });
284
285    metrics
286}

288/// Spans which have been rejected/dropped by dynamic sampling.
289///
290/// Contained spans will only count towards the [`DataCategory::SpanIndexed`] category,
291/// as the total category is counted from now in in metrics.
292struct UnsampledSpans {
293    spans: Vec<Item>,
294    legacy: Vec<Item>,
295    integrations: Vec<Item>,
296}
297
298impl From<SerializedSpans> for UnsampledSpans {
299    fn from(value: SerializedSpans) -> Self {
300        let SerializedSpans {
301            headers: _,
302            spans,
303            legacy,
304            integrations,
305        } = value;
306
307        Self {
308            spans,
309            legacy,
310            integrations,
311        }
312    }
313}
314
315impl Counted for UnsampledSpans {
316    fn quantities(&self) -> Quantities {
317        let quantity = (outcome_count(&self.spans)
318            + outcome_count(&self.legacy)
319            + outcome_count(&self.integrations)) as usize;
320
321        smallvec::smallvec![(DataCategory::SpanIndexed, quantity)]
322    }
323}
