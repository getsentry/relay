15/// Validates all sampling configurations.
16///
17/// The function validates the sampling config of the project and sampling project (from the trace
18/// root) are valid and supported.
19/// This check is not required to run, but makes sure eventual mis-configurations and errors in the
20/// project config are caught and not silently ignored.
21pub fn validate_configs(ctx: Context<'_>) {
22    // It is possible there is a new version of the dynamic sampling configuration rolled out
23    // and only the innermost, processing, Relay supports it, in which case dynamic sampling
24    // will be delayed to the processing Relay.
25    //
26    // If the processing Relay does not support it, there is a real problem.
27    if !ctx.is_processing() {
28        return;
29    }
30
31    if !is_sampling_config_supported(ctx.project_info)
32        || !ctx
33            .sampling_project_info
34            .is_none_or(is_sampling_config_supported)
35    {
36        relay_log::error!(
37            project_id = ?ctx.project_info.project_id,
38            sampling_project_id = ?ctx.sampling_project_info.and_then(|c| c.project_id.as_ref()),
39            "found unsupported dynamic sampling rules in a processing relay"
40        );
41    }
42}

44/// Computes the sampling decision for a batch of spans.
45///
46/// All spans are evaluated in one go as they are required by the protocol to share the same
47/// DSC, which contains all the sampling relevant information.
48pub async fn run(spans: &Managed<SerializedSpans>, ctx: Context<'_>) -> SamplingResult {
49    // The DSC is always required, we need it to evaluate all rules, if it is missing,
50    // no rules can be applied -> we sample the item.
51    let Some(dsc) = spans.headers.dsc() else {
52        return SamplingResult::NoMatch;
53    };
54
55    let project_sampling_config = get_sampling_config(ctx.project_info);
56    let root_sampling_config = ctx.sampling_project_info.and_then(get_sampling_config);
57
58    // The root sampling config is always required for dynamic sampling. It determines the sample
59    // rate which is applied to the item.
60    let Some(root_sampling_config) = root_sampling_config else {
61        return SamplingResult::NoMatch;
62    };
63
64    // TODO: reservoir sampling
65    let mut evaluator = SamplingEvaluator::new(Utc::now());
66
67    // Apply transaction local rules.
68    //
69    // For spans we generally cannot apply transaction local rules, but some of the rules,
70    // specifically the minimum sample rate rule can still be applied to individual spans.
71    //
72    // In the future this rule type will be replaced with an explicit project based rule type.
73    if let Some(sampling_config) = project_sampling_config {
74        let rules = sampling_config.filter_rules(RuleType::Transaction);
75
76        // We need a segment consistent seed to make sure all spans of the same segment in this
77        // project get a proper minimum sample rate applied.
78        //
79        // The trace id gives us this property and it will also have the upside of consistently
80        // sampling multiple segments of the same trace.
81        evaluator = match evaluator.match_rules(*dsc.trace_id, dsc, rules).await {
82            ControlFlow::Continue(evaluator) => evaluator,
83            ControlFlow::Break(sampling_match) => return SamplingResult::Match(sampling_match),
84        }
85    }
86
87    let rules = root_sampling_config.filter_rules(RuleType::Trace);
88    evaluator
89        .match_rules(*dsc.trace_id, dsc, rules)
90        .await
91        .into()
92}

94fn get_sampling_config(info: &ProjectInfo) -> Option<&SamplingConfig> {
95    let config = info.config.sampling.as_ref()?.as_ref().ok()?;
96    (!config.unsupported()).then_some(config)
97}
98
99fn is_sampling_config_supported(project_info: &ProjectInfo) -> bool {
100    let Some(config) = &project_info.config.sampling else {
101        return true;
102    };
103    matches!(config, ErrorBoundary::Ok(config) if !config.unsupported())
104}
