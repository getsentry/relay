12macro_rules! required {
13    ($value:expr) => {{
14        match $value {
15            Annotated(Some(value), _) => value,
16            Annotated(None, meta) => {
17                relay_log::debug!(
18                    "dropping span because of missing required field {} with meta {meta:?}",
19                    stringify!($value),
20                );
21                return Err(Error::Invalid(DiscardReason::InvalidSpan));
22            }
23        }
24    }};
25}

27/// Context required for [`convert`].
28pub struct Context {
29    /// Server side applied sample rate.
30    pub server_sample_rate: Option<f64>,
31    /// Item retention.
32    pub retention: Retention,
33}
34
35/// Converts a processed [`SpanV2`](super::SpanV2) into a [Kafka](crate::services::store::Store) compatible format.
36pub fn convert(span: IndexedSpanOnly, ctx: &Context) -> Result<Box<StoreSpanV2>> {
37    let mut span = required!(span.0.value);
38
39    let routing_key = span.trace_id.value().map(|v| *v.deref());
40
41    inject_server_sample_rate(&mut span.attributes, ctx.server_sample_rate);
42
43    Ok(Box::new(StoreSpanV2 {
44        routing_key,
45        retention_days: ctx.retention.standard,
46        downsampled_retention_days: ctx.retention.downsampled,
47        item: span,
48    }))
49}
50
51/// Injects a server sample rate into a span.
52fn inject_server_sample_rate(
53    attributes: &mut Annotated<Attributes>,
54    server_sample_rate: Option<f64>,
55) {
56    let Some(server_sample_rate) = server_sample_rate.and_then(FiniteF64::new) else {
57        return;
58    };
59
60    let attributes = attributes.get_or_insert_with(Default::default);
61    attributes.insert("sentry.server_sample_rate", server_sample_rate.to_f64());
62}
