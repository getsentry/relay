17pub fn extract_from_event(
18    dsc: Option<&DynamicSamplingContext>,
19    event: &Annotated<Event>,
20    global_config: &GlobalConfig,
21    config: &Config,
22    server_sample_rate: Option<f64>,
23    event_metrics_extracted: EventMetricsExtracted,
24    spans_extracted: SpansExtracted,
25) -> Option<Vec<Result<Item, ()>>> {
26    // Only extract spans from transactions (not errors).
27    if event_type(event) != Some(EventType::Transaction) {
28        return None;
29    };
30
31    if spans_extracted.0 {
32        return None;
33    }
34
35    if let Some(sample_rate) = global_config.options.span_extraction_sample_rate
36        && utils::sample(sample_rate).is_discard()
37    {
38        return None;
39    }
40
41    let client_sample_rate = dsc.and_then(|ctx| ctx.sample_rate);
42
43    let event = event.value()?;
44
45    let transaction_span = processing::transactions::extraction::extract_segment_span(
46        event,
47        config
48            .aggregator_config_for(MetricNamespace::Spans)
49            .max_tag_value_length,
50        &[],
51    )?;
52
53    let mut results = vec![];
54
55    // Add child spans.
56    if let Some(spans) = event.spans.value() {
57        for span in spans {
58            let Some(inner_span) = span.value() else {
59                continue;
60            };
61            // HACK: clone the span to set the segment_id. This should happen
62            // as part of normalization once standalone spans reach wider adoption.
63            let mut new_span = inner_span.clone();
64            new_span.is_segment = Annotated::new(false);
65            new_span.is_remote = Annotated::new(false);
66            new_span.received = transaction_span.received.clone();
67            new_span.segment_id = transaction_span.segment_id.clone();
68            new_span.platform = transaction_span.platform.clone();
69
70            // If a profile is associated with the transaction, also associate it with its
71            // child spans.
72            new_span.profile_id = transaction_span.profile_id.clone();
73
74            results.push(make_span_item(
75                new_span,
76                config,
77                client_sample_rate,
78                server_sample_rate,
79                event_metrics_extracted.0,
80            ));
81        }
82    }
83
84    results.push(make_span_item(
85        transaction_span,
86        config,
87        client_sample_rate,
88        server_sample_rate,
89        event_metrics_extracted.0,
90    ));
91
92    Some(results)
93}

95fn make_span_item(
96    mut span: Span,
97    config: &Config,
98    client_sample_rate: Option<f64>,
99    server_sample_rate: Option<f64>,
100    metrics_extracted: bool,
101) -> Result<Item, ()> {
102    add_sample_rate(
103        &mut span.measurements,
104        "client_sample_rate",
105        client_sample_rate,
106    );
107    add_sample_rate(
108        &mut span.measurements,
109        "server_sample_rate",
110        server_sample_rate,
111    );
112
113    let mut span = Annotated::new(span);
114
115    validate(&mut span)
116        .inspect_err(|e| {
117            relay_log::debug!(
118                error = e as &dyn Error,
119                span = ?span,
120                source = "event",
121                "invalid span"
122            );
123        })
124        .map_err(|_| ())?;
125
126    let mut item = create_span_item(span, config)?;
127    // If metrics extraction happened for the event, it also happened for its spans:
128    item.set_metrics_extracted(metrics_extracted);
129
130    relay_log::trace!("Adding span to envelope");
131    Ok(item)
132}

134/// Any violation of the span schema.
135#[derive(thiserror::Error, Debug)]
136pub enum ValidationError {
137    #[error("empty span")]
138    EmptySpan,
139    #[error("span is missing `trace_id`")]
140    MissingTraceId,
141    #[error("span is missing `span_id`")]
142    MissingSpanId,
143    #[error("span is missing `timestamp`")]
144    MissingTimestamp,
145    #[error("span is missing `start_timestamp`")]
146    MissingStartTimestamp,
147    #[error("span end must be after start")]
148    EndBeforeStartTimestamp,
149    #[error("span is missing `exclusive_time`")]
150    MissingExclusiveTime,
151}
152
153/// We do not extract or ingest spans with missing fields if those fields are required on the Kafka topic.
154pub fn validate(span: &mut Annotated<Span>) -> Result<(), ValidationError> {
155    let inner = span
156        .value_mut()
157        .as_mut()
158        .ok_or(ValidationError::EmptySpan)?;
159    let Span {
160        exclusive_time,
161        tags,
162        sentry_tags,
163        start_timestamp,
164        timestamp,
165        span_id,
166        trace_id,
167        ..
168    } = inner;
169
170    trace_id.value().ok_or(ValidationError::MissingTraceId)?;
171    span_id.value().ok_or(ValidationError::MissingSpanId)?;
172
173    match (start_timestamp.value(), timestamp.value()) {
174        (Some(start), Some(end)) if end < start => Err(ValidationError::EndBeforeStartTimestamp),
175        (Some(_), Some(_)) => Ok(()),
176        (_, None) => Err(ValidationError::MissingTimestamp),
177        (None, _) => Err(ValidationError::MissingStartTimestamp),
178    }?;
179
180    exclusive_time
181        .value()
182        .ok_or(ValidationError::MissingExclusiveTime)?;
183
184    if let Some(sentry_tags) = sentry_tags.value_mut() {
185        if sentry_tags
186            .group
187            .value()
188            .is_some_and(|s| s.len() > 16 || s.chars().any(|c| !c.is_ascii_hexdigit()))
189        {
190            sentry_tags.group.set_value(None);
191        }
192
193        if sentry_tags
194            .status_code
195            .value()
196            .is_some_and(|s| s.parse::<u16>().is_err())
197        {
198            sentry_tags.status_code.set_value(None);
199        }
200    }
201    if let Some(tags) = tags.value_mut() {
202        tags.retain(|_, value| !value.value().is_empty())
203    }
204
205    Ok(())
206}

208/// Serializes the given span into an envelope item.
209///
210/// In processing relays, creates a Span V2 so it can be published via kafka.
211pub fn create_span_item(span: Annotated<Span>, config: &Config) -> Result<Item, ()> {
212    let mut new_item = Item::new(ItemType::Span);
213    if cfg!(feature = "processing") && config.processing_enabled() {
214        let span_v2 = span.map_value(relay_spans::span_v1_to_span_v2);
215        let payload = match span_v2.to_json() {
216            Ok(payload) => payload,
217            Err(err) => {
218                relay_log::error!("failed to serialize span V2: {}", err);
219                return Err(());
220            }
221        };
222        if let Some(trace_id) = span_v2.value().and_then(|s| s.trace_id.value()) {
223            new_item.set_routing_hint(*trace_id.as_ref());
224        }
225
226        new_item.set_payload(ContentType::Json, payload);
227    } else {
228        let payload = match span.to_json() {
229            Ok(payload) => payload,
230            Err(err) => {
231                relay_log::error!("failed to serialize span: {}", err);
232                return Err(());
233            }
234        };
235        new_item.set_payload(ContentType::Json, payload);
236    }
237
238    Ok(new_item)
239}

241fn add_sample_rate(measurements: &mut Annotated<Measurements>, name: &str, value: Option<f64>) {
242    let value = match value {
243        Some(value) if value > 0.0 => value,
244        _ => return,
245    };
246
247    let measurement = Annotated::new(Measurement {
248        value: Annotated::try_from(value),
249        unit: MetricUnit::Fraction(FractionUnit::Ratio).into(),
250    });
251
252    measurements
253        .get_or_insert_with(Measurements::default)
254        .insert(name.to_owned(), measurement);
255}
