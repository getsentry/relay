25/// Errors that occur during transaction processing.
26#[derive(Debug, thiserror::Error)]
27pub enum Error {
28    #[error("invalid JSON")]
29    InvalidJson(#[from] serde_json::Error),
30    #[error("envelope processor failed")]
31    ProcessingFailed(#[from] ProcessingError),
32    #[error("rate limited")]
33    RateLimited(RateLimits),
34}
35
36impl From<RateLimits> for Error {
37    fn from(value: RateLimits) -> Self {
38        Self::RateLimited(value)
39    }
40}
41
42impl OutcomeError for Error {
43    type Error = Self;
44
45    fn consume(self) -> (Option<Outcome>, Self::Error) {
46        let outcome = match &self {
47            Self::InvalidJson(_) => Outcome::Invalid(DiscardReason::InvalidJson),
48            Self::ProcessingFailed(e) => match e {
49                ProcessingError::InvalidTransaction => {
50                    Outcome::Invalid(DiscardReason::InvalidTransaction)
51                }
52                ProcessingError::EventFiltered(key) => Outcome::Filtered(key.clone()),
53                _other => {
54                    relay_log::error!(
55                        error = &self as &dyn std::error::Error,
56                        "internal error: transaction processing failed"
57                    );
58                    Outcome::Invalid(DiscardReason::Internal)
59                }
60            },
61            Self::RateLimited(limits) => {
62                let reason_code = limits.longest().and_then(|limit| limit.reason_code.clone());
63                Outcome::RateLimited(reason_code)
64            }
65        };
66        (Some(outcome), self)
67    }
68}

92impl Processor for TransactionProcessor {
93    type Input = SerializedTransaction;
94    type Output = TransactionOutput;
95    type Error = Error;
96
97    fn prepare_envelope(&self, envelope: &mut ManagedEnvelope) -> Option<Managed<Self::Input>> {
98        let headers = envelope.envelope().headers().clone();
99
100        let mut event = envelope
101            .envelope_mut()
102            .take_item_by(|item| matches!(*item.ty(), ItemType::Transaction))?;
103
104        // Count number of spans by shallow-parsing the event.
105        // Needed for accounting but not in prod, because the event is immediately parsed afterwards.
106        #[cfg(debug_assertions)]
107        event.ensure_span_count();
108
109        let attachments = envelope
110            .envelope_mut()
111            .take_items_by(|item| matches!(*item.ty(), ItemType::Attachment));
112
113        let profiles = envelope
114            .envelope_mut()
115            .take_items_by(|item| matches!(*item.ty(), ItemType::Profile));
116
117        let work = SerializedTransaction {
118            headers,
119            event,
120            attachments,
121            profiles,
122        };
123
124        Some(Managed::with_meta_from(envelope, work))
125    }
126
127    async fn process(
128        &self,
129        tx: Managed<Self::Input>,
130        mut ctx: Context<'_>,
131    ) -> Result<Output<Self::Output>, Rejected<Self::Error>> {
132        let project_id = tx.scoping().project_id;
133        let mut metrics = Metrics::default();
134
135        relay_log::trace!("Expand transaction");
136        let mut tx = process::expand(tx)?;
137
138        relay_log::trace!("Validate attachments");
139        attachments::validate_attachments(&mut tx, |t| &mut t.attachments, ctx);
140
141        relay_log::trace!("Prepare transaction data");
142        process::prepare_data(&mut tx, &mut ctx, &mut metrics)?;
143
144        relay_log::trace!("Normalize transaction");
145        let mut tx = process::normalize(tx, ctx, &self.geoip_lookup)?;
146
147        relay_log::trace!("Filter transaction");
148        let filters_status = process::run_inbound_filters(&tx, ctx)?;
149
150        let quotas_client = self.quotas_client.as_ref();
151
152        relay_log::trace!("Processing profile");
153        process::process_profile(&mut tx, ctx);
154
155        relay_log::trace!("Sample transaction");
156        let (tx, server_sample_rate) =
157            match process::run_dynamic_sampling(tx, ctx, filters_status, quotas_client).await? {
158                SamplingOutput::Keep {
159                    payload,
160                    sample_rate,
161                } => (payload, sample_rate),
162                SamplingOutput::Drop {
163                    metrics,
164                    mut profile,
165                } => {
166                    // Remaining profile needs to be rate limited:
167                    if let Some(p) = profile {
168                        profile = self.limiter.enforce_quotas(p, ctx).await.ok();
169                    }
170                    return Ok(Output {
171                        main: profile.map(TransactionOutput::Profile),
172                        metrics: Some(metrics),
173                    });
174                }
175            };
176
177        // Need to scrub the transaction before extracting spans.
178        relay_log::trace!("Scrubbing transaction");
179        #[allow(unused_mut)]
180        let mut tx = process::scrub(tx, ctx)?;
181
182        tx = process::extract_spans(tx, ctx, server_sample_rate);
183
184        relay_log::trace!("Enforce quotas");
185        let tx = self.limiter.enforce_quotas(tx, ctx).await?;
186        let tx = match tx.transpose() {
187            either::Either::Left(tx) => tx,
188            either::Either::Right(metrics) => return Ok(Output::metrics(metrics)),
189        };
190
191        if ctx.is_processing() {
192            if !tx.flags.fully_normalized {
193                relay_log::error!(
194                    tags.project = %project_id,
195                    tags.ty = event_type(&tx.event).map(|e| e.to_string()).unwrap_or("none".to_owned()),
196                    "ingested event without normalizing"
197                );
198            };
199
200            let (indexed, metrics) = split_indexed_and_total(tx, ctx, SamplingDecision::Keep)?;
201
202            return Ok(Output {
203                main: Some(TransactionOutput::Indexed(indexed)),
204                metrics: Some(metrics),
205            });
206        }
207
208        Ok(Output {
209            main: Some(TransactionOutput::Full(tx)),
210            metrics: None,
211        })
212    }
213}
