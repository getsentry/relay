6/// A transaction in its serialized state, as transported in an envelope.
7#[derive(Debug)]
8pub struct SerializedTransaction {
9    pub headers: EnvelopeHeaders,
10    pub event: Item,
11    pub attachments: Items,
12    pub profiles: SmallVec<[Item; 3]>,
13}
14
15impl Counted for SerializedTransaction {
16    fn quantities(&self) -> Quantities {
17        let Self {
18            headers: _,
19            event,
20            attachments,
21            profiles,
22        } = self;
23        debug_assert!(!event.spans_extracted());
24        let mut quantities = event.quantities(); // counts spans based on `span_count` header.
25        quantities.extend(attachments.quantities());
26        quantities.extend(profiles.quantities());
27
28        quantities
29    }
30}
