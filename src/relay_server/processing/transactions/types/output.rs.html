10/// Output of the transaction processor.
11#[derive(Debug)]
12pub enum TransactionOutput {
13    /// The transaction has not been dropped by dynamic sampling.
14    Full(Managed<Box<ExpandedTransaction>>),
15    /// The transaction has been dropped by dynamic sampling, only an optional profile remains.
16    Profile(Managed<Box<StandaloneProfile>>),
17    /// The transaction has not been dropped by dynamic sampling, and metrics have been extracted.
18    ///
19    /// This is used in processing relays.
20    Indexed {
21        spans: Option<Managed<ExtractedIndexedSpans>>,
22        transaction: Managed<Box<ExpandedTransaction<Indexed, SpansExtracted>>>,
23    },
24}
25
26impl Forward for TransactionOutput {
27    fn serialize_envelope(
28        self,
29        _ctx: ForwardContext<'_>,
30    ) -> Result<Managed<Box<Envelope>>, Rejected<()>> {
31        match self {
32            TransactionOutput::Full(managed) => managed.try_map(|work, _| {
33                work.serialize_envelope()
34                    .map_err(drop)
35                    .with_outcome(Outcome::Invalid(DiscardReason::Internal))
36            }),
37            TransactionOutput::Profile(profile) => {
38                Ok(profile.map(|profile, _| profile.serialize_envelope()))
39            }
40            TransactionOutput::Indexed { spans, transaction } => {
41                if let Some(spans) = spans {
42                    let _ = spans.internal_error("indexed spans can only be stored");
43                };
44                Err(transaction.internal_error("an indexed transaction can only be stored"))
45            }
46        }
47    }
48
49    #[cfg(feature = "processing")]
50    fn forward_store(
51        self,
52        s: crate::processing::StoreHandle<'_>,
53        ctx: ForwardContext<'_>,
54    ) -> Result<(), Rejected<()>> {
55        use crate::services::store::StoreEvent;
56        use relay_dynamic_config::Feature;
57        use relay_filter::FilterStatKey;
58
59        let (spans, transaction) = match self {
60            TransactionOutput::Full(managed) => {
61                return Err(managed.internal_error("only indexed transactions can be stored"));
62            }
63            TransactionOutput::Profile(profile) => {
64                s.send_to_store(profile.map(|p, _| store::convert_profile(p.profile, false, ctx)));
65                return Ok(());
66            }
67            TransactionOutput::Indexed { spans, transaction } => (spans, transaction),
68        };
69
70        let performance_issues_spans = ctx
71            .project_info
72            .has_feature(Feature::PerformanceIssuesSpans);
73
74        if let Some(spans) = spans {
75            let event_id = transaction.headers.event_id();
76            let retention = ctx.retention(|r| r.span.as_ref());
77
78            for span in spans.split(|spans| spans.into_iter()) {
79                if let Ok(mut span) =
80                    span.try_map(|span, _| store::convert_span(span, event_id, retention))
81                {
82                    if performance_issues_spans && *(span.item.is_segment.value().unwrap_or(&false))
83                    {
84                        span.modify(|span, _| {
85                            span.performance_issues_spans = true;
86                        });
87                    }
88                    s.send_to_store(span)
89                };
90            }
91        }
92
93        let (profile, event) = transaction.split_once(|tx, _| {
94            let ExpandedTransaction {
95                headers: _,
96                mut event,
97                flags: _,
98                attachments,
99                profile,
100                category: _,
101                span_extraction: _,
102            } = *tx;
103
104            if performance_issues_spans && let Some(event) = event.value_mut() {
105                event.performance_issues_spans = true.into();
106            }
107
108            let event = Box::new(StoreEvent {
109                event_category: relay_quotas::DataCategory::TransactionIndexed,
110                event,
111                attachments,
112                user_reports: Vec::new(),
113                retention_days: ctx.event_retention().standard,
114            });
115
116            (profile, event)
117        });
118
119        let discard_transactions = ctx.project_info.has_feature(Feature::DiscardTransaction);
120        match discard_transactions {
121            false => s.send_event(event),
122            true => drop(event.reject_err(Outcome::Filtered(FilterStatKey::Discarded))),
123        }
124
125        if let Some(profile) = profile.transpose() {
126            s.send_to_store(profile.map(|p, _| store::convert_profile(p, true, ctx)));
127        }
128
129        Ok(())
130    }
131}

144
145    pub fn convert_profile(
146        profile: ExpandedProfile,
147        sampled: bool,
148        ctx: ForwardContext<'_>,
149    ) -> StoreProfile {
150        let retention_days = ctx.event_retention().standard;
151
152        StoreProfile {
153            retention_days,
154            quantities: profile.quantities(),
155            profile: {
156                let mut item = profile.serialize_item();
157                item.set_sampled(sampled);
158                item
159            },
160        }
161    }
162
163    pub fn convert_span(
164        span: ExtractedIndexedSpan,
165        event_id: Option<EventId>,
166        retentions: Retention,
167    ) -> Result<Box<StoreSpanV2>, Outcome> {
168        let span = match span.0 {
169            Annotated(Some(span), _) => span,
170            Annotated(None, meta) => {
171                relay_log::debug!("dropping empty span with meta {meta:?}");
172                return Err(Outcome::Invalid(DiscardReason::InvalidSpan));
173            }
174        };
175
176        Ok(Box::new(StoreSpanV2 {
177            routing_key: span.trace_id.value().copied().map(Into::into),
178            retention_days: retentions.standard,
179            downsampled_retention_days: retentions.downsampled,
180            event_id,
181            performance_issues_spans: false,
182            item: span,
183        }))
184    }
