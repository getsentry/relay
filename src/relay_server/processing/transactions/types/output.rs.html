17/// Output of the transaction processor.
18#[derive(Debug)]
19pub enum TransactionOutput {
20    /// The transaction has not been dropped by dynamic sampling.
21    Full(Managed<Box<ExpandedTransaction>>),
22    /// The transaction has been dropped by dynamic sampling, only an optional profile remains.
23    Profile(Managed<Box<StandaloneProfile>>),
24    /// The transaction has not been dropped by dynamic sampling, and metrics have been extracted.
25    ///
26    /// This is used in processing relays.
27    Indexed {
28        spans: Option<Managed<ExtractedIndexedSpans>>,
29        transaction: Managed<Box<ExpandedTransaction<Indexed>>>,
30    },
31}

33impl Forward for TransactionOutput {
34    fn serialize_envelope(
35        self,
36        _ctx: ForwardContext<'_>,
37    ) -> Result<Managed<Box<Envelope>>, Rejected<()>> {
38        match self {
39            TransactionOutput::Full(managed) => managed.try_map(|work, _| {
40                work.serialize_envelope()
41                    .map_err(drop)
42                    .with_outcome(Outcome::Invalid(DiscardReason::Internal))
43            }),
44            TransactionOutput::Profile(profile) => {
45                Ok(profile.map(|profile, _| profile.serialize_envelope()))
46            }
47            TransactionOutput::Indexed { spans, transaction } => {
48                if let Some(spans) = spans {
49                    let _ = spans.internal_error("indexed spans can only be stored");
50                };
51                Err(transaction.internal_error("an indexed transaction can only be stored"))
52            }
53        }
54    }
55
56    #[cfg(feature = "processing")]
57    fn forward_store(
58        self,
59        s: StoreHandle<'_>,
60        ctx: ForwardContext<'_>,
61    ) -> Result<(), Rejected<()>> {
62        let (spans, transaction) = match self {
63            TransactionOutput::Full(managed) => {
64                return Err(managed.internal_error("only indexed transactions can be stored"));
65            }
66            TransactionOutput::Profile(profile) => {
67                s.send_to_store(profile.map(|p, _| store::convert_profile(p.profile, false, ctx)));
68                return Ok(());
69            }
70            TransactionOutput::Indexed { spans, transaction } => (spans, transaction),
71        };
72
73        let performance_issues_spans = ctx
74            .project_info
75            .has_feature(Feature::PerformanceIssuesSpans);
76
77        if let Some(spans) = spans {
78            let event_id = transaction.headers.event_id();
79            let retention = ctx.retention(|r| r.span.as_ref());
80
81            for span in spans.split(|spans| spans.into_iter()) {
82                if let Ok(mut span) =
83                    span.try_map(|span, _| store::convert_span(span, event_id, retention))
84                {
85                    if performance_issues_spans && *(span.item.is_segment.value().unwrap_or(&false))
86                    {
87                        span.modify(|span, _| {
88                            span.performance_issues_spans = true;
89                        });
90                    }
91                    s.send_to_store(span)
92                };
93            }
94        }
95
96        let (profile, transaction) = transaction.split_once(|mut tx, _| (tx.profile.take(), tx));
97        if let Some(profile) = profile.transpose() {
98            s.send_to_store(profile.map(|p, _| store::convert_profile(p, true, ctx)));
99        }
100
101        let envelope = transaction.try_map(|mut work, record_keeper| {
102            // TODO: This should raise an error, Indexed output should go straight to Kafka
103            // instead of an envelope. As long as we have this hack, ignore bookkeeping
104            record_keeper.lenient(relay_quotas::DataCategory::Transaction);
105
106            if let Some(event) = work.event.value_mut()
107                && performance_issues_spans
108            {
109                event.performance_issues_spans = Annotated::new(true);
110            }
111
112            work.serialize_envelope()
113                .map_err(drop)
114                .with_outcome(Outcome::Invalid(DiscardReason::Internal))
115        })?;
116
117        s.send_envelope(envelope.into());
118
119        Ok(())
120    }
121}

135    pub fn convert_profile(
136        profile: ExpandedProfile,
137        sampled: bool,
138        ctx: ForwardContext<'_>,
139    ) -> StoreProfile {
140        let retention_days = ctx.event_retention().standard;
141
142        StoreProfile {
143            retention_days,
144            quantities: profile.quantities(),
145            profile: {
146                let mut item = profile.serialize_item();
147                item.set_sampled(sampled);
148                item
149            },
150        }
151    }
152
153    pub fn convert_span(
154        span: ExtractedIndexedSpan,
155        event_id: Option<EventId>,
156        retentions: Retention,
157    ) -> Result<Box<StoreSpanV2>, Outcome> {
158        let span = match span.0 {
159            Annotated(Some(span), _) => span,
160            Annotated(None, meta) => {
161                relay_log::debug!("dropping empty span with meta {meta:?}");
162                return Err(Outcome::Invalid(DiscardReason::InvalidSpan));
163            }
164        };
165
166        Ok(Box::new(StoreSpanV2 {
167            routing_key: span.trace_id.value().copied().map(Into::into),
168            retention_days: retentions.standard,
169            downsampled_retention_days: retentions.downsampled,
170            event_id,
171            performance_issues_spans: false,
172            item: span,
173        }))
174    }
