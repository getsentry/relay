20/// Creates a span from the transaction and applies tag extraction on it.
21///
22/// Returns `None` when [`tag_extraction::extract_span_tags`] clears the span, which it shouldn't.
23pub fn extract_segment_span(
24    event: &Event,
25    max_tag_value_size: usize,
26    span_allowed_hosts: &[String],
27) -> Option<Span> {
28    let mut spans = [Span::from(event).into()];
29
30    tag_extraction::extract_span_tags(event, &mut spans, max_tag_value_size, span_allowed_hosts);
31    tag_extraction::extract_segment_span_tags(event, &mut spans);
32
33    spans.into_iter().next().and_then(Annotated::into_value)
34}

46/// Extracts metrics from a transaction and its spans.
47pub fn extract_metrics(
48    event: &mut Annotated<Event>,
49    extracted_metrics: &mut ProcessingExtractedMetrics,
50    ctx: ExtractMetricsContext,
51) -> Result<EventMetricsExtracted, ProcessingError> {
52    let ExtractMetricsContext {
53        dsc,
54        project_id,
55        ctx,
56        sampling_decision,
57        metrics_extracted,
58        extract_span_metrics,
59    } = ctx;
60    // TODO(follow-up): this function should always extract metrics. Dynamic sampling should validate
61    // the full metrics extraction config and skip sampling if it is incomplete.
62
63    if metrics_extracted {
64        return Ok(EventMetricsExtracted(true));
65    }
66    let Some(event) = event.value_mut() else {
67        // Nothing to extract, but metrics extraction was called.
68        return Ok(EventMetricsExtracted(true));
69    };
70
71    // NOTE: This function requires a `metric_extraction` in the project config. Legacy configs
72    // will upsert this configuration from transaction and conditional tagging fields, even if
73    // it is not present in the actual project config payload.
74    let combined_config = {
75        let config = match &ctx.project_info.config.metric_extraction {
76            ErrorBoundary::Ok(config) if config.is_supported() => config,
77            _ => return Ok(EventMetricsExtracted(false)),
78        };
79        let global_config = match &ctx.global_config.metric_extraction {
80            ErrorBoundary::Ok(global_config) => global_config,
81            #[allow(unused_variables)]
82            ErrorBoundary::Err(e) => {
83                if cfg!(feature = "processing") && ctx.config.processing_enabled() {
84                    // Config is invalid, but we will try to extract what we can with just the
85                    // project config.
86                    relay_log::error!("Failed to parse global extraction config {e}");
87                    MetricExtractionGroups::EMPTY
88                } else {
89                    // If there's an error with global metrics extraction, it is safe to assume that this
90                    // Relay instance is not up-to-date, and we should skip extraction.
91                    relay_log::debug!("Failed to parse global extraction config: {e}");
92                    return Ok(EventMetricsExtracted(false));
93                }
94            }
95        };
96        CombinedMetricExtractionConfig::new(global_config, config)
97    };
98
99    // Require a valid transaction metrics config.
100    let tx_config = match &ctx.project_info.config.transaction_metrics {
101        Some(ErrorBoundary::Ok(tx_config)) => tx_config,
102        Some(ErrorBoundary::Err(e)) => {
103            relay_log::debug!("Failed to parse legacy transaction metrics config: {e}");
104            return Ok(EventMetricsExtracted(false));
105        }
106        None => {
107            relay_log::debug!("Legacy transaction metrics config is missing");
108            return Ok(EventMetricsExtracted(false));
109        }
110    };
111
112    if !tx_config.is_enabled() {
113        static TX_CONFIG_ERROR: Once = Once::new();
114        TX_CONFIG_ERROR.call_once(|| {
115                if ctx.config.processing_enabled() {
116                    relay_log::error!(
117                        "Processing Relay outdated, received tx config in version {}, which is not supported",
118                        tx_config.version
119                    );
120                }
121            });
122
123        return Ok(EventMetricsExtracted(false));
124    }
125
126    let metrics = crate::metrics_extraction::event::extract_metrics(
127        event,
128        crate::metrics_extraction::event::ExtractMetricsConfig {
129            config: combined_config,
130            sampling_decision,
131            target_project_id: project_id,
132            max_tag_value_size: ctx
133                .config
134                .aggregator_config_for(MetricNamespace::Spans)
135                .max_tag_value_length,
136            extract_spans: extract_span_metrics,
137            transaction_from_dsc: dsc.and_then(|dsc| dsc.transaction.as_deref()),
138        },
139    );
140
141    extracted_metrics.extend(metrics, Some(sampling_decision));
142
143    if !ctx.project_info.has_feature(Feature::DiscardTransaction) {
144        let transaction_from_dsc = dsc.and_then(|dsc| dsc.transaction.as_deref());
145
146        let extractor = TransactionExtractor {
147            config: tx_config,
148            generic_config: Some(combined_config),
149            transaction_from_dsc,
150            sampling_decision,
151            target_project_id: project_id,
152        };
153
154        extracted_metrics.extend(extractor.extract(event)?, Some(sampling_decision));
155    }
156
157    Ok(EventMetricsExtracted(true))
158}
