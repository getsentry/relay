29/// Parses the event payload.
30///
31/// This function boxes the resulting [`ExpandedTransaction`] because keeping it on the stack
32/// causes stack overflows in unit tests when run without optimizations.
33pub fn expand(
34    work: Managed<SerializedTransaction>,
35) -> Result<Managed<Box<ExpandedTransaction>>, Rejected<Error>> {
36    work.try_map(|work, record_keeper| {
37        let SerializedTransaction {
38            headers,
39            event: transaction_item,
40            attachments,
41            profiles,
42        } = work;
43        let mut event = metric!(timer(RelayTimers::EventProcessingDeserialize), {
44            Annotated::<Event>::from_json_bytes(&transaction_item.payload())
45        })?;
46        if let Some(event) = event.value_mut() {
47            event.ty = EventType::Transaction.into();
48        }
49        let flags = Flags {
50            metrics_extracted: transaction_item.metrics_extracted(),
51            spans_extracted: transaction_item.spans_extracted(),
52            fully_normalized: headers.meta().request_trust().is_trusted()
53                && transaction_item.fully_normalized(),
54            spans_rate_limited: false,
55        };
56        validate_flags(&flags);
57
58        let profile = expand_profile(profiles, record_keeper);
59
60        #[cfg(debug_assertions)]
61        {
62            // Fix broken span count headers
63            use relay_protocol::get_value;
64            let embedded = get_value!(event.spans).map_or(0, Vec::len);
65            let diff = embedded as isize - transaction_item.span_count() as isize;
66            record_keeper.modify_by(DataCategory::Span, diff);
67            record_keeper.modify_by(DataCategory::SpanIndexed, diff);
68        }
69
70        Ok::<_, Error>(Box::new(ExpandedTransaction {
71            headers,
72            event,
73            flags,
74            attachments,
75            profile,
76            category: TotalAndIndexed,
77        }))
78    })
79}

81/// Validates the following assumption:
82/// 1. Metrics are only extracted in non-processing relays if the sampling decision is "drop".
83/// 2. That means that if we see a new transaction, it cannot yet have metrics extracted.
84fn validate_flags(flags: &Flags) {
85    if flags.metrics_extracted {
86        relay_log::error!("Received a transaction which already had its metrics extracted.");
87    }
88}

90fn expand_profile(
91    profiles: Vec<Item>,
92    record_keeper: &mut RecordKeeper<'_>,
93) -> Option<ExpandedProfile> {
94    let mut profiles = profiles.into_iter();
95
96    // Accept at most one profile:
97    let profile = profiles.next()?;
98    for additional_profile in profiles {
99        record_keeper.reject_err(
100            Outcome::Invalid(DiscardReason::Profiling(relay_profiling::discard_reason(
101                &ProfileError::TooManyProfiles,
102            ))),
103            additional_profile,
104        );
105    }
106
107    let meta = match relay_profiling::parse_metadata(&profile.payload()) {
108        Ok(meta) => meta,
109        Err(err) => {
110            record_keeper.reject_err(
111                Outcome::Invalid(DiscardReason::Profiling(relay_profiling::discard_reason(
112                    &err,
113                ))),
114                profile,
115            );
116            return None;
117        }
118    };
119
120    // If the profile type is new information, we now count the profile in an additional data category.
121    if profile.profile_type().is_none() {
122        record_keeper.modify_by(
123            match meta.profile_type() {
124                ProfileType::Backend => DataCategory::ProfileBackend,
125                ProfileType::Ui => DataCategory::ProfileUi,
126            },
127            1,
128        );
129    }
130
131    Some(ExpandedProfile {
132        meta,
133        item: profile,
134    })
135}

137/// Validates and massages the data.
138pub fn prepare_data(
139    work: &mut Managed<Box<ExpandedTransaction>>,
140    ctx: &mut Context<'_>,
141    metrics: &mut Metrics,
142) -> Result<(), Rejected<Error>> {
143    let scoping = work.scoping();
144    work.try_modify(|work, record_keeper| {
145        profile::filter(work, record_keeper, *ctx);
146        profile::transfer_id(&mut work.event, work.profile.as_ref().map(|p| p.meta.id));
147        profile::remove_context_if_rate_limited(&mut work.event, scoping, *ctx);
148
149        utils::dsc::validate_and_set_dsc(&mut work.headers, &work.event, ctx);
150
151        utils::event::finalize(
152            &work.headers,
153            &mut work.event,
154            work.attachments.iter(),
155            metrics,
156            ctx.config,
157        )
158        .map_err(Error::from)
159    })?;
160    Ok(())
161}

163/// Normalizes the transaction event.
164pub fn normalize(
165    work: Managed<Box<ExpandedTransaction>>,
166    ctx: Context<'_>,
167    geoip_lookup: &GeoIpLookup,
168) -> Result<Managed<Box<ExpandedTransaction>>, Rejected<Error>> {
169    let project_id = work.scoping().project_id;
170    work.try_map(|mut work, r| {
171        let original_span_count = work.count_embedded_spans_and_self();
172
173        work.flags.fully_normalized = utils::event::normalize(
174            &work.headers,
175            &mut work.event,
176            EventFullyNormalized(work.flags.fully_normalized),
177            project_id,
178            ctx,
179            geoip_lookup,
180        )?
181        .0;
182
183        // Normalization may have trimmed spans:
184        debug_assert!(!work.flags.spans_extracted);
185        let new_span_count = work.count_embedded_spans_and_self();
186        if let Some(trimmed) = original_span_count.checked_sub(new_span_count)
187            && trimmed > 0
188        {
189            r.reject_err(
190                Outcome::Invalid(DiscardReason::ItemTooLarge(DiscardItemType::Span)),
191                [
192                    (DataCategory::Span, trimmed),
193                    (DataCategory::SpanIndexed, trimmed),
194                ],
195            );
196        }
197
198        Ok::<_, Error>(work)
199    })
200}

213/// The result of dynamic sampling.
214pub enum SamplingOutput {
215    /// The decision was retain, maintain full transaction.
216    Keep {
217        payload: Managed<Box<ExpandedTransaction>>,
218        sample_rate: Option<f64>,
219    },
220    /// The decision was discard keep only extracted metrics and an optional profile.
221    Drop {
222        metrics: Managed<ExtractedMetrics>,
223        profile: Option<Managed<Box<StandaloneProfile>>>,
224    },
225}
226
227/// Computes the sampling decision for a transaction and associated items.
228pub fn run_dynamic_sampling(
229    payload: Managed<Box<ExpandedTransaction>>,
230    ctx: Context<'_>,
231    filters_status: FiltersStatus,
232) -> SamplingOutput {
233    let sampling_result = make_dynamic_sampling_decision(&payload, ctx, filters_status);
234
235    let sampling_match = match sampling_result {
236        SamplingResult::Match(m) if m.decision().is_drop() => m,
237        keep => {
238            return SamplingOutput::Keep {
239                payload,
240                sample_rate: keep.sample_rate(),
241            };
242        }
243    };
244
245    // At this point the decision is to drop the payload.
246    let (payload, metrics) = split_indexed_and_total(payload, ctx, SamplingDecision::Drop);
247
248    let (payload, profile) = payload.split_once(|mut payload, _| {
249        let profile = payload.profile.take().map(|profile| StandaloneProfile {
250            profile,
251            // Actually no need to clone here, since we do drop the remaining transaction after,
252            // for simplicity sake we clone for now.
253            headers: payload.headers.clone(),
254        });
255
256        (payload, profile)
257    });
258
259    let outcome = Outcome::FilteredSampling(sampling_match.into_matched_rules().into());
260    let _ = payload.reject_err(outcome);
261
262    SamplingOutput::Drop {
263        metrics,
264        profile: profile.transpose().map(Managed::boxed),
265    }
266}

268/// Computes the dynamic sampling decision for the unit of work, but does not perform action on data.
269fn make_dynamic_sampling_decision(
270    work: &Managed<Box<ExpandedTransaction>>,
271    ctx: Context<'_>,
272    filters_status: FiltersStatus,
273) -> SamplingResult {
274    let sampling_result = do_make_dynamic_sampling_decision(work, ctx, filters_status);
275    relay_statsd::metric!(
276        counter(RelayCounters::SamplingDecision) += 1,
277        decision = sampling_result.decision().as_str(),
278        item = "transaction"
279    );
280    sampling_result
281}
282
283fn do_make_dynamic_sampling_decision(
284    work: &Managed<Box<ExpandedTransaction>>,
285    ctx: Context<'_>,
286    filters_status: FiltersStatus,
287) -> SamplingResult {
288    // Always run dynamic sampling on processing Relays,
289    // but delay decision until inbound filters have been fully processed.
290    // Also, we require transaction metrics to be enabled before sampling.
291    let should_run = matches!(filters_status, FiltersStatus::Ok) || ctx.config.processing_enabled();
292    if !should_run {
293        return SamplingResult::Pending;
294    }
295
296    utils::dynamic_sampling::run(work.headers.dsc(), work.event.value(), &ctx)
297}

305/// Splits transaction into indexed payload and metrics representing the total counts.
306///
307/// Like [`split_indexed_and_total`] but works with [`ExtractedSpans`].
308pub fn split_indexed_and_total_with_extracted_spans(
309    transaction: Managed<Box<ExpandedTransaction>>,
310    spans: Option<Managed<ExtractedSpans>>,
311    ctx: Context<'_>,
312) -> IndexedTransactionAndSpanAndMetrics {
313    let scoping = transaction.scoping();
314
315    let (transaction, metrics) = transaction.split_once(|mut tx, r| {
316        r.lenient(DataCategory::MetricBucket);
317
318        let mut metrics = ProcessingExtractedMetrics::new();
319
320        let had_metrics_extracted = tx.flags.metrics_extracted;
321        tx.flags.metrics_extracted = extraction::extract_metrics(
322            &mut tx.event,
323            &mut metrics,
324            ExtractMetricsContext {
325                dsc: tx.headers.dsc(),
326                project_id: scoping.project_id,
327                ctx,
328                sampling_decision: SamplingDecision::Keep,
329                metrics_extracted: tx.flags.metrics_extracted,
330                extract_span_metrics: spans.is_some(),
331            },
332        )
333        .0;
334
335        if had_metrics_extracted || !tx.flags.metrics_extracted {
336            // Invalid config or invalid original transaction
337            r.lenient(DataCategory::Transaction);
338            r.lenient(DataCategory::Span);
339        }
340
341        // This really is a bug, we ignore here.
342        //
343        // Transactions are counted using a span metric, as transaction payloads should
344        // eventually be fully transformed into spans.
345        //
346        // Since there is no span metric extracted for this transaction, as we already extracted
347        // the spans from the transaction, there is now no metric carrying the transaction category.
348        //
349        // After extracting span metrics the count is accurate again, but attached to the span metrics.
350        // Unless, the spans have been rate limited, which is an actual potential bug which we
351        // ignore here for two reasons:
352        //  - Span rate limits should be applied to transaction as well
353        //  - Long-term transactions will no longer exist
354        if spans.is_none() {
355            r.lenient(DataCategory::Transaction);
356        }
357
358        // Since we just extracted span metrics, which account for the total spans, we need also fix
359        // these counts, later we correct this again.
360        if let Some(spans) = &spans {
361            r.modify_by(DataCategory::Span, spans.0.len() as isize);
362        }
363
364        (Box::new(tx.into_indexed()), metrics.into_inner())
365    });
366
367    // In an ideal world we would use these extracted spans to also extract span metrics instead of
368    // re-using the transaction to get the metrics and risking differences in metrics.
369    //
370    // The master plan on how to clean this up:
371    //  1. Migrate dynamic sampling to EAP
372    //  2. Emit total category outcomes in Relay instead of as a metric
373    //  3. Remove all span metrics, including extraction (possible after 1., and 2.)
374    let spans = spans.map(|spans| {
375        spans.map(|spans, r| {
376            if let Some((c, q)) = metrics
377                .quantities()
378                .iter()
379                .find(|(c, _)| *c == DataCategory::Span)
380            {
381                // "Insurance" that metrics extracted from the transaction spans match the extracted
382                // spans.
383                r.modify_by(*c, -(*q as isize));
384            } else if transaction.flags.metrics_extracted {
385                // This `metrics_extracted` flag really isn't necessary anymore and should be deleted.
386                r.lenient(DataCategory::Span);
387            }
388
389            spans.into_indexed()
390        })
391    });
392
393    (transaction, spans, metrics)
394}

401/// Splits transaction into indexed payload and metrics representing the total counts.
402pub fn split_indexed_and_total(
403    mut work: Managed<Box<ExpandedTransaction>>,
404    ctx: Context<'_>,
405    sampling_decision: SamplingDecision,
406) -> IndexedAndMetrics {
407    let scoping = work.scoping();
408
409    let had_metrics_extracted = work.flags.metrics_extracted;
410    let extract_span_metrics = !work.flags.spans_rate_limited;
411
412    let mut metrics = ProcessingExtractedMetrics::new();
413    work.modify(|work, _| {
414        work.flags.metrics_extracted = extraction::extract_metrics(
415            &mut work.event,
416            &mut metrics,
417            ExtractMetricsContext {
418                dsc: work.headers.dsc(),
419                project_id: scoping.project_id,
420                ctx,
421                sampling_decision,
422                metrics_extracted: work.flags.metrics_extracted,
423                extract_span_metrics,
424            },
425        )
426        .0;
427    });
428
429    work.split_once(|work, r| {
430        r.lenient(DataCategory::MetricBucket);
431        if had_metrics_extracted || !work.flags.metrics_extracted {
432            // Invalid config or invalid original transaction
433            r.lenient(DataCategory::Transaction);
434            r.lenient(DataCategory::Span);
435        }
436        if work.flags.spans_rate_limited {
437            // This really is a bug, we ignore here.
438            //
439            // Transactions are counted using a span metric, as transaction payloads should
440            // eventually be fully transformed into spans. But this is for now only the case for
441            // metrics, the payloads are lagging behind.
442            //
443            // If spans are rate limited, there is no metric to attach the transaction to ->
444            // we're losing a transaction here.
445            //
446            //  If and once we apply span rate limits to transactions this case will also be fixed,
447            //  as it can no longer happen.
448            r.lenient(DataCategory::Transaction);
449        }
450
451        (Box::new(work.into_indexed()), metrics.into_inner())
452    })
453}

455/// Processes the profile attached to the transaction.
456pub fn process_profile(work: &mut Managed<Box<ExpandedTransaction>>, ctx: Context<'_>) {
457    work.modify(|work, record_keeper| {
458        if let Some(profile) = work.profile.as_mut()
459            && let Err(outcome) = profile::process(
460                &mut profile.item,
461                work.headers.meta().client_addr(),
462                work.event.value(),
463                &ctx,
464            )
465        {
466            record_keeper.reject_err(outcome, work.profile.take());
467        };
468
469        let profile_id = work.profile.as_ref().map(|profile| profile.meta.id);
470        profile::transfer_id(&mut work.event, profile_id);
471        profile::scrub_profiler_id(&mut work.event);
472    });
473}

475/// Converts the spans embedded in the transaction into top-level span items.
476///
477/// Only extracts spans in processing.
478pub fn extract_spans(
479    transaction: Managed<Box<ExpandedTransaction>>,
480    ctx: Context<'_>,
481    server_sample_rate: Option<f64>,
482) -> (Managed<ExtractedSpans>, Managed<Box<ExpandedTransaction>>) {
483    // This could be validated with additional typing, a different type for transactions which have
484    // spans extracted.
485    debug_assert!(
486        !transaction.flags.spans_extracted,
487        "spans can only be extracted once"
488    );
489
490    transaction.split_once(|mut tx, r| {
491        let spans =
492            spans::extract_from_event(tx.headers.dsc(), &tx.event, ctx.config, server_sample_rate)
493                .into_iter()
494                .filter_map(|span| match span {
495                    Ok(span) => Some(span),
496                    Err(()) => {
497                        r.reject_err(
498                            Outcome::Invalid(DiscardReason::InvalidSpan),
499                            IndexedSpans(1),
500                        );
501                        None
502                    }
503                })
504                .collect();
505
506        // Once spans are extracted, they are no longer counted towards the transaction.
507        tx.flags.spans_extracted = true;
508
509        (ExtractedSpans(spans), tx)
510    })
511}

513/// Runs PiiProcessors on the event and its attachments.
514pub fn scrub(
515    work: Managed<Box<ExpandedTransaction>>,
516    ctx: Context<'_>,
517) -> Result<Managed<Box<ExpandedTransaction>>, Rejected<Error>> {
518    work.try_map(|mut work, records| {
519        utils::event::scrub(&mut work.event, ctx.project_info)?;
520        utils::attachments::scrub(work.attachments.iter_mut(), ctx.project_info, Some(records));
521        Ok::<_, Error>(work)
522    })
523}
