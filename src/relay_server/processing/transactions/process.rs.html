34/// Parses the event payload.
35///
36/// This function boxes the resulting [`ExpandedTransaction`] because keeping it on the stack
37/// causes stack overflows in unit tests when run without optimizations.
38pub fn expand(
39    work: Managed<SerializedTransaction>,
40) -> Result<Managed<Box<ExpandedTransaction>>, Rejected<Error>> {
41    work.try_map(|work, record_keeper| {
42        let SerializedTransaction {
43            headers,
44            event: transaction_item,
45            attachments,
46            profiles,
47        } = work;
48        let mut event = metric!(timer(RelayTimers::EventProcessingDeserialize), {
49            Annotated::<Event>::from_json_bytes(&transaction_item.payload())
50        })?;
51        if let Some(event) = event.value_mut() {
52            event.ty = EventType::Transaction.into();
53        }
54        let flags = Flags {
55            metrics_extracted: transaction_item.metrics_extracted(),
56            spans_extracted: transaction_item.spans_extracted(),
57            fully_normalized: headers.meta().request_trust().is_trusted()
58                && transaction_item.fully_normalized(),
59            spans_rate_limited: false,
60        };
61        validate_flags(&flags);
62
63        let profile = expand_profile(profiles, record_keeper);
64
65        #[cfg(debug_assertions)]
66        {
67            // Fix broken span count headers
68            use relay_protocol::get_value;
69            let embedded = get_value!(event.spans).map_or(0, Vec::len);
70            let diff = embedded as isize - transaction_item.span_count() as isize;
71            record_keeper.modify_by(DataCategory::Span, diff);
72            record_keeper.modify_by(DataCategory::SpanIndexed, diff);
73        }
74
75        Ok::<_, Error>(Box::new(ExpandedTransaction {
76            headers,
77            event,
78            flags,
79            attachments,
80            profile,
81            extracted_spans: vec![],
82            category: TotalAndIndexed,
83        }))
84    })
85}

87/// Validates the following assumption:
88/// 1. Metrics are only extracted in non-processing relays if the sampling decision is "drop".
89/// 2. That means that if we see a new transaction, it cannot yet have metrics extracted.
90fn validate_flags(flags: &Flags) {
91    if flags.metrics_extracted {
92        relay_log::error!("Received a transaction which already had its metrics extracted.");
93    }
94}

96fn expand_profile(
97    profiles: Items,
98    record_keeper: &mut RecordKeeper<'_>,
99) -> Option<ExpandedProfile> {
100    let mut profiles = profiles.into_iter();
101
102    // Accept at most one profile:
103    let profile = profiles.next()?;
104    for additional_profile in profiles {
105        record_keeper.reject_err(
106            Outcome::Invalid(DiscardReason::Profiling(relay_profiling::discard_reason(
107                &ProfileError::TooManyProfiles,
108            ))),
109            additional_profile,
110        );
111    }
112
113    let meta = match relay_profiling::parse_metadata(&profile.payload()) {
114        Ok(meta) => meta,
115        Err(err) => {
116            record_keeper.reject_err(
117                Outcome::Invalid(DiscardReason::Profiling(relay_profiling::discard_reason(
118                    &err,
119                ))),
120                profile,
121            );
122            return None;
123        }
124    };
125
126    // If the profile type is new information, we now count the profile in an additional data category.
127    if profile.profile_type().is_none() {
128        record_keeper.modify_by(
129            match meta.kind {
130                ProfileType::Backend => DataCategory::ProfileBackend,
131                ProfileType::Ui => DataCategory::ProfileUi,
132            },
133            1,
134        );
135    }
136
137    Some(ExpandedProfile {
138        meta,
139        item: profile,
140    })
141}

143/// Validates and massages the data.
144pub fn prepare_data(
145    work: &mut Managed<Box<ExpandedTransaction>>,
146    ctx: &mut Context<'_>,
147    metrics: &mut Metrics,
148) -> Result<(), Rejected<Error>> {
149    let scoping = work.scoping();
150    work.try_modify(|work, record_keeper| {
151        profile::filter(work, record_keeper, *ctx);
152        profile::transfer_id(&mut work.event, work.profile.as_ref().map(|p| p.meta.id));
153        profile::remove_context_if_rate_limited(&mut work.event, scoping, *ctx);
154
155        utils::dsc::validate_and_set_dsc(&mut work.headers, &work.event, ctx);
156
157        utils::event::finalize(
158            &work.headers,
159            &mut work.event,
160            work.attachments.iter(),
161            metrics,
162            ctx.config,
163        )
164        .map_err(Error::from)
165    })?;
166    Ok(())
167}

169/// Normalizes the transaction event.
170pub fn normalize(
171    work: Managed<Box<ExpandedTransaction>>,
172    ctx: Context<'_>,
173    geoip_lookup: &GeoIpLookup,
174) -> Result<Managed<Box<ExpandedTransaction>>, Rejected<Error>> {
175    let project_id = work.scoping().project_id;
176    work.try_map(|mut work, r| {
177        let original_span_count = work.count_embedded_spans_and_self();
178
179        work.flags.fully_normalized = utils::event::normalize(
180            &work.headers,
181            &mut work.event,
182            EventFullyNormalized(work.flags.fully_normalized),
183            project_id,
184            ctx,
185            geoip_lookup,
186        )?
187        .0;
188
189        // Normalization may have trimmed spans:
190        debug_assert!(!work.flags.spans_extracted);
191        let new_span_count = work.count_embedded_spans_and_self();
192        if let Some(trimmed) = original_span_count.checked_sub(new_span_count)
193            && trimmed > 0
194        {
195            r.reject_err(
196                Outcome::Invalid(DiscardReason::TooLarge(DiscardItemType::Span)),
197                [
198                    (DataCategory::Span, trimmed),
199                    (DataCategory::SpanIndexed, trimmed),
200                ],
201            );
202        }
203
204        Ok::<_, Error>(work)
205    })
206}

219/// The result of dynamic sampling.
220pub enum SamplingOutput {
221    /// The decision was retain, maintain full transaction.
222    Keep {
223        payload: Managed<Box<ExpandedTransaction>>,
224        sample_rate: Option<f64>,
225    },
226    /// The decision was discard keep only extracted metrics and an optional profile.
227    Drop {
228        metrics: Managed<ExtractedMetrics>,
229        profile: Option<Managed<Box<StandaloneProfile>>>,
230    },
231}
232
233/// Computes the sampling decision for a transaction and associated items.
234pub async fn run_dynamic_sampling(
235    payload: Managed<Box<ExpandedTransaction>>,
236    ctx: Context<'_>,
237    filters_status: FiltersStatus,
238    quotas_client: Option<&AsyncRedisClient>,
239) -> Result<SamplingOutput, Rejected<Error>> {
240    let sampling_result =
241        make_dynamic_sampling_decision(&payload, ctx, filters_status, quotas_client).await;
242
243    let sampling_match = match sampling_result {
244        SamplingResult::Match(m) if m.decision().is_drop() => m,
245        keep => {
246            return Ok(SamplingOutput::Keep {
247                payload,
248                sample_rate: keep.sample_rate(),
249            });
250        }
251    };
252
253    // At this point the decision is to drop the payload.
254    let (payload, metrics) = split_indexed_and_total(payload, ctx, SamplingDecision::Drop)?;
255
256    let (payload, profile) = payload.split_once(|mut payload, _| {
257        let profile = payload.profile.take().map(|profile| StandaloneProfile {
258            profile,
259            // Actually no need to clone here, since we do drop the remaining transaction after,
260            // for simplicity sake we clone for now.
261            headers: payload.headers.clone(),
262        });
263
264        (payload, profile)
265    });
266
267    let outcome = Outcome::FilteredSampling(sampling_match.into_matched_rules().into());
268    let _ = payload.reject_err(outcome);
269
270    Ok(SamplingOutput::Drop {
271        metrics,
272        profile: profile.transpose().map(Managed::boxed),
273    })
274}

276/// Computes the dynamic sampling decision for the unit of work, but does not perform action on data.
277async fn make_dynamic_sampling_decision(
278    work: &Managed<Box<ExpandedTransaction>>,
279    ctx: Context<'_>,
280    filters_status: FiltersStatus,
281    quotas_client: Option<&AsyncRedisClient>,
282) -> SamplingResult {
283    let sampling_result =
284        do_make_dynamic_sampling_decision(work, ctx, filters_status, quotas_client).await;
285    relay_statsd::metric!(
286        counter(RelayCounters::SamplingDecision) += 1,
287        decision = sampling_result.decision().as_str(),
288        item = "transaction"
289    );
290    sampling_result
291}
292
293async fn do_make_dynamic_sampling_decision(
294    work: &Managed<Box<ExpandedTransaction>>,
295    ctx: Context<'_>,
296    filters_status: FiltersStatus,
297    #[allow(unused)] quotas_client: Option<&AsyncRedisClient>,
298) -> SamplingResult {
299    // Always run dynamic sampling on processing Relays,
300    // but delay decision until inbound filters have been fully processed.
301    // Also, we require transaction metrics to be enabled before sampling.
302    let should_run = matches!(filters_status, FiltersStatus::Ok) || ctx.config.processing_enabled();
303
304    let can_extract_metrics = matches!(&ctx.project_info.config.transaction_metrics, Some(ErrorBoundary::Ok(c)) if c.is_enabled());
305    if !(should_run && can_extract_metrics) {
306        return SamplingResult::Pending;
307    }
308
309    #[allow(unused_mut)]
310    let mut reservoir = ReservoirEvaluator::new(Arc::clone(ctx.reservoir_counters));
311    #[cfg(feature = "processing")]
312    if let Some(quotas_client) = quotas_client {
313        reservoir.set_redis(work.scoping().organization_id, quotas_client);
314    }
315    utils::dynamic_sampling::run(
316        work.headers.dsc(),
317        work.event.value(),
318        &ctx,
319        Some(&reservoir),
320    )
321    .await
322}

329/// Splits transaction into indexed payload and metrics representing the total counts.
330pub fn split_indexed_and_total(
331    mut work: Managed<Box<ExpandedTransaction>>,
332    ctx: Context<'_>,
333    sampling_decision: SamplingDecision,
334) -> Result<IndexedAndMetrics, Rejected<Error>> {
335    let scoping = work.scoping();
336
337    let had_metrics_extracted = work.flags.metrics_extracted;
338    let extract_span_metrics = !work.flags.spans_rate_limited;
339
340    let mut metrics = ProcessingExtractedMetrics::new();
341    work.try_modify(|work, _| {
342        work.flags.metrics_extracted = extraction::extract_metrics(
343            &mut work.event,
344            &mut metrics,
345            ExtractMetricsContext {
346                dsc: work.headers.dsc(),
347                project_id: scoping.project_id,
348                ctx,
349                sampling_decision,
350                metrics_extracted: work.flags.metrics_extracted,
351                extract_span_metrics,
352            },
353        )?
354        .0;
355
356        Ok::<_, Error>(())
357    })?;
358
359    Ok(work.split_once(|work, r| {
360        r.lenient(DataCategory::MetricBucket);
361        if had_metrics_extracted || !work.flags.metrics_extracted {
362            // Invalid config or invalid original transaction
363            r.lenient(DataCategory::Transaction);
364            r.lenient(DataCategory::Span);
365        }
366
367        (Box::new(work.into_indexed()), metrics.into_inner())
368    }))
369}

371/// Processes the profile attached to the transaction.
372pub fn process_profile(work: &mut Managed<Box<ExpandedTransaction>>, ctx: Context<'_>) {
373    work.modify(|work, record_keeper| {
374        if let Some(profile) = work.profile.as_mut()
375            && let Err(outcome) = profile::process(
376                &mut profile.item,
377                work.headers.meta().client_addr(),
378                work.event.value(),
379                &ctx,
380            )
381        {
382            record_keeper.reject_err(outcome, work.profile.take());
383        };
384
385        let profile_id = work.profile.as_ref().map(|profile| profile.meta.id);
386        profile::transfer_id(&mut work.event, profile_id);
387        profile::scrub_profiler_id(&mut work.event);
388    });
389}

391/// Converts the spans embedded in the transaction into top-level span items.
392///
393/// Only extracts spans in processing.
394pub fn extract_spans(
395    mut work: Managed<Box<ExpandedTransaction>>,
396    ctx: Context<'_>,
397    server_sample_rate: Option<f64>,
398) -> Managed<Box<ExpandedTransaction>> {
399    if !ctx.is_processing() {
400        return work;
401    }
402
403    work.modify(|work, r| {
404        if let Some(results) = spans::extract_from_event(
405            work.headers.dsc(),
406            &work.event,
407            ctx.config,
408            server_sample_rate,
409            EventMetricsExtracted(work.flags.metrics_extracted),
410            SpansExtracted(work.flags.spans_extracted),
411        ) {
412            work.flags.spans_extracted = true;
413            for result in results {
414                match result {
415                    Ok(item) => work.extracted_spans.push(item),
416                    Err(_) => {
417                        r.reject_err(
418                            Outcome::Invalid(DiscardReason::InvalidSpan),
419                            IndexedSpans(1),
420                        );
421                    }
422                }
423            }
424        }
425    });
426    work
427}

429/// Runs PiiProcessors on the event and its attachments.
430pub fn scrub(
431    work: Managed<Box<ExpandedTransaction>>,
432    ctx: Context<'_>,
433) -> Result<Managed<Box<ExpandedTransaction>>, Rejected<Error>> {
434    work.try_map(|mut work, _| {
435        utils::event::scrub(&mut work.event, ctx.project_info)?;
436        utils::attachments::scrub(work.attachments.iter_mut(), ctx.project_info);
437        Ok::<_, Error>(work)
438    })
439}
