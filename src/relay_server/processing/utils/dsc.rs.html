9/// Ensures there is a valid dynamic sampling context and corresponding project state.
10///
11/// The dynamic sampling context (DSC) specifies the project_key of the project that initiated
12/// the trace. That project state should have been loaded previously by the project cache and is
13/// available on the `ProcessEnvelopeState`. Under these conditions, this cannot happen:
14///
15///  - There is no DSC in the envelope headers. This occurs with older or third-party SDKs.
16///  - The project key does not exist. This can happen if the project key was disabled, the
17///    project removed, or in rare cases when a project from another Sentry instance is referred
18///    to.
19///  - The project key refers to a project from another organization. In this case the project
20///    cache does not resolve the state and instead leaves it blank.
21///  - The project state could not be fetched. This is a runtime error, but in this case Relay
22///    should fall back to the next-best sampling rule set.
23///
24/// In all of the above cases, this function will compute a new DSC using information from the
25/// event payload, similar to how SDKs do this. The `sampling_project_state` is also switched to
26/// the main project state.
27///
28/// If there is no transaction event in the envelope, this function will do nothing.
29///
30/// The function will return the sampling project information of the root project for the event. If
31/// no sampling project information is specified, the project information of the eventâ€™s project
32/// will be returned.
33pub fn validate_and_set_dsc(
34    headers: &mut EnvelopeHeaders,
35    event: &Annotated<Event>,
36    ctx: &mut Context,
37) {
38    let original_dsc = headers.dsc();
39    if original_dsc.is_some() && ctx.sampling_project_info.is_some() {
40        return;
41    }
42
43    // The DSC can only be computed if there's a transaction event. Note that `dsc_from_event`
44    // below already checks for the event type.
45    if let Some(event) = event.value()
46        && let Some(key_config) = ctx.project_info.get_public_key_config()
47        && let Some(mut dsc) = dsc_from_event(key_config.public_key, event)
48    {
49        // All other information in the DSC must be discarded, but the sample rate was
50        // actually applied by the client and is therefore correct.
51        let original_sample_rate = original_dsc.and_then(|dsc| dsc.sample_rate);
52        dsc.sample_rate = dsc.sample_rate.or(original_sample_rate);
53
54        headers.set_dsc(dsc);
55
56        ctx.sampling_project_info = Some(ctx.project_info);
57        return;
58    }
59
60    // If we cannot compute a new DSC but the old one is incorrect, we need to remove it.
61    headers.remove_dsc();
62}

64/// Computes a dynamic sampling context from a transaction event.
65///
66/// Returns `None` if the passed event is not a transaction event, or if it does not contain a
67/// trace ID in its trace context. All optional fields in the dynamic sampling context are
68/// populated with the corresponding attributes from the event payload if they are available.
69///
70/// Since sampling information is not available in the event payload, the `sample_rate` field
71/// cannot be set when computing the dynamic sampling context from a transaction event.
72pub fn dsc_from_event(public_key: ProjectKey, event: &Event) -> Option<DynamicSamplingContext> {
73    if event.ty.value() != Some(&EventType::Transaction) {
74        return None;
75    }
76
77    let trace = event.context::<TraceContext>()?;
78    let trace_id = *trace.trace_id.value()?;
79    let user = event.user.value();
80
81    Some(DynamicSamplingContext {
82        trace_id,
83        public_key,
84        release: event.release.as_str().map(str::to_owned),
85        environment: event.environment.value().cloned(),
86        transaction: event.transaction.value().cloned(),
87        replay_id: None,
88        sample_rate: None,
89        user: TraceUserContext {
90            user_segment: user
91                .and_then(|u| u.segment.value().cloned())
92                .unwrap_or_default(),
93            user_id: user
94                .and_then(|u| u.id.as_str())
95                .unwrap_or_default()
96                .to_owned(),
97        },
98        sampled: None,
99        other: Default::default(),
100    })
