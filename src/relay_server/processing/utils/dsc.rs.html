11/// Ensures there is a valid dynamic sampling context and corresponding project state.
12///
13/// The dynamic sampling context (DSC) specifies the project_key of the project that initiated
14/// the trace. That project state should have been loaded previously by the project cache and is
15/// available on the `ProcessEnvelopeState`. Under these conditions, this cannot happen:
16///
17///  - There is no DSC in the envelope headers. This occurs with older or third-party SDKs.
18///  - The project key does not exist. This can happen if the project key was disabled, the
19///    project removed, or in rare cases when a project from another Sentry instance is referred
20///    to.
21///  - The project key refers to a project from another organization. In this case the project
22///    cache does not resolve the state and instead leaves it blank.
23///  - The project state could not be fetched. This is a runtime error, but in this case Relay
24///    should fall back to the next-best sampling rule set.
25///
26/// In all of the above cases, this function will compute a new DSC using information from the
27/// event payload, similar to how SDKs do this. The `sampling_project_state` is also switched to
28/// the main project state.
29///
30/// If there is no transaction event in the envelope, this function will do nothing.
31///
32/// The function will use the sampling project information of the trace root project. If
33/// the sampling project information is missing - due to the project being disabled or belonging to
34/// a separate org - the event’s own project information will be used instead.
35pub fn validate_and_set_dsc(
36    headers: &mut EnvelopeHeaders,
37    event: &Annotated<Event>,
38    ctx: &mut Context,
39) {
40    let mut original_dsc = headers.dsc_mut();
41    if let Some(dsc) = original_dsc.as_mut()
42        && let Some(sp) = ctx.sampling_project_info
43    {
44        dsc.project_id = sp.project_id;
45        return;
46    }
47    if ctx.sampling_project_info.is_none() {
48        relay_statsd::metric!(
49            counter(RelayCounters::SamplingProjectUnresolved) += 1,
50            item = "transaction"
51        );
52    }
53
54    // The DSC can only be computed if there's a transaction event. Note that `dsc_from_event`
55    // below already checks for the event type.
56    if let Some(event) = event.value()
57        && let Some(key_config) = ctx.project_info.get_public_key_config()
58        && let Some(mut dsc) =
59            dsc_from_event(key_config.public_key, ctx.project_info.project_id, event)
60    {
61        // All other information in the DSC must be discarded, but the sample rate was
62        // actually applied by the client and is therefore correct.
63        dsc.sample_rate = original_dsc.as_ref().and_then(|dsc| dsc.sample_rate);
64
65        headers.set_dsc(dsc);
66
67        ctx.sampling_project_info = Some(ctx.project_info);
68        return;
69    }
70
71    // If we cannot compute a new DSC but the old one is incorrect, we need to remove it.
72    headers.remove_dsc();
73}

75/// Computes a dynamic sampling context from a transaction event.
76///
77/// Returns `None` if the passed event is not a transaction event, or if it does not contain a
78/// trace ID in its trace context. All optional fields in the dynamic sampling context are
79/// populated with the corresponding attributes from the event payload if they are available.
80///
81/// Since sampling information is not available in the event payload, the `sample_rate` field
82/// cannot be set when computing the dynamic sampling context from a transaction event.
83pub fn dsc_from_event(
84    public_key: ProjectKey,
85    project_id: Option<ProjectId>,
86    event: &Event,
87) -> Option<DynamicSamplingContext> {
88    if event.ty.value() != Some(&EventType::Transaction) {
89        return None;
90    }
91
92    let trace = event.context::<TraceContext>()?;
93    let trace_id = *trace.trace_id.value()?;
94    let user = event.user.value();
95
96    Some(DynamicSamplingContext {
97        trace_id,
98        public_key,
99        project_id,
100        release: event.release.as_str().map(str::to_owned),
101        environment: event.environment.value().cloned(),
102        transaction: event.transaction.value().cloned(),
103        replay_id: None,
104        sample_rate: None,
105        user: TraceUserContext {
106            user_segment: user
107                .and_then(|u| u.segment.value().cloned())
108                .unwrap_or_default(),
109            user_id: user
110                .and_then(|u| u.id.as_str())
111                .unwrap_or_default()
112                .to_owned(),
113        },
114        sampled: None,
115        other: Default::default(),
116    })
117}
