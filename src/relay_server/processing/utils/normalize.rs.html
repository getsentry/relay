10/// Utility to create an [`eap::time::Config`].
11pub fn time_config<F>(headers: &EnvelopeHeaders, f: F, ctx: Context<'_>) -> eap::time::Config
12where
13    F: FnOnce(&RetentionsConfig) -> Option<&RetentionConfig>,
14{
15    eap::time::Config {
16        received_at: headers.meta().received_at(),
17        sent_at: headers.sent_at(),
18        max_in_past: Some(retention_days_to_duration(
19            ctx.to_forward().retention(f).standard,
20        )),
21        max_in_future: ctx
22            .config
23            .max_secs_in_future()
24            .try_into()
25            .ok()
26            .map(Duration::from_secs),
27        minimum_clock_drift: MINIMUM_CLOCK_DRIFT,
28    }
29}
30
31fn retention_days_to_duration(days: u16) -> Duration {
32    const DAYS_TO_SECONDS: u64 = 24 * 60 * 60;
33    Duration::from_secs(days as u64 * DAYS_TO_SECONDS)
34}
