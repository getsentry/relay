1/// Data categories a payload counts towards.
2///
3/// Payloads start out as [`TotalAndIndexed`] and transition to [`Indexed`] once metrics
4/// have been extracted, transferring ownership of the total category to the metrics.
5pub trait TotalCategory {
6    /// Whether the payload still counts towards the total data category.
7    const HAS_TOTAL: bool;
8}
9
10/// The total and indexed category.
11///
12/// This category tracks payloads in the total and indexed data categories.
13/// Until a payload has metrics extracted it owns both categories.
14#[derive(Copy, Clone, Debug)]
15pub struct TotalAndIndexed;
16
17impl TotalCategory for TotalAndIndexed {
18    const HAS_TOTAL: bool = true;
19}
20
21/// The indexed category.
22///
23/// Once metric extraction happened, payloads no longer track/represent the total category,
24/// this was transferred over to the metrics.
25///
26/// Every stored payload, must have metrics extracted and transferred this ownership.
27#[derive(Copy, Clone, Debug)]
28pub struct Indexed;
29
30impl TotalCategory for Indexed {
31    const HAS_TOTAL: bool = false;
32}
