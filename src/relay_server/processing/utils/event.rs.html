41/// Returns the data category if there is an event.
42///
43/// The data category is computed from the event type. Both `Default` and `Error` events map to
44/// the `Error` data category. If there is no Event, `None` is returned.
45pub fn event_category(event: &Annotated<Event>) -> Option<DataCategory> {
46    event_type(event).map(DataCategory::from)
47}
48
49/// Returns the event type if there is an event.
50///
51/// If the event does not have a type, `Some(EventType::Default)` is assumed. If, in contrast, there
52/// is no event, `None` is returned.
53pub fn event_type(event: &Annotated<Event>) -> Option<EventType> {
54    event
55        .value()
56        .map(|event| event.ty.value().copied().unwrap_or_default())
57}

59pub fn finalize<'a>(
60    headers: &EnvelopeHeaders,
61    event: &mut Annotated<Event>,
62    attachments: impl Iterator<Item = &'a Item>,
63    metrics: &mut Metrics,
64    config: &Config,
65) -> Result<(), ProcessingError> {
66    let inner_event = match event.value_mut() {
67        Some(event) => event,
68        None if !config.processing_enabled() => return Ok(()),
69        None => return Err(ProcessingError::NoEventPayload),
70    };
71
72    if !config.processing_enabled() {
73        static MY_VERSION_STRING: OnceLock<String> = OnceLock::new();
74        let my_version = MY_VERSION_STRING.get_or_init(|| RelayVersion::current().to_string());
75
76        inner_event
77            .ingest_path
78            .get_or_insert_with(Default::default)
79            .push(Annotated::new(RelayInfo {
80                version: Annotated::new(my_version.clone()),
81                public_key: config
82                    .public_key()
83                    .map_or(Annotated::empty(), |pk| Annotated::new(pk.to_string())),
84                other: Default::default(),
85            }));
86    }
87
88    // Event id is set statically in the ingest path.
89    let event_id = headers.event_id().unwrap_or_default();
90    debug_assert!(!event_id.is_nil());
91
92    // Ensure that the event id in the payload is consistent with the envelope. If an event
93    // id was ingested, this will already be the case. Otherwise, this will insert a new
94    // event id. To be defensive, we always overwrite to ensure consistency.
95    inner_event.id = Annotated::new(event_id);
96
97    // In processing mode, also write metrics into the event. Most metrics have already been
98    // collected at this state, except for the combined size of all attachments.
99    if config.processing_enabled() {
100        let mut metrics = std::mem::take(metrics);
101
102        let attachment_size = attachments.map(|item| item.len() as u64).sum::<u64>();
103
104        if attachment_size > 0 {
105            metrics.bytes_ingested_event_attachment = Annotated::new(attachment_size);
106        }
107
108        inner_event._metrics = Annotated::new(metrics);
109
110        if inner_event.ty.value() == Some(&EventType::Transaction) {
111            let platform = utils::platform_tag(inner_event);
112            let client_name = utils::client_name_tag(headers.meta().client_name());
113
114            metric!(
115                counter(RelayCounters::EventTransaction) += 1,
116                source = utils::transaction_source_tag(inner_event),
117                platform = platform,
118                contains_slashes = if inner_event
119                    .transaction
120                    .as_str()
121                    .unwrap_or_default()
122                    .contains('/')
123                {
124                    "true"
125                } else {
126                    "false"
127                }
128            );
129
130            let span_count = inner_event.spans.value().map(Vec::len).unwrap_or(0) as u64;
131            metric!(
132                histogram(RelayHistograms::EventSpans) = span_count,
133                sdk = client_name,
134                platform = platform,
135            );
136
137            let has_otel = inner_event
138                .contexts
139                .value()
140                .is_some_and(|contexts| contexts.contains::<OtelContext>());
141
142            if has_otel {
143                metric!(
144                    counter(RelayCounters::OpenTelemetryEvent) += 1,
145                    sdk = client_name,
146                    platform = platform,
147                );
148            }
149        }
150
151        if let Some(dsc) = headers.dsc()
152            && let Ok(Some(value)) = relay_protocol::to_value(dsc)
153        {
154            inner_event._dsc = Annotated::new(value);
155        }
156    }
157
158    let mut processor = ClockDriftProcessor::new(headers.sent_at(), headers.meta().received_at())
159        .at_least(MINIMUM_CLOCK_DRIFT);
160    processor::process_value(event, &mut processor, ProcessingState::root())
161        .map_err(|_| ProcessingError::InvalidTransaction)?;
162
163    // Log timestamp delays for all events after clock drift correction. This happens before
164    // store processing, which could modify the timestamp if it exceeds a threshold. We are
165    // interested in the actual delay before this correction.
166    if let Some(timestamp) = event.value().and_then(|e| e.timestamp.value()) {
167        let event_delay = headers.meta().received_at() - timestamp.into_inner();
168        if event_delay > SignedDuration::minutes(1) {
169            let category = event_category(event).unwrap_or(DataCategory::Unknown);
170            metric!(
171                timer(RelayTimers::TimestampDelay) = event_delay.to_std().unwrap(),
172                category = category.name(),
173            );
174        }
175    }
176
177    Ok(())
178}

180/// Performs event normalization and surrounding bookkeeping.
181pub fn normalize(
182    headers: &EnvelopeHeaders,
183    event: &mut Annotated<Event>,
184    mut event_fully_normalized: EventFullyNormalized,
185    ctx: &Context,
186    geoip_lookup: &GeoIpLookup,
187) -> Result<EventFullyNormalized, ProcessingError> {
188    if event.value().is_empty() {
189        // NOTE(iker): only processing relays create events from
190        // attachments, so these events won't be normalized in
191        // non-processing relays even if the config is set to run full
192        // normalization.
193        return Ok(event_fully_normalized);
194    }
195
196    let full_normalization = match ctx.config.normalization_level() {
197        NormalizationLevel::Full => true,
198        NormalizationLevel::Default => {
199            if ctx.config.processing_enabled() && event_fully_normalized.0 {
200                return Ok(event_fully_normalized);
201            }
202
203            ctx.config.processing_enabled()
204        }
205    };
206
207    let request_meta = headers.meta();
208    let client_ipaddr = request_meta.client_addr().map(IpAddr::from);
209
210    let transaction_aggregator_config = ctx
211        .config
212        .aggregator_config_for(MetricNamespace::Transactions);
213
214    let ai_model_costs = ctx.global_config.ai_model_costs.as_ref().ok();
215    let ai_operation_type_map = ctx.global_config.ai_operation_type_map.as_ref().ok();
216    let http_span_allowed_hosts = ctx.global_config.options.http_span_allowed_hosts.as_slice();
217
218    let project_info = ctx.project_info;
219    let retention_days: i64 = project_info
220        .config
221        .event_retention
222        .unwrap_or(DEFAULT_EVENT_RETENTION)
223        .into();
224
225    utils::log_transaction_name_metrics(event, |event| {
226        let event_validation_config = EventValidationConfig {
227            received_at: Some(headers.meta().received_at()),
228            max_secs_in_past: Some(retention_days * 24 * 3600),
229            max_secs_in_future: Some(ctx.config.max_secs_in_future()),
230            transaction_timestamp_range: Some(transaction_aggregator_config.timestamp_range()),
231            is_validated: false,
232        };
233
234        let key_id = project_info
235            .get_public_key_config()
236            .and_then(|key| Some(key.numeric_id?.to_string()));
237        if full_normalization && key_id.is_none() {
238            relay_log::error!(
239                "project state for key {} is missing key id",
240                headers.meta().public_key()
241            );
242        }
243
244        // TODO: duplicated from `EnvelopeProcessorService::process`. We should pass project_id
245        // in the Context instead since it's already guaranteed to exist at this point.
246        let project_id = ctx
247            .project_info
248            .project_id
249            .or_else(|| headers.meta().project_id())
250            .ok_or(ProcessingError::MissingProjectId)?;
251        let normalization_config = NormalizationConfig {
252            project_id: Some(project_id.value()),
253            client: request_meta.client().map(str::to_owned),
254            key_id,
255            protocol_version: Some(request_meta.version().to_string()),
256            grouping_config: project_info.config.grouping_config.clone(),
257            client_ip: client_ipaddr.as_ref(),
258            // if the setting is enabled we do not want to infer the ip address
259            infer_ip_address: !project_info
260                .config
261                .datascrubbing_settings
262                .scrub_ip_addresses,
263            client_sample_rate: headers.dsc().and_then(|ctx| ctx.sample_rate),
264            user_agent: RawUserAgentInfo {
265                user_agent: request_meta.user_agent(),
266                client_hints: request_meta.client_hints(),
267            },
268            max_name_and_unit_len: Some(
269                transaction_aggregator_config
270                    .max_name_length
271                    .saturating_sub(MeasurementsConfig::MEASUREMENT_MRI_OVERHEAD),
272            ),
273            breakdowns_config: project_info.config.breakdowns_v2.as_ref(),
274            performance_score: project_info.config.performance_score.as_ref(),
275            normalize_user_agent: Some(true),
276            transaction_name_config: TransactionNameConfig {
277                rules: &project_info.config.tx_name_rules,
278            },
279            device_class_synthesis_config: project_info.has_feature(Feature::DeviceClassSynthesis),
280            enrich_spans: true,
281            max_tag_value_length: ctx
282                .config
283                .aggregator_config_for(MetricNamespace::Spans)
284                .max_tag_value_length,
285            is_renormalize: false,
286            remove_other: full_normalization,
287            emit_event_errors: full_normalization,
288            span_description_rules: project_info.config.span_description_rules.as_ref(),
289            geoip_lookup: Some(geoip_lookup),
290            ai_model_costs,
291            ai_operation_type_map,
292            enable_trimming: true,
293            measurements: Some(CombinedMeasurementsConfig::new(
294                ctx.project_info.config().measurements.as_ref(),
295                ctx.global_config.measurements.as_ref(),
296            )),
297            normalize_spans: true,
298            replay_id: headers.dsc().and_then(|ctx| ctx.replay_id),
299            span_allowed_hosts: http_span_allowed_hosts,
300            span_op_defaults: ctx.global_config.span_op_defaults.borrow(),
301            performance_issues_spans: ctx
302                .project_info
303                .has_feature(Feature::PerformanceIssuesSpans),
304        };
305
306        metric!(timer(RelayTimers::EventProcessingNormalization), {
307            validate_event(event, &event_validation_config)
308                .map_err(|_| ProcessingError::InvalidTransaction)?;
309            normalize_event_inner(event, &normalization_config);
310            if full_normalization && has_unprintable_fields(event) {
311                metric!(counter(RelayCounters::EventCorrupted) += 1);
312            }
313            Result::<(), ProcessingError>::Ok(())
314        })
315    })?;
316
317    event_fully_normalized.0 |= full_normalization;
318
319    Ok(event_fully_normalized)
320}

322/// Status for applying some filters that don't drop the event.
323///
324/// The enum represents either the success of running all filters and keeping
325/// the event, [`FiltersStatus::Ok`], or not running all the filters because
326/// some are unsupported, [`FiltersStatus::Unsupported`].
327///
328/// If there are unsuppported filters, Relay should forward the event upstream
329/// so that a more up-to-date Relay can apply filters appropriately. Actions
330/// that depend on the outcome of event filtering, such as metric extraction,
331/// should be postponed until a filtering decision is made.
332#[must_use]
333pub enum FiltersStatus {
334    /// All filters have been applied and the event should be kept.
335    Ok,
336    /// Some filters are not supported and were not applied.
337    ///
338    /// Relay should forward events upstream for a more up-to-date Relay to apply these filters.
339    /// Supported filters were applied and they don't reject the event.
340    Unsupported,
341}
342
343pub fn filter(
344    headers: &EnvelopeHeaders,
345    event: &mut Annotated<Event>,
346    ctx: &Context,
347) -> Result<FiltersStatus, FilterStatKey> {
348    let event = match event.value_mut() {
349        Some(event) => event,
350        // Some events are created by processing relays (e.g. unreal), so they do not yet
351        // exist at this point in non-processing relays.
352        None => return Ok(FiltersStatus::Ok),
353    };
354
355    let client_ip = headers.meta().client_addr();
356    let filter_settings = &ctx.project_info.config.filter_settings;
357
358    metric!(timer(RelayTimers::EventProcessingFiltering), {
359        relay_filter::should_filter(
360            event,
361            client_ip,
362            filter_settings,
363            ctx.global_config.filters(),
364        )
365    })?;
366
367    // Don't extract metrics if relay can't apply generic filters.  A filter
368    // applied in another up-to-date relay in chain may need to drop the event,
369    // and there should not be metrics from dropped events.
370    // In processing relays, always extract metrics to avoid losing them.
371    let supported_generic_filters = ctx.global_config.filters.is_ok()
372        && relay_filter::are_generic_filters_supported(
373            ctx.global_config.filters().map(|f| f.version),
374            ctx.project_info.config.filter_settings.generic.version,
375        );
376    if supported_generic_filters {
377        Ok(FiltersStatus::Ok)
378    } else {
379        Ok(FiltersStatus::Unsupported)
380    }
381}

383/// New type representing the normalization state of the event.
384#[derive(Copy, Clone)]
385pub struct EventFullyNormalized(pub bool);
386
387impl EventFullyNormalized {
388    /// Returns `true` if the event is fully normalized, `false` otherwise.
389    pub fn new(envelope: &Envelope) -> Self {
390        let event_fully_normalized = envelope.meta().request_trust().is_trusted()
391            && envelope
392                .items()
393                .any(|item| item.creates_event() && item.fully_normalized());
394
395        Self(event_fully_normalized)
396    }
397}

407/// Creates a span from the transaction and applies tag extraction on it.
408///
409/// Returns `None` when [`tag_extraction::extract_span_tags`] clears the span, which it shouldn't.
410pub fn extract_transaction_span(
411    event: &Event,
412    max_tag_value_size: usize,
413    span_allowed_hosts: &[String],
414) -> Option<Span> {
415    let mut spans = [Span::from(event).into()];
416
417    tag_extraction::extract_span_tags(event, &mut spans, max_tag_value_size, span_allowed_hosts);
418    tag_extraction::extract_segment_span_tags(event, &mut spans);
419
420    spans.into_iter().next().and_then(Annotated::into_value)
421}

423/// Checks if the Event includes unprintable fields.
424fn has_unprintable_fields(event: &Annotated<Event>) -> bool {
425    fn is_unprintable(value: &&str) -> bool {
426        value.chars().any(|c| {
427            c == '\u{fffd}' // unicode replacement character
428                || (c.is_control() && !c.is_whitespace()) // non-whitespace control characters
429        })
430    }
431    if let Some(event) = event.value() {
432        let env = event.environment.as_str().filter(is_unprintable);
433        let release = event.release.as_str().filter(is_unprintable);
434        env.is_some() || release.is_some()
435    } else {
436        false
437    }
438}
