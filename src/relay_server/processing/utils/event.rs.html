39/// Returns the data category if there is an event.
40///
41/// The data category is computed from the event type. Both `Default` and `Error` events map to
42/// the `Error` data category. If there is no Event, `None` is returned.
43pub fn event_category(event: &Annotated<Event>) -> Option<DataCategory> {
44    event_type(event).map(DataCategory::from)
45}
46
47/// Returns the event type if there is an event.
48///
49/// If the event does not have a type, `Some(EventType::Default)` is assumed. If, in contrast, there
50/// is no event, `None` is returned.
51pub fn event_type(event: &Annotated<Event>) -> Option<EventType> {
52    event
53        .value()
54        .map(|event| event.ty.value().copied().unwrap_or_default())
55}

57pub fn finalize<'a>(
58    headers: &EnvelopeHeaders,
59    event: &mut Annotated<Event>,
60    attachments: impl Iterator<Item = &'a Item>,
61    metrics: &mut Metrics,
62    config: &Config,
63) -> Result<(), ProcessingError> {
64    let inner_event = match event.value_mut() {
65        Some(event) => event,
66        None if !config.processing_enabled() => return Ok(()),
67        None => return Err(ProcessingError::NoEventPayload),
68    };
69
70    if !config.processing_enabled() {
71        static MY_VERSION_STRING: OnceLock<String> = OnceLock::new();
72        let my_version = MY_VERSION_STRING.get_or_init(|| RelayVersion::current().to_string());
73
74        inner_event
75            .ingest_path
76            .get_or_insert_with(Default::default)
77            .push(Annotated::new(RelayInfo {
78                version: Annotated::new(my_version.clone()),
79                public_key: config
80                    .public_key()
81                    .map_or(Annotated::empty(), |pk| Annotated::new(pk.to_string())),
82                other: Default::default(),
83            }));
84    }
85
86    // Event id is set statically in the ingest path.
87    let event_id = headers.event_id().unwrap_or_default();
88    debug_assert!(!event_id.is_nil());
89
90    // Ensure that the event id in the payload is consistent with the envelope. If an event
91    // id was ingested, this will already be the case. Otherwise, this will insert a new
92    // event id. To be defensive, we always overwrite to ensure consistency.
93    inner_event.id = Annotated::new(event_id);
94
95    // In processing mode, also write metrics into the event. Most metrics have already been
96    // collected at this state, except for the combined size of all attachments.
97    if config.processing_enabled() {
98        let mut metrics = std::mem::take(metrics);
99
100        let attachment_size = attachments.map(|item| item.len() as u64).sum::<u64>();
101
102        if attachment_size > 0 {
103            metrics.bytes_ingested_event_attachment = Annotated::new(attachment_size);
104        }
105
106        inner_event._metrics = Annotated::new(metrics);
107
108        if inner_event.ty.value() == Some(&EventType::Transaction) {
109            let platform = utils::platform_tag(inner_event);
110            let client_name = utils::client_name_tag(headers.meta().client_name());
111
112            metric!(
113                counter(RelayCounters::EventTransaction) += 1,
114                source = utils::transaction_source_tag(inner_event),
115                platform = platform,
116                contains_slashes = if inner_event
117                    .transaction
118                    .as_str()
119                    .unwrap_or_default()
120                    .contains('/')
121                {
122                    "true"
123                } else {
124                    "false"
125                }
126            );
127
128            let span_count = inner_event.spans.value().map(Vec::len).unwrap_or(0) as u64;
129            metric!(
130                histogram(RelayHistograms::EventSpans) = span_count,
131                sdk = client_name,
132                platform = platform,
133            );
134
135            let has_otel = inner_event
136                .contexts
137                .value()
138                .is_some_and(|contexts| contexts.contains::<OtelContext>());
139
140            if has_otel {
141                metric!(
142                    counter(RelayCounters::OpenTelemetryEvent) += 1,
143                    sdk = client_name,
144                    platform = platform,
145                );
146            }
147        }
148
149        if let Some(dsc) = headers.dsc()
150            && let Ok(Some(value)) = relay_protocol::to_value(dsc)
151        {
152            inner_event._dsc = Annotated::new(value);
153        }
154    }
155
156    let mut processor = ClockDriftProcessor::new(headers.sent_at(), headers.meta().received_at())
157        .at_least(MINIMUM_CLOCK_DRIFT);
158    processor::process_value(event, &mut processor, ProcessingState::root()).map_err(|err| {
159        relay_log::debug!(
160            error = &err as &dyn std::error::Error,
161            "invalid transaction"
162        );
163        ProcessingError::InvalidTransaction
164    })?;
165
166    // Log timestamp delays for all events after clock drift correction. This happens before
167    // store processing, which could modify the timestamp if it exceeds a threshold. We are
168    // interested in the actual delay before this correction.
169    if let Some(timestamp) = event.value().and_then(|e| e.timestamp.value()) {
170        let event_delay = headers.meta().received_at() - timestamp.into_inner();
171        if event_delay > SignedDuration::minutes(1) {
172            let category = event_category(event).unwrap_or(DataCategory::Unknown);
173            metric!(
174                timer(RelayTimers::TimestampDelay) = event_delay.to_std().unwrap(),
175                category = category.name(),
176            );
177        }
178    }
179
180    Ok(())
181}

183/// Performs event normalization and surrounding bookkeeping.
184pub fn normalize(
185    headers: &EnvelopeHeaders,
186    event: &mut Annotated<Event>,
187    mut event_fully_normalized: EventFullyNormalized,
188    ctx: &Context,
189    geoip_lookup: &GeoIpLookup,
190) -> Result<EventFullyNormalized, ProcessingError> {
191    if event.value().is_empty() {
192        // NOTE(iker): only processing relays create events from
193        // attachments, so these events won't be normalized in
194        // non-processing relays even if the config is set to run full
195        // normalization.
196        return Ok(event_fully_normalized);
197    }
198
199    let full_normalization = match ctx.config.normalization_level() {
200        NormalizationLevel::Full => true,
201        NormalizationLevel::Default => {
202            if ctx.config.processing_enabled() && event_fully_normalized.0 {
203                return Ok(event_fully_normalized);
204            }
205
206            ctx.config.processing_enabled()
207        }
208    };
209
210    let request_meta = headers.meta();
211    let client_ipaddr = request_meta.client_addr().map(IpAddr::from);
212
213    let transaction_aggregator_config = ctx
214        .config
215        .aggregator_config_for(MetricNamespace::Transactions);
216
217    let ai_model_costs = ctx.global_config.ai_model_costs.as_ref().ok();
218    let ai_operation_type_map = ctx.global_config.ai_operation_type_map.as_ref().ok();
219    let http_span_allowed_hosts = ctx.global_config.options.http_span_allowed_hosts.as_slice();
220
221    let project_info = ctx.project_info;
222    let retention_days: i64 = project_info
223        .config
224        .event_retention
225        .unwrap_or(DEFAULT_EVENT_RETENTION)
226        .into();
227
228    utils::log_transaction_name_metrics(event, |event| {
229        let event_validation_config = EventValidationConfig {
230            received_at: Some(headers.meta().received_at()),
231            max_secs_in_past: Some(retention_days * 24 * 3600),
232            max_secs_in_future: Some(ctx.config.max_secs_in_future()),
233            transaction_timestamp_range: Some(transaction_aggregator_config.timestamp_range()),
234            is_validated: false,
235        };
236
237        let key_id = project_info
238            .get_public_key_config()
239            .and_then(|key| Some(key.numeric_id?.to_string()));
240        if full_normalization && key_id.is_none() {
241            relay_log::error!(
242                "project state for key {} is missing key id",
243                headers.meta().public_key()
244            );
245        }
246
247        // TODO: duplicated from `EnvelopeProcessorService::process`. We should pass project_id
248        // in the Context instead since it's already guaranteed to exist at this point.
249        let project_id = ctx
250            .project_info
251            .project_id
252            .or_else(|| headers.meta().project_id())
253            .ok_or(ProcessingError::MissingProjectId)?;
254        let normalization_config = NormalizationConfig {
255            project_id: Some(project_id.value()),
256            client: request_meta.client().map(str::to_owned),
257            key_id,
258            protocol_version: Some(request_meta.version().to_string()),
259            grouping_config: project_info.config.grouping_config.clone(),
260            client_ip: client_ipaddr.as_ref(),
261            // if the setting is enabled we do not want to infer the ip address
262            infer_ip_address: !project_info
263                .config
264                .datascrubbing_settings
265                .scrub_ip_addresses,
266            client_sample_rate: headers.dsc().and_then(|ctx| ctx.sample_rate),
267            user_agent: RawUserAgentInfo {
268                user_agent: request_meta.user_agent(),
269                client_hints: request_meta.client_hints(),
270            },
271            max_name_and_unit_len: Some(
272                transaction_aggregator_config
273                    .max_name_length
274                    .saturating_sub(MeasurementsConfig::MEASUREMENT_MRI_OVERHEAD),
275            ),
276            breakdowns_config: project_info.config.breakdowns_v2.as_ref(),
277            performance_score: project_info.config.performance_score.as_ref(),
278            normalize_user_agent: Some(true),
279            transaction_name_config: TransactionNameConfig {
280                rules: &project_info.config.tx_name_rules,
281            },
282            device_class_synthesis_config: project_info.has_feature(Feature::DeviceClassSynthesis),
283            enrich_spans: true,
284            max_tag_value_length: ctx
285                .config
286                .aggregator_config_for(MetricNamespace::Spans)
287                .max_tag_value_length,
288            is_renormalize: false,
289            remove_other: full_normalization,
290            emit_event_errors: full_normalization,
291            span_description_rules: project_info.config.span_description_rules.as_ref(),
292            geoip_lookup: Some(geoip_lookup),
293            ai_model_costs,
294            ai_operation_type_map,
295            enable_trimming: true,
296            measurements: Some(CombinedMeasurementsConfig::new(
297                ctx.project_info.config().measurements.as_ref(),
298                ctx.global_config.measurements.as_ref(),
299            )),
300            normalize_spans: true,
301            replay_id: headers.dsc().and_then(|ctx| ctx.replay_id),
302            span_allowed_hosts: http_span_allowed_hosts,
303            span_op_defaults: ctx.global_config.span_op_defaults.borrow(),
304            performance_issues_spans: ctx
305                .project_info
306                .has_feature(Feature::PerformanceIssuesSpans),
307        };
308
309        metric!(timer(RelayTimers::EventProcessingNormalization), {
310            validate_event(event, &event_validation_config).map_err(|err| {
311                relay_log::debug!(
312                    error = &err as &dyn std::error::Error,
313                    "invalid transaction"
314                );
315                ProcessingError::InvalidTransaction
316            })?;
317            normalize_event_inner(event, &normalization_config);
318            if full_normalization && has_unprintable_fields(event) {
319                metric!(counter(RelayCounters::EventCorrupted) += 1);
320            }
321            Result::<(), ProcessingError>::Ok(())
322        })
323    })?;
324
325    event_fully_normalized.0 |= full_normalization;
326
327    Ok(event_fully_normalized)
328}

330/// Status for applying some filters that don't drop the event.
331///
332/// The enum represents either the success of running all filters and keeping
333/// the event, [`FiltersStatus::Ok`], or not running all the filters because
334/// some are unsupported, [`FiltersStatus::Unsupported`].
335///
336/// If there are unsuppported filters, Relay should forward the event upstream
337/// so that a more up-to-date Relay can apply filters appropriately. Actions
338/// that depend on the outcome of event filtering, such as metric extraction,
339/// should be postponed until a filtering decision is made.
340#[must_use]
341pub enum FiltersStatus {
342    /// All filters have been applied and the event should be kept.
343    Ok,
344    /// Some filters are not supported and were not applied.
345    ///
346    /// Relay should forward events upstream for a more up-to-date Relay to apply these filters.
347    /// Supported filters were applied and they don't reject the event.
348    Unsupported,
349}
350
351pub fn filter(
352    headers: &EnvelopeHeaders,
353    event: &mut Annotated<Event>,
354    ctx: &Context,
355) -> Result<FiltersStatus, FilterStatKey> {
356    let event = match event.value_mut() {
357        Some(event) => event,
358        // Some events are created by processing relays (e.g. unreal), so they do not yet
359        // exist at this point in non-processing relays.
360        None => return Ok(FiltersStatus::Ok),
361    };
362
363    let client_ip = headers.meta().client_addr();
364    let filter_settings = &ctx.project_info.config.filter_settings;
365
366    metric!(timer(RelayTimers::EventProcessingFiltering), {
367        relay_filter::should_filter(
368            event,
369            client_ip,
370            filter_settings,
371            ctx.global_config.filters(),
372        )
373    })?;
374
375    // Don't extract metrics if relay can't apply generic filters.  A filter
376    // applied in another up-to-date relay in chain may need to drop the event,
377    // and there should not be metrics from dropped events.
378    // In processing relays, always extract metrics to avoid losing them.
379    let supported_generic_filters = ctx.global_config.filters.is_ok()
380        && relay_filter::are_generic_filters_supported(
381            ctx.global_config.filters().map(|f| f.version),
382            ctx.project_info.config.filter_settings.generic.version,
383        );
384    if supported_generic_filters {
385        Ok(FiltersStatus::Ok)
386    } else {
387        Ok(FiltersStatus::Unsupported)
388    }
389}

391/// New type representing the normalization state of the event.
392#[derive(Copy, Clone)]
393pub struct EventFullyNormalized(pub bool);
394
395impl EventFullyNormalized {
396    /// Returns `true` if the event is fully normalized, `false` otherwise.
397    pub fn new(envelope: &Envelope) -> Self {
398        let event_fully_normalized = envelope.meta().request_trust().is_trusted()
399            && envelope
400                .items()
401                .any(|item| item.creates_event() && item.fully_normalized());
402
403        Self(event_fully_normalized)
404    }
405}

415/// Checks if the Event includes unprintable fields.
416fn has_unprintable_fields(event: &Annotated<Event>) -> bool {
417    fn is_unprintable(value: &&str) -> bool {
418        value.chars().any(|c| {
419            c == '\u{fffd}' // unicode replacement character
420                || (c.is_control() && !c.is_whitespace()) // non-whitespace control characters
421        })
422    }
423    if let Some(event) = event.value() {
424        let env = event.environment.as_str().filter(is_unprintable);
425        let release = event.release.as_str().filter(is_unprintable);
426        env.is_some() || release.is_some()
427    } else {
428        false
429    }
430}
