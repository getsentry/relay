10pub fn process(reports: &mut Managed<SerializedUserReports>) {
11    reports.modify(|reports, records| {
12        process_user_reports(&mut reports.reports, records);
13    })
14}

16/// Validates and normalizes all user report items in the envelope.
17///
18/// User feedback items are removed from the envelope if they contain invalid JSON or if the
19/// JSON violates the schema (basic type validation). Otherwise, their normalized representation
20/// is written back into the item.
21pub fn process_user_reports(user_reports: &mut Vec<Item>, records: &mut RecordKeeper<'_>) {
22    for mut user_report in std::mem::take(user_reports) {
23        let data = match process_user_report(user_report.payload()) {
24            Ok(data) => data,
25            Err(err) => {
26                records.reject_err(err, user_report);
27                continue;
28            }
29        };
30        user_report.set_payload(ContentType::Json, data);
31        user_reports.push(user_report);
32    }
33}
34
35fn process_user_report(user_report: Bytes) -> Result<Bytes, ProcessingError> {
36    // There is a customer SDK which sends invalid reports with a trailing `\n`,
37    // strip it here, even if they update/fix their SDK there will still be many old
38    // versions with the broken SDK out there.
39    let user_report = trim_whitespaces(&user_report);
40
41    let report =
42        serde_json::from_slice::<UserReport>(user_report).map_err(ProcessingError::InvalidJson)?;
43
44    serde_json::to_string(&report)
45        .map(Bytes::from)
46        .map_err(ProcessingError::SerializeFailed)
47}
48
49fn trim_whitespaces(data: &[u8]) -> &[u8] {
50    let Some(from) = data.iter().position(|x| !x.is_ascii_whitespace()) else {
51        return &[];
52    };
53    let Some(to) = data.iter().rposition(|x| !x.is_ascii_whitespace()) else {
54        return &[];
55    };
56    &data[from..to + 1]
57}
