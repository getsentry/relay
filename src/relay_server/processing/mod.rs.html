29/// A processor, for an arbitrary unit of work extracted from an envelope.
30///
31/// The processor takes items from an envelope, to process it.
32/// To fully process an envelope, multiple processor may need to take items from the same envelope.
33///
34/// Event based envelopes should only be handled by a single processor, as the protocol
35/// defines all items in an event based envelope to relate to the envelope.
36pub trait Processor {
37    /// A unit of work, the processor can process.
38    type UnitOfWork: Counted;
39    /// The result after processing a [`Self::UnitOfWork`].
40    type Output: Forward;
41    /// The error returned by [`Self::process`].
42    type Error: std::error::Error;
43
44    /// Extracts a [`Self::UnitOfWork`] from a [`ManagedEnvelope`].
45    ///
46    /// This is infallible, if a processor wants to report an error,
47    /// it should return a [`Self::UnitOfWork`] which later, can produce an error when being
48    fn prepare_envelope(&self, envelope: &mut ManagedEnvelope)
49    -> Option<Managed<Self::UnitOfWork>>;
50
51    /// Processes a [`Self::UnitOfWork`].
52    async fn process(
53        &self,
54        work: Managed<Self::UnitOfWork>,
55        ctx: Context<'_>,
56    ) -> Result<Output<Self::Output>, Rejected<Self::Error>>;
57}

59/// Read-only context for processing.
60#[derive(Copy, Clone)]
61pub struct Context<'a> {
62    /// The Relay configuration.
63    pub config: &'a Config,
64    /// A view of the currently active global configuration.
65    pub global_config: &'a GlobalConfig,
66    /// Project configuration associated with the unit of work.
67    pub project_info: &'a ProjectInfo,
68    /// Cached rate limits associated with the unit of work.
69    ///
70    /// The caller needs to ensure the rate limits are not yet expired.
71    pub rate_limits: &'a RateLimits,
72}
73
74impl Context<'_> {
75    /// Returns `true` if Relay is running in processing mode.
76    pub fn is_processing(&self) -> bool {
77        self.config.processing_enabled()
78    }
79
80    /// Checks on-off feature flags for envelope items, like profiles and spans.
81    ///
82    /// It checks for the presence of the passed feature flag, but does not filter items
83    /// when there is no full project config available. This is the case in stat and proxy
84    /// Relays.
85    pub fn should_filter(&self, feature: relay_dynamic_config::Feature) -> bool {
86        use relay_config::RelayMode::*;
87
88        match self.config.relay_mode() {
89            Proxy | Static | Capture => false,
90            Managed => !self.project_info.has_feature(feature),
91        }
92    }
93}

95/// The main processing output and all of its by products.
96pub struct Output<T> {
97    /// The main output of a [`Processor`].
98    pub main: T,
99    /// Metric by products.
100    pub metrics: ProcessingExtractedMetrics,
101}
102
103impl<T> Output<T> {
104    /// Creates a new output with just a main output.
105    pub fn just(main: T) -> Self {
106        Self {
107            main,
108            metrics: ProcessingExtractedMetrics::new(),
109        }
110    }
111}
