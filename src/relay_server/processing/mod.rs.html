26/// A processor, for an arbitrary unit of work extracted from an envelope.
27///
28/// The processor takes items from an envelope, to process it.
29/// To fully process an envelope, multiple processor may need to take items from the same envelope.
30///
31/// Event based envelopes should only be handled by a single processor, as the protocol
32/// defines all items in an event based envelope to relate to the envelope.
33pub trait Processor {
34    /// A unit of work, the processor can process.
35    type UnitOfWork: Counted;
36    /// The result after processing a [`Self::UnitOfWork`].
37    type Output: Forward;
38    /// The error returned by [`Self::process`].
39    type Error: std::error::Error;
40
41    /// Extracts a [`Self::UnitOfWork`] from a [`ManagedEnvelope`].
42    ///
43    /// This is infallible, if a processor wants to report an error,
44    /// it should return a [`Self::UnitOfWork`] which later, can produce an error when being
45    fn prepare_envelope(&self, envelope: &mut ManagedEnvelope)
46    -> Option<Managed<Self::UnitOfWork>>;
47
48    /// Processes a [`Self::UnitOfWork`].
49    async fn process(
50        &self,
51        work: Managed<Self::UnitOfWork>,
52        ctx: Context<'_>,
53    ) -> Result<Output<Self::Output>, Rejected<Self::Error>>;
54}

56/// Read-only context for processing.
57#[derive(Copy, Clone)]
58pub struct Context<'a> {
59    /// The Relay configuration.
60    pub config: &'a Config,
61    /// A view of the currently active global configuration.
62    pub global_config: &'a GlobalConfig,
63    /// Project configuration associated with the unit of work.
64    pub project_info: &'a ProjectInfo,
65    /// Project configuration associated with the root of the trace of the unit of work.
66    pub sampling_project_info: Option<&'a ProjectInfo>,
67    /// Cached rate limits associated with the unit of work.
68    ///
69    /// The caller needs to ensure the rate limits are not yet expired.
70    pub rate_limits: &'a RateLimits,
71}
72
73impl Context<'_> {
74    /// Returns `true` if Relay is running in proxy mode.
75    pub fn is_proxy(&self) -> bool {
76        matches!(self.config.relay_mode(), relay_config::RelayMode::Proxy)
77    }
78
79    /// Returns `true` if Relay has processing enabled.
80    ///
81    /// Processing indicates, this Relay is the final Relay processing this item.
82    pub fn is_processing(&self) -> bool {
83        self.config.processing_enabled()
84    }
85
86    /// Checks on-off feature flags for envelope items, like profiles and spans.
87    ///
88    /// It checks for the presence of the passed feature flag, but does not filter items
89    /// when there is no full project config available. This is the case in stat and proxy
90    /// Relays.
91    pub fn should_filter(&self, feature: relay_dynamic_config::Feature) -> bool {
92        use relay_config::RelayMode::*;
93
94        match self.config.relay_mode() {
95            Proxy => false,
96            Managed => !self.project_info.has_feature(feature),
97        }
98    }
99}

101/// The main processing output and all of its by products.
102#[derive(Debug)]
103pub struct Output<T> {
104    /// The main output of a [`Processor`].
105    pub main: Option<T>,
106    /// Metric by products.
107    pub metrics: Option<Managed<ExtractedMetrics>>,
108}
109
110impl<T> Output<T> {
111    /// Creates a new output with just a main output.
112    pub fn just(main: T) -> Self {
113        Self {
114            main: Some(main),
115            metrics: None,
116        }
117    }
118
119    /// Creates a new output just containing metrics.
120    pub fn metrics(metrics: Managed<ExtractedMetrics>) -> Self {
121        Self {
122            main: None,
123            metrics: Some(metrics),
124        }
125    }
126
127    /// Maps an `Output<T>` to `Output<S>` by applying a function to [`Self::main`].
128    pub fn map<F, S>(self, f: F) -> Output<S>
129    where
130        F: FnOnce(T) -> S,
131    {
132        Output {
133            main: self.main.map(f),
134            metrics: self.metrics,
135        }
136    }
137}

139/// A processor output which can be forwarded to a different destination.
140pub trait Forward {
141    /// Serializes the output into an [`Envelope`].
142    ///
143    /// All output must be serializable as an envelope.
144    fn serialize_envelope(self) -> Result<Managed<Box<Envelope>>, Rejected<()>>;
145
146    /// Serializes the output into a [`crate::services::store::StoreService`] compatible format.
147    ///
148    /// This function must only be called when Relay is configured to be in processing mode.
149    #[cfg(feature = "processing")]
150    fn forward_store(
151        self,
152        s: &relay_system::Addr<crate::services::store::Store>,
153    ) -> Result<(), Rejected<()>>;
154}
