35/// A processor, for an arbitrary unit of work extracted from an envelope.
36///
37/// The processor takes items from an envelope, to process it.
38/// To fully process an envelope, multiple processor may need to take items from the same envelope.
39///
40/// Event based envelopes should only be handled by a single processor, as the protocol
41/// defines all items in an event based envelope to relate to the envelope.
42pub trait Processor {
43    /// A unit of work, the processor can process.
44    type UnitOfWork: Counted;
45    /// The result after processing a [`Self::UnitOfWork`].
46    type Output: Forward;
47    /// The error returned by [`Self::process`].
48    type Error: std::error::Error + 'static;
49
50    /// Extracts a [`Self::UnitOfWork`] from a [`ManagedEnvelope`].
51    ///
52    /// This is infallible, if a processor wants to report an error,
53    /// it should return a [`Self::UnitOfWork`] which later, can produce an error when being processed.
54    ///
55    /// Returns `None` if nothing in the envelope concerns this processor.
56    fn prepare_envelope(&self, envelope: &mut ManagedEnvelope)
57    -> Option<Managed<Self::UnitOfWork>>;
58
59    /// Processes a [`Self::UnitOfWork`].
60    async fn process(
61        &self,
62        work: Managed<Self::UnitOfWork>,
63        ctx: Context<'_>,
64    ) -> Result<Output<Self::Output>, Rejected<Self::Error>>;
65}

67/// Read-only context for processing.
68#[derive(Copy, Clone, Debug)]
69pub struct Context<'a> {
70    /// The Relay configuration.
71    pub config: &'a Config,
72    /// A view of the currently active global configuration.
73    pub global_config: &'a GlobalConfig,
74    /// Project configuration associated with the unit of work.
75    pub project_info: &'a ProjectInfo,
76    /// Project configuration associated with the root of the trace of the unit of work.
77    pub sampling_project_info: Option<&'a ProjectInfo>,
78    /// Cached rate limits associated with the unit of work.
79    ///
80    /// The caller needs to ensure the rate limits are not yet expired.
81    pub rate_limits: &'a RateLimits,
82    /// Reservoir counters for "get more samples" functionality.
83    pub reservoir_counters: &'a ReservoirCounters,
84}
85
86impl<'a> Context<'a> {
87    /// Returns `true` if Relay has processing enabled.
88    ///
89    /// Processing indicates, this Relay is the final Relay processing this item.
90    pub fn is_processing(&self) -> bool {
91        cfg!(feature = "processing") && self.config.processing_enabled()
92    }
93
94    /// Checks on-off feature flags for envelope items, like profiles and spans.
95    ///
96    /// It checks for the presence of the passed feature flag, but does not filter items
97    /// when there is no full project config available. This is the case in stat and proxy
98    /// Relays.
99    pub fn should_filter(&self, feature: relay_dynamic_config::Feature) -> bool {
100        use relay_config::RelayMode::*;
101
102        match self.config.relay_mode() {
103            Proxy => false,
104            Managed => !self.project_info.has_feature(feature),
105        }
106    }
107
108    /// Creates a [`ForwardContext`] from this [`Context`].
109    pub fn to_forward(self) -> ForwardContext<'a> {
110        ForwardContext {
111            config: self.config,
112            global_config: self.global_config,
113            project_info: self.project_info,
114        }
115    }
116}

141/// The main processing output and all of its by products.
142#[derive(Debug)]
143pub struct Output<T> {
144    /// The main output of a [`Processor`].
145    pub main: Option<T>,
146    /// Metric by products.
147    pub metrics: Option<Managed<ExtractedMetrics>>,
148}
149
150impl<T> Output<T> {
151    /// Creates a new output with just a main output.
152    pub fn just(main: T) -> Self {
153        Self {
154            main: Some(main),
155            metrics: None,
156        }
157    }
158
159    /// Creates a new output just containing metrics.
160    pub fn metrics(metrics: Managed<ExtractedMetrics>) -> Self {
161        Self {
162            main: None,
163            metrics: Some(metrics),
164        }
165    }
166
167    /// Maps an `Output<T>` to `Output<S>` by applying a function to [`Self::main`].
168    pub fn map<F, S>(self, f: F) -> Output<S>
169    where
170        F: FnOnce(T) -> S,
171    {
172        Output {
173            main: self.main.map(f),
174            metrics: self.metrics,
175        }
176    }
177}
