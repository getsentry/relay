8#[derive(Debug, thiserror::Error)]
9#[error("the item is not allowed/supported in this envelope")]
10pub struct InvalidItem;
11
12impl OutcomeError for InvalidItem {
13    type Error = Self;
14
15    fn consume(self) -> (Option<crate::services::outcome::Outcome>, Self::Error) {
16        (Some(Outcome::Invalid(DiscardReason::InvalidEnvelope)), self)
17    }
18}

20/// A processor which rejects unhandled invalid items.
21///
22/// This processor accepts certain envelope items which may be mixed into envelopes incorrectly.
23#[derive(Debug)]
24pub struct InvalidUnhandledProcessor {
25    _priv: (),
26}
27
28impl InvalidUnhandledProcessor {
29    /// Creates a new [`Self`].
30    pub fn new() -> Self {
31        Self { _priv: () }
32    }
33}
34
35impl processing::Processor for InvalidUnhandledProcessor {
36    type Input = InvalidUnhandledItems;
37    type Output = Nothing;
38    type Error = InvalidItem;
39
40    fn cogs() -> FeatureWeights {
41        AppFeature::UnattributedEnvelope.into()
42    }
43
44    fn prepare_envelope(&self, envelope: &mut ManagedEnvelope) -> Option<Managed<Self::Input>> {
45        let invalid_items = envelope
46            .envelope_mut()
47            // `FormData` only makes sense together with an item which creates an event,
48            // if it's mixed into an envelope which does not create an event, it is invalid.
49            .take_items_by(|item| matches!(*item.ty(), ItemType::FormData))
50            .into_vec();
51
52        if invalid_items.is_empty() {
53            return None;
54        }
55
56        Some(Managed::with_meta_from_managed_envelope(
57            envelope,
58            InvalidUnhandledItems { invalid_items },
59        ))
60    }
61
62    async fn process(
63        &self,
64        invalid: Managed<Self::Input>,
65        _ctx: Context<'_>,
66    ) -> Result<Output<Self::Output>, Rejected<Self::Error>> {
67        Err(invalid.reject_err(InvalidItem))
68    }
69}

71/// Unknown items extracted from an envelope.
72#[derive(Debug)]
73pub struct InvalidUnhandledItems {
74    /// List of unknown envelope items.
75    invalid_items: Vec<Item>,
76}
77
78impl Counted for InvalidUnhandledItems {
79    fn quantities(&self) -> Quantities {
80        self.invalid_items.quantities()
81    }
