10/// A processor output which can be forwarded to a different destination.
11pub trait Forward {
12    /// Serializes the output into an [`Envelope`].
13    ///
14    /// All output must be serializable as an envelope.
15    fn serialize_envelope(
16        self,
17        ctx: ForwardContext<'_>,
18    ) -> Result<Managed<Box<Envelope>>, Rejected<()>>;
19
20    /// Serializes the output into a [`crate::services::store::StoreService`] compatible format.
21    ///
22    /// This function must only be called when Relay is configured to be in processing mode.
23    #[cfg(feature = "processing")]
24    fn forward_store(
25        self,
26        s: &relay_system::Addr<crate::services::store::Store>,
27        ctx: ForwardContext<'_>,
28    ) -> Result<(), Rejected<()>>;
29}

31/// Context passed to [`Forward`].
32///
33/// A minified version of [`Context`](super::Context), which does not contain processing specific information.
34#[derive(Copy, Clone, Debug)]
35pub struct ForwardContext<'a> {
36    /// The Relay configuration.
37    #[expect(unused, reason = "not yet used")]
38    pub config: &'a Config,
39    /// A view of the currently active global configuration.
40    #[expect(unused, reason = "not yet used")]
41    pub global_config: &'a GlobalConfig,
42    /// Project configuration associated with the unit of work.
43    #[cfg_attr(not(feature = "processing"), expect(unused))]
44    pub project_info: &'a ProjectInfo,
45}
46
47#[cfg(feature = "processing")]
48impl ForwardContext<'_> {
49    /// Returns the [`Retention`] for a specific type/product.
50    pub fn retention<F>(&self, f: F) -> Retention
51    where
52        F: FnOnce(&RetentionsConfig) -> Option<&RetentionConfig>,
53    {
54        if let Some(retention) = f(&self.project_info.config.retentions) {
55            return Retention::from(*retention);
56        }
57
58        Retention::from(RetentionConfig {
59            standard: self
60                .project_info
61                .config
62                .event_retention
63                .unwrap_or(crate::constants::DEFAULT_EVENT_RETENTION),
64            downsampled: self.project_info.config.downsampled_event_retention,
65        })
66    }
67}

98/// Full retention settings to apply to specific payloads.
99#[derive(Debug, Copy, Clone)]
100#[cfg(feature = "processing")]
101pub struct Retention {
102    /// Standard / full fidelity retention policy in days.
103    pub standard: u16,
104    /// Downsampled retention policy in days.
105    pub downsampled: u16,
106}
107
108#[cfg(feature = "processing")]
109impl From<RetentionConfig> for Retention {
110    fn from(value: RetentionConfig) -> Self {
111        Self {
112            standard: value.standard,
113            downsampled: value.downsampled.unwrap_or(value.standard),
114        }
115    }
