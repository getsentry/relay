15/// A transparent handle that dispatches between store-like services.
16#[cfg(feature = "processing")]
17#[derive(Debug, Clone, Copy)]
18pub struct StoreHandle<'a> {
19    store: &'a Addr<Store>,
20    upload: Option<&'a Addr<Upload>>,
21}
22
23#[cfg(feature = "processing")]
24impl<'a> StoreHandle<'a> {
25    pub fn new(store: &'a Addr<Store>, upload: Option<&'a Addr<Upload>>) -> Self {
26        Self { store, upload }
27    }
28
29    /// Sends a message to the [`Store`] service.
30    pub fn store<M>(&self, message: M)
31    where
32        Store: FromMessage<M>,
33    {
34        self.store.send(message);
35    }
36
37    /// Sends a message to the [`Upload`] service.
38    pub fn upload<M>(&self, message: M)
39    where
40        Upload: FromMessage<M>,
41    {
42        if let Some(upload) = self.upload {
43            upload.send(message);
44        } else {
45            relay_log::error!("Upload service not configured. Dropping message.");
46        }
47    }
48}

50/// A processor output which can be forwarded to a different destination.
51pub trait Forward {
52    /// Serializes the output into an [`Envelope`].
53    ///
54    /// All output must be serializable as an envelope.
55    fn serialize_envelope(
56        self,
57        ctx: ForwardContext<'_>,
58    ) -> Result<Managed<Box<Envelope>>, Rejected<()>>;
59
60    /// Serializes the output into a [`crate::services::store::StoreService`] compatible format.
61    ///
62    /// This function must only be called when Relay is configured to be in processing mode.
63    #[cfg(feature = "processing")]
64    fn forward_store(self, s: StoreHandle<'_>, ctx: ForwardContext<'_>)
65    -> Result<(), Rejected<()>>;
66}

68/// Context passed to [`Forward`].
69///
70/// A minified version of [`Context`](super::Context), which does not contain processing specific information.
71#[derive(Copy, Clone, Debug)]
72pub struct ForwardContext<'a> {
73    /// The Relay configuration.
74    #[cfg_attr(
75        not(feature = "processing"),
76        expect(unused, reason = "only used in processing")
77    )]
78    pub config: &'a Config,
79    /// A view of the currently active global configuration.
80    #[expect(unused, reason = "not yet used")]
81    pub global_config: &'a GlobalConfig,
82    /// Project configuration associated with the unit of work.
83    pub project_info: &'a ProjectInfo,
84}
85
86impl ForwardContext<'_> {
87    /// Returns the [`Retention`] for a specific type/product.
88    pub fn retention<F>(&self, f: F) -> Retention
89    where
90        F: FnOnce(&RetentionsConfig) -> Option<&RetentionConfig>,
91    {
92        if let Some(retention) = f(&self.project_info.config.retentions) {
93            return Retention::from(*retention);
94        }
95
96        self.event_retention()
97    }
98
99    /// Returns the event [`Retention`].
100    ///
101    /// This retention is also often used for older products and can be considered a default
102    /// retention for products which do not define their own retention.
103    pub fn event_retention(&self) -> Retention {
104        Retention::from(RetentionConfig {
105            standard: self
106                .project_info
107                .config
108                .event_retention
109                .unwrap_or(crate::constants::DEFAULT_EVENT_RETENTION),
110            downsampled: self.project_info.config.downsampled_event_retention,
111        })
112    }
113}

140/// Full retention settings to apply to specific payloads.
141#[derive(Debug, Copy, Clone)]
142pub struct Retention {
143    /// Standard / full fidelity retention policy in days.
144    pub standard: u16,
145    /// Downsampled retention policy in days.
146    #[cfg_attr(not(feature = "processing"), expect(unused))]
147    pub downsampled: u16,
148}
149
150impl From<RetentionConfig> for Retention {
151    fn from(value: RetentionConfig) -> Self {
152        Self {
153            standard: value.standard,
154            downsampled: value.downsampled.unwrap_or(value.standard),
155        }
156    }
