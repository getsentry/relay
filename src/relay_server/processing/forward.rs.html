16/// A transparent handle that dispatches between store-like services.
17#[cfg(feature = "processing")]
18#[derive(Debug, Clone, Copy)]
19pub struct StoreHandle<'a> {
20    store: &'a Addr<Store>,
21    upload: Option<&'a Addr<Upload>>,
22}
23
24#[cfg(feature = "processing")]
25impl<'a> StoreHandle<'a> {
26    pub fn new(store: &'a Addr<Store>, upload: Option<&'a Addr<Upload>>) -> Self {
27        Self { store, upload }
28    }
29
30    /// Sends a message to the [`Store`] service.
31    pub fn store<M>(&self, message: M)
32    where
33        Store: FromMessage<M>,
34    {
35        self.store.send(message);
36    }
37
38    /// Sends a message to the [`Upload`] service.
39    #[expect(unused)]
40    pub fn upload<M>(&self, message: M)
41    where
42        Upload: FromMessage<M>,
43    {
44        if let Some(upload) = self.upload {
45            upload.send(message);
46        } else {
47            relay_log::error!("Upload service not configured. Dropping message.");
48        }
49    }
50}

52/// A processor output which can be forwarded to a different destination.
53pub trait Forward {
54    /// Serializes the output into an [`Envelope`].
55    ///
56    /// All output must be serializable as an envelope.
57    fn serialize_envelope(
58        self,
59        ctx: ForwardContext<'_>,
60    ) -> Result<Managed<Box<Envelope>>, Rejected<()>>;
61
62    /// Serializes the output into a [`crate::services::store::StoreService`] compatible format.
63    ///
64    /// This function must only be called when Relay is configured to be in processing mode.
65    #[cfg(feature = "processing")]
66    fn forward_store(self, s: StoreHandle<'_>, ctx: ForwardContext<'_>)
67    -> Result<(), Rejected<()>>;
68}

70/// Context passed to [`Forward`].
71///
72/// A minified version of [`Context`](super::Context), which does not contain processing specific information.
73#[derive(Copy, Clone, Debug)]
74pub struct ForwardContext<'a> {
75    /// The Relay configuration.
76    #[expect(unused, reason = "not yet used")]
77    pub config: &'a Config,
78    /// A view of the currently active global configuration.
79    #[expect(unused, reason = "not yet used")]
80    pub global_config: &'a GlobalConfig,
81    /// Project configuration associated with the unit of work.
82    #[cfg_attr(not(feature = "processing"), expect(unused))]
83    pub project_info: &'a ProjectInfo,
84}
85
86#[cfg(feature = "processing")]
87impl ForwardContext<'_> {
88    /// Returns the [`Retention`] for a specific type/product.
89    pub fn retention<F>(&self, f: F) -> Retention
90    where
91        F: FnOnce(&RetentionsConfig) -> Option<&RetentionConfig>,
92    {
93        if let Some(retention) = f(&self.project_info.config.retentions) {
94            return Retention::from(*retention);
95        }
96
97        Retention::from(RetentionConfig {
98            standard: self
99                .project_info
100                .config
101                .event_retention
102                .unwrap_or(crate::constants::DEFAULT_EVENT_RETENTION),
103            downsampled: self.project_info.config.downsampled_event_retention,
104        })
105    }
106}

133/// Full retention settings to apply to specific payloads.
134#[derive(Debug, Copy, Clone)]
135#[cfg(feature = "processing")]
136pub struct Retention {
137    /// Standard / full fidelity retention policy in days.
138    pub standard: u16,
139    /// Downsampled retention policy in days.
140    pub downsampled: u16,
141}
142
143#[cfg(feature = "processing")]
144impl From<RetentionConfig> for Retention {
145    fn from(value: RetentionConfig) -> Self {
146        Self {
147            standard: value.standard,
148            downsampled: value.downsampled.unwrap_or(value.standard),
149        }
150    }
