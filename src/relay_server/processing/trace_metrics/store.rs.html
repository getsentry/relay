21macro_rules! required {
22    ($value:expr) => {{
23        match $value {
24            Annotated(Some(value), _) => value,
25            Annotated(None, meta) => {
26                relay_log::debug!(
27                    "dropping trace metric because of missing required field {} with meta {meta:?}",
28                    stringify!($value),
29                );
30                return Err(Error::Invalid(DiscardReason::InvalidTraceMetric));
31            }
32        }
33    }};
34}

46pub fn convert(metric: WithHeader<TraceMetric>, ctx: &Context) -> Result<StoreTraceItem> {
47    let quantities = metric.quantities();
48
49    let metric = required!(metric.value);
50    let timestamp = required!(metric.timestamp);
51
52    let meta = extract_meta_attributes(&metric, &metric.attributes);
53    let attrs = metric.attributes.0.unwrap_or_default();
54    let fields = FieldAttributes {
55        metric_name: required!(metric.name),
56        metric_type: required!(metric.ty),
57        metric_unit: metric.unit.into_value(),
58        value: extract_numeric_value(required!(metric.value))?,
59        timestamp,
60        span_id: metric.span_id.into_value(),
61    };
62
63    let client_sample_rate = extract_client_sample_rate(&attrs).unwrap_or(1.0);
64
65    let trace_item = TraceItem {
66        item_type: TraceItemType::Metric.into(),
67        organization_id: ctx.scoping.organization_id.value(),
68        project_id: ctx.scoping.project_id.value(),
69        received: Some(ts(ctx.received_at)),
70        retention_days: ctx.retention.standard.into(),
71        downsampled_retention_days: ctx.retention.downsampled.into(),
72        timestamp: Some(ts(timestamp.0)),
73        trace_id: required!(metric.trace_id).to_string(),
74        item_id: uuid_to_item_id(Uuid::new_v7(timestamp.into())),
75        attributes: attributes(meta, attrs, fields),
76        client_sample_rate,
77        server_sample_rate: 1.0,
78    };
79
80    Ok(StoreTraceItem {
81        trace_item,
82        quantities,
83    })
84}

86fn ts(dt: DateTime<Utc>) -> Timestamp {
87    Timestamp {
88        seconds: dt.timestamp(),
89        nanos: i32::try_from(dt.timestamp_subsec_nanos()).unwrap_or(0),
90    }

102fn extract_numeric_value(value: Value) -> Result<f64> {
103    match value {
104        Value::F64(v) => Ok(v),
105        Value::I64(v) => Ok(v as f64),
106        Value::U64(v) => Ok(v as f64),
107        _ => Err(Error::Invalid(DiscardReason::InvalidTraceMetric)),
108    }
109}

111fn attributes(
112    meta: HashMap<String, AnyValue>,
113    attributes: Attributes,
114    fields: FieldAttributes,
115) -> HashMap<String, AnyValue> {
116    let mut result = meta;
117    // +N, one for each field attribute added and some extra for potential meta.
118    result.reserve(attributes.0.len() + 15);
119
120    processing::utils::store::convert_attributes_into(&mut result, attributes);
121
122    let FieldAttributes {
123        metric_name,
124        metric_type,
125        metric_unit,
126        value,
127        timestamp,
128        span_id,
129    } = fields;
130
131    result.insert(
132        "sentry.metric_name".to_owned(),
133        AnyValue {
134            value: Some(any_value::Value::StringValue(metric_name.clone())),
135        },
136    );
137
138    result.insert(
139        "sentry.metric_type".to_owned(),
140        AnyValue {
141            value: Some(any_value::Value::StringValue(metric_type.to_string())),
142        },
143    );
144
145    // Add key names matching metric name and type to workaround current co-occuring attributes limitations.
146    result.insert(
147        format!("sentry._internal.cooccuring.name.{metric_name}"),
148        AnyValue {
149            value: Some(any_value::Value::BoolValue(true)),
150        },
151    );
152    result.insert(
153        format!("sentry._internal.cooccuring.type.{metric_type}"),
154        AnyValue {
155            value: Some(any_value::Value::BoolValue(true)),
156        },
157    );
158
159    if let Some(metric_unit) = metric_unit {
160        result.insert(
161            "sentry.metric_unit".to_owned(),
162            AnyValue {
163                value: Some(any_value::Value::StringValue(metric_unit.to_string())),
164            },
165        );
166
167        result.insert(
168            format!("sentry._internal.cooccuring.unit.{metric_unit}"),
169            AnyValue {
170                value: Some(any_value::Value::BoolValue(true)),
171            },
172        );
173    }
174
175    result.insert(
176        "sentry.value".to_owned(),
177        AnyValue {
178            value: Some(any_value::Value::DoubleValue(value)),
179        },
180    );
181
182    let timestamp_nanos = timestamp
183        .into_inner()
184        .timestamp_nanos_opt()
185        .unwrap_or_default();
186
187    result.insert(
188        "sentry.timestamp_precise".to_owned(),
189        AnyValue {
190            value: Some(any_value::Value::IntValue(timestamp_nanos)),
191        },
192    );
193
194    if let Some(span_id) = span_id {
195        result.insert(
196            "sentry.span_id".to_owned(),
197            AnyValue {
198                value: Some(any_value::Value::StringValue(span_id.to_string())),
199            },
200        );
201    }
202
203    result
204}

281    #[test]
282    fn test_extract_client_sample_rate_function() {
283        let mut attrs_map = BTreeMap::new();
284        let attr = Attribute {
285            value: AttributeValue {
286                ty: Annotated::new(AttributeType::Double),
287                value: Annotated::new(Value::F64(0.5)),
288            },
289            other: Object::new(),
290        };
291        attrs_map.insert("sentry.client_sample_rate".to_owned(), Annotated::new(attr));
292        let attrs = Attributes(attrs_map);
293
294        let sample_rate = extract_client_sample_rate(&attrs);
295        assert_eq!(sample_rate, Some(0.5));
296
297        // Without sample rate attribute
298        let empty_attrs = Attributes(BTreeMap::new());
299        let default_rate = extract_client_sample_rate(&empty_attrs);
300        assert_eq!(default_rate, None);
301
302        // Invalid sample rate
303        let invalid_attrs = Attributes(BTreeMap::from([(
304            "sentry.client_sample_rate".to_owned(),
305            Annotated::new(Attribute {
306                value: AttributeValue {
307                    ty: Annotated::new(AttributeType::Double),
308                    value: Annotated::new(Value::F64(2.0)),
309                },
310                other: Object::new(),
311            }),
312        )]));
313        let invalid_rate = extract_client_sample_rate(&invalid_attrs);
314        assert_eq!(invalid_rate, None);
315    }
