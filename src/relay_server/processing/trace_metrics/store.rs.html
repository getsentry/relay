18macro_rules! required {
19    ($value:expr) => {{
20        match $value {
21            Annotated(Some(value), _) => value,
22            Annotated(None, meta) => {
23                relay_log::debug!(
24                    "dropping trace metric because of missing required field {} with meta {meta:?}",
25                    stringify!($value),
26                );
27                return Err(Error::Invalid(DiscardReason::InvalidTraceMetric));
28            }
29        }
30    }};
31}

43pub fn convert(metric: WithHeader<TraceMetric>, ctx: &Context) -> Result<StoreTraceItem> {
44    let quantities = metric.quantities();
45
46    let metric = required!(metric.value);
47    let timestamp = required!(metric.timestamp);
48
49    let meta = extract_meta_attributes(&metric, &metric.attributes);
50    let attrs = metric.attributes.0.unwrap_or_default();
51    let fields = FieldAttributes {
52        metric_name: required!(metric.name),
53        metric_type: required!(metric.ty),
54        value: extract_numeric_value(required!(metric.value))?,
55        timestamp,
56        span_id: metric.span_id.into_value(),
57    };
58
59    let trace_item = TraceItem {
60        item_type: TraceItemType::Metric.into(),
61        organization_id: ctx.scoping.organization_id.value(),
62        project_id: ctx.scoping.project_id.value(),
63        received: Some(ts(ctx.received_at)),
64        retention_days: ctx.retention.standard.into(),
65        downsampled_retention_days: ctx.retention.downsampled.into(),
66        timestamp: Some(ts(timestamp.0)),
67        trace_id: required!(metric.trace_id).to_string(),
68        item_id: Uuid::new_v7(timestamp.into()).as_bytes().to_vec(),
69        attributes: attributes(meta, attrs, fields),
70        client_sample_rate: 1.0,
71        server_sample_rate: 1.0,
72    };
73
74    Ok(StoreTraceItem {
75        trace_item,
76        quantities,
77    })
78}

80fn ts(dt: DateTime<Utc>) -> Timestamp {
81    Timestamp {
82        seconds: dt.timestamp(),
83        nanos: i32::try_from(dt.timestamp_subsec_nanos()).unwrap_or(0),
84    }

95fn extract_numeric_value(value: Value) -> Result<f64> {
96    match value {
97        Value::F64(v) => Ok(v),
98        Value::I64(v) => Ok(v as f64),
99        Value::U64(v) => Ok(v as f64),
100        _ => Err(Error::Invalid(DiscardReason::InvalidTraceMetric)),
101    }
102}

104fn attributes(
105    meta: HashMap<String, AnyValue>,
106    attributes: Attributes,
107    fields: FieldAttributes,
108) -> HashMap<String, AnyValue> {
109    let mut result = meta;
110    result.reserve(attributes.0.len() + 5);
111
112    for (name, attribute) in attributes {
113        let meta = AttributeMeta {
114            meta: IntoValue::extract_meta_tree(&attribute),
115        };
116        if let Some(meta) = meta.to_any_value() {
117            result.insert(format!("sentry._meta.fields.attributes.{name}"), meta);
118        }
119
120        let value = attribute
121            .into_value()
122            .and_then(|v| v.value.value.into_value());
123
124        let Some(value) = value else {
125            continue;
126        };
127
128        let Some(value) = (match value {
129            Value::Bool(v) => Some(any_value::Value::BoolValue(v)),
130            Value::I64(v) => Some(any_value::Value::IntValue(v)),
131            Value::U64(v) => i64::try_from(v).ok().map(any_value::Value::IntValue),
132            Value::F64(v) => Some(any_value::Value::DoubleValue(v)),
133            Value::String(v) => Some(any_value::Value::StringValue(v)),
134            Value::Array(_) | Value::Object(_) => {
135                debug_assert!(false, "unsupported trace metric value");
136                None
137            }
138        }) else {
139            continue;
140        };
141
142        result.insert(name, AnyValue { value: Some(value) });
143    }
144
145    let FieldAttributes {
146        metric_name,
147        metric_type,
148        value,
149        timestamp,
150        span_id,
151    } = fields;
152
153    result.insert(
154        "sentry.metric_name".to_owned(),
155        AnyValue {
156            value: Some(any_value::Value::StringValue(metric_name)),
157        },
158    );
159
160    result.insert(
161        "sentry.metric_type".to_owned(),
162        AnyValue {
163            value: Some(any_value::Value::StringValue(metric_type.to_string())),
164        },
165    );
166
167    result.insert(
168        "sentry.value".to_owned(),
169        AnyValue {
170            value: Some(any_value::Value::DoubleValue(value)),
171        },
172    );
173
174    let timestamp_nanos = timestamp
175        .into_inner()
176        .timestamp_nanos_opt()
177        .unwrap_or_default();
178
179    result.insert(
180        "sentry.timestamp_nanos".to_owned(),
181        AnyValue {
182            value: Some(any_value::Value::StringValue(timestamp_nanos.to_string())),
183        },
184    );
185    result.insert(
186        "sentry.timestamp_precise".to_owned(),
187        AnyValue {
188            value: Some(any_value::Value::IntValue(timestamp_nanos)),
189        },
190    );
191
192    if let Some(span_id) = span_id {
193        result.insert(
194            "sentry.span_id".to_owned(),
195            AnyValue {
196                value: Some(any_value::Value::StringValue(span_id.to_string())),
197            },
198        );
199    }
200
201    result
202}

240    fn test_trace_metric_conversion() {

243            "trace_id": "5B8EFFF798038103D269B633813FC60C",
244            "span_id": "EEE19B7EC3C1B174",
