10pub fn validate(metrics: &mut Managed<ExpandedTraceMetrics>) {
11    metrics.retain(
12        |metrics| &mut metrics.metrics,
13        |metric, _| {
14            validate_trace_metric(metric).inspect_err(|err| {
15                relay_log::debug!("failed to validate trace metric: {err}");
16            })
17        },
18    );
19}
20
21fn validate_trace_metric(metric: &Annotated<TraceMetric>) -> Result<()> {
22    match metric.value().and_then(|m| m.ty.value()) {
23        Some(MetricType::Gauge | MetricType::Distribution | MetricType::Counter) => {}
24        Some(MetricType::Unknown(_)) | None => {
25            return Err(Error::Invalid(DiscardReason::InvalidTraceMetric));
26        }
27    }
28    Ok(())
29}

31/// Validates that there is only a single trace metric container processed at a time.
32///
33/// Similar to logs, the `TraceMetric` item must always be sent as an `ItemContainer`, currently it is not allowed to
34/// send multiple containers for trace metrics.
35///
36/// This restriction may be lifted in the future.
37///
38/// This limit mostly exists to incentivise SDKs to batch multiple trace metrics into a single container,
39/// technically it can be removed without issues.
40pub fn container(metrics: &Managed<SerializedTraceMetrics>) -> Result<(), Rejected<Error>> {
41    // It's fine if there was no trace metric container, as we may accept OTel trace metrics in the future.
42    if metrics.metrics.len() > 1 {
43        return Err(metrics.reject_err(Error::DuplicateContainer));
44    }
45
46    Ok(())
47}
