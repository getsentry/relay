12/// Validates that there is only a single trace metric container processed at a time.
13///
14/// Similar to logs, the `TraceMetric` item must always be sent as an `ItemContainer`, currently it is not allowed to
15/// send multiple containers for trace metrics.
16///
17/// This restriction may be lifted in the future.
18///
19/// This limit mostly exists to incentivise SDKs to batch multiple trace metrics into a single container,
20/// technically it can be removed without issues.
21pub fn container(metrics: &Managed<SerializedTraceMetrics>) -> Result<(), Rejected<Error>> {
22    // It's fine if there was no trace metric container, as we may accept OTel trace metrics in the future.
23    if metrics.metrics.len() > 1 {
24        return Err(metrics.reject_err(Error::DuplicateContainer));
25    }
26
27    Ok(())
28}

30/// Validates contained metrics do not exceed the maximum size limit.
31///
32/// Currently this only considers the maximum metric size configured in the configuration.
33///
34/// In the future we may want to increase the limit or start trimming excessive
35/// attributes/payloads. For now we drop metrics which exceed our size limit.
36///
37/// This matches the logic defined in [`check_envelope_size_limits`](crate::utils::check_envelope_size_limits),
38/// when validating envelope sizes, the actual size of individual metrics is not known and therefore
39/// must be enforced consistently after parsing again.
40pub fn size(metrics: &mut Managed<ExpandedTraceMetrics>, ctx: Context<'_>) {
41    let max_size_bytes = ctx.config.max_trace_metric_size();
42
43    metrics.retain(
44        |metrics| &mut metrics.metrics,
45        |metric, _| {
46            let size = calculate_size(metric);
47
48            match size > max_size_bytes {
49                true => Err(Error::TooLarge),
50                false => Ok(()),
51            }
52        },
53    );
54}

56/// Calculates the byte size for a trace metric.
57///
58/// Similar to [`relay_ourlogs::calculate_size`].
59fn calculate_size(metric: &WithHeader<TraceMetric>) -> usize {
60    let Some(metric) = metric.value() else {
61        // Same logic we use for logs.
62        return 1;
63    };
64
65    let mut size = 0;
66
67    size += metric.name.value().map_or(0, |s| s.len());
68    size += relay_event_normalization::eap::value_size(&metric.value);
69
70    if let Some(attributes) = metric.attributes.value() {
71        size += relay_event_normalization::eap::attributes_size(attributes);
72    }
73
74    size.max(1)
75}

77/// Validates the semantic validity of a trace metric.
78pub fn validate(metrics: &mut Managed<ExpandedTraceMetrics>) {
79    metrics.retain(
80        |metrics| &mut metrics.metrics,
81        |metric, _| {
82            validate_trace_metric(metric).inspect_err(|err| {
83                relay_log::debug!("failed to validate trace metric: {err}");
84            })
85        },
86    );
87}
88
89fn validate_trace_metric(metric: &Annotated<TraceMetric>) -> Result<()> {
90    match metric.value().and_then(|m| m.ty.value()) {
91        Some(MetricType::Gauge | MetricType::Distribution | MetricType::Counter) => {}
92        Some(MetricType::Unknown(_)) | None => {
93            return Err(Error::Invalid(DiscardReason::InvalidTraceMetric));
94        }
95    }
96    Ok(())
97}
