91    /// Fully processes an envelope and returns the resulting outputs.
92    ///
93    /// Since an envelope may contain payloads which need to be processed independently,
94    /// the function may return multiple results at once.
95    pub async fn run(
96        &self,
97        mut envelope: ManagedEnvelope,
98        ctx: Context<'_>,
99    ) -> Vec<Output<Outputs>> {
100        let mut outputs = Vec::with_capacity(5);
101
102        let pi = ctx.project_info;
103
104        macro_rules! run {
105            ($processor:expr) => {{
106                if let Some(output) = self.run_one(&$processor, &mut envelope, ctx).await {
107                    outputs.push(output.map(Into::into));
108                }
109            }};
110        }
111
112        // The order of processors is deliberate and must not be changed lightly!
113        //
114        // Processors may partially consume items of the envelope and different processors may
115        // process the same item types.
116        //
117        // Currently this allows for never intended combinations of different item types in envelopes,
118        // ideally we make Relay gradually more restrictive and properly define and follow
119        // the rules of item combinations allowed in a single envelope.
120
121        // Item types which can be mixed into any envelope.
122        run!(self.sessions);
123        run!(self.client_reports);
124
125        // Primary item types.
126        run!(self.replays);
127        if !pi.has_feature(Feature::SpanV2ExperimentalProcessing) {
128            // To be fully replaced with the npn-legacy span processor.
129            run!(self.legacy_spans);
130        }
131        run!(self.spans);
132        run!(self.logs);
133        run!(self.trace_metrics);
134        run!(self.profile_chunks);
135        run!(self.check_ins);
136
137        // Event based envelopes.
138        run!(self.errors);
139        run!(self.transactions);
140
141        // Item types which can be sent with a primary item type and can also be sent standalone.
142        //
143        // These need to be processed after their respective primary type.
144        run!(self.trace_attachments);
145        run!(self.attachments);
146        run!(self.user_reports);
147        run!(self.profiles);
148
149        // Fallback for forward compatibility.
150        run!(self.forward_unknown);
151        // All remaining items which make no sense.
152        run!(self.invalid);
153
154        // After processing there must not be any items in the original envelope left.
155        match envelope.envelope().is_empty() {
156            true => envelope.accept(),
157            false => {
158                relay_log::error!(
159                    items = ?envelope.envelope().items().map(Item::ty).collect::<Vec<_>>(),
160                    "Processed envelope has items remaining"
161                );
162                envelope.update();
163                envelope.reject(Outcome::Invalid(DiscardReason::Internal));
164            }
165        }
166
167        outputs
168    }

170    async fn run_one<T: Processor>(
171        &self,
172        processor: &T,
173        envelope: &mut ManagedEnvelope,
174        ctx: Context<'_>,
175    ) -> Option<Output<T::Output>>
176    where
177        T::Input: Debug,
178    {
179        let item = processor.prepare_envelope(envelope)?;
180
181        relay_log::trace!(
182            processor = std::any::type_name::<T>(),
183            "processing item: {:?}",
184            item.as_ref(),
185        );
186
187        let _token = self.cogs.timed(ResourceId::Relay, T::cogs());
188
189        let output = relay_statsd::metric!(
190            timer(RelayTimers::EventProcessingProcess),
191            processor = std::any::type_name::<T>(),
192            { processor.process(item, ctx).await }
193        );
194
195        match output {
196            Ok(output) => Some(output),
197            Err(error) => {
198                // This is not a fatal error case. For example a processor may reject an
199                // item because it was filtered by an inbound filter or rate limit.
200                // This means that other items from the same original envelope must still be processed.
201                relay_log::debug!(
202                    error = &error as &dyn std::error::Error,
203                    processor = std::any::type_name::<T>(),
204                    "item rejected by processor"
205                );
206
207                None
208            }
209        }
210    }
