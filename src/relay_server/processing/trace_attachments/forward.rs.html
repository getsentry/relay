11impl ExpandedAttachments {
12    fn serialize_envelope(self) -> Result<Box<Envelope>, serde_json::Error> {
13        let Self {
14            headers,
15            server_sample_rate: _,
16            attachments,
17        } = self;
18        let mut items = Items::new();
19        for attachment in attachments {
20            items.push(attachment_to_item(attachment)?);
21        }
22        Ok(Envelope::from_parts(headers, items))
23    }
24}
25
26impl Forward for Managed<ExpandedAttachments> {
27    fn serialize_envelope(
28        self,
29        _ctx: ForwardContext<'_>,
30    ) -> Result<Managed<Box<Envelope>>, Rejected<()>> {
31        self.try_map(|this, _| {
32            this.serialize_envelope()
33                .map_err(drop)
34                .with_outcome(Outcome::Invalid(DiscardReason::Internal))
35        })
36    }
37
38    #[cfg(feature = "processing")]
39    fn forward_store(
40        self,
41        s: StoreHandle<'_>,
42        ctx: ForwardContext<'_>,
43    ) -> Result<(), Rejected<()>> {
44        let retention = ctx.retention(|r| r.trace_attachment.as_ref());
45        let server_sample_rate = self.server_sample_rate;
46        for attachment in self.split(|work| work.attachments) {
47            if let Ok(message) = store::convert(attachment, retention, server_sample_rate) {
48                s.upload(message)
49            }
50        }
51        Ok(())
52    }
53}

55/// Converts an expanded trace attachment to an envelope item.
56pub fn attachment_to_item(attachment: ExpandedAttachment) -> Result<Item, serde_json::Error> {
57    let ExpandedAttachment {
58        parent_id,
59        meta,
60        body,
61    } = attachment;
62
63    let meta_json = meta.to_json()?;
64    let meta_bytes = meta_json.into_bytes();
65    let meta_length = meta_bytes.len();
66
67    let mut payload = bytes::BytesMut::with_capacity(meta_length + body.len());
68    payload.extend_from_slice(&meta_bytes);
69    payload.extend_from_slice(&body);
70
71    let mut item = Item::new(ItemType::Attachment);
72    item.set_payload(ContentType::TraceAttachment, payload.freeze());
73    item.set_meta_length(meta_length as u32);
74    item.set_parent_id(parent_id);
75
76    Ok(item)
77}
