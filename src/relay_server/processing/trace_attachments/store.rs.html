18/// Converts an expanded attachment to a storable unit.
19pub fn convert(
20    attachment: Managed<ExpandedAttachment>,
21    retention: Retention,
22    server_sample_rate: Option<f64>,
23) -> Result<Managed<StoreAttachment>, Rejected<()>> {
24    let scoping = attachment.scoping();
25    let received_at = attachment.received_at();
26    attachment.try_map(|attachment, _record_keeper| {
27        let ExpandedAttachment {
28            parent_id,
29            meta,
30            body,
31        } = attachment;
32        let ctx = Context {
33            span_id: parent_id.and_then(|p| p.as_span_id()),
34            received_at,
35            scoping,
36            retention,
37            server_sample_rate,
38        };
39        let trace_item = attachment_to_trace_item(meta, ctx)
40            .ok_or(Outcome::Invalid(DiscardReason::InvalidTraceAttachment))?;
41
42        Ok::<_, Outcome>(StoreAttachment { trace_item, body })
43    })
44}

46/// Context for converting an attachment to a trace item.
47#[derive(Debug, Clone, Copy)]
48struct Context {
49    /// Received time.
50    received_at: DateTime<Utc>,
51    /// Item scoping.
52    scoping: Scoping,
53    /// Item retention.
54    #[cfg(feature = "processing")]
55    retention: Retention,
56    /// Server-side sample rate.
57    server_sample_rate: Option<f64>,
58    /// The ID of the span that owns the attachment.
59    span_id: Option<SpanId>,
60}

62fn attachment_to_trace_item(
63    meta: Annotated<TraceAttachmentMeta>,
64    ctx: Context,
65) -> Option<TraceItem> {
66    let meta = meta.into_value()?;
67    let annotated_meta = extract_meta_attributes(&meta, &meta.attributes);
68    let TraceAttachmentMeta {
69        trace_id,
70        attachment_id,
71        timestamp,
72        filename,
73        content_type,
74        attributes,
75        other: _,
76    } = meta;
77
78    let fields = Fields {
79        content_type: content_type.into_value()?,
80        filename: filename.into_value(),
81        span_id: ctx.span_id,
82    };
83
84    let attributes = attributes.into_value().unwrap_or_default();
85
86    let client_sample_rate = extract_client_sample_rate(&attributes).unwrap_or(1.0);
87
88    let trace_item = TraceItem {
89        organization_id: ctx.scoping.organization_id.value(),
90        project_id: ctx.scoping.project_id.value(),
91        trace_id: trace_id.into_value()?.to_string(),
92        item_id: attachment_id.into_value()?.into_bytes().to_vec(),
93        item_type: TraceItemType::Attachment.into(),
94        timestamp: Some(proto_timestamp(timestamp.into_value()?.0)),
95        attributes: convert_attributes(annotated_meta, attributes, fields),
96        client_sample_rate,
97        server_sample_rate: ctx.server_sample_rate.unwrap_or(1.0),
98        retention_days: ctx.retention.standard as u32,
99        received: Some(proto_timestamp(ctx.received_at)),
100        downsampled_retention_days: ctx.retention.downsampled as u32,
101    };
102    Some(trace_item)
103}

105struct Fields {
106    content_type: String,
107    filename: Option<String>,
108    span_id: Option<SpanId>,
109}

112fn convert_attributes(
113    meta: HashMap<String, AnyValue>,
114    attributes: Attributes,
115    fields: Fields,
116) -> HashMap<String, AnyValue> {
117    let mut result = meta;
118    result.reserve(attributes.0.len() + 5);
119
120    for (name, attribute) in attributes {
121        let meta = AttributeMeta {
122            meta: IntoValue::extract_meta_tree(&attribute),
123        };
124        if let Some(meta) = meta.to_any_value() {
125            result.insert(format!("sentry._meta.fields.attributes.{name}"), meta);
126        }
127
128        let value = attribute
129            .into_value()
130            .and_then(|v| v.value.value.into_value());
131
132        let Some(value) = value else {
133            continue;
134        };
135
136        let Some(value) = (match value {
137            Value::Bool(v) => Some(any_value::Value::BoolValue(v)),
138            Value::I64(v) => Some(any_value::Value::IntValue(v)),
139            Value::U64(v) => i64::try_from(v).ok().map(any_value::Value::IntValue),
140            Value::F64(v) => Some(any_value::Value::DoubleValue(v)),
141            Value::String(v) => Some(any_value::Value::StringValue(v)),
142            Value::Array(_) | Value::Object(_) => {
143                debug_assert!(false, "unsupported attachment attribute value");
144                None
145            }
146        }) else {
147            continue;
148        };
149
150        result.insert(name, AnyValue { value: Some(value) });
151    }
152
153    let Fields {
154        content_type,
155        filename,
156        span_id,
157    } = fields;
158
159    result.insert(
160        "sentry.content-type".to_owned(),
161        AnyValue {
162            value: Some(any_value::Value::StringValue(
163                content_type.as_str().to_owned(),
164            )),
165        },
166    );
167
168    // See https://opentelemetry.io/docs/specs/semconv/registry/attributes/file/#file-name.
169    if let Some(filename) = filename {
170        result.insert(
171            "file.name".to_owned(),
172            AnyValue {
173                value: Some(any_value::Value::StringValue(filename)),
174            },
175        );
176    }
177
178    if let Some(span_id) = span_id {
179        result.insert(
180            "sentry.span_id".to_owned(),
181            AnyValue {
182                value: Some(any_value::Value::StringValue(span_id.to_string())),
183            },
184        );
185    }
186
187    result
188}
