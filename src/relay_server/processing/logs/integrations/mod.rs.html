11/// Expands a list of [`Integration`] items into `result`.
12///
13/// The function expects *only* log item integrations.
14pub fn expand_into(
15    result: &mut ContainerItems<OurLog>,
16    records: &mut RecordKeeper<'_>,
17    items: Vec<Item>,
18) {
19    for item in items {
20        let integration = match item.integration() {
21            Some(Integration::Logs(integration)) => integration,
22            integration => {
23                records.internal_error(InvalidIntegration(integration), item);
24                continue;
25            }
26        };
27
28        let produce = |log: OurLog| {
29            let byte_size = relay_ourlogs::calculate_size(&log);
30
31            records.modify_by(DataCategory::LogItem, 1);
32            records.modify_by(DataCategory::LogByte, byte_size as isize);
33
34            result.push(WithHeader {
35                header: Some(OurLogHeader {
36                    byte_size: Some(byte_size),
37                    other: Default::default(),
38                }),
39                value: log.into(),
40            });
41        };
42
43        let payload = item.payload();
44        let payload = payload.as_slice();
45
46        let result = match integration {
47            LogsIntegration::OtelV1 { format } => otel::expand(format, payload, produce),
48        };
49
50        match result {
51            Err(err) => drop(records.reject_err(err, item)),
52            Ok(()) => {
53                // Undo all the base item quantities, as they will be completely taken over by the parsed
54                // contents, which contains an arbitrary amount of items (even 0).
55                for (category, quantity) in item.quantities() {
56                    records.modify_by(category, -(quantity as isize));
57                }
58            }
59        }
60    }
61}
