1use relay_event_schema::protocol::OurLog;
2use relay_ourlogs::VercelLog;
3
4use crate::integrations::VercelLogDrainFormat;
5use crate::processing::logs::{Error, Result};
6use crate::services::outcome::DiscardReason;
7
8/// Expands Vercel logs into the [`OurLog`] format.
9pub fn expand<F>(format: VercelLogDrainFormat, payload: &[u8], mut produce: F) -> Result<()>
10where
11    F: FnMut(OurLog),
12{
13    let mut count: i32 = 0;
14
15    match format {
16        VercelLogDrainFormat::Json => {
17            let logs = serde_json::from_slice::<Vec<VercelLog>>(payload).map_err(|e| {
18                relay_log::debug!(
19                    error = &e as &dyn std::error::Error,
20                    "Failed to parse logs data as JSON"
21                );
22                Error::Invalid(DiscardReason::InvalidJson)
23            })?;
24
25            for log in logs {
26                count += 1;
27                let ourlog = relay_ourlogs::vercel_log_to_sentry_log(log);
28                produce(ourlog);
29            }
30        }
31        VercelLogDrainFormat::NdJson => {
32            for line in payload.split(|&b| b == b'\n') {
33                if line.is_empty() {
34                    continue;
35                }
36
37                if let Ok(log) = serde_json::from_slice::<VercelLog>(line) {
38                    count += 1;
39                    let ourlog = relay_ourlogs::vercel_log_to_sentry_log(log);
40                    produce(ourlog);
41                }
42            }
43        }
44    }
45
46    if count == 0 {
47        relay_log::debug!("Failed to parse any logs from vercel log drain payload");
48        return Err(Error::Invalid(DiscardReason::InvalidJson));
49    }
50
51    Ok(())
52}
