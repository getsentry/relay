19macro_rules! required {
20    ($value:expr) => {{
21        match $value {
22            Annotated(Some(value), _) => value,
23            Annotated(None, meta) => {
24                relay_log::debug!(
25                    "dropping log because of missing required field {} with meta {meta:?}",
26                    stringify!($value),
27                );
28                return Err(Error::Invalid(DiscardReason::InvalidLog));
29            }
30        }
31    }};
32}

34/// Context parameters for [`convert`].
35#[derive(Debug, Clone, Copy)]
36pub struct Context {
37    /// Received time.
38    pub received_at: DateTime<Utc>,
39    /// Item scoping.
40    pub scoping: Scoping,
41    /// Storage retention in days.
42    pub retention: Option<u16>,
43}
44
45pub fn convert(log: WithHeader<OurLog>, ctx: &Context) -> Result<StoreTraceItem> {
46    let quantities = log.quantities();
47
48    let log = required!(log.value);
49    let timestamp = required!(log.timestamp);
50
51    let meta = extract_meta_attributes(&log);
52    let attrs = log.attributes.0.unwrap_or_default();
53    let fields = FieldAttributes {
54        level: required!(log.level),
55        timestamp,
56        body: required!(log.body),
57        span_id: log.span_id.into_value(),
58    };
59
60    let trace_item = TraceItem {
61        item_type: TraceItemType::Log.into(),
62        organization_id: ctx.scoping.organization_id.value(),
63        project_id: ctx.scoping.project_id.value(),
64        received: Some(ts(ctx.received_at)),
65        retention_days: ctx.retention.unwrap_or(DEFAULT_EVENT_RETENTION).into(),
66        timestamp: Some(ts(timestamp.0)),
67        trace_id: required!(log.trace_id).to_string(),
68        item_id: Uuid::new_v7(timestamp.into()).as_bytes().to_vec(),
69        attributes: attributes(meta, attrs, fields),
70        client_sample_rate: 1.0,
71        server_sample_rate: 1.0,
72    };
73
74    Ok(StoreTraceItem {
75        trace_item,
76        quantities,
77    })
78}

80fn ts(dt: DateTime<Utc>) -> Timestamp {
81    Timestamp {
82        seconds: dt.timestamp(),
83        nanos: i32::try_from(dt.timestamp_subsec_nanos()).unwrap_or(0),
84    }

87/// The schema of a 'metadata attribute' stored in EAP.
88///
89/// A metadata attribute is a regular attribute stored in EAP that contains metadata about a field
90/// of the original payload processed by Relay and other components.
91///
92/// It is also a place to store other, non-processing related, metadata on attributes, for example
93/// the `unit` of an attribute.
94///
95/// The attribute metadata itself is serialized as a JSON string.
96#[derive(Debug, Serialize)]
97struct AttributeMeta {
98    /// Meta as it was extracted from Relay's annotated model.
99    meta: MetaTree,
100}
101
102impl AttributeMeta {
103    fn to_any_value(&self) -> Option<AnyValue> {
104        if self.meta.is_empty() {
105            return None;
106        }
107
108        let s = serde_json::to_string(self)
109            .inspect_err(|err| {
110                relay_log::error!(
111                    error = err as &dyn std::error::Error,
112                    "attribute meta serialization failed"
113                )
114            })
115            .ok()?;
116
117        Some(AnyValue {
118            value: Some(any_value::Value::StringValue(s)),
119        })
120    }
121}

123/// Extracts [`TraceItem`] meta attributes from a [`OurLog`].
124///
125/// The implementation currently piggy backs on [`IntoValue::extract_child_meta`],
126/// a lighter implementation using a [`relay_event_schema::processor::Processor`]
127/// which removes the meta instead of cloning it is thinkable.
128///
129/// All extracted metadata is converted into [`TraceItem::attributes`] compatible values,
130/// by building a metadata representation for each top level field and attribute,
131/// serializing the result into JSON and building an appropriate metadata key.
132///
133/// The schema for metadata keys follows the format `sentry._meta.fields.{key}`,
134/// for attributes respectively `sentry._meta.fields.attributes.{key}`.
135fn extract_meta_attributes(log: &OurLog) -> HashMap<String, AnyValue> {
136    let mut meta = IntoValue::extract_child_meta(log);
137    // Attributes are the only 'nested' meta we allow.
138    let attributes = meta.remove("attributes");
139
140    let mut result = HashMap::with_capacity(
141        meta.len()
142            + attributes.as_ref().map_or(0, size_of_meta_tree)
143            // This len is filled later by the [`attributes`] function, it's just to try initialize
144            // the hashamap once with the correct count.
145            + log.attributes.value().map_or(0, |a| a.0.len()),
146    );
147
148    for (key, meta) in meta {
149        let attr = AttributeMeta { meta };
150        if let Some(value) = attr.to_any_value() {
151            let key = format!("sentry._meta.fields.{key}");
152            result.insert(key, value);
153        }
154    }
155
156    let Some(mut attributes) = attributes else {
157        return result;
158    };
159
160    for (key, meta) in std::mem::take(&mut attributes.children) {
161        let attr = AttributeMeta { meta };
162        if let Some(value) = attr.to_any_value() {
163            let key = format!("sentry._meta.fields.attributes.{key}");
164            result.insert(key, value);
165        }
166    }
167
168    // The `attributes` field itself can have metadata attached,
169    // we already took out all the metadata of the children, so now just emit
170    // the remaining metadata on the `attributes`.
171    let meta = AttributeMeta { meta: attributes };
172    if let Some(value) = meta.to_any_value() {
173        result.insert("sentry._meta.fields.attributes".to_owned(), value);
174    }
175
176    result
177}

179/// Calculates the immediate size of the meta tree passed in.
180///
181/// This only counts non empty meta elements of the passed in meta tree and its children,
182/// it does not recursively traverse the children.
183fn size_of_meta_tree(meta: &MetaTree) -> usize {
184    let mut size = 0;
185
186    if !meta.meta.is_empty() {
187        size += 1;
188    }
189    for meta in meta.children.values() {
190        if !meta.meta.is_empty() {
191            size += 1;
192        }
193    }
194
195    size
196}

198/// Fields on the log message which are stored as fields.
199struct FieldAttributes {
200    /// The log level.
201    ///
202    /// See: [`OurLog::level`].
203    level: OurLogLevel,
204    /// The original timestamp when the log was created.
205    ///
206    /// See: [`OurLog::timestamp`].
207    timestamp: relay_event_schema::protocol::Timestamp,
208    /// The log body.
209    ///
210    /// See: [`OurLog::body`].
211    body: String,
212    /// The optionally associated span id.
213    ///
214    /// See: [`OurLog::span_id`].
215    span_id: Option<SpanId>,
216}
217
218/// Extracts all attributes of a log, combines it with extracted meta attributes.
219fn attributes(
220    meta: HashMap<String, AnyValue>,
221    attributes: Attributes,
222    fields: FieldAttributes,
223) -> HashMap<String, AnyValue> {
224    let mut result = meta;
225    // +N, one for each field attribute added and some extra for potential meta.
226    result.reserve(attributes.0.len() + 5 + 3);
227
228    for (name, attribute) in attributes {
229        let meta = AttributeMeta {
230            meta: IntoValue::extract_meta_tree(&attribute),
231        };
232        if let Some(meta) = meta.to_any_value() {
233            result.insert(format!("sentry._meta.fields.attributes.{name}"), meta);
234        }
235
236        let value = attribute
237            .into_value()
238            .and_then(|v| v.value.value.into_value());
239
240        let Some(value) = value else {
241            // Meta has already been handled, no value -> skip.
242            // There are also no current plans to handle `null` in EAP.
243            continue;
244        };
245
246        let Some(value) = (match value {
247            Value::Bool(v) => Some(any_value::Value::BoolValue(v)),
248            Value::I64(v) => Some(any_value::Value::IntValue(v)),
249            Value::U64(v) => i64::try_from(v).ok().map(any_value::Value::IntValue),
250            Value::F64(v) => Some(any_value::Value::DoubleValue(v)),
251            Value::String(v) => Some(any_value::Value::StringValue(v)),
252            // These cases do not happen, as they are not valid attributes
253            // and they should have been filtered out before already.
254            Value::Array(_) | Value::Object(_) => {
255                debug_assert!(false, "unsupported log value");
256                None
257            }
258        }) else {
259            continue;
260        };
261
262        result.insert(name, AnyValue { value: Some(value) });
263    }
264
265    let FieldAttributes {
266        level,
267        timestamp,
268        body,
269        span_id,
270    } = fields;
271    // Unconditionally override any prior set attributes with the same key, as they should always
272    // come from the log itself.
273    //
274    // Ideally these attributes are marked as private in sentry-conventions and potentially
275    // validated against.
276    result.insert(
277        "sentry.severity_text".to_owned(),
278        AnyValue {
279            value: Some(any_value::Value::StringValue(level.to_string())),
280        },
281    );
282    let timestamp_nanos = timestamp
283        .into_inner()
284        .timestamp_nanos_opt()
285        // We can expect valid timestamps at this point, clock drift correction / normalization
286        // should've taken care of this already.
287        .unwrap_or_default();
288    result.insert(
289        "sentry.timestamp_nanos".to_owned(),
290        AnyValue {
291            value: Some(any_value::Value::StringValue(timestamp_nanos.to_string())),
292        },
293    );
294    result.insert(
295        "sentry.timestamp_precise".to_owned(),
296        AnyValue {
297            value: Some(any_value::Value::IntValue(timestamp_nanos)),
298        },
299    );
300    result.insert(
301        "sentry.body".to_owned(),
302        AnyValue {
303            value: Some(any_value::Value::StringValue(body)),
304        },
305    );
306    if let Some(span_id) = span_id {
307        result.insert(
308            "sentry.span_id".to_owned(),
309            AnyValue {
310                value: Some(any_value::Value::StringValue(span_id.to_string())),
311            },
312        );
313    }
314
315    result
316}
