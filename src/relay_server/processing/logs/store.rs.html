19macro_rules! required {
20    ($value:expr) => {{
21        match $value {
22            Annotated(Some(value), _) => value,
23            Annotated(None, meta) => {
24                relay_log::debug!(
25                    "dropping log because of missing required field {} with meta {meta:?}",
26                    stringify!($value),
27                );
28                return Err(Error::Invalid(DiscardReason::InvalidLog));
29            }
30        }
31    }};
32}

34/// Context parameters for [`convert`].
35#[derive(Debug, Clone, Copy)]
36pub struct Context {
37    /// Received time.
38    pub received_at: DateTime<Utc>,
39    /// Item scoping.
40    pub scoping: Scoping,
41    /// Storage retention in days.
42    pub retention: Option<u16>,
43    /// Storage retention for downsampled data in days
44    pub downsampled_retention: Option<u16>,
45}

47pub fn convert(log: WithHeader<OurLog>, ctx: &Context) -> Result<StoreTraceItem> {
48    let quantities = log.quantities();
49
50    let log = required!(log.value);
51    let timestamp = required!(log.timestamp);
52
53    let meta = extract_meta_attributes(&log);
54    let attrs = log.attributes.0.unwrap_or_default();
55    let fields = FieldAttributes {
56        level: required!(log.level),
57        timestamp,
58        body: required!(log.body),
59        span_id: log.span_id.into_value(),
60    };
61    let retention_days = ctx.retention.unwrap_or(DEFAULT_EVENT_RETENTION);
62
63    let trace_item = TraceItem {
64        item_type: TraceItemType::Log.into(),
65        organization_id: ctx.scoping.organization_id.value(),
66        project_id: ctx.scoping.project_id.value(),
67        received: Some(ts(ctx.received_at)),
68        retention_days: retention_days.into(),
69        downsampled_retention_days: ctx.downsampled_retention.unwrap_or(retention_days).into(),
70        timestamp: Some(ts(timestamp.0)),
71        trace_id: required!(log.trace_id).to_string(),
72        item_id: Uuid::new_v7(timestamp.into()).as_bytes().to_vec(),
73        attributes: attributes(meta, attrs, fields),
74        client_sample_rate: 1.0,
75        server_sample_rate: 1.0,
76    };
77
78    Ok(StoreTraceItem {
79        trace_item,
80        quantities,
81    })
82}

84fn ts(dt: DateTime<Utc>) -> Timestamp {
85    Timestamp {
86        seconds: dt.timestamp(),
87        nanos: i32::try_from(dt.timestamp_subsec_nanos()).unwrap_or(0),
88    }

91/// The schema of a 'metadata attribute' stored in EAP.
92///
93/// A metadata attribute is a regular attribute stored in EAP that contains metadata about a field
94/// of the original payload processed by Relay and other components.
95///
96/// It is also a place to store other, non-processing related, metadata on attributes, for example
97/// the `unit` of an attribute.
98///
99/// The attribute metadata itself is serialized as a JSON string.
100#[derive(Debug, Serialize)]
101struct AttributeMeta {
102    /// Meta as it was extracted from Relay's annotated model.
103    meta: MetaTree,
104}
105
106impl AttributeMeta {
107    fn to_any_value(&self) -> Option<AnyValue> {
108        if self.meta.is_empty() {
109            return None;
110        }
111
112        let s = serde_json::to_string(self)
113            .inspect_err(|err| {
114                relay_log::error!(
115                    error = err as &dyn std::error::Error,
116                    "attribute meta serialization failed"
117                )
118            })
119            .ok()?;
120
121        Some(AnyValue {
122            value: Some(any_value::Value::StringValue(s)),
123        })
124    }
125}

127/// Extracts [`TraceItem`] meta attributes from a [`OurLog`].
128///
129/// The implementation currently piggy backs on [`IntoValue::extract_child_meta`],
130/// a lighter implementation using a [`relay_event_schema::processor::Processor`]
131/// which removes the meta instead of cloning it is thinkable.
132///
133/// All extracted metadata is converted into [`TraceItem::attributes`] compatible values,
134/// by building a metadata representation for each top level field and attribute,
135/// serializing the result into JSON and building an appropriate metadata key.
136///
137/// The schema for metadata keys follows the format `sentry._meta.fields.{key}`,
138/// for attributes respectively `sentry._meta.fields.attributes.{key}`.
139fn extract_meta_attributes(log: &OurLog) -> HashMap<String, AnyValue> {
140    let mut meta = IntoValue::extract_child_meta(log);
141    // Attributes are the only 'nested' meta we allow.
142    let attributes = meta.remove("attributes");
143
144    let mut result = HashMap::with_capacity(
145        meta.len()
146            + attributes.as_ref().map_or(0, size_of_meta_tree)
147            // This len is filled later by the [`attributes`] function, it's just to try initialize
148            // the hashamap once with the correct count.
149            + log.attributes.value().map_or(0, |a| a.0.len()),
150    );
151
152    for (key, meta) in meta {
153        let attr = AttributeMeta { meta };
154        if let Some(value) = attr.to_any_value() {
155            let key = format!("sentry._meta.fields.{key}");
156            result.insert(key, value);
157        }
158    }
159
160    let Some(mut attributes) = attributes else {
161        return result;
162    };
163
164    for (key, meta) in std::mem::take(&mut attributes.children) {
165        let attr = AttributeMeta { meta };
166        if let Some(value) = attr.to_any_value() {
167            let key = format!("sentry._meta.fields.attributes.{key}");
168            result.insert(key, value);
169        }
170    }
171
172    // The `attributes` field itself can have metadata attached,
173    // we already took out all the metadata of the children, so now just emit
174    // the remaining metadata on the `attributes`.
175    let meta = AttributeMeta { meta: attributes };
176    if let Some(value) = meta.to_any_value() {
177        result.insert("sentry._meta.fields.attributes".to_owned(), value);
178    }
179
180    result
181}

183/// Calculates the immediate size of the meta tree passed in.
184///
185/// This only counts non empty meta elements of the passed in meta tree and its children,
186/// it does not recursively traverse the children.
187fn size_of_meta_tree(meta: &MetaTree) -> usize {
188    let mut size = 0;
189
190    if !meta.meta.is_empty() {
191        size += 1;
192    }
193    for meta in meta.children.values() {
194        if !meta.meta.is_empty() {
195            size += 1;
196        }
197    }
198
199    size
200}

202/// Fields on the log message which are stored as fields.
203struct FieldAttributes {
204    /// The log level.
205    ///
206    /// See: [`OurLog::level`].
207    level: OurLogLevel,
208    /// The original timestamp when the log was created.
209    ///
210    /// See: [`OurLog::timestamp`].
211    timestamp: relay_event_schema::protocol::Timestamp,
212    /// The log body.
213    ///
214    /// See: [`OurLog::body`].
215    body: String,
216    /// The optionally associated span id.
217    ///
218    /// See: [`OurLog::span_id`].
219    span_id: Option<SpanId>,
220}
221
222/// Extracts all attributes of a log, combines it with extracted meta attributes.
223fn attributes(
224    meta: HashMap<String, AnyValue>,
225    attributes: Attributes,
226    fields: FieldAttributes,
227) -> HashMap<String, AnyValue> {
228    let mut result = meta;
229    // +N, one for each field attribute added and some extra for potential meta.
230    result.reserve(attributes.0.len() + 5 + 3);
231
232    for (name, attribute) in attributes {
233        let meta = AttributeMeta {
234            meta: IntoValue::extract_meta_tree(&attribute),
235        };
236        if let Some(meta) = meta.to_any_value() {
237            result.insert(format!("sentry._meta.fields.attributes.{name}"), meta);
238        }
239
240        let value = attribute
241            .into_value()
242            .and_then(|v| v.value.value.into_value());
243
244        let Some(value) = value else {
245            // Meta has already been handled, no value -> skip.
246            // There are also no current plans to handle `null` in EAP.
247            continue;
248        };
249
250        let Some(value) = (match value {
251            Value::Bool(v) => Some(any_value::Value::BoolValue(v)),
252            Value::I64(v) => Some(any_value::Value::IntValue(v)),
253            Value::U64(v) => i64::try_from(v).ok().map(any_value::Value::IntValue),
254            Value::F64(v) => Some(any_value::Value::DoubleValue(v)),
255            Value::String(v) => Some(any_value::Value::StringValue(v)),
256            // These cases do not happen, as they are not valid attributes
257            // and they should have been filtered out before already.
258            Value::Array(_) | Value::Object(_) => {
259                debug_assert!(false, "unsupported log value");
260                None
261            }
262        }) else {
263            continue;
264        };
265
266        result.insert(name, AnyValue { value: Some(value) });
267    }
268
269    let FieldAttributes {
270        level,
271        timestamp,
272        body,
273        span_id,
274    } = fields;
275    // Unconditionally override any prior set attributes with the same key, as they should always
276    // come from the log itself.
277    //
278    // Ideally these attributes are marked as private in sentry-conventions and potentially
279    // validated against.
280    result.insert(
281        "sentry.severity_text".to_owned(),
282        AnyValue {
283            value: Some(any_value::Value::StringValue(level.to_string())),
284        },
285    );
286    let timestamp_nanos = timestamp
287        .into_inner()
288        .timestamp_nanos_opt()
289        // We can expect valid timestamps at this point, clock drift correction / normalization
290        // should've taken care of this already.
291        .unwrap_or_default();
292    result.insert(
293        "sentry.timestamp_nanos".to_owned(),
294        AnyValue {
295            value: Some(any_value::Value::StringValue(timestamp_nanos.to_string())),
296        },
297    );
298    result.insert(
299        "sentry.timestamp_precise".to_owned(),
300        AnyValue {
301            value: Some(any_value::Value::IntValue(timestamp_nanos)),
302        },
303    );
304    result.insert(
305        "sentry.body".to_owned(),
306        AnyValue {
307            value: Some(any_value::Value::StringValue(body)),
308        },
309    );
310    if let Some(span_id) = span_id {
311        result.insert(
312            "sentry.span_id".to_owned(),
313            AnyValue {
314                value: Some(any_value::Value::StringValue(span_id.to_string())),
315            },
316        );
317    }
318
319    result
320}
