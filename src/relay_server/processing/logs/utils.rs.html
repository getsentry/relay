2
3use crate::envelope::WithHeader;

5/// Returns the calculated size of a [`OurLog`].
6///
7/// Unlike [`relay_ourlogs::calculate_size`], this accesses the already manifested byte size
8/// of the log, instead of calculating it.
9///
10/// When compiled with debug assertion the function asserts the presence of a manifested byte size.
11pub fn get_calculated_byte_size(log: &WithHeader<OurLog>) -> usize {
12    // Use the remembered byte size here, as this is the one Relay stored when initially
13    // receiving the log item. This will not include any modifications done to the log.
14    let bytes = log.header.as_ref().and_then(|header| header.byte_size);
15
16    debug_assert!(
17        bytes.is_some(),
18        "processed logs should always have a byte size assigned"
19    );
20
21    // Only fall back to a calculated size, when there is nothing stored.
22    // This should never happen, logs should have a size assigned to them immediately after
23    // they have been expanded.
24    let bytes = bytes.unwrap_or_else(|| log.value().map_or(1, relay_ourlogs::calculate_size));
25
26    usize::try_from(bytes).unwrap_or(usize::MAX)
27}
