18/// Parses all serialized logs into their [`ExpandedLogs`] representation.
19///
20/// Individual, invalid logs will be discarded.
21pub fn expand(logs: Managed<SerializedLogs>, _ctx: Context<'_>) -> Managed<ExpandedLogs> {
22    let received_at = logs.received_at();
23    let trust = logs.headers.meta().request_trust();
24
25    logs.map(|logs, records| {
26        records.lenient(DataCategory::LogByte);
27
28        let mut all_logs = Vec::with_capacity(logs.count());
29        for logs in logs.logs {
30            let expanded = expand_log_container(&logs, received_at, trust);
31            let expanded = records.or_default(expanded, logs);
32            all_logs.extend(expanded);
33        }
34
35        for otel_log in logs.otel_logs {
36            match expand_otel_log(&otel_log, received_at) {
37                Ok(log) => all_logs.push(log),
38                Err(err) => {
39                    records.reject_err(err, otel_log);
40                    continue;
41                }
42            }
43        }
44
45        ExpandedLogs {
46            headers: logs.headers,
47            #[cfg(feature = "processing")]
48            retention: _ctx.project_info.config.event_retention,
49            logs: all_logs,
50        }
51    })
52}

54/// Processes expanded logs.
55///
56/// Validates, scrubs, normalizes and enriches individual log entries.
57pub fn process(logs: &mut Managed<ExpandedLogs>, ctx: Context<'_>) {
58    logs.modify(|logs, records| {
59        let meta = logs.headers.meta();
60        logs.logs
61            .retain_mut(|log| records.or_default(process_log(log, meta, ctx).map(|_| true), &*log));
62    });
63}

65fn expand_otel_log(item: &Item, received_at: DateTime<Utc>) -> Result<WithHeader<OurLog>> {
66    let log = serde_json::from_slice::<OtelLog>(&item.payload()).map_err(|err| {
67        relay_log::debug!("failed to parse OTel Log: {err}");
68        Error::Invalid(DiscardReason::InvalidJson)
69    })?;
70
71    let log = relay_ourlogs::otel_to_sentry_log(log, received_at).map_err(|err| {
72        relay_log::debug!("failed to convert OTel Log to Sentry Log: {:?}", err);
73        Error::Invalid(DiscardReason::InvalidLog)
74    })?;
75
76    // The OTel log conversion already adds certain Sentry attributes which are included in the
77    // cost here.
78    //
79    // As OTel logs are deprecated and to be removed, this is okay for now.
80    //
81    // See: <https://github.com/getsentry/relay/issues/4884>.
82    let byte_size = Some(relay_ourlogs::calculate_size(&log));
83    Ok(WithHeader {
84        value: Annotated::new(log),
85        header: Some(OurLogHeader {
86            byte_size,
87            other: Default::default(),
88        }),
89    })
90}

92fn expand_log_container(
93    item: &Item,
94    received_at: DateTime<Utc>,
95    trust: RequestTrust,
96) -> Result<ContainerItems<OurLog>> {
97    let mut logs = ItemContainer::parse(item)
98        .map_err(|err| {
99            relay_log::debug!("failed to parse logs container: {err}");
100            Error::Invalid(DiscardReason::InvalidJson)
101        })?
102        .into_items();
103
104    for log in &mut logs {
105        // Calculate the received byte size and remember it as metadata, in the header.
106        // This is to keep track of the originally received size even as Relay adds, removes or
107        // modifies attributes.
108        //
109        // Since Relay can be deployed in multiple layers, we need to remember the size of the
110        // first Relay which received the log, but at the same time we must be able to trust that
111        // size.
112        //
113        // If there is no size calculated or we cannot trust the source, we re-calculate the size.
114        let byte_size = log.header.as_ref().and_then(|h: &OurLogHeader| h.byte_size);
115        if trust.is_untrusted() || matches!(byte_size, None | Some(0)) {
116            let byte_size = log.value().map(relay_ourlogs::calculate_size).unwrap_or(1);
117            log.header.get_or_insert_default().byte_size = Some(byte_size);
118        }
119
120        relay_ourlogs::ourlog_merge_otel(log, received_at);
121    }
122
123    Ok(logs)
124}

126fn process_log(log: &mut Annotated<OurLog>, meta: &RequestMeta, ctx: Context<'_>) -> Result<()> {
127    scrub(log, ctx).inspect_err(|err| {
128        relay_log::debug!("failed to scrub pii from log: {err}");
129    })?;
130
131    normalize(log, meta).inspect_err(|err| {
132        relay_log::debug!("failed to normalize log: {err}");
133    })?;
134
135    Ok(())
136}

138fn scrub(log: &mut Annotated<OurLog>, ctx: Context<'_>) -> Result<()> {
139    let pii_config = ctx
140        .project_info
141        .config
142        .datascrubbing_settings
143        .pii_config()
144        .map_err(|e| Error::PiiConfig(e.clone()))?;
145
146    if let Some(ref config) = ctx.project_info.config.pii_config {
147        let mut processor = PiiProcessor::new(config.compiled());
148        process_value(log, &mut processor, ProcessingState::root())?;
149    }
150
151    if let Some(config) = pii_config {
152        let mut processor = PiiProcessor::new(config.compiled());
153        process_value(log, &mut processor, ProcessingState::root())?;
154    }
155
156    Ok(())
157}

159fn normalize(log: &mut Annotated<OurLog>, meta: &RequestMeta) -> Result<()> {
160    process_value(log, &mut SchemaProcessor, ProcessingState::root())?;
161
162    let Some(log) = log.value_mut() else {
163        return Err(Error::Invalid(DiscardReason::NoData));
164    };
165
166    process_attribute_types(log);
167    populate_ua_fields(log, meta.user_agent(), meta.client_hints().as_deref());
168
169    Ok(())
170}

172fn populate_ua_fields(log: &mut OurLog, user_agent: Option<&str>, client_hints: ClientHints<&str>) {
173    let attributes = log.attributes.get_or_insert_with(Default::default);
174
175    const BROWSER_NAME: &str = "sentry.browser.name";
176    const BROWSER_VERSION: &str = "sentry.browser.version";
177
178    if attributes.contains_key(BROWSER_NAME) && attributes.contains_key(BROWSER_VERSION) {
179        return;
180    }
181
182    let Some(context) = BrowserContext::from_hints_or_ua(&RawUserAgentInfo {
183        user_agent,
184        client_hints,
185    }) else {
186        return;
187    };
188
189    if let Some(name) = context.name.into_value() {
190        attributes.insert_if_missing(BROWSER_NAME, || name);
191    }
192    if let Some(version) = context.version.into_value() {
193        attributes.insert_if_missing(BROWSER_VERSION, || version);
194    }
195}

197fn process_attribute_types(ourlog: &mut OurLog) {
198    let Some(attributes) = ourlog.attributes.value_mut() else {
199        return;
200    };
201
202    let attributes = attributes.iter_mut().map(|(_, attr)| attr);
203
204    for attribute in attributes {
205        use AttributeType::*;
206
207        let Some(inner) = attribute.value_mut() else {
208            continue;
209        };
210
211        match (&mut inner.value.ty, &mut inner.value.value) {
212            (Annotated(Some(Boolean), _), Annotated(Some(Value::Bool(_)), _)) => (),
213            (Annotated(Some(Integer), _), Annotated(Some(Value::I64(_)), _)) => (),
214            (Annotated(Some(Integer), _), Annotated(Some(Value::U64(_)), _)) => (),
215            (Annotated(Some(Double), _), Annotated(Some(Value::I64(_)), _)) => (),
216            (Annotated(Some(Double), _), Annotated(Some(Value::U64(_)), _)) => (),
217            (Annotated(Some(Double), _), Annotated(Some(Value::F64(_)), _)) => (),
218            (Annotated(Some(String), _), Annotated(Some(Value::String(_)), _)) => (),
219            // Note: currently the mapping to Kafka requires that invalid or unknown combinations
220            // of types and values are removed from the mapping.
221            //
222            // Usually Relay would only modify the offending values, but for now, until there
223            // is better support in the pipeline here, we need to remove the entire attribute.
224            (Annotated(Some(Unknown(_)), _), _) => {
225                let original = attribute.value_mut().take();
226                attribute.meta_mut().add_error(ErrorKind::InvalidData);
227                attribute.meta_mut().set_original_value(original);
228            }
229            (Annotated(Some(_), _), Annotated(Some(_), _)) => {
230                let original = attribute.value_mut().take();
231                attribute.meta_mut().add_error(ErrorKind::InvalidData);
232                attribute.meta_mut().set_original_value(original);
233            }
234            (Annotated(None, _), _) | (_, Annotated(None, _)) => {
235                let original = attribute.value_mut().take();
236                attribute.meta_mut().add_error(ErrorKind::MissingAttribute);
237                attribute.meta_mut().set_original_value(original);
238            }
239        }
240    }

437            &Value::String("131.0.0".to_owned()),
