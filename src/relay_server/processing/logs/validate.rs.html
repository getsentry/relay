1use crate::processing::logs::{Error, SerializedLogs};
2use crate::processing::{Managed, Rejected};
3
4/// Validates that there is only a single log container processed at a time.
5///
6/// The `Log` item must always be sent as an `ItemContainer`, currently it is not allowed to
7/// send multiple containers for logs.
8///
9/// This restriction may be lifted in the future.
10///
11/// This limit mostly exists to incentivise SDKs to batch multiple logs into a single container,
12/// technically it can be removed without issues.
13pub fn container(logs: &Managed<SerializedLogs>) -> Result<(), Rejected<Error>> {
14    // It's fine if there was no log container, as we still accept OTel logs.
15    if logs.logs.len() > 1 {
16        return Err(logs.reject_err(Error::DuplicateContainer));
17    }
18
19    Ok(())
20}
