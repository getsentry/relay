5/// Validates that there is only a single log container processed at a time.
6///
7/// The `Log` item must always be sent as an `ItemContainer`, currently it is not allowed to
8/// send multiple containers for logs.
9///
10/// This restriction may be lifted in the future.
11///
12/// This limit mostly exists to incentivise SDKs to batch multiple logs into a single container,
13/// technically it can be removed without issues.
14pub fn container(logs: &Managed<SerializedLogs>) -> Result<()> {
15    // It's fine if there was no log container, as we still accept OTel logs.
16    if logs.logs.len() > 1 {
17        return Err(Error::DuplicateContainer);
18    }
19
20    Ok(())
21}

23/// Validate that the envelope has no trace context header.
24///
25/// For now, this only emits a metric so we can verify that logs envelopes do not
26/// contain a trace context.
27pub fn dsc(logs: &Managed<SerializedLogs>) {
28    relay_statsd::metric!(
29        counter(RelayCounters::EnvelopeWithLogs) += 1,
30        dsc = match logs.headers.dsc() {
31            Some(_) => "yes",
32            None => "no",
33        }
34    )
35}
