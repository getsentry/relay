7#[derive(Debug, thiserror::Error)]
8#[error("the item is not allowed/supported in this envelope")]
9pub struct UnsupportedItem;
10
11impl OutcomeError for UnsupportedItem {
12    type Error = Self;
13
14    fn consume(self) -> (Option<crate::services::outcome::Outcome>, Self::Error) {
15        (Some(Outcome::Invalid(DiscardReason::InvalidEnvelope)), self)
16    }
17}

19/// A processor which only forwards unknown envelope items.
20///
21/// For compatibility
22#[derive(Debug)]
23pub struct ForwardUnknownProcessor {}
24
25impl ForwardUnknownProcessor {
26    /// Creates a new [`Self`].
27    pub fn new() -> Self {
28        Self {}
29    }
30}
31
32impl processing::Processor for ForwardUnknownProcessor {
33    type Input = UnknownItems;
34    type Output = ForwardUnknownOutput;
35    type Error = UnsupportedItem;
36
37    fn prepare_envelope(&self, envelope: &mut ManagedEnvelope) -> Option<Managed<Self::Input>> {
38        let unknown_items = envelope
39            .envelope_mut()
40            .take_items_by(|item| matches!(*item.ty(), ItemType::Unknown(_)))
41            .into_vec();
42
43        if unknown_items.is_empty() {
44            return None;
45        }
46
47        Some(Managed::with_meta_from(
48            envelope,
49            UnknownItems {
50                headers: envelope.envelope().headers().clone(),
51                unknown_items,
52            },
53        ))
54    }
55
56    async fn process(
57        &self,
58        unknown_items: Managed<Self::Input>,
59        ctx: Context<'_>,
60    ) -> Result<Output<Self::Output>, Rejected<Self::Error>> {
61        if !ctx.config.accept_unknown_items() {
62            return Err(unknown_items.reject_err(UnsupportedItem));
63        }
64
65        Ok(Output::just(ForwardUnknownOutput(unknown_items)))
66    }
67}
68
69/// Output produced by [`ForwardUnknownProcessor`].
70#[derive(Debug)]
71pub struct ForwardUnknownOutput(Managed<UnknownItems>);
72
73impl Forward for ForwardUnknownOutput {
74    fn serialize_envelope(
75        self,
76        _: processing::ForwardContext<'_>,
77    ) -> Result<Managed<Box<Envelope>>, Rejected<()>> {
78        let Self(ui) = self;
79        Ok(ui.map(|ui, _| Envelope::from_parts(ui.headers, Items::from_vec(ui.unknown_items))))
80    }
81
82    #[cfg(feature = "processing")]
83    fn forward_store(
84        self,
85        _s: processing::forward::StoreHandle<'_>,
86        _ctx: processing::ForwardContext<'_>,
87    ) -> Result<(), Rejected<()>> {
88        Err(self.0.reject_err(UnsupportedItem).map(drop))
89    }
90}
