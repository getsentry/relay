9/// Maximum expected segment ID for a replay session, under normal operation.
10const MAX_SEGMENTS_ID: u64 = 720;
11
12/// Filters replays sent for a project which does not allow replay ingestion.
13pub fn feature_flag(
14    replays: Managed<SerializedReplays>,
15    ctx: Context<'_>,
16) -> Result<Managed<SerializedReplays>, Rejected<Error>> {
17    match ctx.should_filter(Feature::SessionReplay)
18        || (ctx
19            .project_info
20            .has_feature(Feature::SessionReplayVideoDisabled)
21            && !replays.videos.is_empty())
22    {
23        true => Err(replays.reject_err(Error::FilterFeatureFlag)),
24        false => Ok(replays),
25    }
26}

28/// Applies inbound filters to individual replays.
29pub fn filter(replays: &mut Managed<ExpandedReplays>, ctx: Context<'_>) {
30    let client_addr = replays.headers.meta().client_addr();
31    let event_id = replays.headers.event_id();
32
33    replays.retain(
34        |replays| &mut replays.replays,
35        |replay, _| {
36            let Some(event) = replay.event_mut() else {
37                return Ok(());
38            };
39            let event = event.value().ok_or(Error::NoEventContent)?;
40
41            relay_filter::should_filter(
42                event,
43                client_addr,
44                &ctx.project_info.config.filter_settings,
45                ctx.global_config.filters(),
46            )
47            .map_err(Error::Filtered)?;
48
49            // Log segments that exceed the hour limit so we can diagnose errant SDKs
50            // or exotic customer implementations.
51            if let Some(segment_id) = event.segment_id.value()
52                && *segment_id > MAX_SEGMENTS_ID
53            {
54                metric!(counter(RelayCounters::ReplayExceededSegmentLimit) += 1);
55
56                relay_log::debug!(
57                    event_id = ?event_id,
58                    project_id = ctx.project_info.project_id.map(|v| v.value()),
59                    organization_id = ctx.project_info.organization_id.map(|o| o.value()),
60                    segment_id = segment_id,
61                    "replay segment-exceeded-limit"
62                );
63            }
64
65            Ok::<_, Error>(())
66        },
67    )
68}
