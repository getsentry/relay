14/// Errors that can occur when serializing an expanded replay into envelope items.
15#[derive(Debug, thiserror::Error)]
16enum SerializeReplayError {
17    /// Failed to serialize the replay event to JSON.
18    #[error("json serialization failed")]
19    Json(#[from] serde_json::Error),
20    /// Failed to serialize the replay video event to MessagePack.
21    #[error("msgpack serialization failed")]
22    MsgPack(#[from] rmp_serde::encode::Error),
23}
24
25impl Forward for ReplaysOutput {
26    fn serialize_envelope(
27        self,
28        _ctx: processing::ForwardContext<'_>,
29    ) -> Result<Managed<Box<Envelope>>, Rejected<()>> {
30        Ok(self.0.map(|replays, records| {
31            let ExpandedReplays { headers, replays } = replays;
32            let mut items = Items::new();
33
34            for replay in replays {
35                match serialize_replay(&replay) {
36                    Ok(replay_items) => items.extend(replay_items),
37                    Err(error) => {
38                        relay_log::error!(
39                            error = &error as &dyn std::error::Error,
40                            event_id = ?headers.event_id(),
41                            "failed to serialize replay"
42                        );
43                        records.reject_err(Error::FailedToSerializeReplay, replay);
44                    }
45                }
46            }
47
48            Envelope::from_parts(headers, items)
49        }))
50    }
51
52    #[cfg(feature = "processing")]
53    fn forward_store(
54        self,
55        s: processing::StoreHandle<'_>,
56        ctx: processing::ForwardContext<'_>,
57    ) -> Result<(), Rejected<()>> {
58        let envelope = self.serialize_envelope(ctx)?;
59        let envelope = ManagedEnvelope::from(envelope).into_processed();
60
61        s.store(crate::services::store::StoreEnvelope { envelope });
62
63        Ok(())
64    }
65}

67fn create_replay_event_item(payload: Bytes) -> Item {
68    let mut item = Item::new(ItemType::ReplayEvent);

85fn serialize_replay(replay: &ExpandedReplay) -> Result<SmallVec<[Item; 2]>, SerializeReplayError> {
86    match replay {
87        ExpandedReplay::WebReplay { event, recording } => {
88            let event_bytes: Bytes = event.to_json()?.into_bytes().into();
89
90            Ok(smallvec![
91                create_replay_event_item(event_bytes),
92                create_replay_recording_item(recording.clone()),
93            ])
94        }
95        ExpandedReplay::NativeReplay {
96            event,
97            recording,
98            video,
99        } => {
100            let event_bytes: Bytes = event.to_json()?.into_bytes().into();
101
102            let video_event = ReplayVideoEvent {
103                replay_event: event_bytes,
104                replay_recording: recording.clone(),
105                replay_video: video.clone(),
106            };
107
108            let payload = rmp_serde::to_vec_named(&video_event)?;
109            Ok(smallvec![create_replay_video_item(payload.into())])
110        }
111    }
