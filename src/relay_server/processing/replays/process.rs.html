20fn expand_video(item: &Item) -> Result<ExpandedReplay, Error> {
21    let ReplayVideoEvent {
22        replay_event: event,
23        replay_recording: recording,
24        replay_video: video,
25    } = rmp_serde::from_slice(&item.payload()).map_err(|_| Error::InvalidReplayVideoEvent)?;
26
27    if video.is_empty() {
28        return Err(Error::InvalidReplayVideoEvent);
29    }
30
31    let event = Annotated::<Replay>::from_json_bytes(&event)?;
32
33    Ok(ExpandedReplay::NativeReplay {
34        event,
35        recording,
36        video,
37    })
38}

40/// Parses all serialized replays into their [`ExpandedReplays`] representation.
41///
42/// Discards items if they are invalid or there is an invalid combination/amount.
43pub fn expand(replays: Managed<SerializedReplays>) -> Managed<ExpandedReplays> {
44    replays.map(|replays, records| {
45        let SerializedReplays {
46            headers,
47            events,
48            recordings,
49            videos,
50        } = replays;
51        let mut replays = Vec::new();
52
53        // There should be at most one event and recording and if there is one there needs to be
54        // one of the other
55        match (events.as_slice(), recordings.as_slice()) {
56            // Valid case (no 'web replays') if the envelope contains some replay_videos ('native replays')
57            ([], []) => (),
58            ([event], [recording]) => {
59                let event_bytes = event.payload();
60                let recording_bytes = recording.payload();
61
62                match Annotated::<Replay>::from_json_bytes(&event_bytes) {
63                    Ok(event) => {
64                        replays.push(ExpandedReplay::WebReplay {
65                            event,
66                            recording: recording_bytes,
67                        });
68                    }
69                    Err(err) => {
70                        records.reject_err(Error::from(err), SmallVec::from([event, recording]));
71                    }
72                }
73            }
74            (a, b) => {
75                relay_log::error!(
76                    sdk = headers.meta().client().unwrap_or("unknown"),
77                    event_count = a.len(),
78                    recording_count = b.len(),
79                    "replay item recording mismatch"
80                );
81
82                for item in a {
83                    records.reject_err(Error::InvalidItemCount, item);
84                }
85                for item in b {
86                    records.reject_err(Error::InvalidItemCount, item);
87                }
88            }
89        }
90
91        // From the SDKs it seems like there will only be one video per envelope but that is not
92        // clearly specified anywhere. Also it seems like their will always only be a video or a
93        // (event, recording).
94        // Currently the logic still allows for multiple videos per envelope as well as 'native' and
95        // 'web' replays in the same envelope, in the future we could be more strict on this.
96        for video in &videos {
97            match expand_video(video) {
98                Ok(replay) => replays.push(replay),
99                Err(err) => drop(records.reject_err(err, video)),
100            }
101        }
102
103        if replays.len() > 1 {
104            relay_log::error!(
105                sdk = headers.meta().client().unwrap_or("unknown"),
106                event_count = events.len(),
107                recording_count = recordings.len(),
108                video_count = videos.len(),
109                "multiple replay items in same envelope"
110            );
111        }
112
113        ExpandedReplays { headers, replays }
114    })
115}

117/// Normalizes individual replay events.
118pub fn normalize(replays: &mut Managed<ExpandedReplays>, geoip_lookup: &GeoIpLookup) {
119    let meta = replays.headers.meta();
120    let client_addr = meta.client_addr();
121    let user_agent = RawUserAgentInfo {
122        user_agent: meta.user_agent().map(String::from),
123        client_hints: meta.client_hints().to_owned(),
124    };
125
126    replays.modify(|replay, _| {
127        for replay in replay.replays.iter_mut() {
128            replay::normalize(
129                replay.get_event(),
130                client_addr,
131                &user_agent.as_deref(),
132                geoip_lookup,
133            )
134        }
135    })
136}

138fn scrub_event(event: &mut Annotated<Replay>, ctx: Context<'_>) -> Result<(), Error> {
139    if let Some(ref config) = ctx.project_info.config.pii_config {
140        let mut processor = PiiProcessor::new(config.compiled());
141        processor::process_value(event, &mut processor, ProcessingState::root())?;
142    }
143
144    let pii_config = ctx
145        .project_info
146        .config
147        .datascrubbing_settings
148        .pii_config()
149        .map_err(|e| Error::PiiConfig(e.clone()))?;
150
151    if let Some(config) = pii_config {
152        let mut processor = PiiProcessor::new(config.compiled());
153        processor::process_value(event, &mut processor, ProcessingState::root())?;
154    };
155    Ok(())
156}

158fn scrub_recordings(
159    replays: &mut Managed<ExpandedReplays>,
160    ctx: Context<'_>,
161) -> Result<(), Rejected<Error>> {
162    let event_id = replays.headers.event_id();
163    let pii_config = match ctx.project_info.config.datascrubbing_settings.pii_config() {
164        Ok(config) => config.as_ref(),
165        Err(e) => return Err(replays.reject_err(Error::PiiConfig(e.clone()))),
166    };
167
168    replays.retain(
169        |replays| &mut replays.replays,
170        |replay, _| {
171            // Has some internal state so don't move out of the retain.
172            let mut scrubber = RecordingScrubber::new(
173                ctx.config.max_replay_uncompressed_size(),
174                ctx.project_info.config.pii_config.as_ref(),
175                pii_config,
176            );
177
178            if scrubber.is_empty() {
179                return Ok::<(), Error>(());
180            }
181
182            let payload = replay.get_recording();
183            *payload = metric!(timer(RelayTimers::ReplayRecordingProcessing), {
184                scrubber.process_recording(payload)
185            })
186            .map(Into::into)
187            .map_err(|error| {
188                relay_log::debug!(
189                    error = &error as &dyn std::error::Error,
190                    event_id = ?event_id,
191                    project_id = ctx.project_info.project_id.map(|v| v.value()),
192                    organization_id = ctx.project_info.organization_id.map(|o| o.value()),
193                    "invalid replay recording"
194                );
195                Error::InvalidReplayRecordingEvent
196            })?;
197            Ok::<(), Error>(())
198        },
199    );
200    Ok(())
201}

203/// Applies PII scrubbing to individual replay events and recordings.
204///
205/// Will reject the entire envelope if loading the pii config fails for replay_recordings.
206pub fn scrub(
207    replays: &mut Managed<ExpandedReplays>,
208    ctx: Context<'_>,
209) -> Result<(), Rejected<Error>> {
210    replays.retain(
211        |replays| &mut replays.replays,
212        |replay, _| {
213            scrub_event(replay.get_event(), ctx)
214                .inspect_err(|err| relay_log::debug!("failed to scrub pii from replay: {err}"))
215        },
216    );
217
218    if ctx
219        .project_info
220        .has_feature(Feature::SessionReplayRecordingScrubbing)
221    {
222        scrub_recordings(replays, ctx)
223    } else {
224        Ok(())
225    }
