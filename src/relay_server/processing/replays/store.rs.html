9/// Reserved bytes for message metadata.
10const MESSAGE_METADATA_OVERHEAD: usize = 2000;
11
12/// Context parameters for [`convert`].
13#[derive(Debug, Clone, Copy)]
14pub struct Context {
15    /// The event ID.
16    pub event_id: EventId,
17    /// Item retention in days.
18    pub retention: u16,
19    /// Maximum allowed size for a replay recording Kafka message.
20    pub max_replay_message_size: usize,
21}
22
23/// Converts an [`ExpandedReplay`] into a storable [`StoreReplay`].
24///
25/// Fails if the event can not be serialized or the created message is too large for the consumer.
26pub fn convert(replay: ExpandedReplay, ctx: &Context) -> Result<StoreReplay, Error> {
27    let (recording, event, video) = match replay {
28        ExpandedReplay::StandaloneRecording { recording } => (recording, None, None),
29        ExpandedReplay::WebReplay { event, recording } => (recording, Some(event), None),
30        ExpandedReplay::NativeReplay {
31            event,
32            recording,
33            video,
34        } => (recording, Some(event), Some(video)),
35    };
36    let event = event.map(serialize_event).transpose()?;
37
38    // Size of the consumer message. We can be reasonably sure this won't overflow because
39    // of the request size validation provided by Nginx and Relay.
40    let mut payload_size = MESSAGE_METADATA_OVERHEAD;
41    payload_size += event.as_ref().map_or(0, |b| b.len());
42    payload_size += video.as_ref().map_or(0, |b| b.len());
43    payload_size += recording.len();
44
45    if payload_size >= ctx.max_replay_message_size {
46        relay_log::debug!("replay_recording over maximum size.");
47        return Err(Error::TooLarge);
48    }
49
50    Ok(StoreReplay {
51        event_id: ctx.event_id,
52        retention_days: ctx.retention,
53        recording,
54        event,
55        video,
56    })
57}

59fn serialize_event(replay: Annotated<Replay>) -> Result<Bytes, Error> {
60    replay
61        .to_json()
62        .map_err(|_| Error::FailedToSerializeReplay)
63        .map(|json| json.into_bytes().into())
64}
