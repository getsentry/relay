30    fn try_expand(items: &mut Vec<Item>, ctx: Context<'_>) -> Result<Option<Expansion<Self>>> {
31        let report = if let Some(item) = utils::take_item_of_type(items, ItemType::UnrealReport) {
32            UnrealReport::Report(item)
33        } else {
34            let items: Vec<Item> = utils::take_items_by(items, |i| i.is_unreal_expanded());
35            if items.is_empty() {
36                return Ok(None);
37            }
38            UnrealReport::Expanded(items)
39        };
40
41        let mut metrics = Default::default();
42
43        let expansion = utils::if_not_processing!(ctx, {
44            Expansion {
45                event: Box::new(utils::take_event_from_crash_items(items, &mut metrics, ctx)?),
46                attachments: utils::take_items_of_type(items, ItemType::Attachment),
47                user_reports: utils::take_items_of_type(items, ItemType::UserReport),
48                error: Self::Forward(report),
49                metrics,
50                fully_normalized: false,
51            }
52        } else {
53            use crate::envelope::AttachmentType;
54            use crate::services::processor::ProcessingError;
55
56            let expansion = match report {
57                UnrealReport::Report(item) => crate::utils::expand_unreal(item.payload(), ctx.processing.config)?,
58                UnrealReport::Expanded(report_items) => crate::utils::expand_unreal_items(report_items.into())?,
59            };
60
61            let event = expansion.event;
62            let mut attachments = expansion.attachments.into_vec();
63            attachments.extend(items.extract_if(.., |item| *item.ty() == ItemType::Attachment));
64
65            // `process` later fills this event in, ideally the event is already filled in here,
66            // during the expansion, it is split into two phases now, to keep compatibility with
67            // the existing unreal code.
68            let mut event = match utils::take_item_of_type(items, ItemType::Event).or(event) {
69                Some(event) => utils::event_from_json_payload(event, None, &mut metrics, ctx)?,
70                None => utils::take_event_from_attachments(&mut attachments, &mut metrics, ctx)?,
71            };
72
73            let mut user_reports: Vec<Item> = utils::take_items_of_type(items, ItemType::UserReport);
74
75            let event_id = ctx.envelope.event_id().unwrap_or_default();
76            debug_assert_ne!(event_id, Default::default(), "event id must always be set");
77
78            let user_header = ctx
79                .envelope
80                .get_header(crate::constants::UNREAL_USER_HEADER)
81                .and_then(|v| v.as_str());
82
83            // After the removal of the old error/event processing pipeline, the `expand_unreal` and
84            // `process_unreal` functions can be significantly reworked to no longer parse the Unreal
85            // context multiple times by merging the functions into one.
86            //
87            // This is currently still split to avoid too many changes and code duplication at once.
88            if let Some(result) =
89                crate::utils::process_unreal(event_id, &mut event, &attachments, user_header)
90                    .map_err(ProcessingError::InvalidUnrealReport)?
91            {
92                user_reports.extend(result.user_reports);
93            }
94
95            let minidump = utils::take_item_by(&mut attachments, |item| {
96                item.attachment_type() == Some(AttachmentType::Minidump)
97            });
98            let apple_crash_report = utils::take_item_by(&mut attachments, |item| {
99                item.attachment_type() == Some(AttachmentType::AppleCrashReport)
100            });
101
102            if let Some(minidump) = &minidump {
103                crate::utils::process_minidump(
104                    event.get_or_insert_with(Default::default),
105                    &minidump.payload(),
106                );
107                metrics.bytes_ingested_event_minidump = (minidump.len() as u64).into();
108            }
109            if let Some(acr) = &apple_crash_report {
110                crate::utils::process_apple_crash_report(
111                    event.get_or_insert_with(Default::default)
112                );
113                metrics.bytes_ingested_event_applecrashreport = (acr.len() as u64).into();
114            }
115
116            Expansion {
117                event: Box::new(event),
118                attachments,
119                user_reports,
120                error: Self::Process {
121                    minidump,
122                    apple_crash_report,
123                },
124                metrics,
125                fully_normalized: false,
126            }
127        });
128
129        Ok(Some(expansion))
130    }

142    #[cfg(feature = "processing")]
143    fn apply_rate_limit(
144        &mut self,
145        _category: relay_quotas::DataCategory,
146        limits: relay_quotas::RateLimits,
147        records: &mut RecordKeeper<'_>,
148    ) -> Result<()> {
149        use crate::processing::errors::Error;
150
151        let Self::Process {
152            minidump,
153            apple_crash_report,
154        } = self
155        else {
156            return Ok(());
157        };
158
159        if let Some(apple_crash_report) = apple_crash_report
160            && !apple_crash_report.rate_limited()
161        {
162            apple_crash_report.set_rate_limited(true);
163            records.reject_err(Error::RateLimited(limits.clone()), &*apple_crash_report);
164        }
165
166        if let Some(minidump) = minidump
167            && !minidump.rate_limited()
168        {
169            minidump.set_rate_limited(true);
170            records.reject_err(Error::RateLimited(limits.clone()), &*minidump);
171        }
172
173        Ok(())
174    }

176    fn serialize_into(self, items: &mut Vec<Item>, _ctx: ForwardContext<'_>) -> Result<()> {
177        match self {
178            Self::Forward(UnrealReport::Report(report)) => items.push(report),
179            Self::Forward(UnrealReport::Expanded(report_items)) => items.extend(report_items),
180            #[cfg(feature = "processing")]
181            Self::Process {
182                minidump,
183                apple_crash_report,
184            } => {
185                items.extend(minidump);
186                items.extend(apple_crash_report);
187            }
188        }
189
190        Ok(())
191    }
192
193    fn minidump_mut(&mut self) -> Option<&mut Item> {
194        match self {
195            Self::Forward(..) => None,
196            #[cfg(feature = "processing")]
197            Self::Process { minidump, .. } => minidump.as_mut(),
198        }
199    }

202impl Counted for Unreal {
203    fn quantities(&self) -> Quantities {
204        match self {
205            Self::Forward(UnrealReport::Report(..)) => Quantities::default(),
206            Self::Forward(UnrealReport::Expanded(items)) => items.quantities(),
207            #[cfg(feature = "processing")]
208            Self::Process {
209                minidump,
210                apple_crash_report,
211            } => {
212                let mut quantities = Quantities::default();
213
214                if let Some(apple_crash_report) = apple_crash_report
215                    && !apple_crash_report.rate_limited()
216                {
217                    quantities.extend(apple_crash_report.quantities());
218                }
219                if let Some(minidump) = minidump
220                    && !minidump.rate_limited()
221                {
222                    quantities.extend(minidump.quantities());
223                }
224
225                quantities
226            }
227        }
228    }
