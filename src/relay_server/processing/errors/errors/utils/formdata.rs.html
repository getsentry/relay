6pub fn merge_formdata(target: &mut SerdeValue, item: &Item) {
7    let payload = item.payload();
8    let mut aggregator = ChunkedFormDataAggregator::new();
9
10    for entry in FormDataIter::new(&payload) {
11        if entry.key() == "sentry" || entry.key().starts_with("sentry___") {
12            // Custom clients can submit longer payloads and should JSON encode event data into
13            // the optional `sentry` field or a `sentry___<namespace>` field.
14            match serde_json::from_str(entry.value()) {
15                Ok(event) => utils::merge_values(target, event),
16                Err(_) => relay_log::debug!("invalid json event payload in sentry form field"),
17            }
18        } else if let Some(index) = utils::get_sentry_chunk_index(entry.key(), "sentry__") {
19            // Electron SDK splits up long payloads into chunks starting at sentry__1 with an
20            // incrementing counter. Assemble these chunks here and then decode them below.
21            aggregator.insert(index, entry.value());
22        } else if let Some(keys) = utils::get_sentry_entry_indexes(entry.key()) {
23            // Try to parse the nested form syntax `sentry[key][key]` This is required for the
24            // Breakpad client library, which only supports string values of up to 64
25            // characters.
26            utils::update_nested_value(target, &keys, entry.value());
27        } else {
28            // Merge additional form fields from the request with `extra` data from the event
29            // payload and set defaults for processing. This is sent by clients like Breakpad or
30            // Crashpad.
31            utils::update_nested_value(target, &["extra", entry.key()], entry.value());
32        }
33    }
34
35    if !aggregator.is_empty() {
36        match serde_json::from_str(&aggregator.join()) {
37            Ok(event) => utils::merge_values(target, event),
38            Err(_) => relay_log::debug!("invalid json event payload in sentry__* form fields"),
39        }
40    }
41}

44mod tests {

46    use crate::envelope::{ContentType, ItemType};
47    use crate::utils::FormDataWriter;

49    fn form_data_item(fields: &[(&str, &str)]) -> Item {
50        let mut writer = FormDataWriter::new();
51        for (key, value) in fields {
52            writer.append(key, value);
53        }
54        let mut item = Item::new(ItemType::FormData);
55        item.set_payload(ContentType::Text, writer.into_inner());
56        item
57    }
58
59    #[test]
60    fn test_merge_formdata_assembles_chunks() {
61        let item = form_data_item(&[("sentry__0", r#"{"message":"#), ("sentry__1", r#""hi"}"#)]);
62        let mut target = SerdeValue::Object(Default::default());
63
64        merge_formdata(&mut target, &item);
65
66        assert_eq!(target, serde_json::json!({ "message": "hi" }));
67    }
68
69    #[test]
70    fn test_merge_formdata_assembles_disjoint_chunks() {
71        let item = form_data_item(&[
72            ("sentry__0", r#"{"message":"#),
73            ("sentry__4000000000", r#""hi"}"#),
74        ]);
75        let mut target = SerdeValue::Object(Default::default());
76
77        merge_formdata(&mut target, &item);
78
79        assert_eq!(target, serde_json::json!({ "message": "hi" }));
80    }
