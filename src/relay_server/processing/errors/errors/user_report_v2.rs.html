13impl SentryError for UserReportV2 {

18    fn try_expand(items: &mut Vec<Item>, ctx: Context<'_>) -> Result<Option<Expansion<Self>>> {
19        let Some(ev) = utils::take_item_of_type(items, ItemType::UserReportV2) else {
20            return Ok(None);
21        };
22
23        let mut metrics = Default::default();
24
25        let attachments: Vec<Item> = utils::take_items_of_type(items, ItemType::Attachment);
26
27        relay_statsd::metric!(
28            counter(RelayCounters::FeedbackAttachments) += attachments.len() as u64
29        );
30
31        Ok(Some(Expansion {
32            event: Box::new(utils::event_from_json_payload(
33                ev,
34                EventType::UserReportV2,
35                &mut metrics,
36                ctx,
37            )?),
38            attachments,
39            user_reports: Default::default(),
40            error: Self,
41            metrics,
42            fully_normalized: false,
43        }))
44    }
