14/// Expands an (error) event envelope.
15///
16/// Identifies the underlying type of error/event and expands it into an event and its parts.
17///
18/// For example an crash report attachment may be expanded into an error event, multiple other
19/// attachments and some user feedback.
20pub fn expand(
21    error: Managed<SerializedError>,
22    ctx: Context<'_>,
23) -> Result<Managed<ExpandedError>, Rejected<Error>> {
24    error.try_map(|error, records| do_expand(error, ctx, records))
25}
26
27fn do_expand(
28    mut error: SerializedError,
29    ctx: Context<'_>,
30    records: &mut RecordKeeper<'_>,
31) -> Result<ExpandedError> {
32    let is_trusted = error.headers.meta().request_trust().is_trusted();
33
34    // Certain attachment types are dissolved into different types (Nintendo Switch),
35    // or are created from other types (unreal).
36    records.lenient(DataCategory::Attachment);
37    records.lenient(DataCategory::AttachmentItem);
38    // User feedback is extracted from unreal reports.
39    records.lenient(DataCategory::UserReportV2);
40
41    let Some(parsed) = ErrorKind::try_expand(
42        &mut error.items,
43        errors::Context {
44            envelope: &error.headers,
45            processing: ctx,
46        },
47    )?
48    else {
49        return Err(ProcessingError::NoEventPayload.into());
50    };
51
52    // All unprocessed items which would create an event but were not turned into an event are
53    // duplicates.
54    //
55    // E.g. if you send an envelope with two error events, you end up here.
56    //
57    // It is reasonable to just reject such envelopes, this is ported over from a prior
58    // implementation of the error processing pipeline.
59    for item in error.items.extract_if(.., |item| item.creates_event()) {
60        records.reject_err(ProcessingError::DuplicateItem(item.ty().clone()), item);
61    }
62
63    if ctx.is_processing() {
64        // In processing, there are cannot be any leftover events due to forward compatibility.
65        for item in std::mem::take(&mut error.items) {
66            records.reject_err(ProcessingError::UnsupportedItem, item);
67        }
68    }
69
70    Ok(ExpandedError {
71        headers: error.headers,
72        fully_normalized: EventFullyNormalized(is_trusted && parsed.fully_normalized),
73        metrics: parsed.metrics,
74        event: parsed.event,
75        attachments: parsed.attachments,
76        user_reports: parsed.user_reports,
77        data: parsed.error,
78        other: error.items,
79    })
80}

82pub fn process(error: &mut Managed<ExpandedError>) -> Result<(), Rejected<Error>> {
83    error.try_modify(|error, records| {
84        process_user_reports(&mut error.user_reports, records);
85
86        Ok::<_, Error>(())
87    })
88}

128pub fn scrub(error: &mut Managed<ExpandedError>, ctx: Context<'_>) -> Result<(), Rejected<Error>> {
129    error.try_modify(|error, records| {
130        processing::utils::event::scrub(&mut error.event, ctx.project_info)?;
131        processing::utils::attachments::scrub(
132            error.attachments.iter_mut(),
133            ctx.project_info,
134            Some(records),
135        );
136        if let Some(minidump) = error.data.minidump_mut() {
137            processing::utils::attachments::scrub(
138                std::iter::once(minidump),
139                ctx.project_info,
140                Some(records),
141            );
142        }
143
144        Ok::<_, Error>(())
145    })
146}

148/// Validates and normalizes all user report items in the envelope.
149///
150/// User feedback items are removed from the envelope if they contain invalid JSON or if the
151/// JSON violates the schema (basic type validation). Otherwise, their normalized representation
152/// is written back into the item.
153fn process_user_reports(user_reports: &mut Vec<Item>, records: &mut RecordKeeper<'_>) {
154    for mut user_report in std::mem::take(user_reports) {
155        let data = match process_user_report(user_report.payload()) {
156            Ok(data) => data,
157            Err(err) => {
158                records.reject_err(err, user_report);
159                continue;
160            }
161        };
162        user_report.set_payload(ContentType::Json, data);
163        user_reports.push(user_report);
164    }
165}

167fn process_user_report(user_report: Bytes) -> Result<Bytes> {
168    // There is a customer SDK which sends invalid reports with a trailing `\n`,
169    // strip it here, even if they update/fix their SDK there will still be many old
170    // versions with the broken SDK out there.
171    let user_report = trim_whitespaces(&user_report);
172
173    let report =
174        serde_json::from_slice::<UserReport>(user_report).map_err(ProcessingError::InvalidJson)?;
175
176    serde_json::to_string(&report)
177        .map(Bytes::from)
178        .map_err(ProcessingError::SerializeFailed)
179        .map_err(Into::into)
180}
181
182fn trim_whitespaces(data: &[u8]) -> &[u8] {
183    let Some(from) = data.iter().position(|x| !x.is_ascii_whitespace()) else {
184        return &[];
185    };
186    let Some(to) = data.iter().rposition(|x| !x.is_ascii_whitespace()) else {
187        return &[];
188    };
189    &data[from..to + 1]
190}
