11/// Expands an (error) event envelope.
12///
13/// Identifies the underlying type of error/event and expands it into an event and its parts.
14///
15/// For example an crash report attachment may be expanded into an error event, multiple other
16/// attachments and some user feedback.
17pub fn expand(
18    error: Managed<SerializedError>,
19    ctx: Context<'_>,
20) -> Result<Managed<ExpandedError>, Rejected<Error>> {
21    error.try_map(|error, records| do_expand(error, ctx, records))
22}
23
24fn do_expand(
25    mut error: SerializedError,
26    ctx: Context<'_>,
27    records: &mut RecordKeeper<'_>,
28) -> Result<ExpandedError> {
29    let is_trusted = error.headers.meta().request_trust().is_trusted();
30
31    // Certain attachment types are dissolved into different types (Nintendo Switch),
32    // or are created from other types (unreal).
33    records.lenient(DataCategory::Attachment);
34    records.lenient(DataCategory::AttachmentItem);
35    // User feedback is extracted from unreal reports.
36    records.lenient(DataCategory::UserReportV2);
37
38    let Some(parsed) = ErrorKind::try_expand(
39        &mut error.items,
40        errors::Context {
41            envelope: &error.headers,
42            processing: ctx,
43        },
44    )?
45    else {
46        return Err(ProcessingError::NoEventPayload.into());
47    };
48
49    // All unprocessed items which would create an event but were not turned into an event are
50    // duplicates.
51    //
52    // E.g. if you send an envelope with two error events, you end up here.
53    //
54    // It is reasonable to just reject such envelopes, this is ported over from a prior
55    // implementation of the error processing pipeline.
56    for item in error.items.extract_if(.., |item| item.creates_event()) {
57        records.reject_err(ProcessingError::DuplicateItem(item.ty().clone()), item);
58    }
59
60    if ctx.is_processing() {
61        // In processing, there are cannot be any leftover events due to forward compatibility.
62        for item in std::mem::take(&mut error.items) {
63            records.reject_err(ProcessingError::UnsupportedItem, item);
64        }
65    }
66
67    Ok(ExpandedError {
68        headers: error.headers,
69        fully_normalized: EventFullyNormalized(is_trusted && parsed.fully_normalized),
70        metrics: parsed.metrics,
71        event: parsed.event,
72        attachments: parsed.attachments,
73        user_reports: parsed.user_reports,
74        data: parsed.error,
75        other: error.items,
76    })
77}

79pub fn process(error: &mut Managed<ExpandedError>) -> Result<(), Rejected<Error>> {
80    error.try_modify(|error, records| {
81        processing::user_reports::process_user_reports(&mut error.user_reports, records);
82
83        Ok::<_, Error>(())
84    })
85}

87pub fn finalize(
88    error: &mut Managed<ExpandedError>,
89    ctx: Context<'_>,
90) -> Result<(), Rejected<Error>> {
91    error.try_modify(|error, _| {
92        processing::utils::event::finalize(
93            &error.headers,
94            &mut error.event,
95            error.attachments.iter(),
96            &mut error.metrics,
97            ctx.config,
98        )?;
99
100        Ok::<_, Error>(())
101    })
102}

104pub fn normalize(
105    error: &mut Managed<ExpandedError>,
106    geoip_lookup: &GeoIpLookup,
107    ctx: Context<'_>,
108) -> Result<(), Rejected<Error>> {
109    let scoping = error.scoping();
110
111    error.try_modify(|error, _| {
112        error.fully_normalized = processing::utils::event::normalize(
113            &error.headers,
114            &mut error.event,
115            error.fully_normalized,
116            scoping.project_id,
117            ctx,
118            geoip_lookup,
119        )?;
120
121        Ok::<_, Error>(())
122    })
123}

125pub fn scrub(error: &mut Managed<ExpandedError>, ctx: Context<'_>) -> Result<(), Rejected<Error>> {
126    error.try_modify(|error, records| {
127        processing::utils::event::scrub(&mut error.event, ctx.project_info)?;
128        processing::utils::attachments::scrub(
129            error.attachments.iter_mut(),
130            ctx.project_info,
131            Some(records),
132        );
133        if let Some(minidump) = error.data.minidump_mut() {
134            processing::utils::attachments::scrub(
135                std::iter::once(minidump),
136                ctx.project_info,
137                Some(records),
138            );
139        }
140
141        Ok::<_, Error>(())
142    })
143}

145/// Validates the attachments and drop any invalid ones.
146///
147/// An attachment might be a placeholder, in which case it needs to be validated.
148pub fn validate_attachments(error: &mut Managed<ExpandedError>, ctx: Context<'_>) {
149    if !ctx.is_processing() {
150        return;
151    }
152
153    error.modify(|error, records| {
154        error.attachments.retain_mut(|attachment| {
155            match processing::utils::attachments::validate(attachment, ctx.config) {
156                Ok(()) => true,
157                Err(err) => {
158                    records.reject_err(err, &*attachment);
159                    false
160                }
161            }
162        });
163    });
