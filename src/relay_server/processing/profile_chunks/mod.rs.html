18#[derive(Debug, thiserror::Error)]
19pub enum Error {
20    /// Error raised in [`relay_profiling`].
21    #[error("Profiling Error: {0}")]
22    Profiling(#[from] relay_profiling::ProfileError),
23    /// The profile chunks are rate limited.
24    #[error("rate limited")]
25    RateLimited(RateLimits),
26    /// Profile chunks filtered because of a missing feature flag.
27    #[error("profile chunks feature flag missing")]
28    FilterFeatureFlag,
29}
30
31impl From<RateLimits> for Error {
32    fn from(value: RateLimits) -> Self {
33        Self::RateLimited(value)
34    }
35}
36
37impl crate::managed::OutcomeError for Error {
38    type Error = Self;
39
40    fn consume(self) -> (Option<Outcome>, Self::Error) {
41        let outcome = match &self {
42            Self::Profiling(relay_profiling::ProfileError::Filtered(f)) => {
43                Some(Outcome::Filtered(f.clone()))
44            }
45            Self::Profiling(err) => Some(Outcome::Invalid(DiscardReason::Profiling(
46                relay_profiling::discard_reason(err),
47            ))),
48
49            Self::RateLimited(limits) => {
50                let reason_code = limits.longest().and_then(|limit| limit.reason_code.clone());
51                Some(Outcome::RateLimited(reason_code))
52            }
53            Self::FilterFeatureFlag => None,
54        };
55        (outcome, self)
56    }
57}

59/// A processor for profile chunks.
60///
61/// It processes items of type: [`ItemType::ProfileChunk`].
62#[derive(Debug)]
63pub struct ProfileChunksProcessor {
64    limiter: Arc<QuotaRateLimiter>,
65}
66
67impl ProfileChunksProcessor {
68    /// Creates a new [`Self`].
69    pub fn new(limiter: Arc<QuotaRateLimiter>) -> Self {
70        Self { limiter }
71    }
72}
73
74impl processing::Processor for ProfileChunksProcessor {
75    type UnitOfWork = SerializedProfileChunks;
76    type Output = ProfileChunkOutput;
77    type Error = Error;
78
79    fn prepare_envelope(
80        &self,
81        envelope: &mut ManagedEnvelope,
82    ) -> Option<Managed<Self::UnitOfWork>> {
83        let profile_chunks = envelope
84            .envelope_mut()
85            .take_items_by(|item| matches!(*item.ty(), ItemType::ProfileChunk))
86            .into_vec();
87
88        if profile_chunks.is_empty() {
89            return None;
90        }
91
92        Some(Managed::with_meta_from(
93            envelope,
94            SerializedProfileChunks {
95                headers: envelope.envelope().headers().clone(),
96                profile_chunks,
97            },
98        ))
99    }
100
101    async fn process(
102        &self,
103        mut profile_chunks: Managed<Self::UnitOfWork>,
104        ctx: Context<'_>,
105    ) -> Result<Output<Self::Output>, Rejected<Error>> {
106        filter::feature_flag(ctx).reject(&profile_chunks)?;
107
108        process::process(&mut profile_chunks, ctx);
109
110        let profile_chunks = self.limiter.enforce_quotas(profile_chunks, ctx).await?;
111
112        Ok(Output::just(ProfileChunkOutput(profile_chunks)))
113    }
114}

116/// Output produced by [`ProfileChunksProcessor`].
117#[derive(Debug)]
118pub struct ProfileChunkOutput(Managed<SerializedProfileChunks>);
119
120impl Forward for ProfileChunkOutput {
121    fn serialize_envelope(
122        self,
123        _: processing::ForwardContext<'_>,
124    ) -> Result<Managed<Box<Envelope>>, Rejected<()>> {
125        let Self(profile_chunks) = self;
126        Ok(profile_chunks
127            .map(|pc, _| Envelope::from_parts(pc.headers, Items::from_vec(pc.profile_chunks))))
128    }
129
130    #[cfg(feature = "processing")]
131    fn forward_store(
132        self,
133        s: processing::forward::StoreHandle<'_>,
134        ctx: processing::ForwardContext<'_>,
135    ) -> Result<(), Rejected<()>> {
136        use crate::services::store::StoreProfileChunk;
137
138        let Self(profile_chunks) = self;
139        let retention_days = ctx.event_retention().standard;
140
141        for item in profile_chunks.split(|pc| pc.profile_chunks) {
142            s.store(item.map(|item, _| StoreProfileChunk {
143                retention_days,
144                payload: item.payload(),
145                quantities: item.quantities(),
146            }));
147        }
148
149        Ok(())
150    }
151}

153/// Serialized profile chunks extracted from an envelope.
154#[derive(Debug)]
155pub struct SerializedProfileChunks {
156    /// Original envelope headers.
157    pub headers: EnvelopeHeaders,
158    /// List of serialized profile chunk items.
159    pub profile_chunks: Vec<Item>,
160}
161
162impl Counted for SerializedProfileChunks {
163    fn quantities(&self) -> Quantities {
164        let mut ui = 0;
165        let mut backend = 0;
166
167        for pc in &self.profile_chunks {
168            match pc.profile_type() {
169                Some(ProfileType::Ui) => ui += 1,
170                Some(ProfileType::Backend) => backend += 1,
171                None => {}
172            }
173        }
174
175        let mut quantities = smallvec![];
176        if ui > 0 {
177            quantities.push((DataCategory::ProfileChunkUi, ui));
178        }
179        if backend > 0 {
180            quantities.push((DataCategory::ProfileChunk, backend));
181        }
182
183        quantities
184    }
185}
