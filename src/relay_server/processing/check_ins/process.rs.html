1use crate::envelope::ContentType;
2use crate::managed::Managed;

5/// Normalizes all check-ins using the [`relay_monitors`] module.
6///
7/// Individual, invalid check-ins will be discarded.
8pub fn normalize(check_ins: &mut Managed<SerializedCheckIns>) {
9    let scoping = check_ins.scoping();
10
11    check_ins.retain(
12        |check_ins| &mut check_ins.check_ins,
13        |check_in| {
14            let payload = check_in.payload();
15            let result = relay_monitors::process_check_in(&payload, scoping.project_id)
16                .inspect_err(|err| {
17                    relay_log::debug!(
18                        error = err as &dyn std::error::Error,
19                        "dropped invalid monitor check-in"
20                    )
21                })?;
22
23            check_in.set_routing_hint(result.routing_hint);
24            check_in.set_payload(ContentType::Json, result.payload);
25
26            Ok::<_, Error>(())
27        },
28    );
29}
