1use axum::extract::OptionalFromRequestParts;
2use axum::http::request::Parts;
3use relay_auth::Signature;
4
5use crate::extractors::SignatureError;
6use crate::service::ServiceState;
7
8impl OptionalFromRequestParts<ServiceState> for Signature {
9    type Rejection = SignatureError;
10
11    async fn from_request_parts(
12        parts: &mut Parts,
13        _: &ServiceState,
14    ) -> Result<Option<Self>, Self::Rejection> {
15        parts
16            .headers
17            .get("x-sentry-relay-signature")
18            .map(|header| {
19                header
20                    .to_str()
21                    .map_err(|_| SignatureError::MalformedHeader("x-sentry-relay-signature"))
22                    .map(|sig| Signature(sig.to_owned()))
23            })
24            .transpose()
25    }
26}
