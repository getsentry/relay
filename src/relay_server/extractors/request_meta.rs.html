30#[derive(Debug, thiserror::Error)]
31pub enum BadEventMeta {
32    #[error("missing authorization information")]
33    MissingAuth,
34
35    #[error("multiple authorization payloads detected")]
36    MultipleAuth,
37
38    #[error("unsupported protocol version ({0})")]
39    UnsupportedProtocolVersion(u16),
40
41    #[error("bad envelope authentication header")]
42    BadEnvelopeAuth(#[source] serde_json::Error),
43
44    #[error("bad project path parameter")]
45    BadProject(#[from] PathRejection),
46
47    #[error("bad x-sentry-auth header")]
48    BadAuth(#[from] ParseAuthError),
49
50    #[error("bad sentry DSN public key")]
51    BadPublicKey(#[from] ParseProjectKeyError),
52}
53
54impl From<Infallible> for BadEventMeta {
55    fn from(infallible: Infallible) -> Self {
56        match infallible {}
57    }
58}
59
60impl IntoResponse for BadEventMeta {
61    fn into_response(self) -> Response {
62        let code = match self {
63            Self::MissingAuth
64            | Self::MultipleAuth
65            | Self::BadAuth(_)
66            | Self::BadEnvelopeAuth(_) => StatusCode::UNAUTHORIZED,
67            Self::UnsupportedProtocolVersion(_) | Self::BadProject(_) | Self::BadPublicKey(_) => {
68                StatusCode::BAD_REQUEST
69            }
70        };
71
72        (code, ApiErrorResponse::from_error(&self)).into_response()
73    }
74}

76/// Wrapper around a Sentry DSN with parsed public key.
77///
78/// The Sentry DSN type carries a plain public key string. However, Relay handles copy `ProjectKey`
79/// types internally. Converting from `String` to `ProjectKey` is fallible and should be caught when
80/// deserializing the request.
81///
82/// This type caches the parsed project key in addition to the DSN. Other than that, it
83/// transparently serializes to and deserializes from a DSN string.
84#[derive(Debug, Clone, Eq, PartialEq)]
85pub struct PartialDsn {
86    pub scheme: Scheme,
87    pub public_key: ProjectKey,
88    pub host: String,
89    pub port: u16,
90    pub path: String,
91    pub project_id: Option<ProjectId>,
92}
93
94impl PartialDsn {
95    /// Ensures a valid public key and project ID in the DSN.
96    fn from_dsn(dsn: Dsn) -> Result<Self, ParseDsnError> {
97        let project_id = dsn
98            .project_id()
99            .value()
100            .parse()
101            .map_err(|_| ParseDsnError::NoProjectId)?;
102
103        let public_key = dsn
104            .public_key()
105            .parse()
106            .map_err(|_| ParseDsnError::NoUsername)?;
107
108        Ok(Self {
109            scheme: dsn.scheme(),
110            public_key,
111            host: dsn.host().to_owned(),
112            port: dsn.port(),
113            path: dsn.path().to_owned(),
114            project_id: Some(project_id),
115        })
116    }
117
118    /// Creates a new [`PartialDsn`] for a Relay outbound request.
119    pub fn outbound(scoping: &Scoping, upstream: &UpstreamDescriptor<'_>) -> Self {
120        Self {
121            scheme: upstream.scheme(),
122            public_key: scoping.project_key,
123            host: upstream.host().to_owned(),
124            port: upstream.port(),
125            path: "".to_owned(),
126            project_id: Some(scoping.project_id),
127        }
128    }
129
130    /// Returns the project identifier that the DSN points to.
131    pub fn project_id(&self) -> Option<ProjectId> {
132        self.project_id
133    }
134
135    /// Returns the public key part of the DSN for authentication.
136    pub fn public_key(&self) -> ProjectKey {
137        self.public_key
138    }
139}

141impl fmt::Display for PartialDsn {
142    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
143        write!(f, "{}://{}:@{}", self.scheme, self.public_key, self.host)?;
144        if self.port != self.scheme.default_port() {
145            write!(f, ":{}", self.port)?;
146        }
147        let project_id = self.project_id.unwrap_or_else(|| ProjectId::new(0));
148        write!(f, "/{}{}", self.path.trim_start_matches('/'), project_id)
149    }
150}

425    /// Completes missing information with complete `RequestMeta`.
426    ///
427    /// All fields that set in this instance will remain.
428    pub fn copy_to(self, mut complete: RequestMeta) -> RequestMeta {
429        // DSN needs to be validated by the caller and will not be copied over.
430
431        if self.client.is_some() {
432            complete.client = self.client;
433        }
434        if self.version != default_version() {
435            complete.version = self.version;
436        }
437        if self.origin.is_some() {
438            complete.origin = self.origin;
439        }
440        if self.remote_addr.is_some() {
441            complete.remote_addr = self.remote_addr;
442        }
443        if !self.forwarded_for.is_empty() {
444            complete.forwarded_for = self.forwarded_for;
445        }
446        if self.user_agent.is_some() {
447            complete.user_agent = self.user_agent;
448        }
449        if self.from_internal_relay {
450            complete.from_internal_relay = self.from_internal_relay;
451        }
452
453        complete.client_hints.copy_from(self.client_hints);
454
455        if self.no_cache {
456            complete.no_cache = true;
457        }
458
459        complete
460    }

463impl FromRequestParts<ServiceState> for PartialMeta {
464    type Rejection = Infallible;
465
466    async fn from_request_parts(
467        parts: &mut Parts,
468        state: &ServiceState,
469    ) -> Result<Self, Self::Rejection> {
470        let mut ua = RawUserAgentInfo::default();
471        for (key, value) in &parts.headers {
472            ua.set_ua_field_from_header(key.as_str(), value.to_str().ok().map(str::to_string));
473        }
474
475        let mut from_internal_relay = false;
476        let relay_id = parts
477            .headers
478            .get("x-sentry-relay-id")
479            .and_then(|h| h.to_str().ok())
480            .and_then(|h| h.parse::<RelayId>().ok());
481
482        if let Some(relay_id) = relay_id {
483            relay_log::configure_scope(|s| s.set_tag("relay_id", relay_id));
484            from_internal_relay = state
485                .config()
486                .static_relays()
487                .get(&relay_id)
488                .is_some_and(|ri| ri.internal);
489        }
490
491        let ReceivedAt(received_at) = ReceivedAt::from_request_parts(parts, state).await?;
492
493        Ok(RequestMeta {
494            dsn: None,
495            version: default_version(),
496            client: None,
497            origin: parse_header_url(parts, header::ORIGIN)
498                .or_else(|| parse_header_url(parts, header::REFERER)),
499            remote_addr: ConnectInfo::<SocketAddr>::from_request_parts(parts, state)
500                .await
501                .map(|ConnectInfo(peer)| peer.ip())
502                .ok(),
503            forwarded_for: ForwardedFor::from_request_parts(parts, state)
504                .await?
505                .into_inner(),
506            user_agent: ua.user_agent,
507            no_cache: false,
508            received_at,
509            client_hints: ua.client_hints,
510            from_internal_relay,
511        })
512    }
513}

515fn get_auth_header(req: &Parts, header_name: impl AsHeaderName) -> Option<&str> {
516    req.headers
517        .get(header_name)
518        .and_then(|x| x.to_str().ok())
519        .filter(|h| h.len() >= 7 && h[..7].eq_ignore_ascii_case("sentry "))
520}

522fn auth_from_parts(req: &Parts, path_key: Option<String>) -> Result<Auth, BadEventMeta> {
523    let mut auth = None;
524
525    // try to extract authentication info from http header "x-sentry-auth"
526    if let Some(header) = get_auth_header(req, "x-sentry-auth") {
527        auth = Some(header.parse::<Auth>()?);
528    }
529
530    // try to extract authentication info from http header "authorization"
531    if let Some(header) = get_auth_header(req, header::AUTHORIZATION) {
532        if auth.is_some() {
533            return Err(BadEventMeta::MultipleAuth);
534        }
535
536        auth = Some(header.parse::<Auth>()?);
537    }
538
539    // try to get authentication info from basic auth
540    if let Some(basic_auth) = req
541        .headers
542        .get("authorization")
543        .and_then(|value| value.to_str().ok())
544        .and_then(|x| {
545            if x.len() >= 6 && x[..6].eq_ignore_ascii_case("basic ") {
546                x.get(6..)
547            } else {
548                None
549            }
550        })
551        .and_then(|value| {
552            let decoded = String::from_utf8(BASE64.decode(value.as_bytes()).ok()?).ok()?;
553            let (public_key, _) = decoded.split_once(':')?;
554            Auth::from_pairs([("sentry_key", public_key)]).ok()
555        })
556    {
557        if auth.is_some() {
558            return Err(BadEventMeta::MultipleAuth);
559        }
560        auth = Some(basic_auth);
561    }
562
563    // try to extract authentication info from URL query_param .../?sentry_...=<key>...
564    let query = req.uri.query().unwrap_or_default();
565    if query.contains("sentry_") {
566        if auth.is_some() {
567            return Err(BadEventMeta::MultipleAuth);
568        }
569
570        auth = Some(Auth::from_querystring(query.as_bytes())?);
571    }
572
573    // try to extract authentication info from URL path segment .../{sentry_key}/...
574    if let Some(sentry_key) = path_key {
575        if auth.is_some() {
576            return Err(BadEventMeta::MultipleAuth);
577        }
578
579        auth = Some(
580            Auth::from_pairs(std::iter::once(("sentry_key", sentry_key)))
581                .map_err(|_| BadEventMeta::MissingAuth)?,
582        );
583    }
584
585    auth.ok_or(BadEventMeta::MissingAuth)
586}

588fn parse_header_url(req: &Parts, header: impl AsHeaderName) -> Option<Url> {
589    req.headers
590        .get(header)
591        .and_then(|h| h.to_str().ok())
592        .and_then(|s| s.parse::<Url>().ok())
593        .and_then(|u| match u.scheme() {
594            "http" | "https" => Some(u),
595            _ => None,
596        })
597}

599/// Path parameters containing authentication information for store endpoints.
600///
601/// These parameters implement part of the authentication mechanism. For more information, see
602/// [`RequestMeta`].
603#[derive(Debug, serde::Deserialize)]
604struct StorePath {
605    /// The numeric identifier of the Sentry project.
606    ///
607    /// This parameter is part of the store endpoint paths, which are generally located under
608    /// `/api/:project_id/*`. By default, all store endpoints have the project ID in the path. To
609    /// resolve the project and associated information, Relay actually uses the DSN's
610    /// [`ProjectKey`]. During ingestion, the stated project ID from the URI path is validated
611    /// against information resolved from the upstream.
612    ///
613    /// The legacy endpoint (`/api/store/`) does not have the project ID. In this case, Relay skips
614    /// ID validation during ingestion.
615    project_id: Option<ProjectId>,
616
617    /// The DSN's public key, also referred to as project key.
618    ///
619    /// Some endpoints require this key in the path. On all other endpoints, the key is either sent
620    /// as header or query parameter.
621    sentry_key: Option<String>,
622}

624impl FromRequestParts<ServiceState> for RequestMeta {
625    type Rejection = BadEventMeta;
626
627    async fn from_request_parts(
628        parts: &mut Parts,
629        state: &ServiceState,
630    ) -> Result<Self, Self::Rejection> {
631        let Path(store_path): Path<StorePath> =
632            parts.extract().await.map_err(BadEventMeta::BadProject)?;
633
634        let auth = auth_from_parts(parts, store_path.sentry_key)?;
635        let partial_meta: PartialMeta = parts.extract_with_state(state).await?;
636        let (public_key, key_flags) = ProjectKey::parse_with_flags(auth.public_key())?;
637
638        let config = state.config();
639        let upstream = config.upstream_descriptor();
640
641        let dsn = PartialDsn {
642            scheme: upstream.scheme(),
643            public_key,
644            host: upstream.host().to_owned(),
645            port: upstream.port(),
646            path: String::new(),
647            project_id: store_path.project_id,
648        };
649
650        // For now, we only handle <= v8 and drop everything else
651        let version = auth.version();
652        if version > relay_event_schema::protocol::PROTOCOL_VERSION {
653            return Err(BadEventMeta::UnsupportedProtocolVersion(version));
654        }
655
656        relay_statsd::metric!(
657            counter(RelayCounters::EventProtocol) += 1,
658            version = &version.to_string()
659        );
660
661        Ok(RequestMeta {
662            dsn,
663            version,
664            client: auth.client_agent().map(str::to_owned),
665            origin: partial_meta.origin,
666            remote_addr: partial_meta.remote_addr,
667            forwarded_for: partial_meta.forwarded_for,
668            user_agent: partial_meta.user_agent,
669            no_cache: key_flags.contains(&"no-cache"),
670            received_at: partial_meta.received_at,
671            client_hints: partial_meta.client_hints,
672            from_internal_relay: partial_meta.from_internal_relay,
673        })
674    }
675}
