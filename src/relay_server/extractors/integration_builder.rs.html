16/// Marker indicating a type has been configured on the [`IntegrationBuilder`].
17pub struct HasType(());
18
19/// A builder for integration endpoints.
20///
21/// The builder can be directly extracted from an axum handler.
22#[derive(Clone, Debug)]
23pub struct IntegrationBuilder<T = ()> {
24    envelope: Box<Envelope>,
25    payload: Bytes,
26    _state: PhantomData<T>,
27}
28
29impl IntegrationBuilder<()> {
30    /// Configures the [`Integration`] type.
31    ///
32    /// Setting the type is required.
33    pub fn with_type(mut self, integration: impl Into<Integration>) -> IntegrationBuilder<HasType> {
34        let integration = integration.into();
35
36        self.envelope.add_item({
37            let mut item = Item::new(ItemType::Integration);
38            item.set_payload(integration.into(), self.payload.clone());
39            item
40        });
41
42        IntegrationBuilder {
43            envelope: self.envelope,
44            payload: self.payload,
45            _state: Default::default(),
46        }
47    }
48}
49
50impl<T> IntegrationBuilder<T> {
51    /// Optionally adds a required feature to the envelope.
52    ///
53    /// This can be used for integrations which are still feature flagged.
54    pub fn with_required_feature(mut self, feature: Feature) -> Self {
55        self.envelope.require_feature(feature);
56        self
57    }
58}
59
60impl IntegrationBuilder<HasType> {
61    /// Returns the built integration envelope.
62    pub fn build(mut self) -> Box<Envelope> {
63        self.envelope
64            .meta_mut()
65            .set_client(crate::constants::CLIENT.to_owned());
66
67        self.envelope
68    }
69}
70
71impl FromRequest<ServiceState> for IntegrationBuilder {
72    type Rejection = IntegrationBuilderRejection;
73
74    async fn from_request(mut req: Request, state: &ServiceState) -> Result<Self, Self::Rejection> {
75        let meta: RequestMeta = req.extract_parts_with_state(state).await?;
76        let payload = req.extract().await?;
77
78        let envelope = Envelope::from_request(None, meta);
79
80        Ok(Self {
81            envelope,
82            payload,
83            _state: Default::default(),
84        })
85    }
86}
87
88#[derive(Debug, thiserror::Error)]
89pub enum IntegrationBuilderRejection {
90    #[error("bad event meta: {0}")]
91    BadEventMeta(#[from] BadEventMeta),
92    #[error("unable to read body: {0}")]
93    Body(#[from] BytesRejection),
94}
95
96impl IntoResponse for IntegrationBuilderRejection {
97    fn into_response(self) -> axum::response::Response {
98        match self {
99            Self::BadEventMeta(err) => err.into_response(),
100            Self::Body(err) => err.into_response(),
101        }
102    }
