14#[derive(Debug, thiserror::Error)]
15pub enum SignatureError {
16    #[error("invalid relay signature")]
17    BadSignature(#[source] UnpackError),
18    #[error("missing header: {0}")]
19    MissingHeader(&'static str),
20    #[error("malformed header: {0}")]
21    MalformedHeader(&'static str),
22    #[error("unknown relay id")]
23    UnknownRelay,
24    #[error(transparent)]
25    MalformedBody(#[from] BytesRejection),
26    #[error("invalid JSON data")]
27    InvalidJson(#[from] serde_json::Error),
28    #[error("internal system is overloaded")]
29    ServiceUnavailable(#[from] relay_system::SendError),
30}
31
32impl IntoResponse for SignatureError {
33    fn into_response(self) -> Response {
34        let status = match self {
35            SignatureError::InvalidJson(_) => StatusCode::BAD_REQUEST,
36            SignatureError::MalformedBody(_) => StatusCode::BAD_REQUEST,
37            SignatureError::ServiceUnavailable(_) => StatusCode::SERVICE_UNAVAILABLE,
38            _ => StatusCode::UNAUTHORIZED,
39        };
40
41        (status, ApiErrorResponse::from_error(&self)).into_response()
42    }
43}

45impl From<UnpackError> for SignatureError {
46    fn from(error: UnpackError) -> Self {
47        match error {
48            UnpackError::BadPayload(json_error) => Self::InvalidJson(json_error),
49            other => Self::BadSignature(other),
50        }
51    }
52}

54fn get_header<'a, B>(
55    request: &'a Request<B>,
56    name: &'static str,
57) -> Result<&'a str, SignatureError> {
58    let value = request
59        .headers()
60        .get(name)
61        .ok_or(SignatureError::MissingHeader(name))?;
62
63    value
64        .to_str()
65        .map_err(|_| SignatureError::MalformedHeader(name))
66}

68#[derive(Debug)]
69pub struct SignedBytes {
70    pub body: Bytes,
71    pub relay: RelayInfo,
72}
73
74impl FromRequest<ServiceState> for SignedBytes {
75    type Rejection = SignatureError;
76
77    async fn from_request(request: Request, state: &ServiceState) -> Result<Self, Self::Rejection> {
78        let relay_id = get_header(&request, "x-sentry-relay-id")?
79            .parse::<RelayId>()
80            .map_err(|_| SignatureError::MalformedHeader("x-sentry-relay-id"))?;
81
82        relay_log::configure_scope(|s| s.set_tag("relay_id", relay_id));
83
84        let signature = get_header(&request, "x-sentry-relay-signature")?.to_owned();
85
86        let relay = state
87            .relay_cache()
88            .send(GetRelay { relay_id })
89            .await?
90            .ok_or(SignatureError::UnknownRelay)?;
91
92        let body = Bytes::from_request(request, state).await?;
93        if !relay.public_key.verify(&body, &signature) {
94            Err(SignatureError::BadSignature(UnpackError::BadSignature))
95        } else {
96            Ok(SignedBytes { body, relay })
97        }
98    }
99}

114        let SignedBytes { body, relay } = SignedBytes::from_request(request, state).await?;
115        let inner = serde_json::from_slice(&body)?;
