3/// The type of an event attachment.
4///
5/// These item types must align with the Sentry processing pipeline.
6#[derive(Clone, Debug, Eq, PartialEq)]
7pub enum AttachmentType {
8    /// A regular attachment without special meaning.
9    Attachment,
10
11    /// A minidump crash report (binary data).
12    Minidump,
13
14    /// An apple crash report (text data).
15    AppleCrashReport,
16
17    /// A msgpack-encoded event payload submitted as part of multipart uploads.
18    ///
19    /// This attachment is processed by Relay immediately and never forwarded or persisted.
20    EventPayload,
21
22    /// A msgpack-encoded list of payloads.
23    ///
24    /// There can be two attachments that the SDK may use as swappable buffers. Both attachments
25    /// will be merged and truncated to the maxmimum number of allowed attachments.
26    ///
27    /// This attachment is processed by Relay immediately and never forwarded or persisted.
28    Breadcrumbs,
29
30    // A prosperodump crash report (binary data)
31    Prosperodump,
32
33    /// This is a binary attachment present in Unreal 4 events containing event context information.
34    ///
35    /// This can be deserialized using the `symbolic` crate see
36    /// [`symbolic_unreal::Unreal4Context`].
37    ///
38    /// [`symbolic_unreal::Unreal4Context`]: https://docs.rs/symbolic/*/symbolic/unreal/struct.Unreal4Context.html
39    UnrealContext,
40
41    /// This is a binary attachment present in Unreal 4 events containing event Logs.
42    ///
43    /// This can be deserialized using the `symbolic` crate see
44    /// [`symbolic_unreal::Unreal4LogEntry`].
45    ///
46    /// [`symbolic_unreal::Unreal4LogEntry`]: https://docs.rs/symbolic/*/symbolic/unreal/struct.Unreal4LogEntry.html
47    UnrealLogs,
48
49    /// An application UI view hierarchy (json payload).
50    ViewHierarchy,
51
52    /// Unknown attachment type, forwarded for compatibility.
53    /// Attachments with this type will be dropped if `accept_unknown_items` is set to false.
54    Unknown(String),
55}
56
57impl Default for AttachmentType {
58    fn default() -> Self {
59        Self::Attachment
60    }
61}
62
63impl fmt::Display for AttachmentType {
64    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
65        match self {
66            AttachmentType::Attachment => write!(f, "event.attachment"),
67            AttachmentType::Minidump => write!(f, "event.minidump"),
68            AttachmentType::AppleCrashReport => write!(f, "event.applecrashreport"),
69            AttachmentType::EventPayload => write!(f, "event.payload"),
70            AttachmentType::Prosperodump => write!(f, "playstation.prosperodump"),
71            AttachmentType::Breadcrumbs => write!(f, "event.breadcrumbs"),
72            AttachmentType::UnrealContext => write!(f, "unreal.context"),
73            AttachmentType::UnrealLogs => write!(f, "unreal.logs"),
74            AttachmentType::ViewHierarchy => write!(f, "event.view_hierarchy"),
75            AttachmentType::Unknown(s) => s.fmt(f),
76        }
77    }
78}
79
80impl std::str::FromStr for AttachmentType {
81    type Err = std::convert::Infallible;
82
83    fn from_str(s: &str) -> Result<Self, Self::Err> {
84        Ok(match s {
85            "event.attachment" => AttachmentType::Attachment,
86            "event.minidump" => AttachmentType::Minidump,
87            "event.applecrashreport" => AttachmentType::AppleCrashReport,
88            "event.payload" => AttachmentType::EventPayload,
89            "playstation.prosperodump" => AttachmentType::Prosperodump,
90            "event.breadcrumbs" => AttachmentType::Breadcrumbs,
91            "event.view_hierarchy" => AttachmentType::ViewHierarchy,
92            "unreal.context" => AttachmentType::UnrealContext,
93            "unreal.logs" => AttachmentType::UnrealLogs,
94            other => AttachmentType::Unknown(other.to_owned()),
95        })
96    }
97}
