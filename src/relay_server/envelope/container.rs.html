21/// Error emitted when failing to parse an [`ItemContainer`].
22#[derive(thiserror::Error, Debug)]
23pub enum ContainerParseError {
24    /// The item container was expected to have a different content type.
25    #[error("expected item with content type {expected} but got {actual:?}")]
26    MismatchedContentType {
27        expected: ContentType,
28        actual: Option<ContentType>,
29    },
30    /// The item container was expected to have a different item type.
31    #[error("expected item with item type {expected} but got {actual:?}")]
32    MismatchedItemType {
33        expected: ItemType,
34        actual: ItemType,
35    },
36    /// The item container specified length does not match the amount of items contained in the
37    /// container.
38    #[error("container was specified with length {expected:?} but contained {actual} items")]
39    MismatchedLength {
40        expected: Option<u32>,
41        actual: usize,
42    },
43    /// The container is malformed and cannot be deserialized.
44    #[error("failed to deserialize item container: {0}")]
45    Deserialize(#[from] serde_json::Error),
46}

48/// Error emitted when failing to write/serialize and [`ItemContainer`].
49#[derive(thiserror::Error, Debug)]
50pub enum ContainerWriteError {
51    /// The item container is too large to serialize.
52    #[error("failed to serialize item container, item count overflow")]
53    Overflow,
54    /// The contained items cannot be serialized.
55    #[error("failed to serialize item container: {0}")]
56    Serialize(#[from] serde_json::Error),
57}

59/// Any item contained in an [`ItemContainer`] needs to implement this trait.
60pub trait ContainerItem: FromValue + IntoValue {
61    /// The expected item type of the container for this type.
62    const ITEM_TYPE: ItemType;
63    /// The expected content type of the container for this type.
64    const CONTENT_TYPE: ContentType;
65}

67/// A list of items in an item container.
68pub type ContainerItems<T> = SmallVec<[Annotated<T>; 3]>;

70/// A container for multiple homogeneous envelope items.
71///
72/// Item containers are used to minimize the amount of single envelope items contained in an
73/// envelope. They massively improve parsing speed of envelopes in Relay but are also used
74/// to minimize metadata duplication on item headers.
75///
76/// Especially for small envelope items with high quantities (e.g. logs), this drastically
77/// improves fast path parsing speeds and minimizes serialization overheads, by minimizing
78/// the amount of items in an envelope.
79///
80/// An item container does not have a special [`super::ItemType`], but is identified by the
81/// content type of the item.
82#[derive(Debug)]
83pub struct ItemContainer<T> {
84    items: ContainerItems<T>,
85}
86
87impl<T> ItemContainer<T> {
88    /// Returns all contained items.
89    ///
90    /// The container can be reconstructed using [`ItemContainer::from`].
91    pub fn into_items(self) -> ContainerItems<T> {
92        self.items
93    }
94}

96impl<T: ContainerItem> ItemContainer<T> {
97    /// Parses an [`ItemContainer`] from an envelope [`Item`].
98    ///
99    /// This function also validates metadata of the container, specifically the content type
100    /// and amount of contained items.
101    pub fn parse(item: &Item) -> Result<Self, ContainerParseError> {
102        if item.content_type() != Some(&T::CONTENT_TYPE) {
103            return Err(ContainerParseError::MismatchedContentType {
104                expected: T::CONTENT_TYPE,
105                actual: item.content_type().cloned(),
106            });
107        }
108
109        if item.ty() != &T::ITEM_TYPE {
110            return Err(ContainerParseError::MismatchedItemType {
111                expected: T::ITEM_TYPE,
112                actual: item.ty().clone(),
113            });
114        }
115
116        let payload = item.payload();
117        // Currently we assume every payload is JSON, but in the future we may allow other formats.
118        let mut de = serde_json::Deserializer::from_slice(&payload);
119        let container = Self::deserialize(&mut de)?;
120
121        if Some(container.items.len()) != item.item_count().map(|u| u as usize) {
122            return Err(ContainerParseError::MismatchedLength {
123                expected: item.item_count(),
124                actual: container.items.len(),
125            });
126        }
127
128        Ok(container)
129    }
130
131    /// Serializes the [`ItemContainer`] into an envelope [`Item`].
132    ///
133    /// This will serialize the contained items into the [`Item::payload`] as well as
134    /// update the [Item::content_type] and [`Item::item_count`].
135    pub fn write_to(&self, item: &mut Item) -> Result<(), ContainerWriteError> {
136        let mut payload = bytes::BytesMut::with_capacity(256).writer();
137        let mut ser = serde_json::Serializer::new(&mut payload);
138
139        self.serialize(&mut ser)?;
140
141        item.set_payload_with_item_count(
142            T::CONTENT_TYPE,
143            payload.into_inner(),
144            u32::try_from(self.items.len()).map_err(|_| ContainerWriteError::Overflow)?,
145        );
146
147        Ok(())
148    }
149
150    /// Checks whether a given item is a container of `T`s, according to its item and content types.
151    pub fn is_container(item: &Item) -> bool {
152        item.ty() == &T::ITEM_TYPE && item.content_type() == Some(&T::CONTENT_TYPE)
153    }
154
155    fn deserialize<'de, D: de::Deserializer<'de>>(deserializer: D) -> Result<Self, D::Error> {
156        #[derive(Deserialize)]
157        struct Layout<T> {
158            #[serde(bound(deserialize = "T: FromValue"))]
159            items: AnnotatedItems<T>,
160        }
161
162        let Layout {
163            items: AnnotatedItems(items),
164        } = Layout::<T>::deserialize(deserializer)?;
165
166        Ok(Self { items })
167    }
168
169    fn serialize<S: ser::Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
170        #[derive(Serialize)]
171        struct Layout<'a, T> {
172            #[serde(bound(serialize = "T: IntoValue"))]
173            items: AnnotatedItemsRef<'a, T>,
174        }
175
176        let layout = Layout {
177            items: AnnotatedItemsRef(&self.items),
178        };
179
180        Serialize::serialize(&layout, serializer)
181    }
182}

184impl<T> From<ContainerItems<T>> for ItemContainer<T> {
185    fn from(items: ContainerItems<T>) -> Self {
186        Self { items }
187    }
188}

200/// (De-)Serializes a list of Annotated items with metadata.
201#[derive(Debug)]
202struct AnnotatedItems<T>(ContainerItems<T>);
203
204impl<'de, T> Deserialize<'de> for AnnotatedItems<T>
205where
206    T: FromValue,
207{
208    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
209    where
210        D: serde::Deserializer<'de>,
211    {
212        struct Visitor<T>(PhantomData<T>);
213
214        impl<'de, T> de::Visitor<'de> for Visitor<T>
215        where
216            T: FromValue,
217        {
218            type Value = AnnotatedItems<T>;
219
220            fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
221                formatter.write_str("a list of envelope items")
222            }
223
224            fn visit_seq<A>(self, mut seq: A) -> Result<Self::Value, A::Error>
225            where
226                A: de::SeqAccess<'de>,
227            {
228                let mut items = ContainerItems::new();
229                if let Some(size) = seq.size_hint() {
230                    items.reserve_exact(size);
231                }
232
233                while let Some(DeserializableAnnotated(item)) = seq.next_element()? {
234                    items.push(item);
235                }
236
237                Ok(AnnotatedItems(items))
238            }
239        }
240
241        deserializer.deserialize_seq(Visitor(Default::default()))
242    }
243}

245struct AnnotatedItemsRef<'a, T>(&'a ContainerItems<T>);
246
247impl<T> Serialize for AnnotatedItemsRef<'_, T>
248where
249    T: IntoValue,
250{
251    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
252    where
253        S: ser::Serializer,
254    {
255        let mut seq = serializer.serialize_seq(Some(self.0.len()))?;
256        for item in self.0 {
257            seq.serialize_element(&SerializableAnnotated(item))?;
258        }
259        seq.end()
260    }
261}
