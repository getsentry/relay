13/// Error emitted when failing to parse an [`ItemContainer`].
14#[derive(thiserror::Error, Debug)]
15pub enum ContainerParseError {
16    /// The item container was expected to have a different content type.
17    #[error("expected item with content type {expected} but got {actual:?}")]
18    MismatchedContentType {
19        expected: ContentType,
20        actual: Option<ContentType>,
21    },
22    /// The item container specified length does not match the amount of items contained in the
23    /// container.
24    #[error("container was specified with length {expected:?} but contained {actual} items")]
25    MismatchedLength {
26        expected: Option<u32>,
27        actual: usize,
28    },
29    /// The container is malformed and cannot be deserialized.
30    #[error("failed to deserialize item container: {0}")]
31    Deserialize(#[from] serde_json::Error),
32}
33
34/// Error emitted when failing to write/serialize and [`ItemContainer`].
35#[derive(thiserror::Error, Debug)]
36pub enum ContainerWriteError {
37    /// The item container is too large to serialize.
38    #[error("failed to serialize item container, item count overflow")]
39    Overflow,
40    /// The contained items cannot be serialized.
41    #[error("failed to serialize item container: {0}")]
42    Serialize(#[from] serde_json::Error),
43}

45/// Any item contained in an [`ItemContainer`] needs to implement this trait.
46pub trait ContainerItem: FromValue + IntoValue {
47    /// The expected content type of the container for this type.
48    const CONTENT_TYPE: ContentType;
49}

51/// A list of items in an item container.
52pub type ContainerItems<T> = SmallVec<[Annotated<T>; 3]>;
53
54/// A container for multiple homogeneous envelope items.
55///
56/// Item containers are used to minimize the amount of single envelope items contained in an
57/// envelope. They massively improve parsing speed of envelopes in Relay but are also used
58/// to minimize metadata duplication on item headers.
59///
60/// Especially for small envelope items with high quantities (e.g. logs), this drastically
61/// improves fast path parsing speeds and minimizes serialization overheads, by minimizing
62/// the amount of items in an envelope.
63///
64/// An item container does not have a special [`super::ItemType`], but is identified by the
65/// content type of the item.
66#[derive(Debug)]
67pub struct ItemContainer<T> {
68    items: ContainerItems<T>,
69}
70
71impl<T> ItemContainer<T> {
72    /// Returns all contained items.
73    ///
74    /// The container can be reconstructed using [`ItemContainer::from`].
75    pub fn into_items(self) -> ContainerItems<T> {
76        self.items
77    }
78}

80impl<T: ContainerItem> ItemContainer<T> {
81    /// Parses an [`ItemContainer`] from an envelope [`Item`].
82    ///
83    /// This function also validates metadata of the container, specifically the content type
84    /// and amount of contained items.
85    pub fn parse(item: &Item) -> Result<Self, ContainerParseError> {
86        if item.content_type() != Some(&T::CONTENT_TYPE) {
87            return Err(ContainerParseError::MismatchedContentType {
88                expected: T::CONTENT_TYPE,
89                actual: item.content_type().cloned(),
90            });
91        }
92
93        let payload = item.payload();
94        // Currently we assume every payload is JSON, but in the future we may allow other formats.
95        let mut de = serde_json::Deserializer::from_slice(&payload);
96        let container = Self::deserialize(&mut de)?;
97
98        if Some(container.items.len()) != item.item_count().map(|u| u as usize) {
99            return Err(ContainerParseError::MismatchedLength {
100                expected: item.item_count(),
101                actual: container.items.len(),
102            });
103        }
104
105        Ok(container)
106    }
107
108    /// Serializes the [`ItemContainer`] into an envelope [`Item`].
109    ///
110    /// This will serialize the contained items into the [`Item::payload`] as well as
111    /// update the [Item::content_type] and [`Item::item_count`].
112    pub fn write_to(&self, item: &mut Item) -> Result<(), ContainerWriteError> {
113        let mut payload = bytes::BytesMut::with_capacity(256).writer();
114        let mut ser = serde_json::Serializer::new(&mut payload);
115
116        self.serialize(&mut ser)?;
117
118        item.set_payload_with_item_count(
119            T::CONTENT_TYPE,
120            payload.into_inner(),
121            u32::try_from(self.items.len()).map_err(|_| ContainerWriteError::Overflow)?,
122        );
123
124        Ok(())
125    }
126
127    fn deserialize<'de, D: de::Deserializer<'de>>(deserializer: D) -> Result<Self, D::Error> {
128        #[derive(Deserialize)]
129        struct Layout<T> {
130            #[serde(bound(deserialize = "T: FromValue"))]
131            items: AnnotatedItems<T>,
132        }
133
134        let Layout {
135            items: AnnotatedItems(items),
136        } = Layout::<T>::deserialize(deserializer)?;
137
138        Ok(Self { items })
139    }
140
141    fn serialize<S: ser::Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
142        #[derive(Serialize)]
143        struct Layout<'a, T> {
144            #[serde(bound(serialize = "T: IntoValue"))]
145            items: AnnotatedItemsRef<'a, T>,
146        }
147
148        let layout = Layout {
149            items: AnnotatedItemsRef(&self.items),
150        };
151
152        Serialize::serialize(&layout, serializer)
153    }
154}

157    fn from(items: ContainerItems<T>) -> Self {
158        Self { items }
159    }

166/// (De-)Serializes a list of Annotated items with metadata.
167#[derive(Debug)]
168struct AnnotatedItems<T>(ContainerItems<T>);
169
170impl<'de, T> Deserialize<'de> for AnnotatedItems<T>
171where
172    T: FromValue,
173{
174    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
175    where
176        D: serde::Deserializer<'de>,
177    {
178        struct Visitor<T>(PhantomData<T>);
179
180        impl<'de, T> de::Visitor<'de> for Visitor<T>
181        where
182            T: FromValue,
183        {
184            type Value = AnnotatedItems<T>;
185
186            fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
187                formatter.write_str("a list of envelope items")
188            }
189
190            fn visit_seq<A>(self, mut seq: A) -> Result<Self::Value, A::Error>
191            where
192                A: de::SeqAccess<'de>,
193            {
194                let mut items = ContainerItems::new();
195                if let Some(size) = seq.size_hint() {
196                    items.reserve_exact(size);
197                }
198
199                while let Some(DeserializableAnnotated(item)) = seq.next_element()? {
200                    items.push(item);
201                }
202
203                Ok(AnnotatedItems(items))
204            }
205        }
206
207        deserializer.deserialize_seq(Visitor(Default::default()))
208    }
209}

211struct AnnotatedItemsRef<'a, T>(&'a ContainerItems<T>);
212
213impl<T> Serialize for AnnotatedItemsRef<'_, T>
214where
215    T: IntoValue,
216{
217    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
218    where
219        S: ser::Serializer,
220    {
221        let mut seq = serializer.serialize_seq(Some(self.0.len()))?;
222        for item in self.0 {
223            seq.serialize_element(&SerializableAnnotated(item))?;
224        }
225        seq.end()
226    }
227}
