59    /// Parses an [`Item`] from raw bytes.
60    pub fn parse(bytes: Bytes) -> Result<(Item, usize), EnvelopeError> {
61        let slice = bytes.as_ref();
62        let mut stream = serde_json::Deserializer::from_slice(slice).into_iter();
63
64        let headers: ItemHeaders = match stream.next() {
65            None => return Err(EnvelopeError::UnexpectedEof),
66            Some(Err(error)) => return Err(EnvelopeError::InvalidItemHeader(error)),
67            Some(Ok(headers)) => headers,
68        };
69
70        // Each header is terminated by a UNIX newline.
71        let headers_end = stream.byte_offset();
72        super::require_termination(slice, headers_end)?;
73
74        // The last header does not require a trailing newline, so `payload_start` may point
75        // past the end of the buffer.
76        let payload_start = std::cmp::min(headers_end + 1, bytes.len());
77        let payload_end = match headers.length {
78            Some(len) => {
79                let payload_end = payload_start + len as usize;
80                if bytes.len() < payload_end {
81                    // NB: `Bytes::slice` panics if the indices are out of range.
82                    return Err(EnvelopeError::UnexpectedEof);
83                }
84
85                // Each payload is terminated by a UNIX newline.
86                super::require_termination(slice, payload_end)?;
87                payload_end
88            }
89            None => match bytes[payload_start..].iter().position(|b| *b == b'\n') {
90                Some(relative_end) => payload_start + relative_end,
91                None => bytes.len(),
92            },
93        };
94
95        let payload = bytes.slice(payload_start..payload_end);
96        let item = Item { headers, payload };
97
98        Ok((item, payload_end + 1))
99    }

101    /// Returns the number used for counting towards rate limits and producing outcomes.

103    /// For attachments, we count the number of bytes. Other items are counted as 1.
104    pub fn quantities(&self) -> SmallVec<[(DataCategory, usize); 1]> {
105        let item_count = self.item_count().unwrap_or(1) as usize;
106
107        match self.ty() {
108            ItemType::Event => smallvec![(DataCategory::Error, item_count)],
109            ItemType::Transaction => smallvec![(DataCategory::Transaction, item_count)],
110            ItemType::Security | ItemType::RawSecurity => {
111                smallvec![(DataCategory::Security, item_count)]
112            }
113            ItemType::Nel => smallvec![],
114            ItemType::UnrealReport => smallvec![(DataCategory::Error, item_count)],
115            ItemType::Attachment => smallvec![
116                (DataCategory::Attachment, self.len().max(1)),
117                (DataCategory::AttachmentItem, item_count)
118            ],
119            ItemType::Session | ItemType::Sessions => {
120                smallvec![(DataCategory::Session, item_count)]
121            }
122            ItemType::Statsd | ItemType::MetricBuckets => smallvec![],
123            ItemType::Log | ItemType::OtelLog => smallvec![
124                (DataCategory::LogByte, self.len().max(1)),
125                (DataCategory::LogItem, item_count)
126            ],
127            ItemType::FormData => smallvec![],
128            ItemType::UserReport => smallvec![],
129            ItemType::UserReportV2 => smallvec![(DataCategory::UserReportV2, item_count)],
130            ItemType::Profile => smallvec![(DataCategory::Profile, item_count)],
131            ItemType::ReplayEvent | ItemType::ReplayRecording | ItemType::ReplayVideo => {
132                smallvec![(DataCategory::Replay, item_count)]
133            }
134            ItemType::ClientReport => smallvec![],
135            ItemType::CheckIn => smallvec![(DataCategory::Monitor, item_count)],
136            ItemType::Span | ItemType::OtelSpan => smallvec![(DataCategory::Span, item_count)],
137            // NOTE: semantically wrong, but too expensive to parse.
138            ItemType::OtelTracesData => smallvec![(DataCategory::Span, item_count)],
139            ItemType::ProfileChunk => match self.profile_type() {
140                Some(ProfileType::Backend) => smallvec![(DataCategory::ProfileChunk, item_count)],
141                Some(ProfileType::Ui) => smallvec![(DataCategory::ProfileChunkUi, item_count)],
142                None => smallvec![],
143            },
144            ItemType::Unknown(_) => smallvec![],

366    /// Determines whether the given item creates an event.
367    ///
368    /// This is only true for literal events and crash report attachments.
369    pub fn creates_event(&self) -> bool {
370        match self.ty() {
371            // These items are direct event types.
372            ItemType::Event
373            | ItemType::Transaction
374            | ItemType::Security
375            | ItemType::RawSecurity
376            | ItemType::Nel
377            | ItemType::UnrealReport
378            | ItemType::UserReportV2 => true,
379
380            // Attachments are only event items if they are crash reports or if they carry partial
381            // event payloads. Plain attachments never create event payloads.
382            ItemType::Attachment => {
383                match self.attachment_type().unwrap_or(&AttachmentType::default()) {
384                    AttachmentType::AppleCrashReport
385                    | AttachmentType::Minidump
386                    | AttachmentType::EventPayload
387                    | AttachmentType::Prosperodump
388                    | AttachmentType::Breadcrumbs => true,
389                    AttachmentType::Attachment
390                    | AttachmentType::UnrealContext
391                    | AttachmentType::UnrealLogs
392                    | AttachmentType::ViewHierarchy => false,
393                    // When an outdated Relay instance forwards an unknown attachment type for compatibility,
394                    // we assume that the attachment does not create a new event. This will make it hard
395                    // to introduce new attachment types which _do_ create a new event.
396                    AttachmentType::Unknown(_) => false,
397                }
398            }
399
400            // Form data items may contain partial event payloads, but those are only ever valid if
401            // they occur together with an explicit event item, such as a minidump or apple crash
402            // report. For this reason, FormData alone does not constitute an event item.
403            ItemType::FormData => false,
404
405            // The remaining item types cannot carry event payloads.
406            ItemType::UserReport
407            | ItemType::Session
408            | ItemType::Sessions
409            | ItemType::Statsd
410            | ItemType::MetricBuckets
411            | ItemType::ClientReport
412            | ItemType::ReplayEvent
413            | ItemType::ReplayRecording
414            | ItemType::ReplayVideo
415            | ItemType::Profile
416            | ItemType::CheckIn
417            | ItemType::Span
418            | ItemType::Log
419            | ItemType::OtelLog
420            | ItemType::OtelSpan
421            | ItemType::OtelTracesData
422            | ItemType::ProfileChunk => false,
423
424            // The unknown item type can observe any behavior, most likely there are going to be no
425            // item types added that create events.
426            ItemType::Unknown(_) => false,
427        }
428    }

430    /// Determines whether the given item requires an event with identifier.
431    pub fn requires_event(&self) -> bool {
432        match self.ty() {
433            ItemType::Event => true,
434            ItemType::Transaction => true,
435            ItemType::Security => true,
436            ItemType::Attachment => true,
437            ItemType::FormData => true,
438            ItemType::RawSecurity => true,
439            ItemType::Nel => false,
440            ItemType::UnrealReport => true,
441            ItemType::UserReport => true,
442            ItemType::UserReportV2 => true,
443            ItemType::ReplayEvent => true,
444            ItemType::Session => false,
445            ItemType::Sessions => false,
446            ItemType::Statsd => false,
447            ItemType::MetricBuckets => false,
448            ItemType::ClientReport => false,
449            ItemType::ReplayRecording => false,
450            ItemType::ReplayVideo => false,
451            ItemType::Profile => true,
452            ItemType::CheckIn => false,
453            ItemType::Span => false,
454            ItemType::Log | ItemType::OtelLog => false,
455            ItemType::OtelSpan => false,
456            ItemType::OtelTracesData => false,
457            ItemType::ProfileChunk => false,
458
459            // Since this Relay cannot interpret the semantics of this item, it does not know
460            // whether it requires an event or not. Depending on the strategy, this can cause two
461            // wrong actions here:
462            //  1. return false, but the item requires an event. It is split off by Relay and
463            //     handled separately. If the event is rate limited or filtered, the item still gets
464            //     ingested and needs to be pruned at a later point in the pipeline. This also
465            //     happens with standalone attachments.
466            //  2. return true, but the item does not require an event. It is kept in the same
467            //     envelope and dropped along with the event, even though it should be ingested.
468            // Realistically, most new item types should be ingested largely independent of events,
469            // and the ingest pipeline needs to assume split submission from clients. This makes
470            // returning `false` the safer option.
471            ItemType::Unknown(_) => false,
472        }
473    }

485/// The type of an envelope item.
486#[derive(Clone, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
487pub enum ItemType {
488    /// Event payload encoded in JSON.
489    Event,
490    /// Transaction event payload encoded in JSON.
491    Transaction,
492    /// Security report event payload encoded in JSON.
493    Security,
494    /// Raw payload of an arbitrary attachment.
495    Attachment,
496    /// Multipart form data collected into a stream of JSON tuples.
497    FormData,
498    /// Security report as sent by the browser in JSON.
499    RawSecurity,
500    /// NEL report as sent by the browser.
501    Nel,
502    /// Raw compressed UE4 crash report.
503    UnrealReport,
504    /// User feedback encoded as JSON.
505    UserReport,
506    /// Session update data.
507    Session,
508    /// Aggregated session data.
509    Sessions,
510    /// Individual metrics in text encoding.
511    Statsd,
512    /// Buckets of preaggregated metrics encoded as JSON.
513    MetricBuckets,
514    /// Client internal report (eg: outcomes).
515    ClientReport,
516    /// Profile event payload encoded as JSON.
517    Profile,
518    /// Replay metadata and breadcrumb payload.
519    ReplayEvent,
520    /// Replay Recording data.
521    ReplayRecording,
522    /// Replay Video data.
523    ReplayVideo,
524    /// Monitor check-in encoded as JSON.
525    CheckIn,
526    /// A log from the [OTEL Log format](https://opentelemetry.io/docs/specs/otel/logs/data-model/#log-and-event-record-definition)
527    OtelLog,
528    /// A log for the log product, not internal logs.
529    Log,
530    /// A standalone span.
531    Span,
532    /// A standalone OpenTelemetry span serialized as JSON.
533    OtelSpan,
534    /// An OTLP TracesData container.
535    OtelTracesData,
536    /// UserReport as an Event
537    UserReportV2,
538    /// ProfileChunk is a chunk of a profiling session.
539    ProfileChunk,
540    /// A new item type that is yet unknown by this version of Relay.
541    ///
542    /// By default, items of this type are forwarded without modification. Processing Relays and
543    /// Relays explicitly configured to do so will instead drop those items. This allows
544    /// forward-compatibility with new item types where we expect outdated Relays.
545    Unknown(String),
546    // Keep `Unknown` last in the list. Add new items above `Unknown`.
547}

698#[derive(Clone, Debug, Deserialize, Serialize)]
699pub struct ItemHeaders {
700    /// The type of the item.
701    #[serde(rename = "type")]
702    ty: ItemType,
703
704    /// Content length of the item.
705    ///
706    /// Can be omitted if the item does not contain new lines. In this case, the item payload is
707    /// parsed until the first newline is encountered.
708    #[serde(skip_serializing_if = "Option::is_none")]
709    length: Option<u32>,
710
711    /// The amount of contained items.
712    ///
713    /// This header is required for all items that are transmitted in an envelope [`super::ItemContainer`].
714    /// The amount specified must match the amount of items contained in the container exactly.
715    ///
716    /// Failing to specify the count or a mismatching count will be treated as an invalid envelope.
717    #[serde(skip_serializing_if = "Option::is_none")]
718    item_count: Option<u32>,
719
720    /// If this is an attachment item, this may contain the attachment type.
721    #[serde(skip_serializing_if = "Option::is_none")]
722    attachment_type: Option<AttachmentType>,
723
724    /// Content type of the payload.
725    #[serde(skip_serializing_if = "Option::is_none")]
726    content_type: Option<ContentType>,
727
728    /// If this is an attachment item, this may contain the original file name.
729    #[serde(skip_serializing_if = "Option::is_none")]
730    filename: Option<String>,
731
732    /// The platform this item was produced for.
733    ///
734    /// Currently only used for [`ItemType::ProfileChunk`].
735    /// It contains the same platform as specified in the profile chunk payload,
736    /// hoisted into the header to be able to determine the correct data category.
737    ///
738    /// This is currently considered optional for profile chunks, but may change
739    /// to required in the future.
740    #[serde(skip_serializing_if = "Option::is_none")]
741    platform: Option<String>,
742
743    /// The routing_hint may be used to specify how the envelpope should be routed in when
744    /// published to kafka.
745    ///
746    /// XXX(epurkhiser): This is currently ONLY used for [`ItemType::CheckIn`]'s when publishing
747    /// the envelope into kafka.
748    #[serde(default, skip_serializing_if = "Option::is_none")]
749    routing_hint: Option<Uuid>,
750
751    /// Indicates that this item is being rate limited.
752    ///
753    /// By default, rate limited items are immediately removed from Envelopes. For processing,
754    /// native crash reports still need to be retained. These attachments are marked with the
755    /// `rate_limited` header, which signals to the processing pipeline that the attachment should
756    /// not be persisted after processing.
757    ///
758    /// NOTE: This is internal-only and not exposed into the Envelope.
759    #[serde(default, skip)]
760    rate_limited: bool,
761
762    /// Contains the amount of events this item was generated and aggregated from.
763    ///
764    /// A [metrics buckets](`ItemType::MetricBuckets`) item contains metrics extracted and
765    /// aggregated from (currently) transactions and profiles.
766    ///
767    /// This information can not be directly inferred from the item itself anymore.
768    /// The amount of events this item/metric represents is instead stored here.
769    ///
770    /// NOTE: This is internal-only and not exposed into the Envelope.
771    #[serde(default, skip)]
772    source_quantities: Option<SourceQuantities>,
773
774    /// Flag indicating if metrics have already been extracted from the item.
775    ///
776    /// In order to only extract metrics once from an item while through a
777    /// chain of Relays, a Relay that extracts metrics from an item (typically
778    /// the first Relay) MUST set this flat to true so that upstream Relays do
779    /// not extract the metric again causing double counting of the metric.
780    #[serde(default, skip_serializing_if = "is_false")]
781    metrics_extracted: bool,
782
783    /// Whether or not spans and span metrics have been extracted from a transaction.
784    ///
785    /// This header is set to `true` after both span extraction and span metrics extraction,
786    /// and can be used to skip extraction.
787    ///
788    /// NOTE: This header is also set to `true` for transactions that are themselves extracted
789    /// from spans (the opposite direction), to prevent going in circles.
790    #[serde(default, skip_serializing_if = "is_false")]
791    spans_extracted: bool,
792
793    /// Whether the event has been _fully_ normalized.
794    ///
795    /// If the event has been partially normalized, this flag is false. By
796    /// default, all Relays run some normalization.
797    ///
798    /// Currently only used for events.
799    #[serde(default, skip_serializing_if = "is_false")]
800    fully_normalized: bool,
801
802    /// `false` if the sampling decision is "drop".
803    ///
804    /// In the most common use case, the item is dropped when the sampling decision is "drop".
805    /// For profiles with the feature enabled, however, we keep all profile items and mark the ones
806    /// for which the transaction was dropped as `sampled: false`.
807    #[serde(default = "default_true", skip_serializing_if = "is_true")]
808    sampled: bool,
809
810    /// Tracks whether the item is a backend or ui profile chunk.
811    ///
812    /// NOTE: This is internal-only and not exposed into the Envelope.
813    #[serde(default, skip)]
814    profile_type: Option<ProfileType>,
815
816    /// Other attributes for forward compatibility.
817    #[serde(flatten)]
818    other: BTreeMap<String, Value>,
819}

821/// Container for item quantities that the item was derived from.
822///
823/// For example a metric bucket may be derived and aggregated from multiple transactions.
824#[derive(Copy, Clone, Debug, Default, PartialEq, Eq)]
825pub struct SourceQuantities {
826    /// Transaction quantity.
827    pub transactions: usize,
828    /// Spans quantity.
829    pub spans: usize,
830    /// Profile quantity.
831    pub profiles: usize,
832    /// Total number of buckets.
833    pub buckets: usize,
834}
835
836impl AddAssign for SourceQuantities {
837    fn add_assign(&mut self, other: Self) {
838        let Self {
839            transactions,
840            spans,
841            profiles,
842            buckets,
843        } = self;
844        *transactions += other.transactions;
845        *spans += other.spans;
846        *profiles += other.profiles;
847        *buckets += other.buckets;
848    }
849}
