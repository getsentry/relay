50/// An error that occurs during upload.
51#[derive(Debug, thiserror::Error)]
52pub enum Error {
53    #[error("send failed: {0}")]
54    Send(#[from] RecvError),
55    #[error("request failed: {0}")]
56    UpstreamRequest(#[from] UpstreamRequestError),
57    #[error("request timeout: {0}")]
58    Timeout(#[from] tokio::time::error::Elapsed),
59    #[error("error response from upstream: {0}")]
60    Upstream(#[source] reqwest::Error),
61    #[error("upstream provided invalid location: {0:?}")]
62    InvalidLocation(Option<HeaderValue>),
63    #[error("serializing location failed: {0}")]
64    SerializeFailed(#[from] serde_urlencoded::ser::Error),
65    #[error("failed to sign location")]
66    SigningFailed,
67    #[error("invalid signature: {0}")]
68    InvalidSignature(#[from] SignatureError),
69    #[error("objectstore service unavailable: {0}")]
70    ObjectstoreServiceUnavailable(#[source] SendError),
71    #[cfg(feature = "processing")]
72    #[error("objectstore service: {0}")]
73    Objectstore(#[from] objectstore::Error),
74    #[error("loadshed")]
75    LoadShed,
76    #[error("internal error")]
77    Internal(#[source] http::header::InvalidHeaderValue),
78}
79
80impl Error {
81    fn variant(&self) -> &'static str {
82        match self {
83            Error::Send(_) => "send_failed",
84            Error::UpstreamRequest(_) => "upstream_request",
85            Error::Timeout(_) => "timeout",
86            Error::Upstream(_) => "upstream_response",
87            Error::InvalidLocation(_) => "invalid_location",
88            Error::SigningFailed => "signing_failed",
89            Error::SerializeFailed(_) => "serialize_failed",
90            Error::InvalidSignature(_) => "invalid_signature",
91            Error::ObjectstoreServiceUnavailable(_) => "service_unavailable",
92            #[cfg(feature = "processing")]
93            Error::Objectstore(_) => "objectstore_error",
94            Error::LoadShed => "load_shed",
95            Error::Internal(_) => "internal",
96        }
97    }
98}

235impl<L: UploadLength> InstrumentedSender<L> {
236    fn send(self, result: Result<SignedLocation<L>, Error>) {
237        let result_msg = match &result {
238            Ok(_) => "success",
239            Err(e) => e.variant(),
240        };
241        relay_statsd::metric!(counter(self.metric) += 1, result = result_msg);
242        self.inner.send(result)
243    }
244}

258impl Service {
259    async fn create(
260        &self,
261        Create {
262            project,
263            length,
264            attachment_type,
265        }: Create,
266    ) -> Result<SignedLocation<Provisional>, Error> {
267        match &self.backend {
268            Backend::Upstream { addr } => {
269                let (request, rx) = UploadRequest::create(project, length, attachment_type);
270                addr.send(SendRequest(request));
271                let response = rx.await??;
272                SignedLocation::try_from_response(response)
273            }
274            #[cfg(feature = "processing")]
275            Backend::Objectstore { addr: _, config } => {
276                // We can create & sign a location right here, no need to query the objectstore service.
277                let key = Uuid::now_v7().as_simple().to_string();
278                Location {
279                    project_id: project.scoping.project_id,
280                    key,
281                    length: Provisional(length),
282                    other: Default::default(),
283                }
284                .try_sign(config)
285            }
286        }
287    }
288
289    async fn upload(&self, stream: Stream) -> Result<SignedLocation<Final>, Error> {
290        let Stream {
291            #[cfg_attr(not(feature = "processing"), expect(unused))]
292            received,
293            project,
294            location,
295            stream,
296        } = stream;
297        match &self.backend {
298            Backend::Upstream { addr } => {
299                let (request, rx) = UploadRequest::upload(project, location.try_to_uri()?, stream);
300                addr.send(SendRequest(request));
301                let response = rx.await??;
302                SignedLocation::try_from_response(response)
303            }
304            #[cfg(feature = "processing")]
305            Backend::Objectstore { addr, config } => {
306                let Location {
307                    project_id,
308                    key,
309                    length,
310                    other,
311                } = location.verify(received, config)?;
312
313                let scoping = project.scoping;
314                debug_assert_eq!(scoping.project_id, project_id);
315                debug_assert!(stream.length().is_none_or(|l| Some(l) == length.value()));
316                let byte_counter = stream.byte_counter();
317
318                let key = addr
319                    .send(objectstore::Stream {
320                        organization_id: scoping.organization_id,
321                        project_id,
322                        key,
323                        stream,
324                    })
325                    .await
326                    .map_err(Error::ObjectstoreServiceUnavailable)??
327                    .into_inner();
328                let length = Final(byte_counter.get());
329
330                Location {
331                    project_id,
332                    key,
333                    length,
334                    other,
335                }
336                .try_sign(config)
337            }
338        }
339    }
340
341    async fn timeout<L, F>(&self, future: F) -> Result<SignedLocation<L>, Error>
342    where
343        L: UploadLength,
344        F: IntoFuture<Output = Result<SignedLocation<L>, Error>>,
345    {
346        tokio::time::timeout(self.timeout, future).await?
347    }
348}

413/// An identifier for the upload.
414///
415/// The location can be converted into a URI to be put in the `Location` HTTP header
416/// used by the TUS protocol.
417///
418/// Calling [`Self::try_sign`] appends an `&upload_signature=` query parameter that can later be used
419/// to validate whether the URI (especially the length) has been tampered with.
420#[derive(Debug)]
421pub struct Location<L> {
422    /// Sentry project ID.
423    pub project_id: ProjectId,
424    /// Objectstore identifier.
425    pub key: String,
426    /// Value of the `Upload-Length` header. `None` if `Upload-Defer-Length: 1`.
427    pub length: L,
428    pub other: UploadParams,
429}
430
431impl<L: UploadLength> Location<L> {
432    fn try_to_uri(&self) -> Result<String, Error> {
433        let Location {
434            project_id,
435            key,
436            length,
437            other,
438        } = self;
439        #[derive(Debug, Serialize)]
440        struct QueryParams<'a> {
441            pub upload_length: Option<usize>,
442            #[serde(flatten)]
443            pub other: &'a UploadParams,
444        }
445        let params = QueryParams {
446            upload_length: length.value(),
447            other,
448        };
449        let query = serde_urlencoded::to_string(params)?;
450        match query.as_str() {
451            "" => Ok(format!("/api/{project_id}/upload/{key}/")),
452            _ => Ok(format!("/api/{project_id}/upload/{key}/?{query}")),
453        }
454    }
455
456    #[cfg(feature = "processing")]
457    fn try_sign(self, config: &Config) -> Result<SignedLocation<L>, Error> {
458        let uri = self.try_to_uri()?;
459        let secret_key = config.upload_signing_key().ok_or(Error::SigningFailed)?;
460        let signature = secret_key.sign_with_header(
461            uri.as_bytes(),
462            &SignatureHeader {
463                timestamp: Utc::now(),
464                signature_algorithm: None,
465            },
466        );
467
468        Ok(SignedLocation {
469            location: self,
470            signature,
471        })
472    }
473}

482/// Query parameters for the upload endpoint.
483#[derive(Debug, Deserialize)]
484#[serde(bound = "L: UploadLength")]
485pub struct LocationQueryParams<L: UploadLength> {
486    #[serde(alias = "length")]
487    pub upload_length: L,
488    #[serde(alias = "signature")]
489    pub upload_signature: String,
490    #[serde(flatten)]
491    pub other: UploadParams,
492}
493
494#[derive(Debug, Default, Serialize)]
495pub struct UploadParams(BTreeMap<String, String>);
496
497impl<'de> Deserialize<'de> for UploadParams {
498    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
499    where
500        D: serde::Deserializer<'de>,
501    {
502        struct UploadParamsVisitor;
503
504        impl<'de> serde::de::Visitor<'de> for UploadParamsVisitor {
505            type Value = UploadParams;
506
507            fn expecting(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
508                formatter.write_str("upload query parameters")
509            }
510
511            fn visit_map<A>(self, mut map: A) -> Result<Self::Value, A::Error>
512            where
513                A: serde::de::MapAccess<'de>,
514            {
515                let mut upload_params = BTreeMap::new();
516
517                while let Some(key) = map.next_key::<&str>()? {
518                    if key.starts_with("upload_") {
519                        upload_params.insert(key.to_owned(), map.next_value()?);
520                    } else {
521                        map.next_value::<serde::de::IgnoredAny>()?;
522                    }
523                }
524
525                Ok(UploadParams(upload_params))
526            }
527        }
528
529        deserializer.deserialize_map(UploadParamsVisitor)
530    }
531}

533/// A verifiable [`Location`] signed by this Relay or an upstream Relay.
534#[derive(Debug)]
535pub struct SignedLocation<L: UploadLength> {
536    location: Location<L>,
537    signature: Signature,
538}
539
540impl<L: UploadLength> SignedLocation<L> {
541    /// Creates an unverified location from path and query params.
542    ///
543    /// Call `verify` to make sure the signature is correct.
544    pub fn from_parts(
545        project_id: ProjectId,
546        key: String,
547        length: L,
548        signature: String,
549        other: UploadParams,
550    ) -> Self {
551        Self {
552            location: Location {
553                project_id,
554                key,
555                length,
556                other,
557            },
558            signature: Signature(signature),
559        }
560    }
561
562    /// Converts the location into an URI for future reference.
563    pub fn into_header_value(self) -> Result<HeaderValue, Error> {
564        HeaderValue::from_str(&self.try_to_uri()?).map_err(Error::Internal)
565    }
566
567    fn try_to_uri(&self) -> Result<String, Error> {
568        let Self {
569            location,
570            signature,
571        } = self;
572
573        let mut uri = location.try_to_uri()?;
574        uri.push(if uri.ends_with('/') { '?' } else { '&' });
575        uri.push_str("upload_signature=");
576        uri.push_str(&signature.to_string());
577        Ok(uri)
578    }
579
580    /// Converts the signed location into a location object.
581    ///
582    /// Fails if the signature is outdated or incorrect.
583    #[cfg(feature = "processing")]
584    pub fn verify(self, received: DateTime<Utc>, config: &Config) -> Result<Location<L>, Error> {
585        let mut result = Err(SignatureError::Unverifiable);
586        let location = self.location.try_to_uri()?;
587        let max_age = chrono::Duration::seconds(config.upload().max_age);
588
589        if let Some(public_key) = &config
590            .upload()
591            .credentials
592            .as_ref()
593            .map(|c| &c.verification_key)
594        {
595            result = self
596                .signature
597                .verify(location.as_bytes(), public_key, received, max_age);
598        }
599
600        // For the transition phase, check the general purpose signature even when there is a
601        // special-purpose upload key, because the URL may have been signed by an old instance.
602        // This can be simplified after the rollout.
603        if matches!(&result, Err(SignatureError::Unverifiable))
604            && let Some(public_key) = config.credentials().map(|c| &c.public_key)
605        {
606            result = self
607                .signature
608                .verify(location.as_bytes(), public_key, received, max_age);
609        }
610
611        result?;
612
613        Ok(self.location)
614    }
615}

617impl<L> SignedLocation<L>
618where
619    L: UploadLength,
620    LocationQueryParams<L>: for<'de> Deserialize<'de>,
621{
622    fn try_from_response(response: Response) -> Result<Self, Error> {
623        match response.0.error_for_status() {
624            Ok(response) => {
625                let header = response
626                    .headers()
627                    .get(hyper::header::LOCATION)
628                    .ok_or(Error::InvalidLocation(None))?;
629                let uri = header
630                    .to_str()
631                    .map_err(|_| Error::InvalidLocation(Some(header.clone())))?;
632                Self::try_from_str(uri).ok_or(Error::InvalidLocation(Some(header.clone())))
633            }
634            Err(e) => Err(Error::Upstream(e)),
635        }
636    }
637
638    pub fn try_from_str(uri: &str) -> Option<Self> {
639        static ROUTER: std::sync::LazyLock<matchit::Router<()>> = std::sync::LazyLock::new(|| {
640            let mut router = matchit::Router::new();
641            router
642                .insert(UPLOAD_PATCH_PATH, ())
643                .expect("valid route pattern");
644            router
645        });
646
647        let (path, query) = uri.split_once('?')?;
648        let matched = ROUTER.at(path).ok()?;
649        let LocationPath { project_id, key } = LocationPath {
650            project_id: matched.params.get("project_id")?.parse().ok()?,
651            key: matched.params.get("key")?.to_owned(),
652        };
653
654        // Parse query parameters.
655        let LocationQueryParams {
656            upload_length,
657            upload_signature,
658            other,
659        } = serde_urlencoded::from_str(query).ok()?;
660
661        Some(Self::from_parts(
662            project_id,
663            key,
664            upload_length,
665            upload_signature,
666            other,
667        ))
668    }
669}

683/// An upstream request made to the `/upload` endpoint.

690impl UploadRequest {
691    fn create(
692        project: ProjectContext,
693        length: Option<usize>,
694        attachment_type: Option<AttachmentType>,
695    ) -> (
696        Self,
697        oneshot::Receiver<Result<Response, UpstreamRequestError>>,
698    ) {
699        let (sender, rx) = oneshot::channel();
700
701        (
702            Self {
703                project,
704                kind: RequestKind::Create {
705                    length,
706                    attachment_type,
707                },
708                sender,
709            },
710            rx,
711        )
712    }
713
714    fn upload(
715        project: ProjectContext,
716        uri: String,
717        stream: BoundedStream<MeteredStream<ByteStream>>,
718    ) -> (
719        Self,
720        oneshot::Receiver<Result<Response, UpstreamRequestError>>,
721    ) {
722        let (sender, rx) = oneshot::channel();
723        (
724            Self {
725                project,
726                kind: RequestKind::Upload {
727                    uri,
728                    stream: TakeOnce::new(stream),
729                    encoding: HttpEncoding::Zstd, // just a default, will be overwritten by .configure()
730                },
731                sender,
732            },
733            rx,
734        )
735    }

757        match self.kind {
758            RequestKind::Create { .. } => Method::POST,
759            RequestKind::Upload { .. } => Method::PATCH,
760        }
761    }
762
763    fn path(&self) -> Cow<'_, str> {
764        let project_id = self.project.scoping.project_id;
765        match &self.kind {
766            RequestKind::Create { .. } => Cow::Owned(format!("/api/{project_id}/upload/")),
767            RequestKind::Upload { uri, .. } => Cow::Borrowed(uri),
768        }
769    }
770
771    fn route(&self) -> &'static str {
772        "upload"
773    }
774
775    fn respond(
776        self: Box<Self>,
777        result: Result<Response, UpstreamRequestError>,
778    ) -> std::pin::Pin<Box<dyn Future<Output = ()> + Send + Sync>> {
779        Box::pin(async move {
780            let _ = self.sender.send(result);
781        })
782    }
783
784    fn retry(&self) -> bool {
785        match &self.kind {
786            RequestKind::Create { .. } => true,
787            // Once the body has been polled, it cannot be replayed — give up instead.
788            RequestKind::Upload { stream, .. } => !stream.is_taken(),
789        }
790    }
791
792    fn intercept_status_errors(&self) -> bool {
793        true
794    }
795
796    fn set_relay_id(&self) -> bool {
797        true // needed for trusted requests with `Upload-Defer-Length: 1`
798    }
799
800    fn build(&mut self, builder: &mut RequestBuilder) -> Result<(), HttpError> {
801        match &mut self.kind {
802            RequestKind::Create {
803                length,
804                attachment_type,
805            } => {
806                tus::add_creation_headers(*length, *attachment_type, builder)?;
807            }
808            RequestKind::Upload {
809                uri: _,
810                stream,
811                encoding,
812            } => {
813                let Some(body) = RetryableStream::new(stream.clone()) else {
814                    relay_log::error!("upload request stream was already consumed");
815                    return Err(HttpError::Misconfigured);
816                };
817                tus::add_upload_headers(builder);
818
819                let body = encode_body(body, *encoding);
820                builder.content_encoding(*encoding);
821
822                builder.body(reqwest::Body::wrap_stream(body));
823            }
824        };
825
826        let project_key = self.project.scoping.project_key;
827        builder.header("X-Sentry-Auth", format!("Sentry sentry_key={project_key}"));
828        builder.timeout(Duration::MAX); // rely on service timeout to cancel requests
829
830        Ok(())
831    }
832
833    fn configure(&mut self, config: &Config) {
834        if let RequestKind::Upload { encoding, .. } = &mut self.kind {
835            *encoding = config.http_encoding();
836        }
837    }
838}

840fn encode_body<S>(stream: S, encoding: HttpEncoding) -> ByteStream
841where
842    S: futures::Stream<Item = std::io::Result<Bytes>> + Send + 'static,
843{
844    let reader = BufReader::new(StreamReader::new(stream));
845    match encoding {
846        HttpEncoding::Identity => ReaderStream::new(reader).boxed(),
847        HttpEncoding::Deflate => ReaderStream::new(DeflateEncoder::new(reader)).boxed(),
848        HttpEncoding::Gzip => ReaderStream::new(GzipEncoder::new(reader)).boxed(),
849        HttpEncoding::Br => ReaderStream::new(BrotliEncoder::new(reader)).boxed(),
850        HttpEncoding::Zstd => ReaderStream::new(ZstdEncoder::new(reader)).boxed(),
851    }
852}
