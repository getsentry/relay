154/// Defines the possible outcomes from processing an event.
155#[derive(Clone, Debug, PartialEq, Eq, Hash)]
156pub enum Outcome {
157    /// The event has been accepted and handled completely.
158    ///
159    /// This is only emitted for items going from Relay to Snuba directly.
160    #[allow(dead_code)]
161    Accepted,
162
163    /// The event has been filtered due to a configured filter.
164    Filtered(FilterStatKey),
165
166    /// The event has been filtered by a Sampling Rule
167    FilteredSampling(RuleCategories),
168
169    /// The event has been rate limited.
170    RateLimited(Option<ReasonCode>),
171
172    /// The event/metric has been cardinality limited.
173    ///
174    /// Contains the [ID](relay_cardinality::CardinalityLimit::id)
175    /// of the most specific [CardinalityLimit](relay_cardinality::CardinalityLimit) that was exceeded.
176    #[cfg(feature = "processing")]
177    CardinalityLimited(String),
178
179    /// The event has been discarded because of invalid data.
180    Invalid(DiscardReason),
181
182    /// Reserved but unused in Relay.
183    #[allow(dead_code)]
184    Abuse,
185
186    /// The event has already been discarded on the client side.
187    ClientDiscard(String),
188}
189
190impl Outcome {
191    /// Returns the raw numeric value of this outcome for the JSON and Kafka schema.
192    pub fn to_outcome_id(&self) -> OutcomeId {
193        match self {
194            Outcome::Filtered(_) | Outcome::FilteredSampling(_) => OutcomeId::FILTERED,
195            Outcome::RateLimited(_) => OutcomeId::RATE_LIMITED,
196            #[cfg(feature = "processing")]
197            Outcome::CardinalityLimited(_) => OutcomeId::CARDINALITY_LIMITED,
198            Outcome::Invalid(_) => OutcomeId::INVALID,
199            Outcome::Abuse => OutcomeId::ABUSE,
200            Outcome::ClientDiscard(_) => OutcomeId::CLIENT_DISCARD,
201            Outcome::Accepted => OutcomeId::ACCEPTED,
202        }
203    }
204
205    /// Returns the `reason` code field of this outcome.
206    pub fn to_reason(&self) -> Option<Cow<'_, str>> {
207        match self {
208            Outcome::Invalid(DiscardReason::TooLarge(too_large_reason)) => Some(Cow::Owned(
209                format!("too_large:{}", too_large_reason.as_str()),
210            )),
211            Outcome::Invalid(discard_reason) => Some(Cow::Borrowed(discard_reason.name())),
212            Outcome::Filtered(filter_key) => Some(filter_key.clone().name()),
213            Outcome::FilteredSampling(rule_ids) => Some(Cow::Owned(format!("Sampled:{rule_ids}"))),
214            Outcome::RateLimited(code_opt) => {
215                code_opt.as_ref().map(|code| Cow::Borrowed(code.as_str()))
216            }
217            #[cfg(feature = "processing")]
218            Outcome::CardinalityLimited(id) => Some(Cow::Borrowed(id)),
219            Outcome::ClientDiscard(discard_reason) => Some(Cow::Borrowed(discard_reason)),
220            Outcome::Abuse => None,
221            Outcome::Accepted => None,
222        }
223    }
224
225    /// Returns true if there is a bug or an infrastructure problem causing event loss.
226    ///
227    /// This can happen when we introduce bugs or during incidents.
228    ///
229    /// During healthy operation, this should always return false.
230    pub fn is_unexpected(&self) -> bool {
231        matches!(
232            self,
233            Outcome::Invalid(
234                DiscardReason::Internal
235                    | DiscardReason::ProjectState
236                    | DiscardReason::ProjectStatePii,
237            )
238        )
239    }
240}

242impl fmt::Display for Outcome {
243    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
244        match self {
245            Outcome::Filtered(key) => write!(f, "filtered by {key}"),
246            Outcome::FilteredSampling(rule_ids) => write!(f, "sampling rule {rule_ids}"),
247            Outcome::RateLimited(None) => write!(f, "rate limited"),
248            Outcome::RateLimited(Some(reason)) => write!(f, "rate limited with reason {reason}"),
249            #[cfg(feature = "processing")]
250            Outcome::CardinalityLimited(id) => write!(f, "cardinality limited ({id})"),
251            Outcome::Invalid(DiscardReason::Internal) => write!(f, "internal error"),
252            Outcome::Invalid(reason) => write!(f, "invalid data ({reason})"),
253            Outcome::Abuse => write!(f, "abuse limit reached"),
254            Outcome::ClientDiscard(reason) => write!(f, "discarded by client ({reason})"),
255            Outcome::Accepted => write!(f, "accepted"),
256        }
257    }

345

472

761/// Raw representation of an outcome for serialization.
762///
763/// The JSON serialization of this structure is placed on the Kafka topic and used in the HTTP
764/// endpoints. To create a new outcome, use [`TrackOutcome`], instead.
765#[derive(Debug, Serialize, Deserialize, Clone)]
766pub struct TrackRawOutcome {
767    /// The timespan of the event outcome.
768    timestamp: String,
769    /// Organization id.
770    #[serde(default, skip_serializing_if = "Option::is_none")]
771    org_id: Option<OrganizationId>,
772    /// Project id.
773    project_id: ProjectId,
774    /// The DSN project key id.
775    #[serde(default, skip_serializing_if = "Option::is_none")]
776    key_id: Option<u64>,
777    /// The outcome.
778    outcome: OutcomeId,
779    /// Reason for the outcome.
780    #[serde(default, skip_serializing_if = "Option::is_none")]
781    reason: Option<String>,
782    /// The event id.
783    #[serde(default, skip_serializing_if = "Option::is_none")]
784    event_id: Option<EventId>,
785    /// The client ip address.
786    #[serde(default, skip_serializing_if = "Option::is_none")]
787    remote_addr: Option<String>,
788    /// The source of the outcome (which Relay sent it)
789    #[serde(default, skip_serializing_if = "Option::is_none")]
790    source: Option<String>,
791    /// The event's data category.
792    #[serde(default, skip_serializing_if = "Option::is_none")]
793    pub category: Option<u8>,
794    /// The number of events or total attachment size in bytes.
795    #[serde(default, skip_serializing_if = "Option::is_none")]
796    pub quantity: Option<u32>,
797}
798
799impl TrackRawOutcome {
800    fn from_outcome(msg: TrackOutcome, config: &Config) -> Self {
801        let reason = msg.outcome.to_reason().map(|reason| reason.to_string());
802
803        // convert to a RFC 3339 formatted date with the shape YYYY-MM-DDTHH:MM:SS.mmmmmmZ
804        // e.g. something like: "2019-09-29T09:46:40.123456Z"
805        let timestamp = msg.timestamp.to_rfc3339_opts(SecondsFormat::Micros, true);
806
807        let org_id = match msg.scoping.organization_id.value() {
808            0 => None,
809            id => Some(OrganizationId::new(id)),
810        };
811
812        // since TrackOutcome objects come only from this Relay (and not any downstream
813        // Relays), set the source to whatever our current outcome source is.
814        let source = config.outcome_source().map(str::to_owned);
815
816        TrackRawOutcome {
817            timestamp,
818            org_id,
819            project_id: msg.scoping.project_id,
820            key_id: msg.scoping.key_id,
821            outcome: msg.outcome.to_outcome_id(),
822            reason,
823            event_id: msg.event_id,
824            remote_addr: msg.remote_addr.map(|addr| addr.to_string()),
825            source,
826            category: msg.category.value(),
827            quantity: Some(msg.quantity),
828        }
829    }
830
831    #[cfg(feature = "processing")]
832    fn is_billing(&self) -> bool {
833        matches!(self.outcome, OutcomeId::ACCEPTED | OutcomeId::RATE_LIMITED)
834    }
835}

867/// Outcome producer backend via HTTP as [`TrackRawOutcome`].
868#[derive(Debug)]
869struct HttpOutcomeProducer {
870    config: Arc<Config>,
871    upstream_relay: Addr<UpstreamRelay>,
872    unsent_outcomes: Vec<TrackRawOutcome>,
873    flush_handle: SleepHandle,
874}
875
876impl HttpOutcomeProducer {
877    pub fn new(config: Arc<Config>, upstream_relay: Addr<UpstreamRelay>) -> Self {
878        Self {
879            config,
880            upstream_relay,
881            unsent_outcomes: Vec::new(),
882            flush_handle: SleepHandle::idle(),
883        }
884    }
885
886    fn send_batch(&mut self) {
887        self.flush_handle.reset();
888
889        if self.unsent_outcomes.is_empty() {
890            relay_log::warn!("unexpected send_batch scheduled with no outcomes to send");
891            return;
892        } else {
893            relay_log::trace!(
894                "sending outcome batch of size {}",
895                self.unsent_outcomes.len()
896            );
897        }
898
899        let request = SendOutcomes {
900            outcomes: mem::take(&mut self.unsent_outcomes),
901        };
902
903        let upstream_relay = self.upstream_relay.clone();
904
905        relay_system::spawn!(async move {
906            match upstream_relay.send(SendQuery(request)).await {
907                Ok(_) => relay_log::trace!("outcome batch sent"),
908                Err(error) => {
909                    relay_log::error!(error = &error as &dyn Error, "outcome batch sending failed")
910                }
911            }
912        });
913    }
914
915    fn handle_message(&mut self, message: TrackRawOutcome) {
916        relay_log::trace!("batching outcome");
917        self.unsent_outcomes.push(message);
918
919        if self.unsent_outcomes.len() >= self.config.outcome_batch_size() {
920            self.send_batch();
921        } else if self.flush_handle.is_idle() {
922            self.flush_handle.set(self.config.outcome_batch_interval());
923        }
924    }
925}
926
927impl Service for HttpOutcomeProducer {
928    type Interface = TrackRawOutcome;
929
930    async fn run(mut self, mut rx: relay_system::Receiver<Self::Interface>) {
931        loop {
932            tokio::select! {
933                // Prioritize flush over receiving messages to prevent starving.
934                biased;
935
936                () = &mut self.flush_handle => self.send_batch(),
937                Some(message) = rx.recv() => self.handle_message(message),
938                else => break,
939            }
940        }
941    }
942}

944/// Outcome producer backend via HTTP as [`ClientReport`].
945#[derive(Debug)]
946struct ClientReportOutcomeProducer {
947    flush_interval: Duration,
948    unsent_reports: BTreeMap<Scoping, Vec<ClientReport>>,
949    flush_handle: SleepHandle,
950    envelope_processor: Addr<EnvelopeProcessor>,
951}
952
953impl ClientReportOutcomeProducer {
954    fn new(config: &Config, envelope_processor: Addr<EnvelopeProcessor>) -> Self {
955        Self {
956            // Use same batch interval as outcome aggregator
957            flush_interval: Duration::from_secs(config.outcome_aggregator().flush_interval),
958            unsent_reports: BTreeMap::new(),
959            flush_handle: SleepHandle::idle(),
960            envelope_processor,
961        }
962    }
963
964    fn flush(&mut self) {
965        relay_log::trace!("flushing client reports");
966        self.flush_handle.reset();
967
968        let unsent_reports = mem::take(&mut self.unsent_reports);
969        for (scoping, client_reports) in unsent_reports.into_iter() {
970            self.envelope_processor.send(SubmitClientReports {
971                client_reports,
972                scoping,
973            });
974        }
975    }
976
977    fn handle_message(&mut self, msg: TrackOutcome) {
978        let mut client_report = ClientReport {
979            timestamp: Some(UnixTimestamp::from_secs(
980                msg.timestamp.timestamp().try_into().unwrap_or(0),
981            )),
982            ..Default::default()
983        };
984
985        // The outcome type determines what field to place the outcome in:
986        let discarded_events = match msg.outcome {
987            Outcome::Filtered(_) => &mut client_report.filtered_events,
988            Outcome::FilteredSampling(_) => &mut client_report.filtered_sampling_events,
989            Outcome::RateLimited(_) => &mut client_report.rate_limited_events,
990            Outcome::Invalid(DiscardReason::InvalidSignature | DiscardReason::MissingSignature) => {
991                &mut client_report.discarded_events
992            }
993            _ => {
994                relay_log::debug!(
995                    "Outcome '{}' cannot be converted to client report",
996                    msg.outcome
997                );
998                return;
999            }
1000        };
1001
1002        // Now that we know where to put it, let's create a DiscardedEvent
1003        let discarded_event = DiscardedEvent {
1004            reason: msg.outcome.to_reason().unwrap_or_default().to_string(),
1005            category: msg.category,
1006            quantity: msg.quantity,
1007        };
1008        discarded_events.push(discarded_event);
1009
1010        self.unsent_reports
1011            .entry(msg.scoping)
1012            .or_default()
1013            .push(client_report);
1014
1015        if self.flush_interval == Duration::ZERO {
1016            // Flush immediately. Useful for integration tests.
1017            self.flush();
1018        } else if self.flush_handle.is_idle() {
1019            self.flush_handle.set(self.flush_interval);
1020        }
1021    }
1022}

1024impl Service for ClientReportOutcomeProducer {
1025    type Interface = TrackOutcome;
1026
1027    async fn run(mut self, mut rx: relay_system::Receiver<Self::Interface>) {
1028        loop {
1029            tokio::select! {
1030                // Prioritize flush over receiving messages to prevent starving.
1031                biased;
1032
1033                () = &mut self.flush_handle => self.flush(),
1034                Some(message) = rx.recv() => self.handle_message(message),
1035                else => break,
1036            }
1037        }
1038    }

1125#[derive(Debug)]
1126enum OutcomeBroker {
1127    ClientReport(Addr<TrackOutcome>),
1128    Http(Addr<TrackRawOutcome>),
1129    #[cfg(feature = "processing")]
1130    Kafka(KafkaOutcomesProducer),
1131    Disabled,
1132}
1133
1134impl OutcomeBroker {
1135    fn handle_message(&self, message: OutcomeProducer, config: &Config) {
1136        match message {
1137            OutcomeProducer::TrackOutcome(msg) => self.handle_track_outcome(msg, config),
1138            OutcomeProducer::TrackRawOutcome(msg) => self.handle_track_raw_outcome(msg),
1139        }
1140    }
1141
1142    #[cfg(feature = "processing")]
1143    fn send_kafka_message(
1144        &self,
1145        producer: &KafkaOutcomesProducer,
1146        message: TrackRawOutcome,
1147    ) -> Result<(), OutcomeError> {
1148        relay_log::trace!("Tracking kafka outcome: {message:?}");
1149
1150        let payload = serde_json::to_string(&message).map_err(OutcomeError::SerializationError)?;
1151
1152        // At the moment, we support outcomes with optional EventId.
1153        // Here we create a fake EventId, when we don't have the real one, so that we can
1154        // create a kafka message key that spreads the events nicely over all the
1155        // kafka consumer groups.
1156        let key = message.event_id.unwrap_or_default().0;
1157
1158        // Dispatch to the correct topic and cluster based on the kind of outcome.
1159        let topic = if message.is_billing() {
1160            KafkaTopic::OutcomesBilling
1161        } else {
1162            KafkaTopic::Outcomes
1163        };
1164
1165        let result = producer.client.send(
1166            topic,
1167            Some(key.into_bytes()),
1168            None,
1169            "outcome",
1170            payload.as_bytes(),
1171        );
1172
1173        match result {
1174            Ok(_) => Ok(()),
1175            Err(kafka_error) => Err(OutcomeError::SendFailed(kafka_error)),
1176        }
1177    }
1178
1179    fn handle_track_outcome(&self, message: TrackOutcome, config: &Config) {
1180        match self {
1181            #[cfg(feature = "processing")]
1182            Self::Kafka(kafka_producer) => {
1183                send_outcome_metric(&message, "kafka");
1184                let raw_message = TrackRawOutcome::from_outcome(message, config);
1185                if let Err(error) = self.send_kafka_message(kafka_producer, raw_message) {
1186                    relay_log::error!(error = &error as &dyn Error, "failed to produce outcome");
1187                }
1188            }
1189            Self::ClientReport(producer) => {
1190                send_outcome_metric(&message, "client_report");
1191                producer.send(message);
1192            }
1193            Self::Http(producer) => {
1194                send_outcome_metric(&message, "http");
1195                producer.send(TrackRawOutcome::from_outcome(message, config));
1196            }
1197            Self::Disabled => (),
1198        }
1199    }
1200
1201    fn handle_track_raw_outcome(&self, message: TrackRawOutcome) {
1202        match self {
1203            #[cfg(feature = "processing")]
1204            Self::Kafka(kafka_producer) => {
1205                send_outcome_metric(&message, "kafka");
1206                if let Err(error) = self.send_kafka_message(kafka_producer, message) {
1207                    relay_log::error!(error = &error as &dyn Error, "failed to produce outcome");
1208                }
1209            }
1210            Self::Http(producer) => {
1211                send_outcome_metric(&message, "http");
1212                producer.send(message);
1213            }
1214            Self::ClientReport(_) => (),
1215            Self::Disabled => (),
1216        }
1217    }
1218}

1243/// Service implementing the [`OutcomeProducer`] interface.
1244#[derive(Debug)]
1245pub struct OutcomeProducerService {
1246    config: Arc<Config>,
1247    inner: ProducerInner,
1248}
1249
1250impl OutcomeProducerService {
1251    pub fn create(
1252        config: Arc<Config>,
1253        upstream_relay: Addr<UpstreamRelay>,
1254        envelope_processor: Addr<EnvelopeProcessor>,
1255    ) -> anyhow::Result<Self> {
1256        let inner = match config.emit_outcomes() {
1257            #[cfg(feature = "processing")]
1258            EmitOutcomes::AsOutcomes if config.processing_enabled() => {
1259                // We emit raw outcomes, and accept raw outcomes emitted by downstream Relays
1260                relay_log::info!("Configured to emit outcomes via kafka");
1261                ProducerInner::Kafka(KafkaOutcomesProducer::create(&config)?)
1262            }
1263            EmitOutcomes::AsOutcomes => {
1264                relay_log::info!("Configured to emit outcomes via http");
1265                ProducerInner::Http(HttpOutcomeProducer::new(
1266                    Arc::clone(&config),
1267                    upstream_relay,
1268                ))
1269            }
1270            EmitOutcomes::AsClientReports => {
1271                // We emit client reports, and we do NOT accept raw outcomes
1272                relay_log::info!("Configured to emit outcomes as client reports");
1273                ProducerInner::ClientReport(ClientReportOutcomeProducer::new(
1274                    &config,
1275                    envelope_processor,
1276                ))
1277            }
1278            EmitOutcomes::None => {
1279                relay_log::info!("Configured to drop all outcomes");
1280                ProducerInner::Disabled
1281            }
1282        };
1283
1284        Ok(Self { config, inner })
1285    }
1286}

1288impl Service for OutcomeProducerService {
1289    type Interface = OutcomeProducer;
1290
1291    async fn run(self, mut rx: relay_system::Receiver<Self::Interface>) {
1292        let Self { config, inner } = self;
1293
1294        let broker = inner.start();
1295
1296        relay_log::info!("OutcomeProducer started.");
1297        while let Some(message) = rx.recv().await {
1298            broker.handle_message(message, &config);
1299        }
1300        relay_log::info!("OutcomeProducer stopped.");
1301    }
