9/// Converts a session [`Bucket`] into an EAP [`TraceItem`].
10pub fn to_trace_item(scoping: Scoping, bucket: Bucket, retention: u16) -> Option<TraceItem> {
11    if bucket.name.namespace() != MetricNamespace::Sessions {
12        return None;
13    }
14
15    // Currently max 5 tags + 1 metric value.
16    let mut attributes = HashMap::with_capacity(6);
17
18    match bucket.name.try_name()? {
19        "session" => {
20            let BucketValue::Counter(v) = bucket.value else {
21                return None;
22            };
23            // The metric counter uses floats to represent a counter, but this value represent an
24            // integer, the total amount of sessions. Restore the original type and store it as
25            // such (an integer).
26            let count = v.to_f64() as i64;
27            attributes.insert(
28                "session_count".to_owned(),
29                AnyValue {
30                    value: Some(any_value::Value::IntValue(count)),
31                },
32            );
33        }
34        "user" => {
35            let BucketValue::Set(set) = &bucket.value else {
36                return None;
37            };
38            attributes.insert("user_id_hash".to_owned(), set_to_attribute_value(set));
39        }
40        "error" => {
41            let BucketValue::Set(set) = &bucket.value else {
42                return None;
43            };
44            attributes.insert(
45                "errored_session_id_hash".to_owned(),
46                set_to_attribute_value(set),
47            );
48        }
49        _ => return None,
50    }
51
52    for (name, value) in bucket.tags.into_iter().filter_map(tag_to_attribute) {
53        attributes.insert(name, value);
54    }
55
56    let uuid = uuid::Uuid::new_v4();
57    Some(TraceItem {
58        organization_id: scoping.organization_id.value(),
59        project_id: scoping.project_id.value(),
60        trace_id: uuid.to_string(),
61        item_id: uuid_to_item_id(uuid),
62        item_type: TraceItemType::UserSession.into(),
63        timestamp: Some(prost_types::Timestamp {
64            seconds: bucket.timestamp.as_secs() as i64,
65            nanos: 0,
66        }),
67        received: bucket
68            .metadata
69            .received_at
70            .map(|ts| prost_types::Timestamp {
71                seconds: ts.as_secs() as i64,
72                nanos: 0,
73            }),
74        retention_days: retention.into(),
75        downsampled_retention_days: retention.into(),
76        attributes,
77        client_sample_rate: 1.0,
78        server_sample_rate: 1.0,
79    })
80}

82fn set_to_attribute_value(set: &BTreeSet<u32>) -> AnyValue {
83    let values = set
84        .iter()
85        .map(|v| i64::from(*v))
86        .map(any_value::Value::IntValue)
87        .map(|value| AnyValue { value: Some(value) })
88        .collect();
89
90    AnyValue {
91        value: Some(any_value::Value::ArrayValue(ArrayValue { values })),
92    }
93}
94
95fn tag_to_attribute((name, value): (String, String)) -> Option<(String, AnyValue)> {
96    let name = match name.as_str() {
97        "session.status" => "status".to_owned(),
98        "release" | "environment" | "sdk" | "abnormal_mechanism" => name,
99        _ => return None,
100    };
101
102    let value = AnyValue {
103        value: Some(any_value::Value::StringValue(value)),
104    };
105
106    Some((name, value))
107}
