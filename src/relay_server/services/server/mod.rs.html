30/// Set the number of keep-alive retransmissions to be carried out before declaring that remote end
31/// is not available.
32const KEEPALIVE_RETRIES: u32 = 5;
33
34/// Set a timeout for reading client request headers. If a client does not transmit the entire
35/// header within this time, the connection is closed.
36const CLIENT_HEADER_TIMEOUT: Duration = Duration::from_secs(5);

38/// Only compress responses above this configured size, in bytes.
39///
40/// Small responses don't benefit from compression,
41/// additionally the envelope endpoint which returns the event id
42/// should not be compressed due to a bug in the Unity SDK.
43const COMPRESSION_MIN_SIZE: u16 = 128;

45/// Indicates the type of failure of the server.

60/// Build the axum application with all routes and middleware.
61fn make_app(service: ServiceState, f: impl FnOnce(&Config) -> axum::Router<ServiceState>) -> App {
62    // Build the router middleware into a single service which runs _after_ routing. Service
63    // builder order defines layers added first will be called first. This means:
64    //  - Requests go from top to bottom
65    //  - Responses go from bottom to top
66    let middleware = ServiceBuilder::new()
67        .layer(BodyTimingLayer)
68        .layer(axum::middleware::from_fn(middlewares::metrics))
69        .layer(CatchPanicLayer::custom(middlewares::handle_panic))
70        .layer(SetResponseHeaderLayer::overriding(
71            header::SERVER,
72            HeaderValue::from_static(constants::SERVER),
73        ))
74        .layer(SetResponseHeaderLayer::overriding(
75            HeaderName::from_static("cross-origin-resource-policy"),
76            HeaderValue::from_static("cross-origin"),
77        ))
78        .layer(NewSentryLayer::new_from_top())
79        .layer(SentryHttpLayer::new().enable_transaction())
80        .layer(middlewares::trace_http_layer())
81        .map_request(middlewares::remove_empty_encoding)
82        .layer(RequestDecompressionLayer::new())
83        .layer(
84            CompressionLayer::new()
85                .compress_when(SizeAbove::new(COMPRESSION_MIN_SIZE).and(DefaultPredicate::new())),
86        );
87
88    let router = f(service.config()).layer(middleware).with_state(service);
89
90    // Add middlewares that need to run _before_ routing, which need to wrap the router. This are
91    // especially middlewares that modify the request path for the router:
92    NormalizePath::new(router)
93}

95fn listen(addr: SocketAddr, config: &Config) -> Result<TcpListener, ServerError> {
96    let socket = match addr {
97        SocketAddr::V4(_) => TcpSocket::new_v4(),
98        SocketAddr::V6(_) => TcpSocket::new_v6(),
99    }?;
100
101    #[cfg(all(unix, not(target_os = "solaris"), not(target_os = "illumos")))]
102    socket.set_reuseport(true)?;
103    socket.bind(addr)?;
104    Ok(socket.listen(config.tcp_listen_backlog())?.into_std()?)
105}

107async fn serve(listener: TcpListener, app: App, config: &Config) -> std::io::Result<()> {
108    let handle = Handle::new();
109
110    let acceptor = self::acceptor::RelayAcceptor::new()
111        .tcp_keepalive(config.keepalive_timeout(), KEEPALIVE_RETRIES)
112        .idle_timeout(config.idle_timeout());
113
114    let mut server = axum_server::from_tcp(listener)
115        .acceptor(acceptor)
116        .handle(handle.clone());
117
118    server
119        .http_builder()
120        .http1()
121        .timer(TokioTimer::new())
122        .half_close(true)
123        .keep_alive(true)
124        .header_read_timeout(CLIENT_HEADER_TIMEOUT)
125        .writev(true);
126
127    server
128        .http_builder()
129        .http2()
130        .timer(TokioTimer::new())
131        .keep_alive_timeout(config.keepalive_timeout());
132
133    let service = ServiceBuilder::new()
134        .layer(concurrency_limit::ConcurrencyLimitLayer::new(
135            config.max_connections(),
136        ))
137        .service(ServiceExt::<Request>::into_make_service_with_connect_info::<SocketAddr>(app));
138
139    relay_system::spawn!(emit_active_connections_metric(
140        config.metrics_periodic_interval(),
141        handle.clone(),
142    ));
143
144    relay_system::spawn!(async move {
145        let Shutdown { timeout } = Controller::shutdown_handle().notified().await;
146        relay_log::info!("Shutting down HTTP server");
147
148        match timeout {
149            Some(timeout) => handle.graceful_shutdown(Some(timeout)),
150            None => handle.shutdown(),
151        }
152    });
153
154    server.serve(service).await
155}

157/// HTTP server service.
158///
159/// This is the main HTTP server of Relay which hosts all [services](ServiceState) and dispatches
160/// incoming traffic to them. The server stops when a [`Shutdown`] is triggered.
161pub struct HttpServer {
162    config: Arc<Config>,
163    service: ServiceState,
164    listener: TcpListener,
165    internal_listener: Option<TcpListener>,
166}
167
168impl HttpServer {
169    pub fn new(config: Arc<Config>, service: ServiceState) -> Result<Self, ServerError> {
170        // Inform the user about a removed feature.
171        if config.tls_listen_addr().is_some()
172            || config.tls_identity_password().is_some()
173            || config.tls_identity_path().is_some()
174        {
175            return Err(ServerError::TlsNotSupported);
176        }
177
178        let listener = listen(config.listen_addr(), &config)?;
179        let internal_listener = match config.listen_addr_internal() {
180            Some(addr) => Some(listen(addr, &config)?),
181            None => None,
182        };
183
184        Ok(Self {
185            config,
186            service,
187            listener,
188            internal_listener,
189        })
190    }
191}
192
193impl Service for HttpServer {
194    type Interface = ();
195
196    async fn run(self, _rx: relay_system::Receiver<Self::Interface>) {
197        let Self {
198            config,
199            service,
200            listener,
201            internal_listener,
202        } = self;
203
204        let listen_addr = config.listen_addr();
205
206        relay_log::info!("spawning http server");
207        relay_log::info!("  listening on http://{listen_addr}/");
208        if let Some(internal_addr) = config.listen_addr_internal() {
209            relay_log::info!("  listening on http://{internal_addr}/ [internal]");
210        }
211        relay_statsd::metric!(counter(RelayCounters::ServerStarting) += 1);
212
213        if let Some(internal_listener) = internal_listener {
214            let public = make_app(service.clone(), crate::endpoints::public_routes);
215            let internal = make_app(service, crate::endpoints::internal_routes);
216
217            tokio::try_join!(
218                serve(listener, public, &config),
219                serve(internal_listener, internal, &config),
220            )
221            .map(drop)
222        } else {
223            let app = make_app(service, crate::endpoints::all_routes);
224            serve(listener, app, &config).await
225        }
226        .expect("axum listener to not fail")
227    }
228}

230async fn emit_active_connections_metric(interval: Option<Duration>, handle: Handle) {
231    let Some(mut ticker) = interval.map(tokio::time::interval) else {
232        return;
233    };
234
235    let addr = handle.listening().await.map(|addr| addr.to_string());
236
237    loop {
238        ticker.tick().await;
239        relay_statsd::metric!(
240            gauge(RelayGauges::ServerActiveConnections) = handle.connection_count() as u64,
241            addr = addr.as_deref().unwrap_or("unknown"),
242        );
243    }
244}
