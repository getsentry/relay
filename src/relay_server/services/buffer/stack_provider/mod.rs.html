9/// State of the initialization of the [`StackProvider`].
10///
11/// This state is necessary for initializing resources whenever a [`StackProvider`] is used.
12#[derive(Debug)]
13pub struct InitializationState {
14    pub project_key_pairs: HashSet<ProjectKeyPair>,
15}
16
17impl InitializationState {
18    /// Create a new [`InitializationState`].
19    pub fn new(project_key_pairs: HashSet<ProjectKeyPair>) -> Self {
20        Self { project_key_pairs }
21    }
22
23    /// Creates a new empty [`InitializationState`].
24    pub fn empty() -> Self {
25        Self {
26            project_key_pairs: HashSet::new(),
27        }
28    }
29}
30
31/// The creation type for the [`EnvelopeStack`].
32pub enum StackCreationType {
33    /// An [`EnvelopeStack`] that is created during initialization.
34    Initialization,
35    /// An [`EnvelopeStack`] that is created when an envelope is received.
36    New,
37}
38
39/// A provider of [`EnvelopeStack`] instances that is responsible for creating them.
40pub trait StackProvider: std::fmt::Debug {
41    /// The implementation of [`EnvelopeStack`] that this manager creates.
42    type Stack: EnvelopeStack;
43
44    /// Initializes the [`StackProvider`].
45    fn initialize(&self) -> impl Future<Output = InitializationState>;
46
47    /// Creates an [`EnvelopeStack`].
48    fn create_stack(
49        &self,
50        stack_creation_type: StackCreationType,
51        project_key_pair: ProjectKeyPair,
52    ) -> Self::Stack;
53
54    /// Returns `true` if the store used by this [`StackProvider`] has space to add new
55    /// stacks or items to the stacks.
56    fn has_store_capacity(&self) -> bool;
57
58    /// Returns the total count of the store used by this [`StackProvider`].
59    fn store_total_count(&self) -> impl Future<Output = u64>;
60
61    /// Returns the number of bytes the storage occupies. Will return `None` if no
62    /// reliable information can be provided.
63    fn total_size(&self) -> Option<u64>;
64
65    /// Returns the string representation of the stack type offered by this [`StackProvider`].
66    fn stack_type<'a>(&self) -> &'a str;
67
68    /// Flushes the supplied [`EnvelopeStack`]s.
69    fn flush(
70        &mut self,
71        envelope_stacks: impl IntoIterator<Item = Self::Stack>,
72    ) -> impl Future<Output = ()>;
73}
