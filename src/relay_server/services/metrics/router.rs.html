16/// Service that routes metrics & metric buckets to the appropriate aggregator.
17///
18/// Each aggregator gets its own configuration.
19/// Metrics are routed to the first aggregator which matches the configuration's [`Condition`].
20/// If no condition matches, the metric/bucket is routed to the `default_aggregator`.
21pub struct RouterService {
22    handle: relay_system::Handle,
23    default: AggregatorService,
24    secondary: Vec<(AggregatorService, Condition)>,
25}
26
27impl RouterService {
28    /// Create a new router service.
29    pub fn new(
30        handle: relay_system::Handle,
31        default_config: AggregatorServiceConfig,
32        secondary_configs: Vec<ScopedAggregatorConfig>,
33        receiver: Option<Recipient<FlushBuckets, NoResponse>>,
34        project_cache: ProjectCacheHandle,
35    ) -> Self {
36        let mut secondary = Vec::new();
37
38        for c in secondary_configs {
39            let service =
40                AggregatorService::named(c.name, c.config, receiver.clone(), project_cache.clone());
41            secondary.push((service, c.condition));
42        }
43
44        let default = AggregatorService::new(default_config, receiver, project_cache);
45        Self {
46            handle,
47            default,
48            secondary,
49        }
50    }
51
52    pub fn handle(&self) -> RouterHandle {
53        let mut handles = vec![self.default.handle()];
54        for (aggregator, _) in &self.secondary {
55            handles.push(aggregator.handle());
56        }
57
58        RouterHandle(handles)
59    }
60}
61
62impl Service for RouterService {
63    type Interface = Aggregator;
64
65    async fn run(self, mut rx: relay_system::Receiver<Self::Interface>) {
66        let mut router = StartedRouter::start(self);
67        relay_log::info!("metrics router started");
68
69        // Note that currently this loop never exits and will run till the tokio runtime shuts
70        // down. This is about to change with the refactoring for the shutdown process.
71        loop {
72            tokio::select! {
73                biased;
74
75                Some(message) = rx.recv() => {
76                    router.handle_message(message)
77                },
78
79                else => break,
80            }
81        }
82        relay_log::info!("metrics router stopped");
83    }
84}

86/// Helper struct that holds the [`Addr`]s of started aggregators.
87struct StartedRouter {
88    default: Addr<Aggregator>,
89    secondary: Vec<(Addr<Aggregator>, Vec<MetricNamespace>)>,
90}
91
92impl StartedRouter {
93    fn start(router: RouterService) -> Self {
94        let RouterService {
95            default,
96            secondary,
97            handle,
98        } = router;
99
100        let secondary = secondary
101            .into_iter()
102            .map(|(aggregator, condition)| {
103                let namespaces: Vec<_> = MetricNamespace::all()
104                    .into_iter()
105                    .filter(|&namespace| condition.matches(Some(namespace)))
106                    .collect();
107
108                (handle.start(aggregator), namespaces)
109            })
110            .collect();
111
112        Self {
113            default: handle.start(default),
114            secondary,
115        }
116    }
117
118    fn handle_message(&mut self, message: Aggregator) {
119        let ty = message.variant();
120        relay_statsd::metric!(
121            timer(RelayTimers::MetricRouterServiceDuration),
122            message = ty,
123            {
124                match message {
125                    Aggregator::MergeBuckets(msg) => self.handle_merge_buckets(msg),
126                }
127            }
128        )
129    }
130
131    fn handle_merge_buckets(&mut self, message: MergeBuckets) {
132        let MergeBuckets {
133            project_key,
134            mut buckets,
135        } = message;
136
137        for (aggregator, namespaces) in &self.secondary {
138            let matching;
139            (buckets, matching) = utils::split_off(buckets, |bucket| {
140                bucket
141                    .name
142                    .try_namespace()
143                    .map(|namespace| namespaces.contains(&namespace))
144                    .unwrap_or(false)
145            });
146
147            if !matching.is_empty() {
148                aggregator.send(MergeBuckets::new(project_key, matching));
149            }
150        }
151
152        if !buckets.is_empty() {
153            self.default.send(MergeBuckets::new(project_key, buckets));
154        }
155    }
156}

158/// Provides sync access to the state of the [`RouterService`].
159#[derive(Clone, Debug)]
160pub struct RouterHandle(Vec<AggregatorHandle>);
161
162impl RouterHandle {
163    /// Returns `true` if all the aggregators can still accept metrics.
164    pub fn can_accept_metrics(&self) -> bool {
165        self.0.iter().all(|ah| ah.can_accept_metrics())
166    }
