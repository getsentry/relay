20/// Service implementing the [`EnvelopeProcessor`] interface.
21///
22/// Analog to [`crate::services::processor::EnvelopeProcessorService`] this service handles messages when Relay is run in
23/// proxy mode.
24pub struct ProxyProcessorService {
25    config: Arc<Config>,
26    project_cache: ProjectCacheHandle,
27    addrs: ProxyAddrs,
28}
29
30/// Contains the addresses of services that the proxy-processor publishes to.
31pub struct ProxyAddrs {
32    /// Address of the service used for tracking outcomes.
33    pub outcome_aggregator: Addr<TrackOutcome>,
34    /// Address of the service used for forwarding envelopes to the upstream.
35    pub upstream_relay: Addr<UpstreamRelay>,
36}
37
38impl ProxyProcessorService {
39    /// Creates a proxy processor.
40    pub fn new(config: Arc<Config>, project_cache: ProjectCacheHandle, addrs: ProxyAddrs) -> Self {
41        Self {
42            project_cache,
43            addrs,
44            config,
45        }
46    }
47
48    fn handle_process_envelope(&self, message: ProcessEnvelope) {
49        let wait_time = message.envelope.age();
50        metric!(timer(RelayTimers::EnvelopeWaitTime) = wait_time);
51        self.submit_upstream(message.envelope);
52    }
53
54    fn handle_submit_client_reports(&self, message: SubmitClientReports) {
55        let SubmitClientReports {
56            client_reports,
57            scoping,
58        } = message;
59
60        let upstream = self.config.upstream_descriptor();
61        let dsn = PartialDsn::outbound(&scoping, upstream);
62
63        let mut envelope = Envelope::from_request(None, RequestMeta::outbound(dsn));
64        for client_report in client_reports {
65            match client_report.serialize() {
66                Ok(payload) => {
67                    let mut item = Item::new(ItemType::ClientReport);
68                    item.set_payload(ContentType::Json, payload);
69                    envelope.add_item(item);
70                }
71                Err(error) => {
72                    relay_log::error!(
73                        error = &error as &dyn std::error::Error,
74                        "failed to serialize client report"
75                    );
76                }
77            }
78        }
79
80        let envelope = ManagedEnvelope::new(envelope, self.addrs.outcome_aggregator.clone());
81        self.submit_upstream(envelope);
82    }
83
84    fn submit_upstream(&self, mut envelope: ManagedEnvelope) {
85        if envelope.envelope_mut().is_empty() {
86            envelope.accept();
87            return;
88        }
89
90        relay_log::trace!("sending envelope to sentry endpoint");
91        let http_encoding = self.config.http_encoding();
92        let result = envelope.envelope().to_vec().and_then(|v| {
93            encode_payload(&v.into(), http_encoding).map_err(EnvelopeError::PayloadIoFailed)
94        });
95
96        match result {
97            Ok(body) => {
98                self.addrs.upstream_relay.send(SendRequest(SendEnvelope {
99                    envelope: envelope.into_processed(),
100                    body,
101                    http_encoding,
102                    project_cache: self.project_cache.clone(),
103                }));
104            }
105            Err(error) => {
106                // Errors are only logged for what we consider an internal discard reason. These
107                // indicate errors in the infrastructure or implementation bugs.
108                relay_log::error!(
109                    error = &error as &dyn Error,
110                    tags.project_key = %envelope.scoping().project_key,
111                    "failed to serialize envelope payload"
112                );
113
114                envelope.reject(Outcome::Invalid(DiscardReason::Internal));
115            }
116        }
117    }
118
119    fn handle_message(&self, message: EnvelopeProcessor) {
120        let ty = message.variant();
121
122        metric!(timer(RelayTimers::ProcessMessageDuration), message = ty, {
123            match message {
124                EnvelopeProcessor::ProcessEnvelope(m) => self.handle_process_envelope(*m),
125                EnvelopeProcessor::SubmitClientReports(m) => self.handle_submit_client_reports(*m),
126                EnvelopeProcessor::ProcessBatchedMetrics(_)
127                | EnvelopeProcessor::ProcessProjectMetrics(_)
128                | EnvelopeProcessor::FlushBuckets(_) => {
129                    relay_log::error!("internal error: Metrics not supported in Proxy mode");
130                }
131            }
132        });
133    }
134}
135
136impl Service for ProxyProcessorService {
137    type Interface = EnvelopeProcessor;
138
139    async fn run(self, mut rx: relay_system::Receiver<Self::Interface>) {
140        while let Some(message) = rx.recv().await {
141            self.handle_message(message);
142        }
143    }
144}
