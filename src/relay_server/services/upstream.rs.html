39/// Rate limits returned by the upstream.
40///
41/// Upstream rate limits can come in two forms:
42///  - `Retry-After` header with a generic timeout for all categories.
43///  - `X-Sentry-Rate-Limits` header with fine-grained information on applied rate limits.
44///
45/// These limits do not carry scope information. Use `UpstreamRateLimits::scope` to attach scope
46/// identifiers and return a fully populated `RateLimits` instance.
47#[derive(Debug, Clone)]
48pub struct UpstreamRateLimits {
49    retry_after: RetryAfter,
50    rate_limits: String,
51}
52
53impl UpstreamRateLimits {
54    /// Creates an empty `UpstreamRateLimits` instance.
55    fn new() -> Self {
56        Self {
57            retry_after: RetryAfter::from_secs(0),
58            rate_limits: String::new(),
59        }
60    }
61
62    /// Adds the `Retry-After` header to this rate limits instance.
63    fn retry_after(mut self, header: Option<&str>) -> Self {
64        if let Some(retry_after) = header.and_then(|s| s.parse().ok()) {
65            self.retry_after = retry_after;
66        }
67        self
68    }
69
70    /// Adds the `X-Sentry-Rate-Limits` header to this instance.
71    ///
72    /// If multiple header values are given, this header should be joined. If the header is empty,
73    /// an empty string should be passed.
74    fn rate_limits(mut self, header: String) -> Self {
75        self.rate_limits = header;
76        self
77    }
78
79    /// Creates a scoped rate limit instance based on the provided `Scoping`.
80    pub fn scope(self, scoping: &Scoping) -> RateLimits {
81        // Try to parse the `X-Sentry-Rate-Limits` header in the most lenient way possible. If
82        // anything goes wrong, skip over the invalid parts.
83        let mut rate_limits = utils::parse_rate_limits(scoping, &self.rate_limits);
84
85        // If there are no new-style rate limits in the header, fall back to the `Retry-After`
86        // header. Create a default rate limit that only applies to the current data category at the
87        // most specific scope (Key).
88        // One example of such a generic rate limit is the anti-abuse nginx layer used by SaaS.
89        if !rate_limits.is_limited() {
90            rate_limits.add(RateLimit {
91                categories: DataCategories::new(),
92                scope: RateLimitScope::for_quota(*scoping, QuotaScope::Key),
93                reason_code: Some(ReasonCode::new("generic")),
94                retry_after: self.retry_after,
95                namespaces: Default::default(),
96            });
97        }
98        rate_limits
99    }
100}

102/// An error returned from [`SendRequest`] and [`SendQuery`].
103#[derive(Debug, thiserror::Error)]
104pub enum UpstreamRequestError {
105    #[error("attempted to send upstream request without credentials configured")]
106    NoCredentials,
107
108    /// As opposed to HTTP variant this contains all network errors.
109    #[error("could not send request to upstream")]
110    SendFailed(#[from] reqwest::Error),
111
112    /// Likely a bad HTTP status code or unparseable response.
113    #[error("could not send request")]
114    Http(#[from] HttpError),
115
116    #[error("upstream requests rate limited")]
117    RateLimited(UpstreamRateLimits),
118
119    #[error("upstream request returned error {0}")]
120    ResponseError(StatusCode, #[source] ApiErrorResponse),
121
122    #[error("channel closed")]
123    ChannelClosed,
124
125    #[error("upstream permanently denied authentication")]
126    AuthDenied,
127}
128
129impl UpstreamRequestError {
130    /// Returns the status code of the HTTP request sent to the upstream.
131    ///
132    /// If this error is the result of sending a request to the upstream, this method returns `Some`
133    /// with the status code. If the request could not be made or the error originates elsewhere,
134    /// this returns `None`.
135    fn status_code(&self) -> Option<StatusCode> {
136        match self {
137            UpstreamRequestError::ResponseError(code, _) => Some(*code),
138            UpstreamRequestError::Http(HttpError::Reqwest(e)) => e.status(),
139            _ => None,
140        }
141    }
142
143    /// Returns `true` if the error indicates a network downtime.
144    fn is_network_error(&self) -> bool {
145        match self {
146            Self::SendFailed(_) => true,
147            Self::ResponseError(code, _) => matches!(code.as_u16(), 502..=504),
148            Self::Http(http) => http.is_network_error(),
149            _ => false,
150        }
151    }
152
153    /// Returns `true` if the upstream has permanently rejected this Relay.
154    ///
155    /// This Relay should cease communication with the upstream and may shut down.
156    fn is_permanent_rejection(&self) -> bool {
157        match self {
158            Self::ResponseError(status_code, response) => {
159                *status_code == StatusCode::FORBIDDEN
160                    && response.relay_action() == RelayErrorAction::Stop
161            }
162            _ => false,
163        }
164    }
165
166    /// Returns `true` if the request was received by the upstream.
167    ///
168    /// Despite resulting in an error, the server has received and acknowledged the request. This
169    /// includes rate limits (status code 429), and bad payloads (4XX), but not network errors
170    /// (502-504).
171    pub fn is_received(&self) -> bool {
172        match self {
173            // Rate limits are a special case of `ResponseError(429, _)`.
174            Self::RateLimited(_) => true,
175            // Everything except network errors indicates the upstream has handled this request.
176            Self::ResponseError(_, _) | Self::Http(_) => !self.is_network_error(),
177            // Remaining kinds indicate a failure to send the request.
178            Self::NoCredentials | Self::SendFailed(_) | Self::ChannelClosed | Self::AuthDenied => {
179                false
180            }
181        }
182    }
183
184    /// Returns a categorized description of the error.
185    ///
186    /// This is used for metrics and logging.
187    fn description(&self) -> &'static str {
188        match self {
189            UpstreamRequestError::NoCredentials => "credentials",
190            UpstreamRequestError::SendFailed(_) => "send_failed",
191            UpstreamRequestError::Http(HttpError::Io(_)) => "payload_failed",
192            UpstreamRequestError::Http(HttpError::Json(_)) => "invalid_json",
193            UpstreamRequestError::Http(HttpError::Reqwest(_)) => "reqwest_error",
194            UpstreamRequestError::Http(HttpError::Overflow) => "overflow",
195            UpstreamRequestError::RateLimited(_) => "rate_limited",
196            UpstreamRequestError::ResponseError(_, _) => "response_error",
197            UpstreamRequestError::ChannelClosed => "channel_closed",
198            UpstreamRequestError::AuthDenied => "auth_denied",
199        }
200    }
201}

203/// Checks the authentication state with the upstream.
204///
205/// In static and proxy mode, Relay does not require authentication and `IsAuthenticated` always
206/// returns `true`. Otherwise, this message retrieves the current state of authentication:
207///
208/// - Initially, Relay is unauthenticated until it has established connection.
209/// - If this Relay is not known by the upstream, it remains unauthenticated indefinitely.
210/// - Once Relay has registered, this message reports `true`.
211/// - In periodic intervals Relay re-authenticates, which may drop authentication temporarily.
212#[derive(Debug)]
213pub struct IsAuthenticated;
214
215/// Returns whether Relay is in an outage state.
216///
217/// On repeated failure to submit requests to the upstream, the upstream service moves into an
218/// outage state. During this phase, no requests or retries are performed and all newly submitted
219/// [`SendRequest`] and [`SendQuery`] messages are put into the queue. Once connection is
220/// reestablished, requests resume in FIFO order.
221///
222/// This message resolves to `true` if Relay is in outage mode and `false` if the service is
223/// performing regular operation.
224#[derive(Debug)]
225pub struct IsNetworkOutage;

254/// Errors during the signature or verification process.
255#[derive(Debug, Clone, PartialEq, thiserror::Error)]
256pub enum SignatureError {
257    /// Credentials for a required signature were missing.
258    #[error("missing credentials")]
259    MissingCredentials,
260}

262/// Represents whether credentials are required for signing a request.
263pub enum Sign {
264    /// Credentials are required for signing. If no credentials are available,
265    /// signature creation will fail.
266    Required(SignatureType),
267    /// Credentials are optional for signing. If no credentials are available,
268    /// signature creation will be skipped.
269    Optional(SignatureType),
270}
271
272impl Sign {
273    /// Creates a signature based on the [`Sign`] variant.
274    ///
275    /// This method signs the input data and returns an optional signature string.
276    ///
277    /// If credentials are required but none are provided, it will fail with
278    /// [`SignatureError::MissingCredentials`].
279    pub fn create_signature(
280        self,
281        secret_key: Option<&SecretKey>,
282    ) -> Result<Option<Signature>, SignatureError> {
283        match self {
284            Sign::Required(signature_type) => {
285                let Some(secret_key) = secret_key else {
286                    return Err(SignatureError::MissingCredentials);
287                };
288                Ok(Some(signature_type.create_signature(secret_key)))
289            }
290            Sign::Optional(signature_type) => {
291                let Some(secret_key) = secret_key else {
292                    return Ok(None);
293                };
294                Ok(Some(signature_type.create_signature(secret_key)))
295            }
296        }
297    }
298}

300/// Types of signatures that are supported by Relay.
301#[derive(Debug)]
302pub enum SignatureType {
303    /// Bytes of an envelope body that are used to produce a signature.
304    Body(Bytes),
305    /// No data is needed for this signature because we only want to see if
306    /// the receiving relay can verify the signature correctly.
307    RequestSign,
308}
309
310impl SignatureType {
311    /// Creates a signature data based on the variant.
312    pub fn create_signature(self, secret_key: &SecretKey) -> Signature {
313        match self {
314            SignatureType::Body(data) => secret_key.sign(data.as_ref()),
315            SignatureType::RequestSign => secret_key.sign(&[]),
316        }
317    }
318}

320/// Represents a generic HTTP request to be sent to the upstream.
321pub trait UpstreamRequest: Send + Sync + fmt::Debug {
322    /// The HTTP method of the request.
323    fn method(&self) -> Method;
324
325    /// The path relative to the upstream.
326    fn path(&self) -> Cow<'_, str>;
327
328    /// Whether this request should retry on network errors.
329    ///
330    /// Defaults to `true` and should be disabled if there is an external retry mechanism. Note that
331    /// failures other than network errors will **not** be retried.
332    fn retry(&self) -> bool {
333        true
334    }
335
336    /// The queueing priority of the request.
337    ///
338    ///  - High priority requests are always sent and retried first.
339    ///  - Low priority requests are sent if no high-priority messages are pending in the queue.
340    ///    This also applies to retries: A low-priority message is only sent if there are no
341    ///    high-priority requests waiting.
342    ///
343    /// Within the same priority, requests are delivered in FIFO order.
344    ///
345    /// Defaults to [`Low`](RequestPriority::Low).
346    fn priority(&self) -> RequestPriority {
347        RequestPriority::Low
348    }
349
350    /// Controls whether request errors should be intercepted.
351    ///
352    /// By default, error codes from responses will be intercepted and returned as
353    /// [`UpstreamRequestError`]. This also includes parsing of the request body for diagnostics.
354    /// Return `false` to disable this behavior and receive the verbatim response.
355    fn intercept_status_errors(&self) -> bool {
356        true
357    }
358
359    /// Add the `X-Sentry-Relay-Id` header to the outgoing request.
360    ///
361    /// This header is used for authentication with the upstream and should be enabled only for
362    /// endpoints that require it.
363    ///
364    /// Defaults to `true`.
365    fn set_relay_id(&self) -> bool {
366        true
367    }
368
369    /// Add the `X-Sentry-Relay-Signature` header to the outgoing request.
370    ///
371    /// When no signature should be added, this method should return `None`. Otherwise, this method
372    /// should return the payload to sign. For requests with content encoding, this must be the
373    /// **uncompressed** payload.
374    ///
375    /// If the variant of [`SignatureType`] forces a signature and no Relay credentials are configured,
376    /// the request will fail with [`UpstreamRequestError::NoCredentials`].
377    ///
378    /// Defaults to `None`.
379    fn sign(&mut self) -> Option<Sign> {
380        None
381    }
382
383    /// Returns the name of the logical route.
384    ///
385    /// This is used for internal metrics and logging. Other than the path, this cannot contain
386    /// dynamic elements and should be globally unique.
387    fn route(&self) -> &'static str;
388
389    /// Callback to apply configuration to the request.
390    ///
391    /// This hook is called at least once before `build`. It can be used to include additional
392    /// properties from Relay's config in the Request before it is sent or handled if during request
393    /// creation time the configuration is not available.
394    ///
395    /// This method is optional and defaults to a no-op.
396    fn configure(&mut self, _config: &Config) {}
397
398    /// Callback to build the outgoing web request.
399    ///
400    /// This callback populates the initialized request with headers and a request body.
401    ///
402    /// Note that this function can be called repeatedly if [`retry`](UpstreamRequest::retry)
403    /// returns `true`. This function should therefore not move out of the request struct, but can
404    /// use it to memoize heavy computation.
405    fn build(&mut self, _builder: &mut RequestBuilder) -> Result<(), HttpError> {
406        Ok(())
407    }
408
409    /// Callback to complete an HTTP request.
410    ///
411    /// This callback receives the response or error. At time of invocation, the response body has
412    /// not been consumed. The response body or derived information can then be sent into a channel.
413    fn respond(
414        self: Box<Self>,
415        result: Result<Response, UpstreamRequestError>,
416    ) -> Pin<Box<dyn Future<Output = ()> + Send + Sync>>;
417}

471/// Memoized implementation of [`UpstreamRequest`] for an [`UpstreamQuery`].
472///
473/// This can be used to send queries as requests to the upstream. The request wraps an internal
474/// channel to send responses to.
475#[derive(Debug)]
476struct UpstreamQueryRequest<T: UpstreamQuery> {
477    query: T,
478    body: Option<Bytes>,
479    max_response_size: usize,
480    sender: QuerySender<T>,
481}
482
483impl<T> UpstreamQueryRequest<T>
484where
485    T: UpstreamQuery + 'static,
486{
487    /// Wraps the given `query` in an [`UpstreamQuery`] implementation.
488    pub fn new(query: T, sender: QuerySender<T>) -> Self {
489        Self {
490            query,
491            body: None,
492            max_response_size: 0,
493            sender,
494        }
495    }
496
497    /// Memoize the serialized body for retries and signing.
498    fn body(&mut self) -> Result<Bytes, HttpError> {
499        let body = match self.body {
500            Some(ref body) => body,
501            None => self.body.insert(serde_json::to_vec(&self.query)?.into()),
502        };
503
504        Ok(body.clone())
505    }
506}

508impl<T> UpstreamRequest for UpstreamQueryRequest<T>
509where
510    T: UpstreamQuery + 'static,
511{
512    fn retry(&self) -> bool {
513        T::retry()
514    }
515
516    fn priority(&self) -> RequestPriority {
517        T::priority()
518    }
519
520    fn intercept_status_errors(&self) -> bool {
521        true
522    }
523
524    fn set_relay_id(&self) -> bool {
525        true
526    }
527
528    fn sign(&mut self) -> Option<Sign> {
529        // Computing the body is practically infallible since we're serializing standard structures
530        // into a string. Even if it fails, `sign` is called after `build` and the error will be
531        // reported there.
532        self.body()
533            .ok()
534            .map(SignatureType::Body)
535            .map(Sign::Required)
536    }
537
538    fn method(&self) -> Method {
539        self.query.method()
540    }
541
542    fn path(&self) -> Cow<'_, str> {
543        self.query.path()
544    }
545
546    fn route(&self) -> &'static str {
547        self.query.route()
548    }
549
550    fn configure(&mut self, config: &Config) {
551        // This config attribute is needed during `respond`, which does not have access to the
552        // config. For this reason, we need to store it on the request struct.
553        self.max_response_size = config.max_api_payload_size();
554    }
555
556    fn build(&mut self, builder: &mut RequestBuilder) -> Result<(), HttpError> {
557        let body = self.body()?;
558
559        relay_statsd::metric!(
560            histogram(RelayHistograms::UpstreamQueryBodySize) = body.len() as u64
561        );
562
563        builder
564            .header(header::CONTENT_TYPE, b"application/json")
565            .body(body.clone());
566
567        Ok(())
568    }
569
570    fn respond(
571        self: Box<Self>,
572        result: Result<Response, UpstreamRequestError>,
573    ) -> Pin<Box<dyn Future<Output = ()> + Send + Sync>> {
574        Box::pin(async move {
575            let result = match result {
576                Ok(response) => response
577                    .json(self.max_response_size)
578                    .await
579                    .map_err(UpstreamRequestError::Http),
580                Err(error) => Err(error),
581            };
582
583            self.sender.send(result)
584        })
585    }
586}

595/// Communication with the upstream via HTTP.
596///
597/// This service can send two main types of requests to the upstream, which can in turn be a Relay
598/// or the Sentry webserver:
599///
600///  - [`SendRequest`] sends a plain HTTP request to the upstream that can be configured and handled
601///    freely. Request implementations specify their priority and whether they should be retried
602///    automatically by the upstream service.
603///  - [`SendQuery`] sends a higher-level request with a standardized JSON body and resolves to a
604///    JSON response. The upstream service will automatically sign the message with its private key
605///    for authentication.
606///
607/// The upstream is also responsible to maintain the connection with the upstream. There are two
608/// main messages to inquire about the connection state:
609///
610///  - [`IsAuthenticated`]
611///  - [`IsNetworkOutage`]
612#[derive(Debug)]
613pub enum UpstreamRelay {
614    /// Checks the authentication state with the upstream.
615    IsAuthenticated(IsAuthenticated, Sender<bool>),
616    /// Returns whether Relay is in an outage state.
617    IsNetworkOutage(IsNetworkOutage, Sender<bool>),
618    /// Sends a [request](SendRequest) or [query](SendQuery) to the upstream.
619    SendRequest(Box<dyn UpstreamRequest>),
620}
621
622impl Interface for UpstreamRelay {}

664/// Captures statsd metrics for a completed upstream request.
665fn emit_response_metrics(
666    send_start: Instant,
667    entry: &Entry,
668    send_result: &Result<Response, UpstreamRequestError>,
669) {
670    let description = match send_result {
671        Ok(_) => "success",
672        Err(e) => e.description(),
673    };
674    let status_code = match send_result {
675        Ok(response) => Some(response.status()),
676        Err(error) => error.status_code(),
677    };
678    let status_str = status_code.as_ref().map(|c| c.as_str()).unwrap_or("-");
679
680    relay_statsd::metric!(
681        timer(RelayTimers::UpstreamRequestsDuration) = send_start.elapsed(),
682        result = description,
683        status_code = status_str,
684        route = entry.request.route(),
685        retries = match entry.retries {
686            0 => "0",
687            1 => "1",
688            2 => "2",
689            3..=10 => "few",
690            _ => "many",
691        },
692    );
693
694    relay_statsd::metric!(

795/// A shared, asynchronous client to build and execute requests.
796///
797/// The main way to send a request through this client is [`send`](Self::send).
798///
799/// This instance holds a shared reference internally and can be cloned directly, so it does not
800/// have to be placed in an `Arc`.
801#[derive(Debug, Clone)]
802struct SharedClient {
803    config: Arc<Config>,
804    reqwest: reqwest::Client,
805}
806
807impl SharedClient {
808    /// Creates a new `SharedClient` instance.
809    pub fn build(config: Arc<Config>) -> Self {
810        let reqwest = reqwest::ClientBuilder::new()
811            .connect_timeout(config.http_connection_timeout())
812            .timeout(config.http_timeout())
813            // In the forward endpoint, this means that content negotiation is done twice, and the
814            // response body is first decompressed by the client, then re-compressed by the server.
815            .gzip(true)
816            // Enables async resolver through the `hickory-dns` crate, which uses an LRU cache for
817            // the resolved entries. This helps to limit the amount of requests made to upstream DNS
818            // server (important for K8s infrastructure).
819            .hickory_dns(true)
820            .build()
821            .unwrap();
822
823        Self { config, reqwest }
824    }
825
826    /// Builds the request in a non-blocking fashion.
827    ///
828    /// This creates the request, adds internal headers, and invokes [`UpstreamRequest::build`]. The
829    /// build is invoked in a non-blocking fashion internally, so it can be called from an
830    /// asynchronous runtime.
831    fn build_request(
832        &self,
833        request: &mut dyn UpstreamRequest,
834    ) -> Result<reqwest::Request, UpstreamRequestError> {
835        tokio::task::block_in_place(|| {
836            let url = self
837                .config
838                .upstream_descriptor()
839                .get_url(request.path().as_ref());
840
841            let host_header = self
842                .config
843                .http_host_header()
844                .unwrap_or_else(|| self.config.upstream_descriptor().host());
845
846            let mut builder = RequestBuilder::reqwest(self.reqwest.request(request.method(), url));
847            builder.header("Host", host_header.as_bytes());
848
849            if request.set_relay_id() {
850                if let Some(credentials) = self.config.credentials() {
851                    builder.header("X-Sentry-Relay-Id", credentials.id.to_string());
852                }
853            }
854
855            request.build(&mut builder)?;
856
857            if let Some(payload) = request.sign() {
858                if let Some(signature) = payload
859                    .create_signature(self.config.credentials().map(|cred| &cred.secret_key))
860                    .map_err(|_| UpstreamRequestError::NoCredentials)?
861                {
862                    builder.header("x-sentry-relay-signature", &signature.0);
863                }
864            }
865
866            match builder.finish() {
867                Ok(Request(client_request)) => Ok(client_request),
868                Err(e) => Err(e.into()),
869            }
870        })
871    }
872
873    /// Handles an HTTP response returned from the upstream.
874    ///
875    /// If the response indicates success via 2XX status codes, `Ok(response)` is returned.
876    /// Otherwise, the response is consumed and an error is returned. If `intercept_status_errors`
877    /// is set to `true` on the request, depending on the status code and details provided in the
878    /// payload, one of the following errors is returned:
879    ///
880    ///  1. `RateLimited` for a `429` status code.
881    ///  2. `ResponseError` in all other cases, containing the status and details.
882    async fn transform_response(
883        &self,
884        request: &dyn UpstreamRequest,
885        response: Response,
886    ) -> Result<Response, UpstreamRequestError> {
887        let status = response.status();
888
889        if !request.intercept_status_errors() || status.is_success() {
890            return Ok(response);
891        }
892
893        let upstream_limits = if status == StatusCode::TOO_MANY_REQUESTS {
894            let retry_after = response
895                .get_header(header::RETRY_AFTER)
896                .and_then(|v| std::str::from_utf8(v).ok());
897
898            let rate_limits = response
899                .get_all_headers(utils::RATE_LIMITS_HEADER)
900                .iter()
901                .filter_map(|v| std::str::from_utf8(v).ok())
902                .join(", ");
903
904            let upstream_limits = UpstreamRateLimits::new()
905                .retry_after(retry_after)
906                .rate_limits(rate_limits);
907
908            Some(upstream_limits)
909        } else {
910            None
911        };
912
913        // At this point, we consume the Response. This means we need to consume the response
914        // payload stream, regardless of the status code. Parsing the JSON body may fail, which is a
915        // non-fatal failure as the upstream is not expected to always include a valid JSON
916        // response.
917        let json_result = response.json(self.config.max_api_payload_size()).await;
918
919        if let Some(upstream_limits) = upstream_limits {
920            Err(UpstreamRequestError::RateLimited(upstream_limits))
921        } else {
922            // Coerce the result into an empty `ApiErrorResponse` if parsing JSON did not succeed.
923            let api_response = json_result.unwrap_or_default();
924            Err(UpstreamRequestError::ResponseError(status, api_response))
925        }
926    }
927
928    /// Builds and sends a request to the upstream, returning either a response or the error.
929    pub async fn send(
930        &self,
931        request: &mut dyn UpstreamRequest,
932    ) -> Result<Response, UpstreamRequestError> {
933        request.configure(&self.config);
934        let client_request = self.build_request(request)?;
935        let response = self.reqwest.execute(client_request).await?;
936        self.transform_response(request, Response(response)).await
937    }
938
939    /// Convenience method to send a query to the upstream and await the result.
940    pub async fn send_query<T>(&self, query: T) -> Result<T::Response, UpstreamRequestError>
941    where
942        T: UpstreamQuery + 'static,
943    {
944        let (sender, receiver) = AsyncResponse::channel();
945
946        let mut request = Box::new(UpstreamQueryRequest::new(query, sender));
947        let result = self.send(request.as_mut()).await;
948        request.respond(result).await;
949
950        receiver
951            .await
952            .unwrap_or(Err(UpstreamRequestError::ChannelClosed))
953    }
954}

970impl Entry {

975            retries: 0,

980/// Queue utility for the [`UpstreamRelayService`].
981///
982/// Requests are queued and delivered according to their [`UpstreamRequest::priority`]. This queue
983/// is synchronous and managed by the [`UpstreamBroker`].
984#[derive(Debug)]
985struct UpstreamQueue {
986    /// High priority queue.
987    high: VecDeque<Entry>,
988    /// Low priority queue.
989    low: VecDeque<Entry>,
990    /// High priority retry queue.
991    retry_high: VecDeque<Entry>,
992    /// Low priority retry queue.
993    retry_low: VecDeque<Entry>,
994    /// Retries should not be dequeued before this instant.
995    ///
996    /// This retry increments by the constant `retry_after` instead of backoff,
997    /// since it only kicks in for a short period of time before Relay gets into
998    /// network outage mode (see [`IsNetworkOutage`]).
999    next_retry: Instant,
1000    /// Time to wait before retrying another request from the retry queue.
1001    retry_interval: Duration,
1002}
1003
1004impl UpstreamQueue {
1005    /// Creates an empty upstream queue.
1006    pub fn new(retry_interval: Duration) -> Self {
1007        Self {
1008            high: VecDeque::new(),
1009            low: VecDeque::new(),
1010            retry_high: VecDeque::new(),
1011            retry_low: VecDeque::new(),
1012            next_retry: Instant::now(),
1013            retry_interval,
1014        }
1015    }
1016
1017    /// Returns the number of entries in the queue.
1018    pub fn len(&self) -> usize {
1019        self.high.len() + self.low.len() + self.retry_high.len() + self.retry_low.len()
1020    }
1021
1022    /// Places an entry at the back of the queue.
1023    ///
1024    /// Since entries are dequeued in FIFO order, this entry will be dequeued
1025    /// last within its priority class; see
1026    /// [`dequeue`][`UpstreamQueue::dequeue`] for more details.
1027    pub fn enqueue(&mut self, entry: Entry) {
1028        let priority = entry.request.priority();
1029        match priority {
1030            RequestPriority::High => self.high.push_back(entry),
1031            RequestPriority::Low => self.low.push_back(entry),
1032        }
1033        relay_statsd::metric!(
1034            histogram(RelayHistograms::UpstreamMessageQueueSize) = self.len() as u64,
1035            priority = priority.name(),
1036            attempt = "first"
1037        );
1038    }
1039
1040    /// Places an entry in the retry queue.
1041    ///
1042    /// Entries are dequeued by (1) high/low priority and (2) FIFO order.
1043    ///
1044    /// It also schedules the next retry time, based on the retry back off. The
1045    /// retry queue is not dequeued until the next retry has elapsed.
1046    pub fn retry(&mut self, entry: Entry) {
1047        let priority = entry.request.priority();
1048        match priority {
1049            RequestPriority::High => self.retry_high.push_back(entry),
1050            RequestPriority::Low => self.retry_low.push_back(entry),
1051        };
1052
1053        self.next_retry = Instant::now() + self.retry_interval;
1054
1055        relay_statsd::metric!(
1056            histogram(RelayHistograms::UpstreamMessageQueueSize) = self.len() as u64,
1057            priority = priority.name(),
1058            attempt = "retry"
1059        );
1060    }
1061
1062    /// Dequeues the entry with highest priority.
1063    ///
1064    /// Highest priority entry is determined by (1) request priority and (2)
1065    /// retries first.
1066    pub fn dequeue(&mut self) -> Option<Entry> {
1067        let should_retry = self.next_retry <= Instant::now();
1068
1069        if let Some(Some(entry)) = should_retry.then(|| self.retry_high.pop_front()) {
1070            Some(entry)
1071        } else if let Some(entry) = self.high.pop_front() {
1072            Some(entry)
1073        } else if let Some(Some(entry)) = should_retry.then(|| self.retry_low.pop_front()) {
1074            Some(entry)
1075        } else {
1076            self.low.pop_front()
1077        }
1078    }
1079
1080    /// Starts retrying queued requests.
1081    pub fn trigger_retries(&mut self) {
1082        self.next_retry = Instant::now();
1083    }
1084}

1139/// Internal message of the upstream's [`UpstreamBroker`].
1140///
1141/// These messages are used to serialize state mutations to the broker's internals. They are emitted

1143#[derive(Debug)]

1166/// Service that establishes and maintains authentication.
1167///
1168/// In regular intervals, the service checks for authentication and notifies the upstream if the key
1169/// is no longer valid. This allows Sentry to reject registered Relays during runtim without
1170/// restarts.
1171///
1172/// The monitor updates subscribers via the an `Action` channel of all changes to the authentication
1173/// states.
1174#[derive(Debug)]
1175struct AuthMonitor {
1176    config: Arc<Config>,
1177    client: SharedClient,
1178    state: AuthState,
1179    tx: ActionTx,
1180}
1181
1182impl AuthMonitor {
1183    /// Returns the interval at which this Relay should renew authentication.
1184    ///
1185    /// Returns `Some` if authentication should be retried. Returns `None` if authentication is
1186    /// permanent.
1187    fn renew_auth_interval(&self) -> Option<std::time::Duration> {
1188        if self.config.processing_enabled() {
1189            // processing relays do NOT re-authenticate
1190            None
1191        } else {
1192            // only relays that have a configured auth-interval reauthenticate
1193            self.config.http_auth_interval()
1194        }
1195    }
1196
1197    /// Updates the monitor's internal state and subscribers.
1198    fn send_state(&mut self, state: AuthState) -> Result<(), UpstreamRequestError> {
1199        self.state = state;
1200        self.tx
1201            .send(Action::UpdateAuth(state))
1202            .map_err(|_| UpstreamRequestError::ChannelClosed)
1203    }
1204
1205    /// Performs a single authentication pass.
1206    ///
1207    /// Authentication consists of an initial request, a challenge, and a signed response including
1208    /// the challenge. Throughout this sequence, the auth monitor transitions the authentication
1209    /// state. If authentication succeeds, the state is set to [`AuthState::Registered`] at the end.
1210    ///
1211    /// If any of the requests fail, this method returns an `Err` and leaves the last authentication
1212    /// state in place.
1213    async fn authenticate(
1214        &mut self,
1215        credentials: &Credentials,
1216    ) -> Result<(), UpstreamRequestError> {
1217        relay_log::info!(
1218            descriptor = %self.config.upstream_descriptor(),
1219            "registering with upstream"
1220        );
1221
1222        self.send_state(if self.state.is_authenticated() {
1223            AuthState::Renewing
1224        } else {
1225            AuthState::Registering
1226        })?;
1227
1228        let request = RegisterRequest::new(&credentials.id, &credentials.public_key);
1229        let challenge = self.client.send_query(request).await?;
1230        relay_log::debug!(token = challenge.token(), "got register challenge");
1231
1232        let response = challenge.into_response();
1233        relay_log::debug!("sending register challenge response");
1234        self.client.send_query(response).await?;
1235
1236        relay_log::info!("relay successfully registered with upstream");
1237        self.send_state(AuthState::Registered)?;
1238
1239        Ok(())
1240    }
1241
1242    /// Starts the authentication monitor's cycle.
1243    ///
1244    /// Authentication starts immediately and then enters a loop of recurring reauthentication until
1245    /// one of the following conditions is met:
1246    ///
1247    ///  - Authentication is not required based on the Relay's mode configuration.
1248    ///  - The upstream responded with a permanent rejection (auth denied).
1249    ///  - All subscibers have shut down and the action channel is closed.
1250    pub async fn run(mut self) {
1251        if self.config.relay_mode() != RelayMode::Managed {
1252            return;
1253        }
1254
1255        let config = self.config.clone();
1256        let Some(credentials) = config.credentials() else {
1257            // This is checked during setup by `check_config` and should never happen.
1258            relay_log::error!("authentication called without credentials");
1259            return;
1260        };
1261
1262        let mut backoff = RetryBackoff::new(self.config.http_max_retry_interval());
1263
1264        loop {
1265            match self.authenticate(credentials).await {
1266                Ok(_) => {
1267                    backoff.reset();
1268
1269                    match self.renew_auth_interval() {
1270                        Some(interval) => tokio::time::sleep(interval).await,
1271                        None => return,
1272                    }
1273                }
1274                Err(err) => {
1275                    if backoff.attempt() > 1 {
1276                        relay_log::error!(
1277                            error = &err as &dyn std::error::Error,
1278                            tags.attempts = backoff.attempt(),
1279                            "authentication encountered error",
1280                        );
1281                    }
1282
1283                    // ChannelClosed indicates that there are no more listeners, so we stop
1284                    // authenticating.
1285                    if let UpstreamRequestError::ChannelClosed = err {
1286                        return;
1287                    }
1288
1289                    if err.is_permanent_rejection() {
1290                        self.send_state(AuthState::Denied).ok();
1291                        return;
1292                    }
1293
1294                    // If the authentication request fails due to any reason other than a network
1295                    // error, go back to `Registering` which indicates that this Relay is not
1296                    // authenticated, in case the state was `Renewing` before.
1297                    if !err.is_network_error() {
1298                        self.send_state(AuthState::Registering).ok();
1299                    }
1300
1301                    // Even on network errors, retry authentication independently.
1302                    let backoff = backoff.next_backoff();
1303                    relay_log::debug!(
1304                        "scheduling authentication retry in {} seconds",
1305                        backoff.as_secs()
1306                    );
1307                    tokio::time::sleep(backoff).await;
1308                }
1309            };
1310        }
1311    }
1312}

1314/// Internal state of the [`ConnectionMonitor`].

1316enum ConnectionState {

1318    Connected,

1320    /// Network errors have been observed during the grace period.
1321    ///

1324

1331/// Maintains outage state of the connection to the upstream.
1332///
1333///  Use [`notify_error`](Self::notify_error) and [`reset_error`](Self::reset_error) to inform the
1334/// monitor of successful and failed requests. If errors persist throughout a grace period, the
1335/// monitor spawns a background task to re-establish connections. During this period,
1336/// [`is_stable`](Self::is_stable) returns `false` and no other requests should be made to the
1337/// upstream.
1338///
1339/// This state is synchronous and managed by the [`UpstreamBroker`].
1340#[derive(Debug)]
1341struct ConnectionMonitor {
1342    state: ConnectionState,
1343    client: SharedClient,
1344}
1345
1346impl ConnectionMonitor {
1347    /// Creates a new `ConnectionMonitor` in connected state.
1348    pub fn new(client: SharedClient) -> Self {
1349        Self {
1350            state: ConnectionState::Connected,
1351            client,
1352        }
1353    }
1354
1355    /// Resets `Reconnecting` if the connection task has completed.
1356    fn clean_state(&mut self) -> &ConnectionState {
1357        if let ConnectionState::Reconnecting(ref task) = self.state {
1358            if task.is_finished() {
1359                self.state = ConnectionState::Connected;
1360            }
1361        }
1362
1363        &self.state
1364    }
1365
1366    /// Returns `true` if the connection is not in outage state.
1367    pub fn is_stable(&mut self) -> bool {
1368        match self.clean_state() {
1369            ConnectionState::Connected => true,
1370            ConnectionState::Interrupted(_) => true,
1371            ConnectionState::Reconnecting(_) => false,
1372        }
1373    }
1374
1375    /// Returns `true` if the connection is in outage state.
1376    pub fn is_outage(&mut self) -> bool {
1377        !self.is_stable()
1378    }

1380    /// Performs connection attempts with exponential backoff until successful.
1381    async fn connect(client: SharedClient, tx: ActionTx) {
1382        let mut backoff = RetryBackoff::new(client.config.http_max_retry_interval());
1383
1384        loop {
1385            let next_backoff = backoff.next_backoff();
1386            relay_log::warn!("network outage, scheduling another check in {next_backoff:?}");
1387
1388            tokio::time::sleep(next_backoff).await;
1389            match client.send(&mut GetHealthCheck).await {
1390                // All errors that are not connection errors are considered a successful attempt
1391                Err(e) if e.is_network_error() => continue,
1392                _ => break,
1393            }
1394        }
1395
1396        tx.send(Action::Connected).ok();
1397    }
1398
1399    /// Notifies the monitor of a request that resulted in a network error.
1400    ///
1401    /// This starts a grace period if not already started. If a prior grace period has been
1402    /// exceeded, the monitor spawns a background job to reestablish connection and notifies the
1403    /// given `return_tx` on success.
1404    ///
1405    /// This method does not block.
1406    pub fn notify_error(&mut self, return_tx: &ActionTx) {
1407        let now = Instant::now();
1408
1409        let first_error = match self.clean_state() {
1410            ConnectionState::Connected => now,
1411            ConnectionState::Interrupted(first) => *first,
1412            ConnectionState::Reconnecting(_) => return,
1413        };
1414
1415        self.state = ConnectionState::Interrupted(first_error);
1416
1417        // Only take action if we exceeded the grace period.
1418        if first_error + self.client.config.http_outage_grace_period() <= now {
1419            let return_tx = return_tx.clone();
1420            let task = relay_system::spawn!(Self::connect(self.client.clone(), return_tx));
1421            self.state = ConnectionState::Reconnecting(task);
1422        }
1423    }
1424
1425    /// Notifies the monitor of a request that was delivered to the upstream.
1426    ///
1427    /// Resets the outage grace period and aborts connect background tasks.
1428    pub fn reset_error(&mut self) {
1429        if let ConnectionState::Reconnecting(ref task) = self.state {
1430            task.abort();
1431        }
1432
1433        self.state = ConnectionState::Connected;
1434    }
1435}

1437/// Main broker of the [`UpstreamRelayService`].
1438///
1439/// This handles incoming public messages, internal actions, and maintains the upstream queue.
1440#[derive(Debug)]
1441struct UpstreamBroker {
1442    client: SharedClient,
1443    queue: UpstreamQueue,
1444    auth_state: AuthState,
1445    conn: ConnectionMonitor,
1446    permits: usize,
1447    action_tx: ActionTx,
1448}
1449
1450impl UpstreamBroker {
1451    /// Returns the next entry from the queue if the upstream is in a healthy state.
1452    ///
1453    /// This returns `None` in any of the following conditions:
1454    ///  - Maximum request concurrency has been reached. A slot will be reclaimed through
1455    ///    [`Action::Complete`].
1456    ///  - The connection is in outage state and all outgoing requests are suspended. Outage state
1457    ///    will be reset through [`Action::Connected`].
1458    ///  - Relay is not authenticated, including failed renewals. Auth state will be updated through
1459    ///    [`Action::UpdateAuth`].
1460    ///  - The request queue is empty. New requests will be added through [`SendRequest`] or
1461    ///    [`SendQuery`] in the main message loop.
1462    async fn next_request(&mut self) -> Option<Entry> {
1463        if self.permits == 0 || self.conn.is_outage() || !self.auth_state.is_authenticated() {
1464            return None;
1465        }
1466
1467        let entry = self.queue.dequeue()?;
1468        self.permits -= 1;
1469        Some(entry)
1470    }
1471
1472    /// Attempts to place a new request into the queue.
1473    ///
1474    /// If authentication is permanently denied, the request will be failed immediately. In all
1475    /// other cases, the request is enqueued and will wait for submission.
1476    async fn enqueue(&mut self, request: Box<dyn UpstreamRequest>) {
1477        if let AuthState::Denied = self.auth_state {
1478            // This respond is near-instant because it should just send the error into the request's
1479            // response channel. We do not expect that this blocks the broker.
1480            request.respond(Err(UpstreamRequestError::AuthDenied)).await;
1481        } else {
1482            self.queue.enqueue(Entry::new(request));
1483        }
1484    }
1485
1486    /// Handler of the main message loop.
1487    async fn handle_message(&mut self, message: UpstreamRelay) {
1488        match message {
1489            UpstreamRelay::IsAuthenticated(_, sender) => {
1490                sender.send(self.auth_state.is_authenticated())
1491            }
1492            UpstreamRelay::IsNetworkOutage(_, sender) => sender.send(self.conn.is_outage()),
1493            UpstreamRelay::SendRequest(request) => self.enqueue(request).await,
1494        }
1495    }
1496
1497    /// Spawns a request attempt.
1498    ///
1499    /// The request will run concurrently with other spawned requests and notify the action channel
1500    /// on completion.
1501    fn execute(&self, mut entry: Entry) {
1502        let client = self.client.clone();
1503        let action_tx = self.action_tx.clone();
1504
1505        relay_system::spawn!(async move {
1506            let send_start = Instant::now();
1507            let result = client.send(entry.request.as_mut()).await;
1508            emit_response_metrics(send_start, &entry, &result);
1509
1510            let status = match result {
1511                Err(ref err) if err.is_network_error() => RequestOutcome::Dropped,
1512                _ => RequestOutcome::Received,
1513            };
1514
1515            match status {
1516                RequestOutcome::Dropped if entry.request.retry() => {
1517                    entry.retries += 1;
1518                    action_tx.send(Action::Retry(entry)).ok();
1519                }
1520                _ => entry.request.respond(result).await,
1521            }
1522
1523            // Send an action back to the action channel of the broker, which will invoke
1524            // `handle_action`. This is to let the broker know in a synchronized fashion that the
1525            // request has finished and may need to be retried (above).
1526            action_tx.send(Action::Complete(status)).ok();
1527        });
1528    }
1529
1530    /// Marks completion of a running request and reclaims its slot.
1531    fn complete(&mut self, status: RequestOutcome) {
1532        self.permits += 1;
1533
1534        match status {
1535            RequestOutcome::Dropped => self.conn.notify_error(&self.action_tx),
1536            RequestOutcome::Received => {
1537                self.conn.reset_error();
1538                self.queue.trigger_retries();
1539            }
1540        }
1541    }
1542
1543    /// Handler of the internal action channel.
1544    fn handle_action(&mut self, action: Action) {
1545        match action {
1546            Action::Retry(request) => self.queue.retry(request),
1547            Action::Complete(status) => self.complete(status),
1548            Action::Connected => self.conn.reset_error(),
1549            Action::UpdateAuth(state) => self.auth_state = state,
1550        }
1551    }
1552}

1554/// Implementation of the [`UpstreamRelay`] interface.
1555#[derive(Debug)]
1556pub struct UpstreamRelayService {
1557    config: Arc<Config>,
1558}
1559
1560impl UpstreamRelayService {
1561    /// Creates a new `UpstreamRelay` instance.
1562    pub fn new(config: Arc<Config>) -> Self {
1563        // Broker and other actual components are implemented in the Service's `spawn_handler`.
1564        Self { config }
1565    }
1566}
1567
1568impl Service for UpstreamRelayService {
1569    type Interface = UpstreamRelay;
1570
1571    async fn run(self, mut rx: relay_system::Receiver<Self::Interface>) {
1572        let Self { config } = self;
1573
1574        let client = SharedClient::build(config.clone());
1575
1576        // Channel for serialized communication from the auth monitor, connection monitor, and
1577        // concurrent requests back to the broker.
1578        let (action_tx, mut action_rx) = mpsc::unbounded_channel();
1579
1580        // Spawn a recurring background check for authentication. It terminates automatically if
1581        // authentication is not required or rejected.
1582        let auth = AuthMonitor {
1583            config: config.clone(),
1584            client: client.clone(),
1585            state: AuthState::Unknown,
1586            tx: action_tx.clone(),
1587        };
1588        relay_system::spawn!(auth.run());
1589
1590        // Main broker that serializes public and internal messages, as well as maintains connection
1591        // and authentication state.
1592        let mut broker = UpstreamBroker {
1593            client: client.clone(),
1594            queue: UpstreamQueue::new(config.http_retry_delay()),
1595            auth_state: AuthState::init(&config),
1596            conn: ConnectionMonitor::new(client),
1597            permits: config.max_concurrent_requests(),
1598            action_tx,
1599        };
1600
1601        loop {
1602            tokio::select! {
1603                biased;
1604
1605                Some(action) = action_rx.recv() => broker.handle_action(action),
1606                Some(request) = broker.next_request() => broker.execute(request),
1607                Some(message) = rx.recv() => broker.handle_message(message).await,
1608
1609                else => break,
1610            }
1611        }
1612    }
1613}
