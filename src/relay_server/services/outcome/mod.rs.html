67/// Defines the possible outcomes from processing an event.
68#[derive(Clone, Debug, PartialEq, Eq, Hash)]
69pub enum Outcome {
70    /// The event has been accepted and handled completely.
71    ///
72    /// This is only emitted for items going from Relay to Snuba directly.
73    #[allow(dead_code)]
74    Accepted,
75
76    /// The event has been filtered due to a configured filter.
77    Filtered(FilterStatKey),
78
79    /// The event has been filtered by a Sampling Rule
80    FilteredSampling(RuleCategories),
81
82    /// The event has been rate limited.
83    RateLimited(Option<ReasonCode>),
84
85    /// The event has been discarded because of invalid data.
86    Invalid(DiscardReason),
87
88    /// Reserved but unused in Relay.
89    #[allow(dead_code)]
90    Abuse,
91
92    /// The event has already been discarded on the client side.
93    ClientDiscard(String),
94}
95
96impl Outcome {
97    /// Returns the raw numeric value of this outcome for the JSON and Kafka schema.
98    pub fn to_outcome_id(&self) -> OutcomeId {
99        match self {
100            Outcome::Filtered(_) | Outcome::FilteredSampling(_) => OutcomeId::FILTERED,
101            Outcome::RateLimited(_) => OutcomeId::RATE_LIMITED,
102            Outcome::Invalid(_) => OutcomeId::INVALID,
103            Outcome::Abuse => OutcomeId::ABUSE,
104            Outcome::ClientDiscard(_) => OutcomeId::CLIENT_DISCARD,
105            Outcome::Accepted => OutcomeId::ACCEPTED,
106        }
107    }
108
109    /// Returns the `reason` code field of this outcome.
110    pub fn to_reason(&self) -> Option<Cow<'_, str>> {
111        match self {
112            Outcome::Invalid(DiscardReason::ItemTooLarge(too_large_reason)) => Some(Cow::Owned(
113                format!("too_large:{}", too_large_reason.as_str()),
114            )),
115            Outcome::Invalid(discard_reason) => Some(Cow::Borrowed(discard_reason.name())),
116            Outcome::Filtered(filter_key) => Some(filter_key.clone().name()),
117            Outcome::FilteredSampling(rule_ids) => Some(Cow::Owned(format!("Sampled:{rule_ids}"))),
118            Outcome::RateLimited(code_opt) => {
119                code_opt.as_ref().map(|code| Cow::Borrowed(code.as_str()))
120            }
121            Outcome::ClientDiscard(discard_reason) => Some(Cow::Borrowed(discard_reason)),
122            Outcome::Abuse => None,
123            Outcome::Accepted => None,
124        }
125    }
126
127    /// Returns true if there is a bug or an infrastructure problem causing event loss.
128    ///
129    /// This can happen when we introduce bugs or during incidents.
130    ///
131    /// During healthy operation, this should always return false.
132    pub fn is_unexpected(&self) -> bool {
133        matches!(
134            self,
135            Outcome::Invalid(
136                DiscardReason::Internal
137                    | DiscardReason::ProjectState
138                    | DiscardReason::ProjectStatePii,
139            )
140        )
141    }
142}

144impl fmt::Display for Outcome {
145    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
146        match self {
147            Outcome::Filtered(key) => write!(f, "filtered by {key}"),
148            Outcome::FilteredSampling(rule_ids) => write!(f, "sampling rule {rule_ids}"),
149            Outcome::RateLimited(None) => write!(f, "rate limited"),
150            Outcome::RateLimited(Some(reason)) => write!(f, "rate limited with reason {reason}"),
151            Outcome::Invalid(DiscardReason::Internal) => write!(f, "internal error"),
152            Outcome::Invalid(reason) => write!(f, "invalid data ({reason})"),
153            Outcome::Abuse => write!(f, "abuse limit reached"),
154            Outcome::ClientDiscard(reason) => write!(f, "discarded by client ({reason})"),
155            Outcome::Accepted => write!(f, "accepted"),
156        }
157    }
158}

512/// Similar to [`ItemType`] but it does not have any additional information in the

673/// Similar to [`AttachmentType`] but it does not have any additional information in the
674/// Unknown variant so that it can derive [`Copy`] and be used from [`DiscardReason`].
675/// The variants should be the same as [`AttachmentType`].
676#[derive(Copy, Clone, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
677pub enum DiscardAttachmentType {
678    /// A regular attachment without special meaning.
679    Attachment,
680    /// A minidump crash report (binary data).
681    Minidump,
682    /// An apple crash report (text data).
683    AppleCrashReport,
684    /// A msgpack-encoded event payload submitted as part of multipart uploads.
685    EventPayload,
686    /// A msgpack-encoded list of payloads.
687    Breadcrumbs,
688    /// A Nintendo switch dying message.
689    NnswitchDyingMessage,
690    // A prosperodump crash report (binary data)
691    Prosperodump,
692    /// Binary attachment present in Unreal 4 events containing event context information.
693    UnrealContext,
694    /// Binary attachment present in Unreal 4 events containing event Logs.
695    UnrealLogs,
696    /// An application UI view hierarchy (json payload).
697    ViewHierarchy,
698}
699
700impl From<&AttachmentType> for DiscardAttachmentType {
701    fn from(value: &AttachmentType) -> Self {
702        match value {
703            AttachmentType::Attachment => Self::Attachment,
704            AttachmentType::Minidump => Self::Minidump,
705            AttachmentType::AppleCrashReport => Self::AppleCrashReport,
706            AttachmentType::EventPayload => Self::EventPayload,
707            AttachmentType::Breadcrumbs => Self::Breadcrumbs,
708            AttachmentType::Prosperodump => Self::Prosperodump,
709            AttachmentType::NintendoSwitchDyingMessage => Self::NnswitchDyingMessage,
710            AttachmentType::UnrealContext => Self::UnrealContext,
711            AttachmentType::UnrealLogs => Self::UnrealLogs,
712            AttachmentType::ViewHierarchy => Self::ViewHierarchy,
713        }
714    }
715}
716
717impl From<AttachmentType> for DiscardAttachmentType {
718    fn from(value: AttachmentType) -> Self {
719        From::from(&value)
720    }
721}
