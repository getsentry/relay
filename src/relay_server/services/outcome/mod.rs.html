19/// The numerical identifier of the outcome category (Accepted, Filtered, ...)
20#[derive(Debug, Serialize, Deserialize, Clone, Copy, Eq, PartialEq)]
21pub struct OutcomeId(u8);
22
23impl OutcomeId {
24    // This is not an enum because we still want to forward unknown outcome IDs transparently
25    const ACCEPTED: OutcomeId = OutcomeId(0);
26    const FILTERED: OutcomeId = OutcomeId(1);
27    const RATE_LIMITED: OutcomeId = OutcomeId(2);
28    const INVALID: OutcomeId = OutcomeId(3);
29    const ABUSE: OutcomeId = OutcomeId(4);
30    const CLIENT_DISCARD: OutcomeId = OutcomeId(5);
31    #[cfg_attr(not(any(test, feature = "processing")), expect(unused))]
32    const CARDINALITY_LIMITED: OutcomeId = OutcomeId(6);
33
34    pub fn as_u8(self) -> u8 {
35        self.0
36    }
37
38    /// Returns `true` for outcomes which are critical for billing.
39    pub fn is_billing(self) -> bool {
40        matches!(self, OutcomeId::ACCEPTED | OutcomeId::RATE_LIMITED)
41    }
42}

44/// Defines the possible outcomes from processing an event.
45#[derive(Clone, Debug, PartialEq, Eq, Hash)]
46pub enum Outcome {
47    /// The event has been accepted and handled completely.
48    ///
49    /// This is only emitted for items going from Relay to Snuba directly.
50    #[allow(dead_code)]
51    Accepted,
52
53    /// The event has been filtered due to a configured filter.
54    Filtered(FilterStatKey),
55
56    /// The event has been filtered by a Sampling Rule
57    FilteredSampling(RuleCategories),
58
59    /// The event has been rate limited.
60    RateLimited(Option<ReasonCode>),
61
62    /// The event has been discarded because of invalid data.
63    Invalid(DiscardReason),
64
65    /// Reserved but unused in Relay.
66    #[allow(dead_code)]
67    Abuse,
68
69    /// The event has already been discarded on the client side.
70    ClientDiscard(String),
71}
72
73impl Outcome {
74    /// Returns the raw numeric value of this outcome for the JSON and Kafka schema.
75    pub fn to_outcome_id(&self) -> OutcomeId {
76        match self {
77            Outcome::Filtered(_) | Outcome::FilteredSampling(_) => OutcomeId::FILTERED,
78            Outcome::RateLimited(_) => OutcomeId::RATE_LIMITED,
79            Outcome::Invalid(_) => OutcomeId::INVALID,
80            Outcome::Abuse => OutcomeId::ABUSE,
81            Outcome::ClientDiscard(_) => OutcomeId::CLIENT_DISCARD,
82            Outcome::Accepted => OutcomeId::ACCEPTED,
83        }
84    }
85
86    /// Returns the `reason` code field of this outcome.
87    pub fn to_reason(&self) -> Option<Cow<'_, str>> {
88        match self {
89            Outcome::Invalid(DiscardReason::ItemTooLarge(too_large_reason)) => Some(Cow::Owned(
90                format!("too_large:{}", too_large_reason.as_str()),
91            )),
92            Outcome::Invalid(discard_reason) => Some(Cow::Borrowed(discard_reason.name())),
93            Outcome::Filtered(filter_key) => Some(filter_key.clone().name()),
94            Outcome::FilteredSampling(rule_ids) => Some(Cow::Owned(format!("Sampled:{rule_ids}"))),
95            Outcome::RateLimited(code_opt) => {
96                code_opt.as_ref().map(|code| Cow::Borrowed(code.as_str()))
97            }
98            Outcome::ClientDiscard(discard_reason) => Some(Cow::Borrowed(discard_reason)),
99            Outcome::Abuse => None,
100            Outcome::Accepted => None,
101        }
102    }
103
104    /// Returns true if there is a bug or an infrastructure problem causing event loss.
105    ///
106    /// This can happen when we introduce bugs or during incidents.
107    ///
108    /// During healthy operation, this should always return false.
109    pub fn is_unexpected(&self) -> bool {
110        matches!(
111            self,
112            Outcome::Invalid(
113                DiscardReason::Internal
114                    | DiscardReason::ProjectState
115                    | DiscardReason::ProjectStatePii,
116            )
117        )
118    }
119}

121impl fmt::Display for Outcome {
122    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
123        match self {
124            Outcome::Filtered(key) => write!(f, "filtered by {key}"),
125            Outcome::FilteredSampling(rule_ids) => write!(f, "sampling rule {rule_ids}"),
126            Outcome::RateLimited(None) => write!(f, "rate limited"),
127            Outcome::RateLimited(Some(reason)) => write!(f, "rate limited with reason {reason}"),
128            Outcome::Invalid(DiscardReason::Internal) => write!(f, "internal error"),
129            Outcome::Invalid(reason) => write!(f, "invalid data ({reason})"),
130            Outcome::Abuse => write!(f, "abuse limit reached"),
131            Outcome::ClientDiscard(reason) => write!(f, "discarded by client ({reason})"),
132            Outcome::Accepted => write!(f, "accepted"),
133        }
134    }
135}

651/// Unknown variant so that it can derive [`Copy`] and be used from [`DiscardReason`].
652/// The variants should be the same as [`AttachmentType`].
653#[derive(Copy, Clone, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
654pub enum DiscardAttachmentType {
655    /// A regular attachment without special meaning.
656    Attachment,
657    /// A minidump crash report (binary data).
658    Minidump,
659    /// An apple crash report (text data).
660    AppleCrashReport,
661    /// A msgpack-encoded event payload submitted as part of multipart uploads.
662    EventPayload,
663    /// A msgpack-encoded list of payloads.
664    Breadcrumbs,
665    /// A Nintendo switch dying message.
666    NnswitchDyingMessage,
667    // A prosperodump crash report (binary data)
668    Prosperodump,
669    /// Binary attachment present in Unreal 4 events containing event context information.
670    UnrealContext,
671    /// Binary attachment present in Unreal 4 events containing event Logs.
672    UnrealLogs,
673    /// An application UI view hierarchy (json payload).
674    ViewHierarchy,
675}
676
677impl From<&AttachmentType> for DiscardAttachmentType {
678    fn from(value: &AttachmentType) -> Self {
679        match value {
680            AttachmentType::Attachment => Self::Attachment,
681            AttachmentType::Minidump => Self::Minidump,
682            AttachmentType::AppleCrashReport => Self::AppleCrashReport,
683            AttachmentType::EventPayload => Self::EventPayload,
684            AttachmentType::Breadcrumbs => Self::Breadcrumbs,
685            AttachmentType::Prosperodump => Self::Prosperodump,
686            AttachmentType::NintendoSwitchDyingMessage => Self::NnswitchDyingMessage,
687            AttachmentType::UnrealContext => Self::UnrealContext,
688            AttachmentType::UnrealLogs => Self::UnrealLogs,
689            AttachmentType::ViewHierarchy => Self::ViewHierarchy,
690        }
691    }
692}
693
694impl From<AttachmentType> for DiscardAttachmentType {
695    fn from(value: AttachmentType) -> Self {
696        From::from(&value)
697    }
698}
