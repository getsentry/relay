1//! This module contains the outcomes aggregator, which collects similar outcomes into groups
2//! and flushed them periodically.
3
4use std::collections::HashMap;
5use std::time::Duration;
6
7use chrono::Utc;
8use relay_common::time::UnixTimestamp;
9use relay_config::{Config, EmitOutcomes};
10use relay_quotas::{DataCategory, Scoping};
11use relay_statsd::metric;
12use relay_system::{Addr, Controller, Service, Shutdown};
13
14use crate::services::outcome::{Outcome, OutcomeProducer, TrackOutcome};
15use crate::statsd::RelayTimers;
16use crate::utils::SleepHandle;

18/// Contains everything to construct a `TrackOutcome`, except quantity
19#[derive(Debug, PartialEq, Eq, Hash)]
20struct BucketKey {
21    /// The time slot for which outcomes are aggregated. timestamp = offset * bucket_interval
22    offset: u64,
23    /// Scoping of the request.
24    pub scoping: Scoping,
25    /// The outcome.
26    pub outcome: Outcome,
27    /// The event's data category.
28    pub category: DataCategory,
29}
30
31/// Aggregates [`Outcome`]s into buckets and flushes them periodically.
32///
33/// This service handles a single message [`TrackOutcome`].
34pub struct OutcomeAggregator {
35    /// Whether or not to produce outcomes.
36    ///
37    /// If `true`, all outcomes will be dropped.
38    disabled: bool,
39    /// Whether outcomes should be aggregated.
40    ///
41    /// If `false` outcomes are not aggregated and just immediately forwarded.
42    aggregate: bool,
43    /// The width of each aggregated bucket in seconds
44    bucket_interval: u64,
45    /// The number of seconds between flushes of all buckets
46    flush_interval: u64,
47    /// Mapping from bucket key to quantity.
48    buckets: HashMap<BucketKey, u32>,
49    /// The recipient of the aggregated outcomes
50    outcome_producer: Addr<OutcomeProducer>,
51    /// An optional timeout to the next scheduled flush.
52    flush_handle: SleepHandle,
53}
54
55impl OutcomeAggregator {
56    pub fn new(config: &Config, outcome_producer: Addr<OutcomeProducer>) -> Self {
57        let disabled = matches!(config.emit_outcomes(), EmitOutcomes::None);
58        let aggregate = !matches!(config.emit_outcomes(), EmitOutcomes::AsOutcomes);
59
60        Self {
61            disabled,
62            aggregate,
63            bucket_interval: config.outcome_aggregator().bucket_interval,
64            flush_interval: config.outcome_aggregator().flush_interval,
65            buckets: HashMap::new(),
66            outcome_producer,
67            flush_handle: SleepHandle::idle(),
68        }
69    }
70
71    fn handle_shutdown(&mut self, message: Shutdown) {
72        if message.timeout.is_some() {
73            self.flush();
74            relay_log::info!("outcome aggregator stopped");
75        }
76    }
77
78    fn handle_track_outcome(&mut self, msg: TrackOutcome) {
79        if self.disabled {
80            return;
81        }
82
83        // This is a temporary measure until the aggregator can be entirely removed in favor of
84        // using metrics as the primary aggregator and forward mechanism for outcomes.
85        if !self.aggregate {
86            self.outcome_producer.send(msg);
87            return;
88        }
89
90        let offset = msg.timestamp.timestamp() as u64 / self.bucket_interval;
91
92        let bucket_key = BucketKey {
93            offset,
94            scoping: msg.scoping,
95            outcome: msg.outcome,
96            category: msg.category,
97        };
98
99        let counter = self.buckets.entry(bucket_key).or_insert(0);
100        *counter += msg.quantity;
101
102        if self.flush_interval == 0 {
103            // Flush immediately. This is useful for integration tests.
104            self.do_flush();
105        } else if self.flush_handle.is_idle() {
106            self.flush_handle
107                .set(Duration::from_secs(self.flush_interval));
108        }
109    }
110
111    fn do_flush(&mut self) {
112        self.flush_handle.reset();
113
114        let bucket_interval = self.bucket_interval;
115        let outcome_producer = self.outcome_producer.clone();
116
117        for (bucket_key, quantity) in self.buckets.drain() {
118            let BucketKey {
119                offset,
120                scoping,
121                outcome,
122                category,
123            } = bucket_key;
124
125            // In case the timestamp cannot be extracted, we fallback to the current UTC `DateTime`
126            let timestamp = UnixTimestamp::from_secs(offset * bucket_interval)
127                .as_datetime()
128                .unwrap_or_else(Utc::now);
129
130            let outcome = TrackOutcome {
131                timestamp,
132                scoping,
133                outcome,
134                event_id: None,
135                remote_addr: None,
136                category,
137                quantity,
138            };
139
140            outcome_producer.send(outcome);
141        }
142    }
143
144    fn flush(&mut self) {
145        metric!(timer(RelayTimers::OutcomeAggregatorFlushTime), {
146            self.do_flush();
147        });
148    }
149}
150
151impl Service for OutcomeAggregator {
152    type Interface = TrackOutcome;
153
154    async fn run(mut self, mut rx: relay_system::Receiver<Self::Interface>) {
155        let mut shutdown = Controller::shutdown_handle();
156        relay_log::info!("outcome aggregator started");
157
158        loop {
159            tokio::select! {
160                // Prioritize flush over receiving messages to prevent starving. Shutdown can be
161                // last since it is not vital if there are still messages in the channel.
162                biased;
163
164                () = &mut self.flush_handle => self.flush(),
165                Some(message) = rx.recv() => self.handle_track_outcome(message),
166                shutdown = shutdown.notified() => self.handle_shutdown(shutdown),
167                else => break,
168            }
169        }
170
171        relay_log::info!("outcome aggregator stopped");
172    }
173}
