27/// Converts a [`TrackOutcome`] to a metric [`Bucket`].
28pub fn to_metric(outcome: &TrackOutcome, config: &Config) -> Bucket {
29    static ACCEPTED: LazyLock<MetricName> = LazyLock::new(|| OUTCOME_ACCEPTED_MRI.into());
30    static FILTERED: LazyLock<MetricName> = LazyLock::new(|| FILTERED_MRI.into());
31    static RATE_LIMITED: LazyLock<MetricName> = LazyLock::new(|| RATE_LIMITED_MRI.into());
32    static INVALID: LazyLock<MetricName> = LazyLock::new(|| INVALID_MRI.into());
33    static ABUSE: LazyLock<MetricName> = LazyLock::new(|| ABUSE_MRI.into());
34    static CLIENT_DISCARD: LazyLock<MetricName> = LazyLock::new(|| CLIENT_DISCARD_MRI.into());
35
36    let TrackOutcome {
37        timestamp,
38        // Metrics are attached to the project key already, no need to add information from the scoping.
39        scoping: _,
40        outcome,
41        event_id,
42        remote_addr,
43        category,
44        quantity,
45    } = outcome;
46
47    let name = match outcome {
48        Outcome::Accepted => ACCEPTED.clone(),
49        Outcome::Filtered(_) => FILTERED.clone(),
50        Outcome::FilteredSampling(_) => FILTERED.clone(),
51        Outcome::RateLimited(_) => RATE_LIMITED.clone(),
52        Outcome::Invalid(_) => INVALID.clone(),
53        Outcome::Abuse => ABUSE.clone(),
54        Outcome::ClientDiscard(_) => CLIENT_DISCARD.clone(),
55    };
56
57    let tags = {
58        let mut tags = BTreeMap::new();
59        // `TrackOutcome` can only be created within this Relay -> set the source.
60        if let Some(source) = config.outcome_source() {
61            tags.insert("source".to_owned(), source.to_owned());
62        }
63        if let Some(reason) = outcome.to_reason() {
64            tags.insert("reason".to_owned(), reason.to_string());
65        }
66        // We stopped adding event id and remote addr to outcomes for cardinality reasons.
67        let _ = event_id;
68        let _ = remote_addr;
69        if let Some(category) = category.value() {
70            tags.insert("category".to_owned(), category.to_string());
71        }
72        tags
73    };
74
75    let timestamp = UnixTimestamp::from_datetime(*timestamp).unwrap_or_else(UnixTimestamp::now);
76
77    Bucket {
78        name,
79        value: BucketValue::Counter((*quantity).into()),
80        timestamp,
81        tags,
82        width: 0,
83        metadata: BucketMetadata::new(timestamp),
84    }
85}

87/// Converts a outcome metric name to a [`OutcomeId`].
88///
89/// Returns `None` for unknown or invalid metric names.
90#[cfg(any(test, feature = "processing"))]
91pub fn to_outcome_id(name: &MetricName) -> Option<OutcomeId> {
92    Some(match name.as_ref() {
93        OUTCOME_ACCEPTED_MRI => OutcomeId::ACCEPTED,
94        FILTERED_MRI => OutcomeId::FILTERED,
95        RATE_LIMITED_MRI => OutcomeId::RATE_LIMITED,
96        INVALID_MRI => OutcomeId::INVALID,
97        ABUSE_MRI => OutcomeId::ABUSE,
98        CLIENT_DISCARD_MRI => OutcomeId::CLIENT_DISCARD,
99        CARDINALITY_LIMITED_MRI => OutcomeId::CARDINALITY_LIMITED,
100        _ => return None,
101    })
102}

235    #[test]
236    fn test_to_outcome_id_roundtrip() {
237        let outcomes = [
238            Outcome::Accepted,
239            Outcome::Filtered(FilterStatKey::IpAddress),
240            Outcome::FilteredSampling(RuleCategories(Default::default())),
241            Outcome::RateLimited(None),
242            Outcome::Invalid(DiscardReason::Duplicate),
243            Outcome::Abuse,
244            Outcome::ClientDiscard("foo".to_owned()),
245        ];
246
247        for outcome in outcomes {
248            let bucket = to_metric(
249                &TrackOutcome {
250                    timestamp: chrono::DateTime::from_timestamp_nanos(123_000_000_000),
251                    scoping: scoping(),
252                    outcome: outcome.clone(),
253                    event_id: None,
254                    remote_addr: None,
255                    category: DataCategory::Error,
256                    quantity: 42,
257                },
258                &config(),
259            );
260
261            let id = to_outcome_id(&bucket.name).unwrap();
262            assert_eq!(id, outcome.to_outcome_id(), "{} | {outcome:?}", bucket.name);
263        }
264    }
