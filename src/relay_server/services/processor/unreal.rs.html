13/// Expands Unreal 4 items inside an envelope.
14///
15/// If the envelope does NOT contain an `UnrealReport` item, it doesn't do anything. If the
16/// envelope contains an `UnrealReport` item, it removes it from the envelope and inserts new
17/// items for each of its contents.
18///
19/// The envelope may be dropped if it exceeds size limits after decompression. Particularly,
20/// this includes cases where a single attachment file exceeds the maximum file size. This is in
21/// line with the behavior of the envelope endpoint.
22///
23/// After this, [`crate::services::processor::EnvelopeProcessorService`] should be able to process the envelope the same
24/// way it processes any other envelopes.
25pub fn expand(
26    managed_envelope: &mut TypedEnvelope<ErrorGroup>,
27    config: &Config,
28) -> Result<(), ProcessingError> {
29    let envelope = &mut managed_envelope.envelope_mut();
30
31    if let Some(item) = envelope.take_item_by(|item| item.ty() == &ItemType::UnrealReport) {
32        utils::expand_unreal_envelope(item, envelope, config)?;
33    }
34
35    Ok(())
36}

38/// Extracts event information from an unreal context.
39///
40/// If the event does not contain an unreal context, this function does not perform any action.
41/// If there was no event payload prior to this function, it is created.
42pub fn process(
43    managed_envelope: &mut TypedEnvelope<ErrorGroup>,
44    event: &mut Annotated<Event>,
45) -> Result<Option<EventFullyNormalized>, ProcessingError> {
46    if utils::process_unreal_envelope(event, managed_envelope.envelope_mut())
47        .map_err(ProcessingError::InvalidUnrealReport)?
48    {
49        return Ok(Some(EventFullyNormalized(false)));
50    }
51
52    Ok(None)
53}
