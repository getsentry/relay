42/// Extracts the primary event payload from an envelope.
43///
44/// The event is obtained from only one source in the following precedence:
45///  1. An explicit event item. This is also the case for JSON uploads.
46///  2. A security report item.
47///  3. Attachments `__sentry-event` and `__sentry-breadcrumb1/2`.
48///  4. A multipart form data body.
49///  5. If none match, `Annotated::empty()`.
50pub fn extract<Group: EventProcessing>(
51    managed_envelope: &mut TypedEnvelope<Group>,
52    metrics: &mut Metrics,
53    event_fully_normalized: EventFullyNormalized,
54    config: &Config,
55) -> Result<ExtractionResult, ProcessingError> {
56    let envelope = managed_envelope.envelope_mut();
57
58    // Remove all items first, and then process them. After this function returns, only
59    // attachments can remain in the envelope. The event will be added again at the end of
60    // `process_event`.
61    let event_item = envelope.take_item_by(|item| item.ty() == &ItemType::Event);
62    let transaction_item = envelope.take_item_by(|item| item.ty() == &ItemType::Transaction);
63    let security_item = envelope.take_item_by(|item| item.ty() == &ItemType::Security);
64    let raw_security_item = envelope.take_item_by(|item| item.ty() == &ItemType::RawSecurity);
65    let nel_item = envelope.take_item_by(|item| item.ty() == &ItemType::Nel);
66    let user_report_v2_item = envelope.take_item_by(|item| item.ty() == &ItemType::UserReportV2);
67    let form_item = envelope.take_item_by(|item| item.ty() == &ItemType::FormData);
68    let attachment_item =
69        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::EventPayload));
70    let breadcrumbs1 =
71        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::Breadcrumbs));
72    let breadcrumbs2 =
73        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::Breadcrumbs));
74
75    // Event items can never occur twice in an envelope.
76    if let Some(duplicate) =
77        envelope.get_item_by(|item| is_duplicate(item, config.processing_enabled()))
78    {
79        return Err(ProcessingError::DuplicateItem(duplicate.ty().clone()));
80    }
81
82    let skip_normalization = config.processing_enabled() && event_fully_normalized.0;
83
84    let mut event_metrics_extracted = None;
85    let mut spans_extracted = None;
86    let (event, event_len) = if let Some(item) = event_item.or(security_item) {
87        relay_log::trace!("processing json event");
88        metric!(timer(RelayTimers::EventProcessingDeserialize), {
89            let (mut annotated_event, len) = event_from_json_payload(item, None)?;
90            // Event items can never include transactions, so retain the event type and let
91            // inference deal with this during normalization.
92            if let Some(event) = annotated_event.value_mut() {
93                if !skip_normalization {
94                    event.ty.set_value(None);
95                }
96            }
97            (annotated_event, len)
98        })
99    } else if let Some(item) = transaction_item {
100        relay_log::trace!("processing json transaction");
101
102        event_metrics_extracted = Some(EventMetricsExtracted(item.metrics_extracted()));
103        spans_extracted = Some(SpansExtracted(item.spans_extracted()));
104
105        metric!(timer(RelayTimers::EventProcessingDeserialize), {
106            // Transaction items can only contain transaction events. Force the event type to
107            // hint to normalization that we're dealing with a transaction now.
108            event_from_json_payload(item, Some(EventType::Transaction))?
109        })
110    } else if let Some(item) = user_report_v2_item {
111        relay_log::trace!("processing user_report_v2");
112        event_from_json_payload(item, Some(EventType::UserReportV2))?
113    } else if let Some(item) = raw_security_item {
114        relay_log::trace!("processing security report");
115        event_from_security_report(item, envelope.meta()).map_err(|error| {
116            if !matches!(error, ProcessingError::UnsupportedSecurityType) {
117                relay_log::error!(
118                    error = &error as &dyn Error,
119                    "failed to extract security report"
120                );
121            }
122            error
123        })?
124    } else if let Some(item) = nel_item {
125        relay_log::trace!("processing nel report");
126        event_from_nel_item(item, envelope.meta()).inspect_err(|error| {
127            relay_log::error!(error = error as &dyn Error, "failed to extract NEL report");
128        })?
129    } else if attachment_item.is_some() || breadcrumbs1.is_some() || breadcrumbs2.is_some() {
130        relay_log::trace!("extracting attached event data");
131        event_from_attachments(config, attachment_item, breadcrumbs1, breadcrumbs2)?
132    } else if let Some(item) = form_item {
133        relay_log::trace!("extracting form data");
134        let len = item.len();
135
136        let mut value = SerdeValue::Object(Default::default());
137        merge_formdata(&mut value, item);
138        let event = Annotated::deserialize_with_meta(value).unwrap_or_default();
139
140        (event, len)
141    } else {
142        relay_log::trace!("no event in envelope");
143        (Annotated::empty(), 0)
144    };
145
146    metrics.bytes_ingested_event = Annotated::new(event_len as u64);
147
148    Ok(ExtractionResult {
149        event,
150        event_metrics_extracted,
151        spans_extracted,
152    })
153}

155pub fn finalize<Group: EventProcessing>(
156    managed_envelope: &mut TypedEnvelope<Group>,
157    event: &mut Annotated<Event>,
158    metrics: &mut Metrics,
159    config: &Config,
160) -> Result<(), ProcessingError> {
161    let envelope = managed_envelope.envelope_mut();
162
163    let inner_event = match event.value_mut() {
164        Some(event) => event,
165        None if !config.processing_enabled() => return Ok(()),
166        None => return Err(ProcessingError::NoEventPayload),
167    };
168
169    if !config.processing_enabled() {
170        static MY_VERSION_STRING: OnceLock<String> = OnceLock::new();
171        let my_version = MY_VERSION_STRING.get_or_init(|| RelayVersion::current().to_string());
172
173        inner_event
174            .ingest_path
175            .get_or_insert_with(Default::default)
176            .push(Annotated::new(RelayInfo {
177                version: Annotated::new(my_version.clone()),
178                public_key: config
179                    .public_key()
180                    .map_or(Annotated::empty(), |pk| Annotated::new(pk.to_string())),
181                other: Default::default(),
182            }));
183    }
184
185    // Event id is set statically in the ingest path.
186    let event_id = envelope.event_id().unwrap_or_default();
187    debug_assert!(!event_id.is_nil());
188
189    // Ensure that the event id in the payload is consistent with the envelope. If an event
190    // id was ingested, this will already be the case. Otherwise, this will insert a new
191    // event id. To be defensive, we always overwrite to ensure consistency.
192    inner_event.id = Annotated::new(event_id);
193
194    // In processing mode, also write metrics into the event. Most metrics have already been
195    // collected at this state, except for the combined size of all attachments.
196    if config.processing_enabled() {
197        let mut metrics = std::mem::take(metrics);
198
199        let attachment_size = envelope
200            .items()
201            .filter(|item| item.attachment_type() == Some(&AttachmentType::Attachment))
202            .map(|item| item.len() as u64)
203            .sum::<u64>();
204
205        if attachment_size > 0 {
206            metrics.bytes_ingested_event_attachment = Annotated::new(attachment_size);
207        }
208
209        inner_event._metrics = Annotated::new(metrics);
210
211        if inner_event.ty.value() == Some(&EventType::Transaction) {
212            metric!(
213                counter(RelayCounters::EventTransaction) += 1,
214                source = utils::transaction_source_tag(inner_event),
215                platform =
216                    PlatformTag::from(inner_event.platform.as_str().unwrap_or("other")).name(),
217                contains_slashes = if inner_event
218                    .transaction
219                    .as_str()
220                    .unwrap_or_default()
221                    .contains('/')
222                {
223                    "true"
224                } else {
225                    "false"
226                }
227            );
228
229            let span_count = inner_event.spans.value().map(Vec::len).unwrap_or(0) as u64;
230            metric!(
231                histogram(RelayHistograms::EventSpans) = span_count,
232                sdk = envelope.meta().client_name().name(),
233                platform = inner_event.platform.as_str().unwrap_or("other"),
234            );
235
236            let has_otel = inner_event
237                .contexts
238                .value()
239                .is_some_and(|contexts| contexts.contains::<OtelContext>());
240
241            if has_otel {
242                metric!(
243                    counter(RelayCounters::OpenTelemetryEvent) += 1,
244                    sdk = envelope.meta().client_name().name(),
245                    platform = inner_event.platform.as_str().unwrap_or("other"),
246                );
247            }
248        }
249
250        if let Some(dsc) = envelope.dsc() {
251            if let Ok(Some(value)) = relay_protocol::to_value(dsc) {
252                inner_event._dsc = Annotated::new(value);
253            }
254        }
255    }
256
257    let mut processor =
258        ClockDriftProcessor::new(envelope.sent_at(), managed_envelope.received_at())
259            .at_least(MINIMUM_CLOCK_DRIFT);
260    processor::process_value(event, &mut processor, ProcessingState::root())
261        .map_err(|_| ProcessingError::InvalidTransaction)?;
262
263    // Log timestamp delays for all events after clock drift correction. This happens before
264    // store processing, which could modify the timestamp if it exceeds a threshold. We are
265    // interested in the actual delay before this correction.
266    if let Some(timestamp) = event.value().and_then(|e| e.timestamp.value()) {
267        let event_delay = managed_envelope.received_at() - timestamp.into_inner();
268        if event_delay > SignedDuration::minutes(1) {
269            let category = event_category(event).unwrap_or(DataCategory::Unknown);
270            metric!(
271                timer(RelayTimers::TimestampDelay) = event_delay.to_std().unwrap(),
272                category = category.name(),
273            );
274        }
275    }
276
277    Ok(())
278}

280/// Status for applying some filters that don't drop the event.
281///
282/// The enum represents either the success of running all filters and keeping
283/// the event, [`FiltersStatus::Ok`], or not running all the filters because
284/// some are unsupported, [`FiltersStatus::Unsupported`].
285///
286/// If there are unsuppported filters, Relay should forward the event upstream
287/// so that a more up-to-date Relay can apply filters appropriately. Actions
288/// that depend on the outcome of event filtering, such as metric extraction,
289/// should be postponed until a filtering decision is made.
290#[must_use]
291pub enum FiltersStatus {
292    /// All filters have been applied and the event should be kept.
293    Ok,
294    /// Some filters are not supported and were not applied.
295    ///
296    /// Relay should forward events upstream for a more up-to-date Relay to apply these filters.
297    /// Supported filters were applied and they don't reject the event.
298    Unsupported,
299}
300
301pub fn filter<Group: EventProcessing>(
302    managed_envelope: &mut TypedEnvelope<Group>,
303    event: &mut Annotated<Event>,
304    project_info: Arc<ProjectInfo>,
305    global_config: &GlobalConfig,
306) -> Result<FiltersStatus, ProcessingError> {
307    let event = match event.value_mut() {
308        Some(event) => event,
309        // Some events are created by processing relays (e.g. unreal), so they do not yet
310        // exist at this point in non-processing relays.
311        None => return Ok(FiltersStatus::Ok),
312    };
313
314    let client_ip = managed_envelope.envelope().meta().client_addr();
315    let filter_settings = &project_info.config.filter_settings;
316
317    metric!(timer(RelayTimers::EventProcessingFiltering), {
318        relay_filter::should_filter(event, client_ip, filter_settings, global_config.filters())
319            .map_err(|err| {
320                managed_envelope.reject(Outcome::Filtered(err.clone()));
321                ProcessingError::EventFiltered(err)
322            })
323    })?;
324
325    // Don't extract metrics if relay can't apply generic filters.  A filter
326    // applied in another up-to-date relay in chain may need to drop the event,
327    // and there should not be metrics from dropped events.
328    // In processing relays, always extract metrics to avoid losing them.
329    let supported_generic_filters = global_config.filters.is_ok()
330        && relay_filter::are_generic_filters_supported(
331            global_config.filters().map(|f| f.version),
332            project_info.config.filter_settings.generic.version,
333        );
334    if supported_generic_filters {
335        Ok(FiltersStatus::Ok)
336    } else {
337        Ok(FiltersStatus::Unsupported)
338    }
339}

341/// Apply data privacy rules to the event payload.
342///
343/// This uses both the general `datascrubbing_settings`, as well as the the PII rules.
344pub fn scrub(
345    event: &mut Annotated<Event>,
346    project_info: Arc<ProjectInfo>,
347) -> Result<(), ProcessingError> {
348    let config = &project_info.config;
349
350    if config.datascrubbing_settings.scrub_data {
351        if let Some(event) = event.value_mut() {
352            relay_pii::scrub_graphql(event);
353        }
354    }
355
356    metric!(timer(RelayTimers::EventProcessingPii), {
357        if let Some(ref config) = config.pii_config {
358            let mut processor = PiiProcessor::new(config.compiled());
359            processor::process_value(event, &mut processor, ProcessingState::root())?;
360        }
361        let pii_config = config
362            .datascrubbing_settings
363            .pii_config()
364            .map_err(|e| ProcessingError::PiiConfigError(e.clone()))?;
365        if let Some(config) = pii_config {
366            let mut processor = PiiProcessor::new(config.compiled());
367            processor::process_value(event, &mut processor, ProcessingState::root())?;
368        }
369    });
370
371    Ok(())
372}

374pub fn serialize<Group: EventProcessing>(
375    managed_envelope: &mut TypedEnvelope<Group>,
376    event: &mut Annotated<Event>,
377    event_fully_normalized: EventFullyNormalized,
378    event_metrics_extracted: EventMetricsExtracted,
379    spans_extracted: SpansExtracted,
380) -> Result<(), ProcessingError> {
381    if event.is_empty() {
382        relay_log::error!("Cannot serialize empty event");
383        return Ok(());
384    }
385
386    let data = metric!(timer(RelayTimers::EventProcessingSerialization), {
387        event.to_json().map_err(ProcessingError::SerializeFailed)?
388    });
389
390    let event_type = event_type(event).unwrap_or_default();
391    let mut event_item = Item::new(ItemType::from_event_type(event_type));
392    event_item.set_payload(ContentType::Json, data);
393
394    // TODO: The state should simply maintain & update an `ItemHeaders` object.
395    // If transaction metrics were extracted, set the corresponding item header
396    event_item.set_metrics_extracted(event_metrics_extracted.0);
397    event_item.set_spans_extracted(spans_extracted.0);
398    event_item.set_fully_normalized(event_fully_normalized.0);
399
400    managed_envelope.envelope_mut().add_item(event_item);
401
402    Ok(())
403}

405/// Checks if the Event includes unprintable fields.
406pub fn has_unprintable_fields(event: &Annotated<Event>) -> bool {
407    fn is_unprintable(value: &&str) -> bool {
408        value.chars().any(|c| {
409            c == '\u{fffd}' // unicode replacement character
410                || (c.is_control() && !c.is_whitespace()) // non-whitespace control characters
411        })
412    }
413    if let Some(event) = event.value() {
414        let env = event.environment.as_str().filter(is_unprintable);
415        let release = event.release.as_str().filter(is_unprintable);
416        env.is_some() || release.is_some()
417    } else {
418        false
419    }
420}

422/// Computes and emits metrics for monitoring user feedback (UserReportV2) ingest
423pub fn emit_feedback_metrics(envelope: &Envelope) {
424    let mut has_feedback = false;
425    let mut num_attachments = 0;
426    for item in envelope.items() {
427        match item.ty() {
428            ItemType::UserReportV2 => has_feedback = true,
429            ItemType::Attachment => num_attachments += 1,

435    }

438/// Checks for duplicate items in an envelope.
439///
440/// An item is considered duplicate if it was not removed by sanitation in `process_event` and
441/// `extract_event`. This partially depends on the `processing_enabled` flag.
442fn is_duplicate(item: &Item, processing_enabled: bool) -> bool {
443    match item.ty() {
444        // These should always be removed by `extract_event`:
445        ItemType::Event => true,
446        ItemType::Transaction => true,
447        ItemType::Security => true,
448        ItemType::FormData => true,
449        ItemType::RawSecurity => true,
450        ItemType::UserReportV2 => true,
451
452        // These should be removed conditionally:
453        ItemType::UnrealReport => processing_enabled,
454
455        // These may be forwarded to upstream / store:
456        ItemType::Attachment => false,
457        ItemType::Nel => false,
458        ItemType::UserReport => false,
459
460        // Aggregate data is never considered as part of deduplication
461        ItemType::Session => false,
462        ItemType::Sessions => false,
463        ItemType::Statsd => false,
464        ItemType::MetricBuckets => false,
465        ItemType::ClientReport => false,
466        ItemType::Profile => false,
467        ItemType::ReplayEvent => false,
468        ItemType::ReplayRecording => false,
469        ItemType::ReplayVideo => false,
470        ItemType::CheckIn => false,
471        ItemType::Log => false,
472        ItemType::OtelLog => false,
473        ItemType::Span => false,
474        ItemType::OtelSpan => false,
475        ItemType::OtelTracesData => false,
476        ItemType::ProfileChunk => false,
477
478        // Without knowing more, `Unknown` items are allowed to be repeated
479        ItemType::Unknown(_) => false,
480    }
481}

483fn event_from_json_payload(
484    item: Item,
485    event_type: Option<EventType>,
486) -> Result<ExtractedEvent, ProcessingError> {
487    let mut event = Annotated::<Event>::from_json_bytes(&item.payload())
488        .map_err(ProcessingError::InvalidJson)?;
489
490    if let Some(event_value) = event.value_mut() {
491        if event_type.is_some() {
492            event_value.ty.set_value(event_type);
493        }
494    }
495
496    Ok((event, item.len()))
497}

499fn event_from_security_report(
500    item: Item,
501    meta: &RequestMeta,
502) -> Result<ExtractedEvent, ProcessingError> {
503    let len = item.len();
504    let mut event = Event::default();
505
506    let bytes = item.payload();
507    let data = &bytes;
508    let Some(report_type) =
509        SecurityReportType::from_json(data).map_err(ProcessingError::InvalidJson)?
510    else {
511        return Err(ProcessingError::InvalidSecurityType(bytes));
512    };
513
514    let (apply_result, event_type) = match report_type {
515        SecurityReportType::Csp => (Csp::apply_to_event(data, &mut event), EventType::Csp),
516        SecurityReportType::ExpectCt => (
517            ExpectCt::apply_to_event(data, &mut event),
518            EventType::ExpectCt,
519        ),
520        SecurityReportType::ExpectStaple => (
521            ExpectStaple::apply_to_event(data, &mut event),
522            EventType::ExpectStaple,
523        ),
524        SecurityReportType::Hpkp => (Hpkp::apply_to_event(data, &mut event), EventType::Hpkp),
525        SecurityReportType::Unsupported => return Err(ProcessingError::UnsupportedSecurityType),
526    };
527
528    if let Err(json_error) = apply_result {
529        // logged in extract_event
530        relay_log::configure_scope(|scope| {
531            scope.set_extra("payload", String::from_utf8_lossy(data).into());
532        });
533
534        return Err(ProcessingError::InvalidSecurityReport(json_error));
535    }
536
537    if let Some(release) = item.get_header("sentry_release").and_then(Value::as_str) {
538        event.release = Annotated::from(LenientString(release.to_owned()));
539    }
540
541    if let Some(env) = item
542        .get_header("sentry_environment")
543        .and_then(Value::as_str)
544    {
545        event.environment = Annotated::from(env.to_owned());
546    }
547
548    if let Some(origin) = meta.origin() {
549        event
550            .request
551            .get_or_insert_with(Default::default)
552            .headers
553            .get_or_insert_with(Default::default)
554            .insert("Origin".into(), Annotated::new(origin.to_string().into()));
555    }
556
557    // Explicitly set the event type. This is required so that a `Security` item can be created
558    // instead of a regular `Event` item.
559    event.ty = Annotated::new(event_type);
560
561    Ok((Annotated::new(event), len))
562}

564fn event_from_nel_item(item: Item, _meta: &RequestMeta) -> Result<ExtractedEvent, ProcessingError> {
565    let len = item.len();
566    let mut event = Event {
567        ty: Annotated::new(EventType::Nel),
568        ..Default::default()
569    };
570    let data: &[u8] = &item.payload();
571
572    // Try to get the raw network report.
573    let report = Annotated::from_json_bytes(data).map_err(NetworkReportError::InvalidJson);
574
575    match report {
576        // If the incoming payload could be converted into the raw network error, try
577        // to use it to normalize the event.
578        Ok(report) => {
579            nel::enrich_event(&mut event, report);
580        }
581        Err(err) => {
582            // logged in extract_event
583            relay_log::configure_scope(|scope| {
584                scope.set_extra("payload", String::from_utf8_lossy(data).into());
585            });
586            return Err(ProcessingError::InvalidNelReport(err));
587        }
588    }
589
590    Ok((Annotated::new(event), len))
591}

593fn extract_attached_event(
594    config: &Config,
595    item: Option<Item>,
596) -> Result<Annotated<Event>, ProcessingError> {
597    let item = match item {
598        Some(item) if !item.is_empty() => item,
599        _ => return Ok(Annotated::new(Event::default())),
600    };
601
602    // Protect against blowing up during deserialization. Attachments can have a significantly
603    // larger size than regular events and may cause significant processing delays.
604    if item.len() > config.max_event_size() {
605        return Err(ProcessingError::PayloadTooLarge(item.ty().to_owned()));
606    }
607
608    let payload = item.payload();
609    let deserializer = &mut rmp_serde::Deserializer::from_read_ref(payload.as_ref());
610    Annotated::deserialize_with_meta(deserializer).map_err(ProcessingError::InvalidMsgpack)
611}

613fn parse_msgpack_breadcrumbs(
614    config: &Config,
615    item: Option<Item>,
616) -> Result<Array<Breadcrumb>, ProcessingError> {
617    let mut breadcrumbs = Array::new();
618    let item = match item {
619        Some(item) if !item.is_empty() => item,
620        _ => return Ok(breadcrumbs),
621    };
622
623    // Validate that we do not exceed the maximum breadcrumb payload length. Breadcrumbs are
624    // truncated to a maximum of 100 in event normalization, but this is to protect us from
625    // blowing up during deserialization. As approximation, we use the maximum event payload
626    // size as bound, which is roughly in the right ballpark.
627    if item.len() > config.max_event_size() {
628        return Err(ProcessingError::PayloadTooLarge(item.ty().to_owned()));
629    }
630
631    let payload = item.payload();
632    let mut deserializer = rmp_serde::Deserializer::new(payload.as_ref());
633
634    while !deserializer.get_ref().is_empty() {
635        let breadcrumb = Annotated::deserialize_with_meta(&mut deserializer)?;
636        breadcrumbs.push(breadcrumb);
637    }
638
639    Ok(breadcrumbs)
640}

642fn event_from_attachments(
643    config: &Config,
644    event_item: Option<Item>,
645    breadcrumbs_item1: Option<Item>,
646    breadcrumbs_item2: Option<Item>,
647) -> Result<ExtractedEvent, ProcessingError> {
648    let len = event_item.as_ref().map_or(0, |item| item.len())
649        + breadcrumbs_item1.as_ref().map_or(0, |item| item.len())
650        + breadcrumbs_item2.as_ref().map_or(0, |item| item.len());
651
652    let mut event = extract_attached_event(config, event_item)?;
653    let mut breadcrumbs1 = parse_msgpack_breadcrumbs(config, breadcrumbs_item1)?;
654    let mut breadcrumbs2 = parse_msgpack_breadcrumbs(config, breadcrumbs_item2)?;
655
656    let timestamp1 = breadcrumbs1
657        .iter()
658        .rev()
659        .find_map(|breadcrumb| breadcrumb.value().and_then(|b| b.timestamp.value()));
660
661    let timestamp2 = breadcrumbs2
662        .iter()
663        .rev()
664        .find_map(|breadcrumb| breadcrumb.value().and_then(|b| b.timestamp.value()));
665
666    // Sort breadcrumbs by date. We presume that last timestamp from each row gives the
667    // relative sequence of the whole sequence, i.e., we don't need to splice the sequences
668    // to get the breadrumbs sorted.
669    if timestamp1 > timestamp2 {
670        std::mem::swap(&mut breadcrumbs1, &mut breadcrumbs2);
671    }
672
673    // Limit the total length of the breadcrumbs. We presume that if we have both
674    // breadcrumbs with items one contains the maximum number of breadcrumbs allowed.
675    let max_length = std::cmp::max(breadcrumbs1.len(), breadcrumbs2.len());
676
677    breadcrumbs1.extend(breadcrumbs2);
678
679    if breadcrumbs1.len() > max_length {
680        // Keep only the last max_length elements from the vectors
681        breadcrumbs1.drain(0..(breadcrumbs1.len() - max_length));
682    }
683
684    if !breadcrumbs1.is_empty() {
685        event.get_or_insert_with(Event::default).breadcrumbs = Annotated::new(Values {
686            values: Annotated::new(breadcrumbs1),
687            other: Object::default(),
688        });
689    }
690
691    Ok((event, len))
692}

694fn merge_formdata(target: &mut SerdeValue, item: Item) {
695    let payload = item.payload();
696    let mut aggregator = ChunkedFormDataAggregator::new();
697
698    for entry in FormDataIter::new(&payload) {
699        if entry.key() == "sentry" || entry.key().starts_with("sentry___") {
700            // Custom clients can submit longer payloads and should JSON encode event data into
701            // the optional `sentry` field or a `sentry___<namespace>` field.
702            match serde_json::from_str(entry.value()) {
703                Ok(event) => utils::merge_values(target, event),
704                Err(_) => relay_log::debug!("invalid json event payload in sentry form field"),
705            }
706        } else if let Some(index) = utils::get_sentry_chunk_index(entry.key(), "sentry__") {
707            // Electron SDK splits up long payloads into chunks starting at sentry__1 with an
708            // incrementing counter. Assemble these chunks here and then decode them below.
709            aggregator.insert(index, entry.value());
710        } else if let Some(keys) = utils::get_sentry_entry_indexes(entry.key()) {
711            // Try to parse the nested form syntax `sentry[key][key]` This is required for the
712            // Breakpad client library, which only supports string values of up to 64
713            // characters.
714            utils::update_nested_value(target, &keys, entry.value());
715        } else {
716            // Merge additional form fields from the request with `extra` data from the event
717            // payload and set defaults for processing. This is sent by clients like Breakpad or
718            // Crashpad.
719            utils::update_nested_value(target, &["extra", entry.key()], entry.value());
720        }
721    }
722
723    if !aggregator.is_empty() {
724        match serde_json::from_str(&aggregator.join()) {
725            Ok(event) => utils::merge_values(target, event),
726            Err(_) => relay_log::debug!("invalid json event payload in sentry__* form fields"),
727        }
728    }
729}

740    fn create_breadcrumbs_item(breadcrumbs: &[(Option<DateTime<Utc>>, &str)]) -> Item {
741        let mut data = Vec::new();
742
743        for (date, message) in breadcrumbs {
744            let mut breadcrumb = BTreeMap::new();
745            breadcrumb.insert("message", (*message).to_string());
746            if let Some(date) = date {
747                breadcrumb.insert("timestamp", date.to_rfc3339());
748            }
749
750            rmp_serde::encode::write(&mut data, &breadcrumb).expect("write msgpack");
751        }
752
753        let mut item = Item::new(ItemType::Attachment);
754        item.set_payload(ContentType::MsgPack, data);
755        item
756    }
