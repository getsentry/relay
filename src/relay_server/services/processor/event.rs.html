29    pub event_metrics_extracted: Option<EventMetricsExtracted>,
30    pub spans_extracted: Option<SpansExtracted>,
31}

33/// Extracts the primary event payload from an envelope.
34///
35/// The event is obtained from only one source in the following precedence:
36///  1. An explicit event item. This is also the case for JSON uploads.
37///  2. A security report item.
38///  3. Attachments `__sentry-event` and `__sentry-breadcrumb1/2`.
39///  4. A multipart form data body.
40///  5. If none match, `Annotated::empty()`.
41pub fn extract<Group: EventProcessing>(
42    managed_envelope: &mut TypedEnvelope<Group>,
43    metrics: &mut Metrics,
44    event_fully_normalized: EventFullyNormalized,
45    config: &Config,
46) -> Result<ExtractionResult, ProcessingError> {
47    let envelope = managed_envelope.envelope_mut();
48
49    // Remove all items first, and then process them. After this function returns, only
50    // attachments can remain in the envelope. The event will be added again at the end of
51    // `process_event`.
52    let event_item = envelope.take_item_by(|item| item.ty() == &ItemType::Event);
53    let transaction_item = envelope.take_item_by(|item| item.ty() == &ItemType::Transaction);
54    let security_item = envelope.take_item_by(|item| item.ty() == &ItemType::Security);
55    let raw_security_item = envelope.take_item_by(|item| item.ty() == &ItemType::RawSecurity);
56    let user_report_v2_item = envelope.take_item_by(|item| item.ty() == &ItemType::UserReportV2);
57    let form_item = envelope.take_item_by(|item| item.ty() == &ItemType::FormData);
58    let attachment_item =
59        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::EventPayload));
60    let breadcrumbs1 =
61        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::Breadcrumbs));
62    let breadcrumbs2 =
63        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::Breadcrumbs));
64
65    // Event items can never occur twice in an envelope.
66    if let Some(duplicate) =
67        envelope.get_item_by(|item| is_duplicate(item, config.processing_enabled()))
68    {
69        return Err(ProcessingError::DuplicateItem(duplicate.ty().clone()));
70    }
71
72    let skip_normalization = config.processing_enabled() && event_fully_normalized.0;
73
74    let mut event_metrics_extracted = None;
75    let mut spans_extracted = None;
76    let (event, event_len) = if let Some(item) = event_item.or(security_item) {
77        relay_log::trace!("processing json event");
78        metric!(timer(RelayTimers::EventProcessingDeserialize), {
79            let (mut annotated_event, len) = event_from_json_payload(item, None)?;
80            // Event items can never include transactions, so retain the event type and let
81            // inference deal with this during normalization.
82            if let Some(event) = annotated_event.value_mut()
83                && !skip_normalization
84            {
85                event.ty.set_value(None);
86            }
87            (annotated_event, len)
88        })
89    } else if let Some(item) = transaction_item {
90        relay_log::trace!("processing json transaction");
91
92        event_metrics_extracted = Some(EventMetricsExtracted(item.metrics_extracted()));
93        spans_extracted = Some(SpansExtracted(item.spans_extracted()));
94
95        metric!(timer(RelayTimers::EventProcessingDeserialize), {
96            // Transaction items can only contain transaction events. Force the event type to
97            // hint to normalization that we're dealing with a transaction now.
98            event_from_json_payload(item, Some(EventType::Transaction))?
99        })
100    } else if let Some(item) = user_report_v2_item {
101        relay_log::trace!("processing user_report_v2");
102        event_from_json_payload(item, Some(EventType::UserReportV2))?
103    } else if let Some(item) = raw_security_item {
104        relay_log::trace!("processing security report");
105        event_from_security_report(item, envelope.meta()).map_err(|error| {
106            if !matches!(error, ProcessingError::UnsupportedSecurityType) {
107                relay_log::debug!(
108                    error = &error as &dyn Error,
109                    "failed to extract security report"
110                );
111            }
112            error
113        })?
114    } else if attachment_item.is_some() || breadcrumbs1.is_some() || breadcrumbs2.is_some() {
115        relay_log::trace!("extracting attached event data");
116        event_from_attachments(config, attachment_item, breadcrumbs1, breadcrumbs2)?
117    } else if let Some(item) = form_item {
118        relay_log::trace!("extracting form data");
119        let len = item.len();
120
121        let mut value = SerdeValue::Object(Default::default());
122        merge_formdata(&mut value, item);
123        let event = Annotated::deserialize_with_meta(value).unwrap_or_default();
124
125        (event, len)
126    } else {
127        relay_log::trace!("no event in envelope");
128        (Annotated::empty(), 0)
129    };
130
131    metrics.bytes_ingested_event = Annotated::new(event_len as u64);
132
133    Ok(ExtractionResult {
134        event,
135        event_metrics_extracted,
136        spans_extracted,
137    })
138}

140pub fn serialize<Group: EventProcessing>(
141    managed_envelope: &mut TypedEnvelope<Group>,
142    event: &mut Annotated<Event>,
143    event_fully_normalized: EventFullyNormalized,
144    event_metrics_extracted: EventMetricsExtracted,
145    spans_extracted: SpansExtracted,
146) -> Result<(), ProcessingError> {
147    if event.is_empty() {
148        relay_log::debug!("Cannot serialize empty event");
149        return Ok(());
150    }
151
152    let data = metric!(timer(RelayTimers::EventProcessingSerialization), {
153        event.to_json().map_err(ProcessingError::SerializeFailed)?
154    });
155
156    let event_type = event_type(event).unwrap_or_default();
157    let mut event_item = Item::new(ItemType::from_event_type(event_type));
158    event_item.set_payload(ContentType::Json, data);
159
160    // TODO: The state should simply maintain & update an `ItemHeaders` object.
161    // If transaction metrics were extracted, set the corresponding item header
162    event_item.set_metrics_extracted(event_metrics_extracted.0);
163    event_item.set_spans_extracted(spans_extracted.0);
164    event_item.set_fully_normalized(event_fully_normalized.0);
165    event_item.set_span_count(event.value().and_then(|e| e.spans.value()).map(Vec::len));
166
167    managed_envelope.envelope_mut().add_item(event_item);
168
169    Ok(())
170}

172/// Computes and emits metrics for monitoring user feedback (UserReportV2) ingest
173pub fn emit_feedback_metrics(envelope: &Envelope) {
174    let mut has_feedback = false;
175    let mut num_attachments = 0;
176    for item in envelope.items() {
177        match item.ty() {
178            ItemType::UserReportV2 => has_feedback = true,
179            ItemType::Attachment => num_attachments += 1,
180            _ => (),
181        }
182    }
183    if has_feedback {
184        metric!(counter(RelayCounters::FeedbackAttachments) += num_attachments);
185    }
186}

188/// Checks for duplicate items in an envelope.
189///
190/// An item is considered duplicate if it was not removed by sanitation in `process_event` and
191/// `extract_event`. This partially depends on the `processing_enabled` flag.
192fn is_duplicate(item: &Item, processing_enabled: bool) -> bool {
193    match item.ty() {
194        // These should always be removed by `extract_event`:
195        ItemType::Event => true,
196        ItemType::Transaction => true,
197        ItemType::Security => true,
198        ItemType::FormData => true,
199        ItemType::RawSecurity => true,
200        ItemType::UserReportV2 => true,
201
202        // These should be removed conditionally:
203        ItemType::UnrealReport => processing_enabled,
204
205        // These may be forwarded to upstream / store:
206        ItemType::Attachment => false,
207        ItemType::Nel => false,
208        ItemType::UserReport => false,
209
210        // Aggregate data is never considered as part of deduplication
211        ItemType::Session => false,
212        ItemType::Sessions => false,
213        ItemType::Statsd => false,
214        ItemType::MetricBuckets => false,
215        ItemType::ClientReport => false,
216        ItemType::Profile => false,
217        ItemType::ReplayEvent => false,
218        ItemType::ReplayRecording => false,
219        ItemType::ReplayVideo => false,
220        ItemType::CheckIn => false,
221        ItemType::Log => false,
222        ItemType::TraceMetric => false,
223        ItemType::Span => false,
224        ItemType::ProfileChunk => false,
225        ItemType::Integration => false,
226
227        // Without knowing more, `Unknown` items are allowed to be repeated
228        ItemType::Unknown(_) => false,
229    }
230}

232fn event_from_json_payload(
233    item: Item,
234    event_type: Option<EventType>,
235) -> Result<ExtractedEvent, ProcessingError> {
236    let mut event = Annotated::<Event>::from_json_bytes(&item.payload())
237        .map_err(ProcessingError::InvalidJson)?;
238
239    if let Some(event_value) = event.value_mut()
240        && event_type.is_some()
241    {
242        event_value.ty.set_value(event_type);
243    }
244
245    Ok((event, item.len()))
246}

248fn event_from_security_report(
249    item: Item,
250    meta: &RequestMeta,
251) -> Result<ExtractedEvent, ProcessingError> {
252    let len = item.len();
253    let mut event = Event::default();
254
255    let bytes = item.payload();
256    let data = &bytes;
257    let Some(report_type) =
258        SecurityReportType::from_json(data).map_err(ProcessingError::InvalidJson)?
259    else {
260        return Err(ProcessingError::InvalidSecurityType(bytes));
261    };
262
263    let (apply_result, event_type) = match report_type {
264        SecurityReportType::Csp => (Csp::apply_to_event(data, &mut event), EventType::Csp),
265        SecurityReportType::ExpectCt => (
266            ExpectCt::apply_to_event(data, &mut event),
267            EventType::ExpectCt,
268        ),
269        SecurityReportType::ExpectStaple => (
270            ExpectStaple::apply_to_event(data, &mut event),
271            EventType::ExpectStaple,
272        ),
273        SecurityReportType::Hpkp => (Hpkp::apply_to_event(data, &mut event), EventType::Hpkp),
274        SecurityReportType::Unsupported => return Err(ProcessingError::UnsupportedSecurityType),
275    };
276
277    if let Err(json_error) = apply_result {
278        // logged in extract_event
279        relay_log::configure_scope(|scope| {
280            scope.set_extra("payload", String::from_utf8_lossy(data).into());
281        });
282
283        return Err(ProcessingError::InvalidSecurityReport(json_error));
284    }
285
286    if let Some(release) = item.get_header("sentry_release").and_then(Value::as_str) {
287        event.release = Annotated::from(LenientString(release.to_owned()));
288    }
289
290    if let Some(env) = item
291        .get_header("sentry_environment")
292        .and_then(Value::as_str)
293    {
294        event.environment = Annotated::from(env.to_owned());
295    }
296
297    if let Some(origin) = meta.origin() {
298        event
299            .request
300            .get_or_insert_with(Default::default)
301            .headers
302            .get_or_insert_with(Default::default)
303            .insert("Origin".into(), Annotated::new(origin.to_string().into()));
304    }
305
306    // Explicitly set the event type. This is required so that a `Security` item can be created
307    // instead of a regular `Event` item.
308    event.ty = Annotated::new(event_type);
309
310    Ok((Annotated::new(event), len))
311}

313fn extract_attached_event(
314    config: &Config,
315    item: Option<Item>,
316) -> Result<Annotated<Event>, ProcessingError> {
317    let item = match item {
318        Some(item) if !item.is_empty() => item,
319        _ => return Ok(Annotated::new(Event::default())),
320    };
321
322    // Protect against blowing up during deserialization. Attachments can have a significantly
323    // larger size than regular events and may cause significant processing delays.
324    if item.len() > config.max_event_size() {
325        return Err(ProcessingError::PayloadTooLarge(
326            item.attachment_type()
327                .map(|t| t.into())
328                .unwrap_or_else(|| item.ty().into()),
329        ));
330    }
331
332    let payload = item.payload();
333    let deserializer = &mut rmp_serde::Deserializer::from_read_ref(payload.as_ref());
334    Annotated::deserialize_with_meta(deserializer).map_err(ProcessingError::InvalidMsgpack)
335}

337fn parse_msgpack_breadcrumbs(
338    config: &Config,
339    item: Option<Item>,
340) -> Result<Array<Breadcrumb>, ProcessingError> {
341    let mut breadcrumbs = Array::new();
342    let item = match item {
343        Some(item) if !item.is_empty() => item,
344        _ => return Ok(breadcrumbs),
345    };
346
347    // Validate that we do not exceed the maximum breadcrumb payload length. Breadcrumbs are
348    // truncated to a maximum of 100 in event normalization, but this is to protect us from
349    // blowing up during deserialization. As approximation, we use the maximum event payload
350    // size as bound, which is roughly in the right ballpark.
351    if item.len() > config.max_event_size() {
352        return Err(ProcessingError::PayloadTooLarge(
353            item.attachment_type()
354                .map(|t| t.into())
355                .unwrap_or_else(|| item.ty().into()),
356        ));
357    }
358
359    let payload = item.payload();
360    let mut deserializer = rmp_serde::Deserializer::new(payload.as_ref());
361
362    while !deserializer.get_ref().is_empty() {
363        let breadcrumb = Annotated::deserialize_with_meta(&mut deserializer)?;
364        breadcrumbs.push(breadcrumb);
365    }
366
367    Ok(breadcrumbs)
368}

370fn event_from_attachments(
371    config: &Config,
372    event_item: Option<Item>,
373    breadcrumbs_item1: Option<Item>,
374    breadcrumbs_item2: Option<Item>,
375) -> Result<ExtractedEvent, ProcessingError> {
376    let len = event_item.as_ref().map_or(0, |item| item.len())
377        + breadcrumbs_item1.as_ref().map_or(0, |item| item.len())
378        + breadcrumbs_item2.as_ref().map_or(0, |item| item.len());
379
380    let mut event = extract_attached_event(config, event_item)?;
381    let mut breadcrumbs1 = parse_msgpack_breadcrumbs(config, breadcrumbs_item1)?;
382    let mut breadcrumbs2 = parse_msgpack_breadcrumbs(config, breadcrumbs_item2)?;
383
384    let timestamp1 = breadcrumbs1
385        .iter()
386        .rev()
387        .find_map(|breadcrumb| breadcrumb.value().and_then(|b| b.timestamp.value()));
388
389    let timestamp2 = breadcrumbs2
390        .iter()
391        .rev()
392        .find_map(|breadcrumb| breadcrumb.value().and_then(|b| b.timestamp.value()));
393
394    // Sort breadcrumbs by date. We presume that last timestamp from each row gives the
395    // relative sequence of the whole sequence, i.e., we don't need to splice the sequences
396    // to get the breadrumbs sorted.
397    if timestamp1 > timestamp2 {
398        std::mem::swap(&mut breadcrumbs1, &mut breadcrumbs2);
399    }
400
401    // Limit the total length of the breadcrumbs. We presume that if we have both
402    // breadcrumbs with items one contains the maximum number of breadcrumbs allowed.
403    let max_length = std::cmp::max(breadcrumbs1.len(), breadcrumbs2.len());
404
405    breadcrumbs1.extend(breadcrumbs2);
406
407    if breadcrumbs1.len() > max_length {
408        // Keep only the last max_length elements from the vectors
409        breadcrumbs1.drain(0..(breadcrumbs1.len() - max_length));
410    }
411
412    if !breadcrumbs1.is_empty() {
413        event.get_or_insert_with(Event::default).breadcrumbs = Annotated::new(Values {
414            values: Annotated::new(breadcrumbs1),
415            other: Object::default(),
416        });
417    }
418
419    Ok((event, len))
420}

422fn merge_formdata(target: &mut SerdeValue, item: Item) {
423    let payload = item.payload();
424    let mut aggregator = ChunkedFormDataAggregator::new();
425
426    for entry in FormDataIter::new(&payload) {
427        if entry.key() == "sentry" || entry.key().starts_with("sentry___") {
428            // Custom clients can submit longer payloads and should JSON encode event data into
429            // the optional `sentry` field or a `sentry___<namespace>` field.
430            match serde_json::from_str(entry.value()) {
431                Ok(event) => utils::merge_values(target, event),
432                Err(_) => relay_log::debug!("invalid json event payload in sentry form field"),
433            }
434        } else if let Some(index) = utils::get_sentry_chunk_index(entry.key(), "sentry__") {
435            // Electron SDK splits up long payloads into chunks starting at sentry__1 with an
436            // incrementing counter. Assemble these chunks here and then decode them below.
437            aggregator.insert(index, entry.value());
438        } else if let Some(keys) = utils::get_sentry_entry_indexes(entry.key()) {
439            // Try to parse the nested form syntax `sentry[key][key]` This is required for the
440            // Breakpad client library, which only supports string values of up to 64
441            // characters.
442            utils::update_nested_value(target, &keys, entry.value());
443        } else {
444            // Merge additional form fields from the request with `extra` data from the event
445            // payload and set defaults for processing. This is sent by clients like Breakpad or
446            // Crashpad.
447            utils::update_nested_value(target, &["extra", entry.key()], entry.value());
448        }
449    }
450
451    if !aggregator.is_empty() {
452        match serde_json::from_str(&aggregator.join()) {
453            Ok(event) => utils::merge_values(target, event),
454            Err(_) => relay_log::debug!("invalid json event payload in sentry__* form fields"),
455        }
456    }
457}

468    fn create_breadcrumbs_item(breadcrumbs: &[(Option<DateTime<Utc>>, &str)]) -> Item {
469        let mut data = Vec::new();
470
471        for (date, message) in breadcrumbs {
472            let mut breadcrumb = BTreeMap::new();
473            breadcrumb.insert("message", (*message).to_owned());
474            if let Some(date) = date {
475                breadcrumb.insert("timestamp", date.to_rfc3339());
476            }
477
478            rmp_serde::encode::write(&mut data, &breadcrumb).expect("write msgpack");
479        }
480
481        let mut item = Item::new(ItemType::Attachment);
482        item.set_payload(ContentType::MsgPack, data);
483        item
484    }

488            .value()

494            .value()
495            .unwrap()
496    }
