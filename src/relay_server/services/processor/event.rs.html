34    pub event_metrics_extracted: Option<EventMetricsExtracted>,
35    pub spans_extracted: Option<SpansExtracted>,
36}

38/// Extracts the primary event payload from an envelope.
39///
40/// The event is obtained from only one source in the following precedence:
41///  1. An explicit event item. This is also the case for JSON uploads.
42///  2. A security report item.
43///  3. Attachments `__sentry-event` and `__sentry-breadcrumb1/2`.
44///  4. A multipart form data body.
45///  5. If none match, `Annotated::empty()`.
46pub fn extract<Group: EventProcessing>(
47    managed_envelope: &mut TypedEnvelope<Group>,
48    metrics: &mut Metrics,
49    event_fully_normalized: EventFullyNormalized,
50    config: &Config,
51) -> Result<ExtractionResult, ProcessingError> {
52    let envelope = managed_envelope.envelope_mut();
53
54    // Remove all items first, and then process them. After this function returns, only
55    // attachments can remain in the envelope. The event will be added again at the end of
56    // `process_event`.
57    let event_item = envelope.take_item_by(|item| item.ty() == &ItemType::Event);
58    let transaction_item = envelope.take_item_by(|item| item.ty() == &ItemType::Transaction);
59    let security_item = envelope.take_item_by(|item| item.ty() == &ItemType::Security);
60    let raw_security_item = envelope.take_item_by(|item| item.ty() == &ItemType::RawSecurity);
61    let user_report_v2_item = envelope.take_item_by(|item| item.ty() == &ItemType::UserReportV2);
62    let form_item = envelope.take_item_by(|item| item.ty() == &ItemType::FormData);
63    let attachment_item =
64        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::EventPayload));
65    let breadcrumbs1 =
66        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::Breadcrumbs));
67    let breadcrumbs2 =
68        envelope.take_item_by(|item| item.attachment_type() == Some(&AttachmentType::Breadcrumbs));
69
70    // Event items can never occur twice in an envelope.
71    if let Some(duplicate) =
72        envelope.get_item_by(|item| is_duplicate(item, config.processing_enabled()))
73    {
74        return Err(ProcessingError::DuplicateItem(duplicate.ty().clone()));
75    }
76
77    let skip_normalization = config.processing_enabled() && event_fully_normalized.0;
78
79    let mut event_metrics_extracted = None;
80    let mut spans_extracted = None;
81    let (event, event_len) = if let Some(item) = event_item.or(security_item) {
82        relay_log::trace!("processing json event");
83        metric!(timer(RelayTimers::EventProcessingDeserialize), {
84            let (mut annotated_event, len) = event_from_json_payload(item, None)?;
85            // Event items can never include transactions, so retain the event type and let
86            // inference deal with this during normalization.
87            if let Some(event) = annotated_event.value_mut()
88                && !skip_normalization
89            {
90                event.ty.set_value(None);
91            }
92            (annotated_event, len)
93        })
94    } else if let Some(item) = transaction_item {
95        relay_log::trace!("processing json transaction");
96
97        event_metrics_extracted = Some(EventMetricsExtracted(item.metrics_extracted()));
98        spans_extracted = Some(SpansExtracted(item.spans_extracted()));
99
100        metric!(timer(RelayTimers::EventProcessingDeserialize), {
101            // Transaction items can only contain transaction events. Force the event type to
102            // hint to normalization that we're dealing with a transaction now.
103            event_from_json_payload(item, Some(EventType::Transaction))?
104        })
105    } else if let Some(item) = user_report_v2_item {
106        relay_log::trace!("processing user_report_v2");
107        event_from_json_payload(item, Some(EventType::UserReportV2))?
108    } else if let Some(item) = raw_security_item {
109        relay_log::trace!("processing security report");
110        event_from_security_report(item, envelope.meta()).map_err(|error| {
111            if !matches!(error, ProcessingError::UnsupportedSecurityType) {
112                relay_log::error!(
113                    error = &error as &dyn Error,
114                    "failed to extract security report"
115                );
116            }
117            error
118        })?
119    } else if attachment_item.is_some() || breadcrumbs1.is_some() || breadcrumbs2.is_some() {
120        relay_log::trace!("extracting attached event data");
121        event_from_attachments(config, attachment_item, breadcrumbs1, breadcrumbs2)?
122    } else if let Some(item) = form_item {
123        relay_log::trace!("extracting form data");
124        let len = item.len();
125
126        let mut value = SerdeValue::Object(Default::default());
127        merge_formdata(&mut value, item);
128        let event = Annotated::deserialize_with_meta(value).unwrap_or_default();
129
130        (event, len)
131    } else {
132        relay_log::trace!("no event in envelope");
133        (Annotated::empty(), 0)
134    };
135
136    metrics.bytes_ingested_event = Annotated::new(event_len as u64);
137
138    Ok(ExtractionResult {
139        event,
140        event_metrics_extracted,
141        spans_extracted,
142    })
143}

145/// Status for applying some filters that don't drop the event.
146///
147/// The enum represents either the success of running all filters and keeping
148/// the event, [`FiltersStatus::Ok`], or not running all the filters because
149/// some are unsupported, [`FiltersStatus::Unsupported`].
150///
151/// If there are unsuppported filters, Relay should forward the event upstream
152/// so that a more up-to-date Relay can apply filters appropriately. Actions
153/// that depend on the outcome of event filtering, such as metric extraction,
154/// should be postponed until a filtering decision is made.
155#[must_use]
156pub enum FiltersStatus {
157    /// All filters have been applied and the event should be kept.
158    Ok,
159    /// Some filters are not supported and were not applied.
160    ///
161    /// Relay should forward events upstream for a more up-to-date Relay to apply these filters.
162    /// Supported filters were applied and they don't reject the event.
163    Unsupported,
164}

166pub fn filter<Group: EventProcessing>(
167    managed_envelope: &mut TypedEnvelope<Group>,
168    event: &mut Annotated<Event>,
169    project_info: &ProjectInfo,
170    global_config: &GlobalConfig,
171) -> Result<FiltersStatus, ProcessingError> {
172    let event = match event.value_mut() {
173        Some(event) => event,
174        // Some events are created by processing relays (e.g. unreal), so they do not yet
175        // exist at this point in non-processing relays.
176        None => return Ok(FiltersStatus::Ok),
177    };
178
179    let client_ip = managed_envelope.envelope().meta().client_addr();
180    let filter_settings = &project_info.config.filter_settings;
181
182    metric!(timer(RelayTimers::EventProcessingFiltering), {
183        relay_filter::should_filter(event, client_ip, filter_settings, global_config.filters())
184            .map_err(|err| {
185                managed_envelope.reject(Outcome::Filtered(err.clone()));
186                ProcessingError::EventFiltered(err)
187            })
188    })?;
189
190    // Don't extract metrics if relay can't apply generic filters.  A filter
191    // applied in another up-to-date relay in chain may need to drop the event,
192    // and there should not be metrics from dropped events.
193    // In processing relays, always extract metrics to avoid losing them.
194    let supported_generic_filters = global_config.filters.is_ok()
195        && relay_filter::are_generic_filters_supported(
196            global_config.filters().map(|f| f.version),
197            project_info.config.filter_settings.generic.version,
198        );
199    if supported_generic_filters {
200        Ok(FiltersStatus::Ok)
201    } else {
202        Ok(FiltersStatus::Unsupported)
203    }
204}

206/// Apply data privacy rules to the event payload.
207///
208/// This uses both the general `datascrubbing_settings`, as well as the the PII rules.
209pub fn scrub(
210    event: &mut Annotated<Event>,
211    project_info: &ProjectInfo,
212) -> Result<(), ProcessingError> {
213    let config = &project_info.config;
214
215    if config.datascrubbing_settings.scrub_data
216        && let Some(event) = event.value_mut()
217    {
218        relay_pii::scrub_graphql(event);
219    }
220
221    metric!(timer(RelayTimers::EventProcessingPii), {
222        if let Some(ref config) = config.pii_config {
223            let mut processor = PiiProcessor::new(config.compiled());
224            processor::process_value(event, &mut processor, ProcessingState::root())?;
225        }
226        let pii_config = config
227            .datascrubbing_settings
228            .pii_config()
229            .map_err(|e| ProcessingError::PiiConfigError(e.clone()))?;
230        if let Some(config) = pii_config {
231            let mut processor = PiiProcessor::new(config.compiled());
232            processor::process_value(event, &mut processor, ProcessingState::root())?;
233        }
234    });
235
236    Ok(())
237}

239pub fn serialize<Group: EventProcessing>(
240    managed_envelope: &mut TypedEnvelope<Group>,
241    event: &mut Annotated<Event>,
242    event_fully_normalized: EventFullyNormalized,
243    event_metrics_extracted: EventMetricsExtracted,
244    spans_extracted: SpansExtracted,
245) -> Result<(), ProcessingError> {
246    if event.is_empty() {
247        relay_log::error!("Cannot serialize empty event");
248        return Ok(());
249    }
250
251    let data = metric!(timer(RelayTimers::EventProcessingSerialization), {
252        event.to_json().map_err(ProcessingError::SerializeFailed)?
253    });
254
255    let event_type = event_type(event).unwrap_or_default();
256    let mut event_item = Item::new(ItemType::from_event_type(event_type));
257    event_item.set_payload(ContentType::Json, data);
258
259    // TODO: The state should simply maintain & update an `ItemHeaders` object.
260    // If transaction metrics were extracted, set the corresponding item header
261    event_item.set_metrics_extracted(event_metrics_extracted.0);
262    event_item.set_spans_extracted(spans_extracted.0);
263    event_item.set_fully_normalized(event_fully_normalized.0);
264
265    managed_envelope.envelope_mut().add_item(event_item);
266
267    Ok(())
268}

270/// Computes and emits metrics for monitoring user feedback (UserReportV2) ingest
271pub fn emit_feedback_metrics(envelope: &Envelope) {
272    let mut has_feedback = false;
273    let mut num_attachments = 0;
274    for item in envelope.items() {
275        match item.ty() {
276            ItemType::UserReportV2 => has_feedback = true,
277            ItemType::Attachment => num_attachments += 1,
278            _ => (),
279        }
280    }
281    if has_feedback {
282        metric!(counter(RelayCounters::FeedbackAttachments) += num_attachments);
283    }

286/// Checks for duplicate items in an envelope.
287///
288/// An item is considered duplicate if it was not removed by sanitation in `process_event` and
289/// `extract_event`. This partially depends on the `processing_enabled` flag.
290fn is_duplicate(item: &Item, processing_enabled: bool) -> bool {
291    match item.ty() {
292        // These should always be removed by `extract_event`:
293        ItemType::Event => true,
294        ItemType::Transaction => true,
295        ItemType::Security => true,
296        ItemType::FormData => true,
297        ItemType::RawSecurity => true,
298        ItemType::UserReportV2 => true,
299
300        // These should be removed conditionally:
301        ItemType::UnrealReport => processing_enabled,
302
303        // These may be forwarded to upstream / store:
304        ItemType::Attachment => false,
305        ItemType::Nel => false,
306        ItemType::UserReport => false,
307
308        // Aggregate data is never considered as part of deduplication
309        ItemType::Session => false,
310        ItemType::Sessions => false,
311        ItemType::Statsd => false,
312        ItemType::MetricBuckets => false,
313        ItemType::ClientReport => false,
314        ItemType::Profile => false,
315        ItemType::ReplayEvent => false,
316        ItemType::ReplayRecording => false,
317        ItemType::ReplayVideo => false,
318        ItemType::CheckIn => false,
319        ItemType::Log => false,
320        ItemType::TraceMetric => false,
321        ItemType::Span => false,
322        ItemType::ProfileChunk => false,
323        ItemType::Integration => false,
324
325        // Without knowing more, `Unknown` items are allowed to be repeated
326        ItemType::Unknown(_) => false,
327    }
328}

330fn event_from_json_payload(
331    item: Item,
332    event_type: Option<EventType>,
333) -> Result<ExtractedEvent, ProcessingError> {
334    let mut event = Annotated::<Event>::from_json_bytes(&item.payload())
335        .map_err(ProcessingError::InvalidJson)?;
336
337    if let Some(event_value) = event.value_mut()
338        && event_type.is_some()
339    {
340        event_value.ty.set_value(event_type);
341    }
342
343    Ok((event, item.len()))
344}

346fn event_from_security_report(
347    item: Item,
348    meta: &RequestMeta,
349) -> Result<ExtractedEvent, ProcessingError> {
350    let len = item.len();
351    let mut event = Event::default();
352
353    let bytes = item.payload();
354    let data = &bytes;
355    let Some(report_type) =
356        SecurityReportType::from_json(data).map_err(ProcessingError::InvalidJson)?
357    else {
358        return Err(ProcessingError::InvalidSecurityType(bytes));
359    };
360
361    let (apply_result, event_type) = match report_type {
362        SecurityReportType::Csp => (Csp::apply_to_event(data, &mut event), EventType::Csp),
363        SecurityReportType::ExpectCt => (
364            ExpectCt::apply_to_event(data, &mut event),
365            EventType::ExpectCt,
366        ),
367        SecurityReportType::ExpectStaple => (
368            ExpectStaple::apply_to_event(data, &mut event),
369            EventType::ExpectStaple,
370        ),
371        SecurityReportType::Hpkp => (Hpkp::apply_to_event(data, &mut event), EventType::Hpkp),
372        SecurityReportType::Unsupported => return Err(ProcessingError::UnsupportedSecurityType),
373    };
374
375    if let Err(json_error) = apply_result {
376        // logged in extract_event
377        relay_log::configure_scope(|scope| {
378            scope.set_extra("payload", String::from_utf8_lossy(data).into());
379        });
380
381        return Err(ProcessingError::InvalidSecurityReport(json_error));
382    }
383
384    if let Some(release) = item.get_header("sentry_release").and_then(Value::as_str) {
385        event.release = Annotated::from(LenientString(release.to_owned()));
386    }
387
388    if let Some(env) = item
389        .get_header("sentry_environment")
390        .and_then(Value::as_str)
391    {
392        event.environment = Annotated::from(env.to_owned());
393    }
394
395    if let Some(origin) = meta.origin() {
396        event
397            .request
398            .get_or_insert_with(Default::default)
399            .headers
400            .get_or_insert_with(Default::default)
401            .insert("Origin".into(), Annotated::new(origin.to_string().into()));
402    }
403
404    // Explicitly set the event type. This is required so that a `Security` item can be created
405    // instead of a regular `Event` item.
406    event.ty = Annotated::new(event_type);
407
408    Ok((Annotated::new(event), len))
409}

411fn extract_attached_event(
412    config: &Config,
413    item: Option<Item>,
414) -> Result<Annotated<Event>, ProcessingError> {
415    let item = match item {
416        Some(item) if !item.is_empty() => item,
417        _ => return Ok(Annotated::new(Event::default())),
418    };
419
420    // Protect against blowing up during deserialization. Attachments can have a significantly
421    // larger size than regular events and may cause significant processing delays.
422    if item.len() > config.max_event_size() {
423        return Err(ProcessingError::PayloadTooLarge(
424            item.attachment_type()
425                .map(|t| t.into())
426                .unwrap_or_else(|| item.ty().into()),
427        ));
428    }
429
430    let payload = item.payload();
431    let deserializer = &mut rmp_serde::Deserializer::from_read_ref(payload.as_ref());
432    Annotated::deserialize_with_meta(deserializer).map_err(ProcessingError::InvalidMsgpack)
433}

435fn parse_msgpack_breadcrumbs(
436    config: &Config,
437    item: Option<Item>,
438) -> Result<Array<Breadcrumb>, ProcessingError> {
439    let mut breadcrumbs = Array::new();
440    let item = match item {
441        Some(item) if !item.is_empty() => item,
442        _ => return Ok(breadcrumbs),
443    };
444
445    // Validate that we do not exceed the maximum breadcrumb payload length. Breadcrumbs are
446    // truncated to a maximum of 100 in event normalization, but this is to protect us from
447    // blowing up during deserialization. As approximation, we use the maximum event payload
448    // size as bound, which is roughly in the right ballpark.
449    if item.len() > config.max_event_size() {
450        return Err(ProcessingError::PayloadTooLarge(
451            item.attachment_type()
452                .map(|t| t.into())
453                .unwrap_or_else(|| item.ty().into()),
454        ));
455    }
456
457    let payload = item.payload();
458    let mut deserializer = rmp_serde::Deserializer::new(payload.as_ref());
459
460    while !deserializer.get_ref().is_empty() {
461        let breadcrumb = Annotated::deserialize_with_meta(&mut deserializer)?;
462        breadcrumbs.push(breadcrumb);
463    }
464
465    Ok(breadcrumbs)
466}

468fn event_from_attachments(
469    config: &Config,
470    event_item: Option<Item>,
471    breadcrumbs_item1: Option<Item>,
472    breadcrumbs_item2: Option<Item>,
473) -> Result<ExtractedEvent, ProcessingError> {
474    let len = event_item.as_ref().map_or(0, |item| item.len())
475        + breadcrumbs_item1.as_ref().map_or(0, |item| item.len())
476        + breadcrumbs_item2.as_ref().map_or(0, |item| item.len());
477
478    let mut event = extract_attached_event(config, event_item)?;
479    let mut breadcrumbs1 = parse_msgpack_breadcrumbs(config, breadcrumbs_item1)?;
480    let mut breadcrumbs2 = parse_msgpack_breadcrumbs(config, breadcrumbs_item2)?;
481
482    let timestamp1 = breadcrumbs1
483        .iter()
484        .rev()
485        .find_map(|breadcrumb| breadcrumb.value().and_then(|b| b.timestamp.value()));
486
487    let timestamp2 = breadcrumbs2
488        .iter()
489        .rev()
490        .find_map(|breadcrumb| breadcrumb.value().and_then(|b| b.timestamp.value()));
491
492    // Sort breadcrumbs by date. We presume that last timestamp from each row gives the
493    // relative sequence of the whole sequence, i.e., we don't need to splice the sequences
494    // to get the breadrumbs sorted.
495    if timestamp1 > timestamp2 {
496        std::mem::swap(&mut breadcrumbs1, &mut breadcrumbs2);
497    }
498
499    // Limit the total length of the breadcrumbs. We presume that if we have both
500    // breadcrumbs with items one contains the maximum number of breadcrumbs allowed.
501    let max_length = std::cmp::max(breadcrumbs1.len(), breadcrumbs2.len());
502
503    breadcrumbs1.extend(breadcrumbs2);
504
505    if breadcrumbs1.len() > max_length {
506        // Keep only the last max_length elements from the vectors
507        breadcrumbs1.drain(0..(breadcrumbs1.len() - max_length));
508    }
509
510    if !breadcrumbs1.is_empty() {
511        event.get_or_insert_with(Event::default).breadcrumbs = Annotated::new(Values {
512            values: Annotated::new(breadcrumbs1),
513            other: Object::default(),
514        });
515    }
516
517    Ok((event, len))
518}

520fn merge_formdata(target: &mut SerdeValue, item: Item) {
521    let payload = item.payload();
522    let mut aggregator = ChunkedFormDataAggregator::new();
523
524    for entry in FormDataIter::new(&payload) {
525        if entry.key() == "sentry" || entry.key().starts_with("sentry___") {
526            // Custom clients can submit longer payloads and should JSON encode event data into
527            // the optional `sentry` field or a `sentry___<namespace>` field.
528            match serde_json::from_str(entry.value()) {
529                Ok(event) => utils::merge_values(target, event),
530                Err(_) => relay_log::debug!("invalid json event payload in sentry form field"),
531            }
532        } else if let Some(index) = utils::get_sentry_chunk_index(entry.key(), "sentry__") {
533            // Electron SDK splits up long payloads into chunks starting at sentry__1 with an
534            // incrementing counter. Assemble these chunks here and then decode them below.
535            aggregator.insert(index, entry.value());
536        } else if let Some(keys) = utils::get_sentry_entry_indexes(entry.key()) {
537            // Try to parse the nested form syntax `sentry[key][key]` This is required for the
538            // Breakpad client library, which only supports string values of up to 64
539            // characters.
540            utils::update_nested_value(target, &keys, entry.value());
541        } else {
542            // Merge additional form fields from the request with `extra` data from the event
543            // payload and set defaults for processing. This is sent by clients like Breakpad or
544            // Crashpad.
545            utils::update_nested_value(target, &["extra", entry.key()], entry.value());
546        }
547    }
548
549    if !aggregator.is_empty() {
550        match serde_json::from_str(&aggregator.join()) {
551            Ok(event) => utils::merge_values(target, event),
552            Err(_) => relay_log::debug!("invalid json event payload in sentry__* form fields"),
553        }
554    }
555}
