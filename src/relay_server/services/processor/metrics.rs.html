11/// Checks if the namespace of the passed bucket is valid.
12///
13/// This is returns `true` for most namespaces except:
14///  - [`MetricNamespace::Unsupported`]: Equal to invalid/unknown namespaces.
15///  - [`MetricNamespace::Stats`]: Metric stats are only allowed if the `source` is [`BucketSource::Internal`].
16pub fn is_valid_namespace(bucket: &Bucket, source: BucketSource) -> bool {
17    match bucket.name.namespace() {
18        MetricNamespace::Sessions => true,
19        MetricNamespace::Transactions => true,
20        MetricNamespace::Spans => true,
21        MetricNamespace::Custom => true,
22        MetricNamespace::Stats => source == BucketSource::Internal,
23        MetricNamespace::Unsupported => false,
24    }
25}

27pub fn apply_project_info(
28    mut buckets: Vec<Bucket>,
29    metric_outcomes: &MetricOutcomes,
30    project_info: &ProjectInfo,
31    scoping: Scoping,
32) -> Vec<Bucket> {
33    let mut disabled_namespace_buckets = Vec::new();
34
35    buckets = buckets
36        .into_iter()
37        .filter_map(|bucket| {
38            if !is_metric_namespace_valid(project_info, bucket.name.namespace()) {
39                relay_log::trace!(mri = &*bucket.name, "dropping metric in disabled namespace");
40                disabled_namespace_buckets.push(bucket);
41                return None;
42            };
43
44            Some(bucket)
45        })
46        .collect();
47
48    if !disabled_namespace_buckets.is_empty() {
49        metric_outcomes.track(
50            scoping,
51            &disabled_namespace_buckets,
52            Outcome::Filtered(FilterStatKey::DisabledNamespace),
53        );
54    }
55
56    buckets
57}

59fn is_metric_namespace_valid(state: &ProjectInfo, namespace: MetricNamespace) -> bool {
60    match namespace {
61        MetricNamespace::Sessions => true,
62        MetricNamespace::Transactions => true,
63        MetricNamespace::Spans => true,
64        MetricNamespace::Custom => state.has_feature(Feature::CustomMetrics),
65        MetricNamespace::Stats => true,
66        MetricNamespace::Unsupported => false,
67    }
68}
