30pub fn filter(
31    managed_envelope: &mut TypedEnvelope<SpanGroup>,
32    config: Arc<Config>,
33    project_info: Arc<ProjectInfo>,
34) {
35    let disabled = should_filter(&config, &project_info, Feature::StandaloneSpanIngestion);
36    let otel_disabled = should_filter(&config, &project_info, Feature::OtelEndpoint);
37
38    managed_envelope.retain_items(|item| {
39        if disabled && item.is_span() {
40            relay_log::debug!("dropping span because feature is disabled");
41            ItemAction::DropSilently
42        } else if otel_disabled && item.ty() == &ItemType::OtelTracesData {
43            relay_log::debug!("dropping otel trace because feature is disabled");
44            ItemAction::DropSilently
45        } else {
46            ItemAction::Keep
47        }
48    });
49}

51/// Expands V2 spans to V1 spans.
52///
53/// This expands one item (contanining multiple V2 spans) into several
54/// (containing one V1 span each).
55pub fn expand_v2_spans(
56    managed_envelope: &mut TypedEnvelope<SpanGroup>,
57) -> Result<(), ProcessingError> {
58    let span_v2_items = managed_envelope
59        .envelope_mut()
60        .take_items_by(ItemContainer::<SpanV2>::is_container);
61
62    // V2 spans must always be sent as an `ItemContainer`, currently it is not allowed to
63    // send multiple containers for V2 spans.
64    //
65    // This restriction may be lifted in the future, this is why this validation only happens
66    // when processing is enabled, allowing it to be changed easily in the future.
67    //
68    // This limit mostly exists to incentivise SDKs to batch multiple spans into a single container,
69    // technically it can be removed without issues.
70    if span_v2_items.len() > 1 {
71        return Err(ProcessingError::DuplicateItem(ItemType::Span));
72    }
73    for span_v2_item in span_v2_items {
74        let spans_v2 = match ItemContainer::parse(&span_v2_item) {
75            Ok(spans_v2) => spans_v2,
76            Err(err) => {
77                relay_log::debug!("failed to parse V2 spans: {err}");
78                track_invalid(
79                    managed_envelope,
80                    DiscardReason::InvalidSpan,
81                    span_v2_item.item_count().unwrap_or(1) as usize,
82                );
83                continue;
84            }
85        };
86
87        for span_v2 in spans_v2.into_items() {
88            let span_v1 = span_v2.map_value(relay_spans::span_v2_to_span_v1);
89            match span_v1.to_json() {
90                Ok(payload) => {
91                    let mut new_item = Item::new(ItemType::Span);
92                    new_item.set_payload(ContentType::Json, payload);
93                    managed_envelope.envelope_mut().add_item(new_item);
94                }
95                Err(err) => {
96                    relay_log::debug!("failed to serialize span: {}", err);
97                    track_invalid(managed_envelope, DiscardReason::Internal, 1);
98                }
99            }
100        }
101    }
102
103    Ok(())
104}

106pub fn convert_otel_traces_data(managed_envelope: &mut TypedEnvelope<SpanGroup>) {
107    let envelope = managed_envelope.envelope_mut();
108
109    for item in envelope.take_items_by(|item| item.ty() == &ItemType::OtelTracesData) {
110        convert_traces_data(item, managed_envelope);
111    }
112}
113
114fn convert_traces_data(item: Item, managed_envelope: &mut TypedEnvelope<SpanGroup>) {
115    let traces_data = match parse_traces_data(item) {
116        Ok(traces_data) => traces_data,
117        Err(reason) => {
118            // NOTE: logging quantity=1 is semantically wrong, but we cannot know the real quantity
119            // without parsing.
120            track_invalid(managed_envelope, reason, 1);
121            return;
122        }
123    };
124    for resource_spans in traces_data.resource_spans {
125        for scope_spans in resource_spans.scope_spans {
126            for mut span in scope_spans.spans {
127                // Denormalize instrumentation scope and resource attributes into every span.
128                if let Some(ref scope) = scope_spans.scope {
129                    if !scope.name.is_empty() {
130                        span.attributes.push(KeyValue {
131                            key: "instrumentation.name".to_owned(),
132                            value: Some(AnyValue {
133                                value: Some(Value::StringValue(scope.name.clone())),
134                            }),
135                        })
136                    }
137                    if !scope.version.is_empty() {
138                        span.attributes.push(KeyValue {
139                            key: "instrumentation.version".to_owned(),
140                            value: Some(AnyValue {
141                                value: Some(Value::StringValue(scope.version.clone())),
142                            }),
143                        })
144                    }
145                    scope.attributes.iter().for_each(|a| {
146                        span.attributes.push(KeyValue {
147                            key: format!("instrumentation.{}", a.key),
148                            value: a.value.clone(),
149                        });
150                    });
151                }
152                if let Some(ref resource) = resource_spans.resource {
153                    resource.attributes.iter().for_each(|a| {
154                        span.attributes.push(KeyValue {
155                            key: format!("resource.{}", a.key),
156                            value: a.value.clone(),
157                        });
158                    });
159                }
160
161                let Ok(payload) = serde_json::to_vec(&span) else {
162                    track_invalid(managed_envelope, DiscardReason::Internal, 1);
163                    continue;
164                };
165                let mut item = Item::new(ItemType::OtelSpan);
166                item.set_payload(ContentType::Json, payload);
167                managed_envelope.envelope_mut().add_item(item);
168            }
169        }
170    }
171    managed_envelope.update(); // update envelope summary
172}

174fn track_invalid(
175    managed_envelope: &mut TypedEnvelope<SpanGroup>,
176    reason: DiscardReason,
177    quantity: usize,
178) {
179    managed_envelope.track_outcome(Outcome::Invalid(reason), DataCategory::Span, quantity);
180    managed_envelope.track_outcome(
181        Outcome::Invalid(reason),
182        DataCategory::SpanIndexed,
183        quantity,
184    );
185}

187fn parse_traces_data(item: Item) -> Result<TracesData, DiscardReason> {
188    match item.content_type() {
189        Some(&ContentType::Json) => serde_json::from_slice(&item.payload()).map_err(|e| {
190            relay_log::debug!(
191                error = &e as &dyn std::error::Error,
192                "Failed to parse traces data as JSON"
193            );
194            DiscardReason::InvalidJson
195        }),
196        Some(&ContentType::Protobuf) => TracesData::decode(item.payload()).map_err(|e| {
197            relay_log::debug!(
198                error = &e as &dyn std::error::Error,
199                "Failed to parse traces data as protobuf"
200            );
201            DiscardReason::InvalidProtobuf
202        }),
203        _ => Err(DiscardReason::ContentType),
204    }
205}

207/// Creates a span from the transaction and applies tag extraction on it.
208///
209/// Returns `None` when [`tag_extraction::extract_span_tags`] clears the span, which it shouldn't.
210pub fn extract_transaction_span(
211    event: &Event,
212    max_tag_value_size: usize,
213    span_allowed_hosts: &[String],
214) -> Option<Span> {
215    let mut spans = [Span::from(event).into()];
216
217    tag_extraction::extract_span_tags(event, &mut spans, max_tag_value_size, span_allowed_hosts);
218    tag_extraction::extract_segment_span_tags(event, &mut spans);
219
220    spans.into_iter().next().and_then(Annotated::into_value)
221}

235    #[test]
236    fn attribute_denormalization() {
237        // Construct an OTLP trace payload with:
238        // - a resource with one attribute, containing:
239        // - an instrumentation scope with one attribute, containing:
240        // - a span with one attribute
241        let traces_data = r#"
242        {
243            "resourceSpans": [
244                {
245                    "resource": {
246                        "attributes": [
247                            {
248                                "key": "resource_key",
249                                "value": {
250                                    "stringValue": "resource_value"
251                                }
252                            }
253                        ]
254                    },
255                    "scopeSpans": [
256                        {
257                            "scope": {
258                                "name": "test_instrumentation",
259                                "version": "0.0.1",
260                                "attributes": [
261                                    {
262                                        "key": "scope_key",
263                                        "value": {
264                                            "stringValue": "scope_value"
265                                        }
266                                    }
267                                ]
268                            },
269                            "spans": [
270                                {
271                                    "attributes": [
272                                        {
273                                            "key": "span_key",
274                                            "value": {
275                                                "stringValue": "span_value"
276                                            }
277                                        }
278                                    ]
279                                }
280                            ]
281                        }
282                    ]
283                }
284            ]
285        }
286        "#;
287
288        // Build an envelope containing the OTLP trace data.
289        let bytes =
290            Bytes::from(r#"{"dsn":"https://e12d836b15bb49d7bbf99e64295d995b:@sentry.io/42"}"#);
291        let envelope = Envelope::parse_bytes(bytes).unwrap();
292        let (test_store, _) = Addr::custom();
293        let (outcome_aggregator, _) = Addr::custom();
294        let managed_envelope = ManagedEnvelope::new(envelope, outcome_aggregator, test_store);
295        let mut typed_envelope: TypedEnvelope<_> = (managed_envelope, ProcessingGroup::Span)
296            .try_into()
297            .unwrap();
298        let mut item = Item::new(ItemType::OtelTracesData);
299        item.set_payload(ContentType::Json, traces_data);
300        typed_envelope.envelope_mut().add_item(item.clone());
301
302        // Convert the OTLP trace data into `OtelSpan` item(s).
303        convert_traces_data(item, &mut typed_envelope);
304
305        // Assert that the attributes from the resource and instrumentation
306        // scope were copied.
307        let item = typed_envelope
308            .envelope()
309            .items()
310            .find(|i| *i.ty() == ItemType::OtelSpan)
311            .expect("converted span missing from envelope");
312        let attributes = serde_json::from_slice::<OtelSpan>(&item.payload())
313            .expect("unable to deserialize otel span")
314            .attributes
315            .into_iter()
316            .map(|kv| (kv.key, kv.value.unwrap()))
317            .collect::<BTreeMap<_, _>>();
318        let attribute_value = |key: &str| -> String {
319            match attributes
320                .get(key)
321                .unwrap_or_else(|| panic!("attribute {} missing", key))
322                .to_owned()
323                .value
324            {
325                Some(Value::StringValue(str)) => str,
326                _ => panic!("attribute {} not a string", key),
327            }
328        };
329        assert_eq!(
330            attribute_value("span_key"),
331            "span_value".to_owned(),
332            "original span attribute should be present"
333        );
334        assert_eq!(
335            attribute_value("instrumentation.name"),
336            "test_instrumentation".to_owned(),
337            "instrumentation name should be in attributes"
338        );
339        assert_eq!(
340            attribute_value("instrumentation.version"),
341            "0.0.1".to_owned(),
342            "instrumentation version should be in attributes"
343        );
344        assert_eq!(
345            attribute_value("resource.resource_key"),
346            "resource_value".to_owned(),
347            "resource attribute should be copied with prefix"
348        );
349        assert_eq!(
350            attribute_value("instrumentation.scope_key"),
351            "scope_value".to_owned(),
352            "instruementation scope attribute should be copied with prefix"
353        );
354    }
