18/// Limit the size of the decompressed data to prevent an invalid frame blowing up memory usage.
19const MAX_DECOMPRESSED_SIZE: usize = 100_1024;

21type Result<T> = std::result::Result<T, SwitchProcessingError>;

23/// Expands Nintendo Switch crash-reports.
24///
25/// If the envelope does NOT contain a `dying_message.dat` attachment, it doesn't do anything.
26/// If the attachment item is found and it matches the expected format, it parses the attachment
27/// and updates the envelope with the content. The processed attachment is then removed from the
28/// envelope.
29///
30/// The envelope may be dropped if it exceeds size limits after decompression. Particularly,
31/// this includes cases where a single attachment file exceeds the maximum file size. This is in
32/// line with the behavior of the envelope endpoint.
33///
34/// After this, [`crate::services::processor::EnvelopeProcessorService`] should be able to process
35/// the envelope the same way it processes any other envelopes.
36///
37/// Note: in case of an error, we don't fail but leave the envelope as is.
38pub fn expand(
39    managed_envelope: &mut TypedEnvelope<ErrorGroup>,
40) -> std::result::Result<(), ProcessingError> {
41    let envelope = managed_envelope.envelope_mut();
42
43    if let Some(item) = envelope.take_item_by(|item| {
44        item.attachment_type() == Some(AttachmentType::NintendoSwitchDyingMessage)
45    }) && let Err(e) = expand_dying_message(item.payload(), envelope)
46    {
47        // If we fail to process the dying message, we need to add back the original attachment.
48        envelope.add_item(item);
49        return Err(ProcessingError::InvalidNintendoDyingMessage(e));
50    }
51
52    Ok(())
53}

55/// Parses DyingMessage contents and updates the envelope.
56/// See dying_message.md for the documentation.
57fn expand_dying_message(mut payload: Bytes, envelope: &mut Envelope) -> Result<()> {
58    payload.advance(NNSWITCH_SENTRY_MAGIC.len());
59    let version = payload
60        .try_get_u8()
61        .map_err(|_| SwitchProcessingError::UnexpectedEof {
62            expected: "version",
63        })?;
64    match version {
65        0 => expand_dying_message_v0(payload, envelope),
66        _ => Err(SwitchProcessingError::InvalidValue(
67            "version",
68            version as usize,
69        )),
70    }
71}

73/// DyingMessage protocol v0 parser.
74fn expand_dying_message_v0(mut payload: Bytes, envelope: &mut Envelope) -> Result<()> {
75    let encoding_byte = payload
76        .try_get_u8()
77        .map_err(|_| SwitchProcessingError::UnexpectedEof {
78            expected: "encoding",
79        })?;
80    let format = (encoding_byte >> 6) & 0b0000_0011;
81    let compression = (encoding_byte >> 4) & 0b0000_0011;
82    let compression_arg = encoding_byte & 0b0000_1111;
83
84    let compressed_length =
85        payload
86            .try_get_u16()
87            .map_err(|_| SwitchProcessingError::UnexpectedEof {
88                expected: "compressed data length",
89            })?;
90    let data = decompress_data(
91        payload,
92        compressed_length as usize,
93        compression,
94        compression_arg,
95    )?;
96
97    match format {
98        0 => expand_dying_message_from_envelope_items(data, envelope),
99        _ => Err(SwitchProcessingError::InvalidValue(
100            "payload format",
101            format as usize,
102        )),
103    }
104}

106/// Merges envelope items with the ones contained in the DyingMessage
107fn expand_dying_message_from_envelope_items(data: Bytes, envelope: &mut Envelope) -> Result<()> {
108    let items =
109        Envelope::parse_items_bytes(data).map_err(SwitchProcessingError::EnvelopeParsing)?;
110    for item in items {
111        // If it's an event type, merge it with the main event one already in the envelope.
112        if item.ty() == &ItemType::Event
113            && let Some(event) = envelope.get_item_by_mut(|it| it.ty() == &ItemType::Event)
114        {
115            update_event(item, event).map_err(SwitchProcessingError::InvalidJson)?;
116            // Don't add this item as a new envelope item now that it's merged.
117            continue;
118        }
119        envelope.add_item(item);
120    }
121    Ok(())
122}
123
124fn update_event(item: Item, event: &mut Item) -> std::result::Result<(), serde_json::Error> {
125    let original_json = serde_json::from_slice::<serde_json::Value>(&event.payload())?;
126    let mut new_json = serde_json::from_slice(&item.payload())?;
127    utils::merge_values(&mut new_json, original_json);
128    let new_payload = serde_json::to_vec(&new_json)?;
129    event.set_payload(ContentType::Json, new_payload);
130    Ok(())
131}

133fn decompress_data(
134    payload: Bytes,
135    compressed_length: usize,
136    compression: u8,
137    compression_arg: u8,
138) -> Result<Bytes> {
139    if payload.len() < compressed_length {
140        return Err(SwitchProcessingError::InvalidValue(
141            "compressed data length",
142            compressed_length,
143        ));
144    }
145
146    let data = payload.slice(0..compressed_length);
147    match compression {
148        // No compression
149        0 => Ok(data),
150        // Zstandard
151        1 => decompress_data_zstd(data, compression_arg)
152            .map(Bytes::from)
153            .map_err(SwitchProcessingError::Zstandard),
154        _ => Err(SwitchProcessingError::InvalidValue(
155            "compression format",
156            compression as usize,
157        )),
158    }
159}

161fn get_zstd_dictionary(id: usize) -> Option<&'static zstd::dict::DecoderDictionary<'static>> {
162    // Inlined dictionary binary data.
163    static ZSTD_DICTIONARIES: &[&[u8]] = &[
164        // index 0 = empty dictionary (a.k.a "none")
165        b"",
166    ];
167
168    // We initialize dictionaries (from their binary representation) only once and reuse them when decompressing.
169    static ZSTD_DEC_DICTIONARIES: OnceLock<
170        [zstd::dict::DecoderDictionary; ZSTD_DICTIONARIES.len()],
171    > = OnceLock::new();
172    let dictionaries = ZSTD_DEC_DICTIONARIES.get_or_init(|| {
173        let mut dictionaries: [zstd::dict::DecoderDictionary; ZSTD_DICTIONARIES.len()] =
174            [zstd::dict::DecoderDictionary::new(ZSTD_DICTIONARIES[0])];
175        for i in 0..ZSTD_DICTIONARIES.len() {
176            dictionaries[i] = zstd::dict::DecoderDictionary::new(ZSTD_DICTIONARIES[i]);
177        }
178        dictionaries
179    });
180
181    dictionaries.get(id)
182}
183
184fn decompress_data_zstd(data: Bytes, dictionary_id: u8) -> std::io::Result<Vec<u8>> {
185    let dictionary = get_zstd_dictionary(dictionary_id as usize)
186        .ok_or(std::io::Error::other("Unknown compression dictionary"))?;
187
188    let mut decompressor = ZstdDecompressor::with_prepared_dictionary(dictionary)?;
189    decompressor.decompress(data.as_ref(), MAX_DECOMPRESSED_SIZE)
190}

203    fn create_envelope(dying_message: Bytes) -> TypedEnvelope<ErrorGroup> {
204        // Note: the attachment length specified in the "outer" envelope attachment is very important.
205        //       Otherwise parsing would fail because the inner one can contain line-breaks.
206        let envelope =
207        r#"{"event_id":"9ec79c33ec9942ab8353589fcb2e04dc","dsn":"https://e12d836b15bb49d7bbf99e64295d995b:@sentry.io/42"}
208{"type":"event"}
209{"message":"hello world","level":"error","map":{"a":"val"}}
210{"type":"attachment","filename":"dying_message.dat","length":<len>}
211"#.replace("<len>", &dying_message.len().to_string());
212
213        let envelope = ManagedEnvelope::new(
214            Envelope::parse_bytes([Bytes::from(envelope), dying_message].concat().into()).unwrap(),
215            Addr::dummy(),
216        );
217        (envelope, ProcessingGroup::Error).try_into().unwrap()
218    }

220    #[test]
221    fn test_expand_uncompressed_envelope_items() {
222        // The attachment content is as follows:
223        // - 4 bytes magic = sntr
224        // - 1 byte version = 0
225        // - 1 byte encoding = 0b0000_0000 - i.e. envelope items, uncompressed
226        // - 2 bytes data length = 98 bytes - 0x0062 in big endian representation
227        // - 98 bytes of content
228        let mut envelope = create_envelope(Bytes::from(
229            "sntr\0\0\0\x62\
230            {\"type\":\"event\"}\n\
231            {\"foo\":\"bar\",\"level\":\"info\",\"map\":{\"b\":\"c\"}}\n\
232            {\"type\":\"attachment\",\"length\":2}\n\
233            Hi\n",
234        ));
235
236        let items: Vec<_> = envelope.envelope().items().collect();
237        assert_eq!(items.len(), 2);
238        assert_eq!(items[0].ty(), &ItemType::Event);
239        assert_eq!(
240            items[0].payload(),
241            "{\"message\":\"hello world\",\"level\":\"error\",\"map\":{\"a\":\"val\"}}"
242        );
243        assert_eq!(items[1].ty(), &ItemType::Attachment);
244        assert_eq!(items[1].filename(), Some(NNSWITCH_DYING_MESSAGE_FILENAME));
245        assert_eq!(items[1].payload().len(), 106);
246
247        expand(&mut envelope).unwrap();
248
249        let items: Vec<_> = envelope.envelope().items().collect();
250        assert_eq!(items.len(), 2);
251        assert_eq!(items[0].ty(), &ItemType::Event);
252        assert_eq!(
253            items[0].payload(),
254            "{\"foo\":\"bar\",\"level\":\"info\",\"map\":{\"a\":\"val\",\"b\":\"c\"},\"message\":\"hello world\"}"
255        );
256        assert_eq!(items[1].ty(), &ItemType::Attachment);
257        assert_eq!(items[1].filename(), None);
258        assert_eq!(items[1].payload(), "Hi".as_bytes());
259    }

292    fn create_compressed_dying_message(encoding: u8) -> Vec<u8> {
293        // The attachment content is as follows:
294        // - 4 bytes magic = sntr
295        // - 1 byte version = 0
296        // - 1 byte encoding
297        // - 2 bytes data length = N bytes - in big endian representation
298        // - N bytes of compressed content (Zstandard)
299        let mut compressor = ZstdCompressor::new(3).unwrap();
300        let compressed_data = compressor
301            .compress(
302                b"\
303                {\"type\":\"event\"}\n\
304                {\"foo\":\"bar\",\"level\":\"info\",\"map\":{\"b\":\"c\"}}\n\
305                {\"type\":\"attachment\",\"length\":2}\n\
306                Hi\n\
307                ",
308            )
309            .unwrap();
310        let mut dying_message: Vec<u8> = Vec::new();
311        dying_message.write_all(b"sntr\0").unwrap();
312        dying_message.write_all(&[encoding]).unwrap();
313        dying_message
314            .write_all(&(compressed_data.len() as u16).to_be_bytes())
315            .unwrap();
316        dying_message.write_all(&compressed_data).unwrap();
317        dying_message
318    }
