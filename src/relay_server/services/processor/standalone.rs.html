5/// Processes a standalone envelope by removing unnecessary items.
6///
7/// This function removes form data items from the envelope since they are not allowed in
8/// standalone processing.
9pub fn process(managed_envelope: &mut TypedEnvelope<StandaloneGroup>) {
10    managed_envelope.retain_items(|i| match i.ty() {
11        ItemType::FormData => ItemAction::DropSilently,
12        _ => ItemAction::Keep,
13    });
14}
