9/// Bucket which parses only the minimally required information to implement [`TrackableBucket`].
10///
11/// Note: this can not be used to parse untrusted buckets, there is no normalization of data
12/// happening, e.g. a missing namespace will not turn the bucket into a custom metric bucket.
13#[derive(Deserialize)]
14pub struct MinimalTrackableBucket {
15    name: MetricName,
16    #[serde(flatten)]
17    value: MinimalValue,
18    #[serde(default)]
19    metadata: BucketMetadata,
20}
21
22impl TrackableBucket for MinimalTrackableBucket {
23    fn name(&self) -> &relay_metrics::MetricName {
24        &self.name
25    }
26
27    fn ty(&self) -> relay_metrics::MetricType {
28        self.value.ty()
29    }
30
31    fn summary(&self) -> BucketSummary {
32        let mri = match MetricResourceIdentifier::parse(self.name()) {
33            Ok(mri) => mri,
34            Err(_) => return BucketSummary::default(),
35        };
36
37        match mri.namespace {
38            MetricNamespace::Transactions => {
39                let count = match self.value {
40                    MinimalValue::Counter(c) if mri.name == "usage" => c.to_f64() as usize,
41                    _ => 0,
42                };
43                BucketSummary::Transactions(count)
44            }
45            MetricNamespace::Spans => BucketSummary::Spans(match self.value {
46                MinimalValue::Counter(c) if mri.name == "usage" => c.to_f64() as usize,
47                _ => 0,
48            }),
49            _ => BucketSummary::default(),
50        }
51    }
52
53    fn metadata(&self) -> BucketMetadata {
54        self.metadata
55    }
56}
57
58#[derive(Clone, Copy, Debug, Deserialize)]
59#[serde(tag = "type", content = "value")]
60enum MinimalValue {
61    #[serde(rename = "c")]
62    Counter(CounterType),
63    #[serde(rename = "d")]
64    Distribution(IgnoredAny),
65    #[serde(rename = "s")]
66    Set(IgnoredAny),
67    #[serde(rename = "g")]
68    Gauge(IgnoredAny),
69}
70
71impl MinimalValue {
72    fn ty(self) -> MetricType {
73        match self {
74            MinimalValue::Counter(_) => MetricType::Counter,
75            MinimalValue::Distribution(_) => MetricType::Distribution,
76            MinimalValue::Set(_) => MetricType::Set,
77            MinimalValue::Gauge(_) => MetricType::Gauge,
78        }
79    }
80}
