11/// [`MetricOutcomes`] takes care of creating the right outcomes for metrics at the end of their
12/// lifecycle.
13///
14/// It is aware of surrogate metrics like transaction- and span-duration as well as pure metrics
15/// like custom.
16#[derive(Debug, Clone)]
17pub struct MetricOutcomes {
18    outcomes: Addr<TrackOutcome>,
19}
20
21impl MetricOutcomes {
22    /// Creates a new [`MetricOutcomes`].
23    pub fn new(outcomes: Addr<TrackOutcome>) -> Self {
24        Self { outcomes }
25    }
26
27    /// Tracks an outcome for a list of buckets and generates the necessary outcomes.
28    pub fn track(&self, scoping: Scoping, buckets: &[impl TrackableBucket], outcome: Outcome) {
29        let timestamp = Utc::now();
30
31        // Accepted outcomes go through `track_accepted_outcome`, which does
32        // additional work to prevent billing double-counting.
33        if !matches!(outcome, Outcome::Accepted) {
34            let SourceQuantities {
35                transactions,
36                spans,
37                buckets,
38            } = extract_quantities(buckets);
39
40            let categories = [
41                (DataCategory::Transaction, transactions as _),
42                (DataCategory::Span, spans as _),
43                (DataCategory::MetricBucket, buckets as _),
44            ];
45
46            for (category, quantity) in categories {
47                if quantity > 0 {
48                    self.outcomes.send(TrackOutcome {
49                        timestamp,
50                        scoping,
51                        outcome: outcome.clone(),
52                        event_id: None,
53                        remote_addr: None,
54                        category,
55                        quantity,
56                    });
57                }
58            }
59        }
60    }
61
62    /// Emits accepted outcomes, for the provided list of buckets.
63    ///
64    /// Additionally, adds a marker tag `billing_outcome_emitted` to all buckets for which an
65    /// outcome has been emitted.
66    #[cfg(feature = "processing")]
67    pub fn track_accepted_outcome(&self, scoping: Scoping, buckets: &mut [Bucket]) {
68        let timestamp = Utc::now();
69        for bucket in buckets {
70            let summary = bucket.summary();
71            match summary {
72                BucketSummary::Spans {
73                    count,
74                    is_segment,
75                    was_transaction: _,
76                } => {
77                    if count == 0 {
78                        continue;
79                    }
80
81                    let categories = match is_segment {
82                        true => [DataCategory::Span, DataCategory::Transaction].as_slice(),
83                        false => [DataCategory::Span].as_slice(),
84                    };
85
86                    bucket
87                        .tags
88                        .insert("billing_outcome_emitted".to_owned(), "true".to_owned());
89
90                    for category in categories {
91                        self.outcomes.send(TrackOutcome {
92                            timestamp,
93                            scoping,
94                            outcome: Outcome::Accepted,
95                            event_id: None,
96                            remote_addr: None,
97                            category: *category,
98                            quantity: count as _,
99                        });
100                    }
101                }
102                BucketSummary::None => continue,
103            };
104        }
105    }
106}

108/// The return value of [`TrackableBucket::summary`].
109///
110/// Contains the count of total transactions or spans that went into this bucket.
111#[derive(Debug, Default, Copy, Clone, PartialEq, Eq, PartialOrd, Ord)]
112pub enum BucketSummary {
113    Spans {
114        count: usize,
115        is_segment: bool,
116        was_transaction: bool,
117    },
118    #[default]
119    None,
120}

122/// Minimum information required to track outcomes for a metric bucket.
123pub trait TrackableBucket {
124    /// Full mri of the bucket.
125    fn name(&self) -> &MetricName;
126
127    /// Extracts quota information from the metric bucket.
128    ///
129    /// If the metric was extracted from one or more transactions or spans, it returns the amount
130    /// of datapoints contained in the bucket.
131    fn summary(&self) -> BucketSummary;
132}
133
134impl<T: TrackableBucket> TrackableBucket for &T {
135    fn name(&self) -> &MetricName {
136        (**self).name()
137    }
138
139    fn summary(&self) -> BucketSummary {
140        (**self).summary()
141    }
142}
143
144impl TrackableBucket for Bucket {
145    fn name(&self) -> &MetricName {
146        &self.name
147    }
148
149    fn summary(&self) -> BucketSummary {
150        BucketView::new(self).summary()
151    }
152}
153
154impl TrackableBucket for BucketView<'_> {
155    fn name(&self) -> &MetricName {
156        self.name()
157    }
158
159    fn summary(&self) -> BucketSummary {
160        let mri = match MetricResourceIdentifier::parse(self.name()) {
161            Ok(mri) => mri,
162            Err(_) => return BucketSummary::default(),
163        };
164
165        match mri.namespace {
166            MetricNamespace::Spans => match self.value() {
167                BucketViewValue::Counter(c) if mri.name == "usage" => BucketSummary::Spans {
168                    count: c.to_f64() as usize,
169                    is_segment: self.tags().get("is_segment").is_some_and(|s| s == "true"),
170                    was_transaction: self
171                        .tags()
172                        .get("was_transaction")
173                        .is_some_and(|s| s == "true"),
174                },
175                _ => BucketSummary::Spans {
176                    count: 0,
177                    is_segment: false,
178                    was_transaction: false,
179                },
180            },
181            _ => BucketSummary::default(),
182        }
183    }
184}

186/// Extracts quota information from a list of metric buckets.
187pub fn extract_quantities<I, T>(buckets: I) -> SourceQuantities
188where
189    I: IntoIterator<Item = T>,
190    T: TrackableBucket,
191{
192    let mut quantities = SourceQuantities::default();
193
194    for bucket in buckets {
195        let namespace = bucket.name().namespace();
196        // Never count unsupported metrics, they are considered invalid.
197        // Never count outcomes, as that would create more outcomes creating a loop of outcomes.
198        if namespace != MetricNamespace::Unsupported && namespace != MetricNamespace::Outcomes {
199            quantities.buckets += 1;
200        }
201
202        // Only count metrics for outcomes, where the indexed payload no longer exists.
203        let summary = bucket.summary();
204        match summary {
205            BucketSummary::Spans {
206                count,
207                is_segment,
208                was_transaction,
209            } => {
210                quantities.spans += count;
211                if is_segment && was_transaction {
212                    quantities.transactions += count;
213                }
214            }
215            BucketSummary::None => continue,
216        };
217    }
218
219    quantities
220}
