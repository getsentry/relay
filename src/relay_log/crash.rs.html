3/// Returns `true` if the current process is the crash reporting process.
4pub fn is_crash_reporter_process() -> bool {
5    cfg!(feature = "crash-handler") && std::env::var_os(CRASH_REPORTER_PROCESS_ENV_VAR).is_some()
6}

8/// Initializes a secondary watcher process for reporting fatal crashes.
9///
10/// This creates a secondary process, spawned from the same binary, communicating
11/// with the original process via IPC. If the primary process crashes,
12/// the secondary process will capture a minidump and report it to Sentry, delaying
13/// the shutdown of the crashed process until the minidump has been uploaded.
14#[cfg(feature = "crash-handler")]
15pub fn init(config: &crate::SentryConfig, client: crate::sentry::Client) {
16    use crate::sentry::{self, protocol};
17
18    let Some(db) = config._crash_db.clone() else {
19        return;
20    };
21
22    let child = minidumper_child::MinidumperChild::new()
23        .with_server_env_var(CRASH_REPORTER_PROCESS_ENV_VAR.to_owned())
24        .with_crashes_dir(db.clone())
25        .on_process(|command| {
26            #[cfg(unix)]
27            {
28                use std::os::unix::process::CommandExt;
29                command.arg0("relay-crash");
30            }
31            // Skip arg0, the process name.
32            command.args(std::env::args().skip(1));
33        })
34        .on_minidump(move |buffer, path| {
35            crate::info!("Minidump captured ({})", path.display());
36
37            crate::with_scope(
38                |scope| {
39                    // Remove event.process because this event came from the
40                    // main app process
41                    scope.remove_extra("event.process");
42
43                    let filename = path
44                        .file_name()
45                        .map(|s| s.to_string_lossy().into_owned())
46                        .unwrap_or_else(|| "minidump.dmp".to_owned());
47
48                    scope.add_attachment(protocol::Attachment {
49                        buffer,
50                        filename,
51                        ty: Some(protocol::AttachmentType::Minidump),
52                        ..Default::default()
53                    });
54                },
55                || {
56                    sentry::capture_event(protocol::Event {
57                        level: protocol::Level::Fatal,
58                        ..Default::default()
59                    })
60                },
61            );
62
63            // We need to flush because the server will exit after this closure returns
64            client.flush(Some(std::time::Duration::from_secs(15)));
65        });
66
67    // Make sure the implementations agree on the detection.
68    debug_assert_eq!(
69        is_crash_reporter_process(),
70        child.is_crash_reporter_process()
71    );
72
73    if child.is_crash_reporter_process() {
74        // Set the `event.process` so that it's obvious when Rust events come from
75        // the crash reporter process rather than the main app process
76        sentry::configure_scope(|scope| {
77            scope.set_extra("event.process", "crash-reporter".to_owned().into());
78        });
79    } else {
80        crate::info!("Initializing crash handler in {}", db.display());
81    }
82
83    match child.spawn() {
84        Ok(handle) => {
85            crate::info!("Crash handler initialized");
86            // Dropping the handle would detach the crash handler not catching any crashes.
87            std::mem::forget(handle);
88        }
89        Err(err) if is_crash_reporter_process() => {
90            crate::warn!(
91                error = &err as &dyn std::error::Error,
92                "Failed to initialize crash reporter"
93            );
94            std::process::exit(1);
95        }
96        Err(err) => {
97            crate::warn!(
98                error = &err as &dyn std::error::Error,
99                "Failed to initialize crash reporting process"
100            );
101        }
102    }
103}
