8/// An error that occurs if a number cannot be converted into a [`DataCategory`].
9#[derive(Debug, PartialEq, thiserror::Error)]
10#[error("Unknown numeric data category {0} can not be converted into a DataCategory.")]
11pub struct UnknownDataCategory(pub u8);

13/// Classifies the type of data that is being ingested.
14#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
15#[repr(i8)]
16pub enum DataCategory {
17    /// Reserved and unused.
18    ///
19    /// SDK rate limiting behavior: ignore.
20    Default = 0,
21    /// Error events and Events with an `event_type` not explicitly listed below.
22    ///
23    /// SDK rate limiting behavior: apply to the entire envelope if it contains an item type `event`.
24    Error = 1,
25    /// Transaction events.
26    ///
27    /// SDK rate limiting behavior: apply to the entire envelope if it contains an item `transaction`.
28    Transaction = 2,
29    /// Events with an event type of `csp`, `hpkp`, `expectct` and `expectstaple`.
30    ///
31    /// SDK rate limiting behavior: ignore.
32    Security = 3,
33    /// An attachment. Quantity is the size of the attachment in bytes.
34    ///
35    /// SDK rate limiting behavior: apply to all attachments.
36    Attachment = 4,
37    /// Session updates. Quantity is the number of updates in the batch.
38    ///
39    /// SDK rate limiting behavior: apply to all sessions and session aggregates.
40    Session = 5,
41    /// Profile
42    ///
43    /// This is the category for processed profiles (all profiles, whether or not we store them).
44    ///
45    /// SDK rate limiting behavior: apply to all profiles.
46    Profile = 6,
47    /// Session Replays
48    ///
49    /// SDK rate limiting behavior: apply to all Session Replay data.
50    Replay = 7,
51    /// DEPRECATED: A transaction for which metrics were extracted.
52    ///
53    /// This category is now obsolete because the `Transaction` variant will represent
54    /// processed transactions from now on.
55    ///
56    /// SDK rate limiting behavior: ignore.
57    TransactionProcessed = 8,
58    /// Indexed transaction events.
59    ///
60    /// This is the category for transaction payloads that were accepted and stored in full. In
61    /// contrast, `transaction` only guarantees that metrics have been accepted for the transaction.
62    ///
63    /// SDK rate limiting behavior: ignore.
64    TransactionIndexed = 9,
65    /// Monitor check-ins.
66    ///
67    /// SDK rate limiting behavior: apply to items of type `check_in`.
68    Monitor = 10,
69    /// Indexed Profile
70    ///
71    /// This is the category for indexed profiles that will be stored later.
72    ///
73    /// SDK rate limiting behavior: ignore.
74    ProfileIndexed = 11,
75    /// Span
76    ///
77    /// This is the category for spans from which we extracted metrics from.
78    ///
79    /// SDK rate limiting behavior: apply to spans that are not sent in a transaction.
80    Span = 12,
81    /// Monitor Seat
82    ///
83    /// Represents a monitor job that has scheduled monitor checkins. The seats are not ingested
84    /// but we define it here to prevent clashing values since this data category enumeration
85    /// is also used outside of Relay via the Python package.
86    ///
87    /// SDK rate limiting behavior: ignore.
88    MonitorSeat = 13,
89    /// User Feedback
90    ///
91    /// Represents a User Feedback processed.
92    /// Currently standardized on name UserReportV2 to avoid clashing with the old UserReport.
93    /// TODO(jferg): Rename this to UserFeedback once old UserReport is deprecated.
94    ///
95    /// SDK rate limiting behavior: apply to items of type 'feedback'.
96    UserReportV2 = 14,
97    /// Metric buckets.
98    ///
99    /// SDK rate limiting behavior: apply to `statsd` and `metrics` items.
100    MetricBucket = 15,
101    /// SpanIndexed
102    ///
103    /// This is the category for spans we store in full.
104    ///
105    /// SDK rate limiting behavior: ignore.
106    SpanIndexed = 16,
107    /// ProfileDuration
108    ///
109    /// This data category is used to count the number of milliseconds per indexed profile chunk,
110    /// excluding UI profile chunks.
111    ///
112    /// SDK rate limiting behavior: apply to profile chunks.
113    ProfileDuration = 17,
114    /// ProfileChunk
115    ///
116    /// This is a count of profile chunks received. It will not be used for billing but will be
117    /// useful for customers to track what's being dropped.
118    ///
119    /// SDK rate limiting behavior: apply to profile chunks.
120    ProfileChunk = 18,
121    /// MetricSecond
122    ///
123    /// Reserved by billing to summarize the bucketed product of metric volume
124    /// and metric cardinality. Defined here so as not to clash with future
125    /// categories.
126    ///
127    /// SDK rate limiting behavior: ignore.
128    MetricSecond = 19,
129    /// Replay Video
130    ///
131    /// This is the data category for Session Replays produced via a video recording.
132    ///
133    /// SDK rate limiting behavior: ignore.
134    DoNotUseReplayVideo = 20,
135    /// This is the data category for Uptime monitors.
136    ///
137    /// SDK rate limiting behavior: ignore.
138    Uptime = 21,
139    /// Counts the number of individual attachments, as opposed to the number of bytes in an attachment.
140    ///
141    /// SDK rate limiting behavior: apply to attachments.
142    AttachmentItem = 22,
143    /// LogItem
144    ///
145    /// This is the category for logs for which we store the count log events for users for measuring
146    /// missing breadcrumbs, and count of logs for rate limiting purposes.
147    ///
148    /// SDK rate limiting behavior: apply to logs.
149    LogItem = 23,
150    /// LogByte
151    ///
152    /// This is the category for logs for which we store log event total bytes for users.
153    ///
154    /// SDK rate limiting behavior: apply to logs.
155    LogByte = 24,
156    /// Profile duration of a UI profile.
157    ///
158    /// This data category is used to count the number of milliseconds per indexed UI profile
159    /// chunk.
160    ///
161    /// See also: [`Self::ProfileDuration`]
162    ///
163    /// SDK rate limiting behavior: apply to profile chunks.
164    ProfileDurationUi = 25,
165    /// UI Profile Chunk.
166    ///
167    /// This data category is used to count the number of milliseconds per indexed UI profile
168    /// chunk.
169    ///
170    /// See also: [`Self::ProfileChunk`]
171    ///
172    /// SDK rate limiting behavior: apply to profile chunks.
173    ProfileChunkUi = 26,
174    /// This is the data category to count Seer Autofix run events.
175    ///
176    /// SDK rate limiting behavior: ignore.
177    SeerAutofix = 27,
178    /// This is the data category to count Seer Scanner run events.
179    ///
180    /// SDK rate limiting behavior: ignore.
181    SeerScanner = 28,
182    /// DEPRECATED: Use SeerUser instead.
183    ///
184    /// PreventUser
185    ///
186    /// This is the data category to count the number of assigned Prevent Users.
187    ///
188    /// SDK rate limiting behavior: ignore.
189    PreventUser = 29,
190    /// PreventReview
191    ///
192    /// This is the data category to count the number of Prevent review events.
193    ///
194    /// SDK rate limiting behavior: ignore.
195    PreventReview = 30,
196    /// Size analysis
197    ///
198    /// This is the data category to count the number of size analyses performed.
199    /// 'Size analysis' a static binary analysis of a preprod build artifact
200    /// (e.g. the .apk of an Android app or MacOS .app).
201    /// When enabled there will typically be one such analysis per uploaded artifact.
202    ///
203    /// SDK rate limiting behavior: ignore.
204    SizeAnalysis = 31,
205    /// InstallableBuild
206    ///
207    /// This is the data category to count the number of installable builds.
208    /// It counts the number of artifacts uploaded *not* the number of times the
209    /// artifacts are downloaded for installation.
210    /// When enabled there will typically be one 'InstallableBuild' per uploaded artifact.
211    ///
212    /// SDK rate limiting behavior: ignore.
213    InstallableBuild = 32,
214    /// TraceMetric
215    ///
216    /// This is the data category to count the number of trace metric items.
217    TraceMetric = 33,
218    /// SeerUser
219    ///
220    /// This is the data category to count the number of Seer users.
221    ///
222    /// SDK rate limiting behavior: ignore.
223    SeerUser = 34,
224    //
225    // IMPORTANT: After adding a new entry to DataCategory, go to the `relay-cabi` subfolder and run
226    // `make header` to regenerate the C-binding. This allows using the data category from Python.
227    // Rerun this step every time the **code name** of the variant is updated.
228    //
229    /// Any other data category not known by this Relay.
230    Unknown = -1,
231}

319    /// Returns true if the DataCategory refers to an error (i.e an error event).
320    pub fn is_error(self) -> bool {
321        matches!(self, Self::Error | Self::Default | Self::Security)
322    }
323
324    /// Returns the numeric value for this outcome.
325    pub fn value(self) -> Option<u8> {
326        // negative values (Internal and Unknown) cannot be sent as
327        // outcomes (internally so!)
328        (self as i8).try_into().ok()
329    }
330
331    /// Returns a dedicated category for indexing if this data can be converted to metrics.
332    ///
333    /// This returns `None` for most data categories.
334    pub fn index_category(self) -> Option<Self> {
335        match self {
336            Self::Transaction => Some(Self::TransactionIndexed),
337            Self::Span => Some(Self::SpanIndexed),
338            Self::Profile => Some(Self::ProfileIndexed),
339            _ => None,
340        }
341    }
342
343    /// Returns `true` if this data category is an indexed data category.
344    pub fn is_indexed(self) -> bool {
345        matches!(
346            self,
347            Self::TransactionIndexed | Self::SpanIndexed | Self::ProfileIndexed
348        )
349    }

360impl FromStr for DataCategory {
361    type Err = std::convert::Infallible;
362
363    fn from_str(string: &str) -> Result<Self, Self::Err> {
364        Ok(Self::from_name(string))
365    }
366}

368impl From<EventType> for DataCategory {
369    fn from(ty: EventType) -> Self {
370        match ty {
371            EventType::Default | EventType::Error | EventType::Nel => Self::Error,
372            EventType::Transaction => Self::Transaction,
373            EventType::Csp | EventType::Hpkp | EventType::ExpectCt | EventType::ExpectStaple => {
374                Self::Security
375            }
376            EventType::UserReportV2 => Self::UserReportV2,
377        }
378    }
379}

423    }
