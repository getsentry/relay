52/// Conditionally enables metrics extraction groups from the global config.
53///
54/// Depending on feature flags, groups are either enabled or not.
55/// This configuration is temporarily hard-coded here. It will later be provided by the upstream.
56pub fn add_span_metrics(project_config: &mut ProjectConfig) {
57    let config = project_config
58        .metric_extraction
59        .get_or_insert_with(MetricExtractionConfig::empty);
60
61    if !config.is_supported() || config._span_metrics_extended {
62        return;
63    }
64
65    let features = &project_config.features;
66
67    // If there are any spans in the system, extract the usage metric for them:
68    if features.produces_spans() {
69        config.metrics.push(MetricSpec {
70            category: DataCategory::Span,
71            mri: "c:spans/usage@none".into(),
72            field: None,
73            condition: None,
74            tags: vec![],
75        });
76    }
77
78    // Common span metrics is a requirement for everything else:
79    if !features.has(Feature::ExtractCommonSpanMetricsFromEvent) {
80        return;
81    }
82
83    // Enable the common modules group:
84    config
85        .global_groups
86        .entry(GroupKey::SpanMetricsCommon)
87        .or_default()
88        .is_enabled = true;
89
90    if features.has(Feature::ExtractAddonsSpanMetricsFromEvent) {
91        config
92            .global_groups
93            .entry(GroupKey::SpanMetricsAddons)
94            .or_default()
95            .is_enabled = true;
96    }
97
98    let span_metrics_tx = config
99        .global_groups
100        .entry(GroupKey::SpanMetricsTx)
101        .or_default();
102    span_metrics_tx.is_enabled = true;
103
104    config._span_metrics_extended = true;
105    if config.version == 0 {
106        config.version = MetricExtractionConfig::MAX_SUPPORTED_VERSION;
107    }
108}
