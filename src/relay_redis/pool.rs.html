43/// Manages Redis cluster connections and their lifecycle.
44///
45/// This manager handles the creation and recycling of Redis cluster connections,
46/// ensuring proper connection health through periodic PING checks. It supports both
47/// primary and replica nodes, with optional read-from-replicas functionality.
48#[derive(Debug)]
49pub struct CustomClusterManager {
50    manager: ClusterManager,
51    recycle_check_frequency: usize,
52}
53
54impl CustomClusterManager {
55    /// Creates a new cluster manager for the specified Redis nodes.
56    ///
57    /// The manager will attempt to connect to each node in the provided list and
58    /// maintain connections to the Redis cluster.
59    pub fn new<T: IntoConnectionInfo>(
60        params: Vec<T>,
61        read_from_replicas: bool,
62        recycle_check_frequency: usize,
63    ) -> RedisResult<Self> {
64        Ok(Self {
65            manager: ClusterManager::new(params, read_from_replicas)?,
66            recycle_check_frequency,
67        })
68    }
69}
70
71impl Manager for CustomClusterManager {
72    type Type = ClusterConnection;
73    type Error = RedisError;
74
75    async fn create(&self) -> Result<ClusterConnection, RedisError> {
76        self.manager.create().await
77    }
78
79    async fn recycle(
80        &self,
81        conn: &mut ClusterConnection,
82        metrics: &Metrics,
83    ) -> RecycleResult<RedisError> {
84        // If the interval has been reached, we optimistically assume the connection is active
85        // without doing an actual `PING`.
86        if metrics.recycle_count % self.recycle_check_frequency != 0 {
87            return Ok(());
88        }
89
90        self.manager.recycle(conn, metrics).await
91    }
92}
93
94impl From<Object<CustomClusterManager>> for CustomClusterConnection {
95    fn from(conn: Object<CustomClusterManager>) -> Self {
96        Self(conn)
97    }
98}
