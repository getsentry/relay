3/// Counter metrics for Relay system components.
4pub enum SystemCounters {
5    /// Number of active runtime tasks.
6    ///
7    /// Every call to [`spawn`](`crate::spawn()`) increases this counter by one,
8    /// and decrements the counter by one on termination.
9    ///
10    /// This metric is tagged with:
11    ///  - `id`: A unique identifier for the task, derived from its location in code.
12    ///  - `file`: The source filename where the task is created.
13    ///  - `line`: The source line where the task is created within the file.
14    RuntimeTaskCount,
15}
16
17impl CounterMetric for SystemCounters {
18    fn name(&self) -> &'static str {
19        match self {
20            Self::RuntimeTaskCount => "runtime.task.count",
21        }
22    }
23}

25/// Gauge metrics for Relay system components.
26pub enum SystemGauges {
27    /// A number of messages queued in a services inbound message channel.
28    ///
29    /// This metric is emitted once per second for every running service. Without backlogs, this
30    /// number should be close to `0`. If this number is monotonically increasing, the service is
31    /// not able to process the inbound message volume.
32    ///
33    /// This metric is tagged with:
34    ///  - `service`: The fully qualified type name of the service implementation.
35    ServiceBackPressure,
36}
37
38impl GaugeMetric for SystemGauges {
39    fn name(&self) -> &'static str {
40        match *self {
41            Self::ServiceBackPressure => "service.back_pressure",
42        }
43    }
44}
