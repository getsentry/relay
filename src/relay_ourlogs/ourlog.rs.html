12fn otel_value_to_log_attribute(value: OtelValue) -> Option<Attribute> {
13    match value {
14        OtelValue::BoolValue(v) => Some(Attribute::new(AttributeType::Boolean, Value::Bool(v))),
15        OtelValue::DoubleValue(v) => Some(Attribute::new(AttributeType::Double, Value::F64(v))),
16        OtelValue::IntValue(v) => Some(Attribute::new(AttributeType::Integer, Value::I64(v))),
17        OtelValue::StringValue(v) => Some(Attribute::new(AttributeType::String, Value::String(v))),
18        OtelValue::BytesValue(v) => String::from_utf8(v).map_or(None, |str| {
19            Some(Attribute::new(AttributeType::String, Value::String(str)))
20        }),
21        OtelValue::ArrayValue(_) => None,
22        OtelValue::KvlistValue(_) => None,
23    }
24}

26/// Transform an OtelLog to a Sentry log.
27pub fn otel_to_sentry_log(otel_log: OtelLog, received_at: DateTime<Utc>) -> Result<OurLog, Error> {
28    let OtelLog {
29        severity_number,
30        severity_text,
31        body,
32        attributes,
33        trace_id,
34        span_id,
35        time_unix_nano,
36        ..
37    } = otel_log;
38    let span_id = SpanId::try_from(span_id.as_slice())?;
39    let trace_id = TraceId::try_from(trace_id.as_slice())?;
40    let timestamp = Utc.timestamp_nanos(time_unix_nano as i64);
41    let body = body
42        .and_then(|v| v.value)
43        .and_then(|v| match v {
44            OtelValue::StringValue(s) => Some(s),
45            _ => None,
46        })
47        .unwrap_or_else(String::new);
48
49    let received_at_nanos = received_at
50        .timestamp_nanos_opt()
51        .unwrap_or_else(|| UnixTimestamp::now().as_nanos() as i64);
52
53    let mut attribute_data = Attributes::default();
54
55    attribute_data.insert("sentry.severity_number".to_owned(), severity_number as i64);
56    attribute_data.insert(
57        "sentry.observed_timestamp_nanos".to_owned(),
58        received_at_nanos.to_string(),
59    );
60
61    for attribute in attributes.into_iter() {
62        if let Some(value) = attribute.value.and_then(|v| v.value) {
63            let key = attribute.key;
64            if let Some(v) = otel_value_to_log_attribute(value) {
65                attribute_data.insert_raw(key, Annotated::new(v));
66            }
67        }
68    }
69
70    // Map severity_number to OurLogLevel, falling back to severity_text if it's not a number.
71    // Finally default to Info if severity_number is not in range and severity_text is not a valid
72    // log level.
73    let level = match severity_number {
74        1..=4 => OurLogLevel::Trace,
75        5..=8 => OurLogLevel::Debug,
76        9..=12 => OurLogLevel::Info,
77        13..=16 => OurLogLevel::Warn,
78        17..=20 => OurLogLevel::Error,
79        21..=24 => OurLogLevel::Fatal,
80        _ => match severity_text.as_str() {
81            "trace" => OurLogLevel::Trace,
82            "debug" => OurLogLevel::Debug,
83            "info" => OurLogLevel::Info,
84            "warn" => OurLogLevel::Warn,
85            "error" => OurLogLevel::Error,
86            "fatal" => OurLogLevel::Fatal,
87            _ => OurLogLevel::Info,
88        },
89    };
90    let ourlog = OurLog {
91        timestamp: Annotated::new(Timestamp(timestamp)),
92        trace_id: Annotated::new(trace_id),
93        span_id: Annotated::new(span_id),
94        level: Annotated::new(level),
95        attributes: Annotated::new(attribute_data),
96        body: Annotated::new(body),
97        other: Object::default(),
98    };
99
100    Ok(ourlog)
101}

103/// This fills attributes with OTel specific fields to be compatible with the OTel schema.
104pub fn ourlog_merge_otel(ourlog: &mut Annotated<OurLog>, received_at: DateTime<Utc>) {
105    let Some(ourlog_value) = ourlog.value_mut() else {
106        return;
107    };
108
109    let attributes = ourlog_value.attributes.value_mut().get_or_insert_default();
110
111    let received_at_nanos = received_at
112        .timestamp_nanos_opt()
113        .unwrap_or_else(|| UnixTimestamp::now().as_nanos() as i64);
114
115    attributes.insert_if_missing("sentry.observed_timestamp_nanos", || {
116        received_at_nanos.to_string()
117    });
118}
