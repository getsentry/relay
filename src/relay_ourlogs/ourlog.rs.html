11fn otel_value_to_log_attribute(value: OtelValue) -> Option<OurLogAttribute> {
12    match value {
13        OtelValue::BoolValue(v) => Some(OurLogAttribute::new(
14            OurLogAttributeType::Boolean,
15            Value::Bool(v),
16        )),
17        OtelValue::DoubleValue(v) => Some(OurLogAttribute::new(
18            OurLogAttributeType::Double,
19            Value::F64(v),
20        )),
21        OtelValue::IntValue(v) => Some(OurLogAttribute::new(
22            OurLogAttributeType::Integer,
23            Value::I64(v),
24        )),
25        OtelValue::StringValue(v) => Some(OurLogAttribute::new(
26            OurLogAttributeType::String,
27            Value::String(v),
28        )),
29        OtelValue::BytesValue(v) => String::from_utf8(v).map_or(None, |str| {
30            Some(OurLogAttribute::new(
31                OurLogAttributeType::String,
32                Value::String(str),
33            ))
34        }),
35        OtelValue::ArrayValue(_) => None,
36        OtelValue::KvlistValue(_) => None,
37    }
38}

40/// Transform an OtelLog to a Sentry log.
41pub fn otel_to_sentry_log(otel_log: OtelLog) -> OurLog {
42    let OtelLog {
43        severity_number,
44        severity_text,
45        body,
46        attributes,
47        trace_id,
48        span_id,
49        time_unix_nano,
50        ..
51    } = otel_log;
52
53    let span_id = SpanId(hex::encode(span_id));
54    let trace_id = TraceId(hex::encode(trace_id));
55    let nanos = time_unix_nano;
56    let timestamp = Utc.timestamp_nanos(nanos as i64);
57
58    let body = body
59        .and_then(|v| v.value)
60        .and_then(|v| match v {
61            OtelValue::StringValue(s) => Some(s),
62            _ => None,
63        })
64        .unwrap_or_else(String::new);
65
66    let mut attribute_data = Object::new();
67
68    // We ignore the passed observed time since Relay always acts as the collector in Sentry.
69    // We may change this in the future with forwarding Relays.
70    let observed_time_unix_nano = UnixTimestamp::now().as_nanos();
71
72    attribute_data.insert(
73        "sentry.severity_text".to_string(),
74        Annotated::new(OurLogAttribute::new(
75            OurLogAttributeType::String,
76            Value::String(severity_text.clone()),
77        )),
78    );
79    attribute_data.insert(
80        "sentry.severity_number".to_string(),
81        Annotated::new(OurLogAttribute::new(
82            OurLogAttributeType::Integer,
83            Value::I64(severity_number as i64),
84        )),
85    );
86    attribute_data.insert(
87        "sentry.timestamp_nanos".to_string(),
88        Annotated::new(OurLogAttribute::new(
89            OurLogAttributeType::String,
90            Value::String(time_unix_nano.to_string()),
91        )),
92    );
93    attribute_data.insert(
94        "sentry.observed_timestamp_nanos".to_string(),
95        Annotated::new(OurLogAttribute::new(
96            OurLogAttributeType::String,
97            Value::String(observed_time_unix_nano.to_string()),
98        )),
99    );
100    attribute_data.insert(
101        "sentry.trace_flags".to_string(),
102        Annotated::new(OurLogAttribute::new(
103            OurLogAttributeType::Integer,
104            Value::I64(0),
105        )),
106    );
107
108    for attribute in attributes.into_iter() {
109        if let Some(value) = attribute.value.and_then(|v| v.value) {
110            let key = attribute.key;
111            if let Some(v) = otel_value_to_log_attribute(value) {
112                attribute_data.insert(key, Annotated::new(v));
113            }
114        }
115    }
116
117    // Map severity_number to OurLogLevel, falling back to severity_text if it's not a number.
118    // Finally default to Info if severity_number is not in range and severity_text is not a valid
119    // log level.
120    let level = match severity_number {
121        1..=4 => OurLogLevel::Trace,
122        5..=8 => OurLogLevel::Debug,
123        9..=12 => OurLogLevel::Info,
124        13..=16 => OurLogLevel::Warn,
125        17..=20 => OurLogLevel::Error,
126        21..=24 => OurLogLevel::Fatal,
127        _ => match severity_text.as_str() {
128            "trace" => OurLogLevel::Trace,
129            "debug" => OurLogLevel::Debug,
130            "info" => OurLogLevel::Info,
131            "warn" => OurLogLevel::Warn,
132            "error" => OurLogLevel::Error,
133            "fatal" => OurLogLevel::Fatal,
134            _ => OurLogLevel::Info,
135        },
136    };
137
138    let mut other = Object::default();
139    other.insert(
140        "severity_text".to_string(),
141        Annotated::new(Value::String(severity_text)),
142    );
143    other.insert(
144        "severity_number".to_string(),
145        Annotated::new(Value::I64(severity_number as i64)),
146    );
147    other.insert("trace_flags".to_string(), Annotated::new(Value::I64(0)));
148    other.insert(
149        "timestamp_nanos".to_string(),
150        Annotated::new(Value::U64(otel_log.time_unix_nano)),
151    );
152    other.insert(
153        "observed_timestamp_nanos".to_string(),
154        Annotated::new(Value::U64(observed_time_unix_nano)),
155    );
156
157    OurLog {
158        timestamp: Annotated::new(Timestamp(timestamp)),
159        trace_id: Annotated::new(trace_id),
160        span_id: Annotated::new(span_id),
161        level: Annotated::new(level),
162        attributes: Annotated::new(attribute_data),
163        body: Annotated::new(body),
164        other,
165    }
166}

168/// This fills attributes with OTel specific fields to be compatible with the otel schema.
169///
170/// This also currently backfills data into deprecated fields (other) on the OurLog protocol in order to continue working with the snuba consumers.
171///
172/// This will need to transform all fields into attributes to be ported to using the generic trace items consumers once they're done.
173pub fn ourlog_merge_otel(ourlog: Annotated<OurLog>) -> Annotated<OurLog> {
174    let mut ourlog = ourlog;
175    if let Some(ourlog_value) = ourlog.value_mut() {
176        let attributes = ourlog_value.attributes.value_mut().get_or_insert_default();
177        attributes.insert(
178            "sentry.severity_number".to_string(),
179            Annotated::new(OurLogAttribute::new(
180                OurLogAttributeType::Integer,
181                Value::I64(level_to_otel_severity_number(
182                    ourlog_value.level.value().cloned(),
183                )),
184            )),
185        );
186        attributes.insert(
187            "sentry.severity_text".to_owned(),
188            Annotated::new(OurLogAttribute::new(
189                OurLogAttributeType::String,
190                Value::String(
191                    ourlog_value
192                        .level
193                        .value()
194                        .map(|level| level.to_string())
195                        .unwrap_or_else(|| "info".to_string()),
196                ),
197            )),
198        );
199
200        if let Some(value) = ourlog_value.attribute("sentry.severity_text") {
201            if let Some(s) = value.as_str() {
202                ourlog_value.other.insert(
203                    "severity_text".to_string(),
204                    Annotated::new(Value::String(s.to_string())),
205                );
206            }
207        }
208
209        if let Some(value) = ourlog_value.attribute("sentry.severity_number") {
210            if let Some(v) = value.value() {
211                ourlog_value
212                    .other
213                    .insert("severity_number".to_string(), Annotated::new(v.clone()));
214            }
215        }
216
217        if let Some(value) = ourlog_value.attribute("sentry.trace_flags") {
218            if let Some(v) = value.value() {
219                ourlog_value
220                    .other
221                    .insert("trace_flags".to_string(), Annotated::new(v.clone()));
222            }
223        }
224
225        if let Some(value) = ourlog_value.attribute("sentry.observed_timestamp_nanos") {
226            if let Some(s) = value.as_str() {
227                if let Ok(nanos) = s.parse::<u64>() {
228                    ourlog_value.other.insert(
229                        "observed_timestamp_nanos".to_string(),
230                        Annotated::new(Value::U64(nanos)),
231                    );
232                }
233            }
234        }
235
236        if let Some(value) = ourlog_value.attribute("sentry.timestamp_nanos") {
237            if let Some(s) = value.as_str() {
238                if let Ok(nanos) = s.parse::<u64>() {
239                    ourlog_value.other.insert(
240                        "timestamp_nanos".to_string(),
241                        Annotated::new(Value::U64(nanos)),
242                    );
243                }
244            }
245        }
246
247        // We ignore the passed observed time since Relay always acts as the collector in Sentry.
248        // We may change this in the future with forwarding Relays.
249        let observed_time_unix_nano = UnixTimestamp::now().as_nanos();
250        ourlog_value.other.insert(
251            "observed_timestamp_nanos".to_string(),
252            Annotated::new(Value::U64(observed_time_unix_nano)),
253        );
254
255        if let Some(timestamp_nanos) = ourlog_value
256            .timestamp
257            .value()
258            .and_then(|timestamp| timestamp.0.timestamp_nanos_opt())
259        {
260            ourlog_value.other.insert(
261                "timestamp_nanos".to_string(),
262                Annotated::new(Value::U64(timestamp_nanos as u64)),
263            );
264        }
265    }
266    ourlog
267}

269fn level_to_otel_severity_number(level: Option<OurLogLevel>) -> i64 {
270    match level {
271        Some(OurLogLevel::Trace) => 1,
272        Some(OurLogLevel::Debug) => 5,
273        Some(OurLogLevel::Info) => 9,
274        Some(OurLogLevel::Warn) => 13,
275        Some(OurLogLevel::Error) => 17,
276        Some(OurLogLevel::Fatal) => 21,
277        // 0 is the default value.
278        // https://github.com/open-telemetry/opentelemetry-collector-contrib/blob/68e1d6cd94bfca9bdf725327d4221f97ce0e0564/pkg/stanza/docs/types/severity.md
279        _ => 0,
280    }
281}
