12fn otel_value_to_log_attribute(value: OtelValue) -> Option<Attribute> {
13    match value {
14        OtelValue::BoolValue(v) => Some(Attribute::new(AttributeType::Boolean, Value::Bool(v))),
15        OtelValue::DoubleValue(v) => Some(Attribute::new(AttributeType::Double, Value::F64(v))),
16        OtelValue::IntValue(v) => Some(Attribute::new(AttributeType::Integer, Value::I64(v))),
17        OtelValue::StringValue(v) => Some(Attribute::new(AttributeType::String, Value::String(v))),
18        OtelValue::BytesValue(v) => String::from_utf8(v).map_or(None, |str| {
19            Some(Attribute::new(AttributeType::String, Value::String(str)))
20        }),
21        OtelValue::ArrayValue(_) => None,
22        OtelValue::KvlistValue(_) => None,
23    }
24}

26/// Transform an OtelLog to a Sentry log.
27pub fn otel_to_sentry_log(otel_log: OtelLog, received_at: DateTime<Utc>) -> Result<OurLog, Error> {
28    let OtelLog {
29        severity_number,
30        severity_text,
31        body,
32        attributes,
33        trace_id,
34        span_id,
35        time_unix_nano,
36        ..
37    } = otel_log;
38    let span_id = SpanId::try_from(span_id.as_slice())?;
39    let trace_id = TraceId::try_from(trace_id.as_slice())?;
40    let timestamp = Utc.timestamp_nanos(time_unix_nano as i64);
41    let body = body
42        .and_then(|v| v.value)
43        .and_then(|v| match v {
44            OtelValue::StringValue(s) => Some(s),
45            _ => None,
46        })
47        .unwrap_or_else(String::new);
48
49    let received_at_nanos = received_at
50        .timestamp_nanos_opt()
51        .unwrap_or_else(|| UnixTimestamp::now().as_nanos() as i64);
52
53    let mut attribute_data = Attributes::default();
54
55    attribute_data.insert("sentry.severity_text".to_owned(), severity_text.clone());
56    attribute_data.insert("sentry.severity_number".to_owned(), severity_number as i64);
57    attribute_data.insert(
58        "sentry.timestamp_nanos".to_owned(),
59        time_unix_nano.to_string(),
60    );
61    attribute_data.insert("sentry.timestamp_precise".to_owned(), time_unix_nano as i64);
62    attribute_data.insert(
63        "sentry.observed_timestamp_nanos".to_owned(),
64        received_at_nanos.to_string(),
65    );
66    attribute_data.insert("sentry.trace_flags".to_owned(), 0);
67
68    for attribute in attributes.into_iter() {
69        if let Some(value) = attribute.value.and_then(|v| v.value) {
70            let key = attribute.key;
71            if let Some(v) = otel_value_to_log_attribute(value) {
72                attribute_data.insert_raw(key, Annotated::new(v));
73            }
74        }
75    }
76
77    // Map severity_number to OurLogLevel, falling back to severity_text if it's not a number.
78    // Finally default to Info if severity_number is not in range and severity_text is not a valid
79    // log level.
80    let level = match severity_number {
81        1..=4 => OurLogLevel::Trace,
82        5..=8 => OurLogLevel::Debug,
83        9..=12 => OurLogLevel::Info,
84        13..=16 => OurLogLevel::Warn,
85        17..=20 => OurLogLevel::Error,
86        21..=24 => OurLogLevel::Fatal,
87        _ => match severity_text.as_str() {
88            "trace" => OurLogLevel::Trace,
89            "debug" => OurLogLevel::Debug,
90            "info" => OurLogLevel::Info,
91            "warn" => OurLogLevel::Warn,
92            "error" => OurLogLevel::Error,
93            "fatal" => OurLogLevel::Fatal,
94            _ => OurLogLevel::Info,
95        },
96    };
97    let ourlog = OurLog {
98        timestamp: Annotated::new(Timestamp(timestamp)),
99        trace_id: Annotated::new(trace_id),
100        span_id: Annotated::new(span_id),
101        level: Annotated::new(level),
102        attributes: Annotated::new(attribute_data),
103        body: Annotated::new(body),
104        other: Object::default(),
105    };
106
107    Ok(ourlog)
108}

110/// This fills attributes with OTel specific fields to be compatible with the OTel schema.
111pub fn ourlog_merge_otel(ourlog: &mut Annotated<OurLog>, received_at: DateTime<Utc>) {
112    let Some(ourlog_value) = ourlog.value_mut() else {
113        return;
114    };
115    let attributes = ourlog_value.attributes.value_mut().get_or_insert_default();
116    // We can only extract microseconds as the conversion from float to Timestamp
117    // messes up with the precision and nanoseconds are never preserved.
118    let timestamp_nanos = ourlog_value
119        .timestamp
120        .value()
121        .map(|timestamp| {
122            ((datetime_to_timestamp(timestamp.into_inner()) * 1e6).round() as i64) * 1000
123        })
124        .unwrap_or_default();
125
126    let received_at_nanos = received_at
127        .timestamp_nanos_opt()
128        .unwrap_or_else(|| UnixTimestamp::now().as_nanos() as i64);
129
130    attributes.insert(
131        "sentry.severity_text".to_owned(),
132        ourlog_value
133            .level
134            .value()
135            .map(|level| level.to_string())
136            .unwrap_or_else(|| "info".to_owned()),
137    );
138    attributes.insert(
139        "sentry.severity_number".to_owned(),
140        level_to_otel_severity_number(ourlog_value.level.value().cloned()),
141    );
142    attributes.insert(
143        "sentry.timestamp_nanos".to_owned(),
144        timestamp_nanos.to_string(),
145    );
146    attributes.insert("sentry.timestamp_precise".to_owned(), timestamp_nanos);
147    attributes.insert(
148        "sentry.observed_timestamp_nanos".to_owned(),
149        received_at_nanos.to_string(),
150    );
151    attributes.insert("sentry.trace_flags".to_owned(), 0);
152}
153
154fn level_to_otel_severity_number(level: Option<OurLogLevel>) -> i64 {
155    match level {
156        Some(OurLogLevel::Trace) => 1,
157        Some(OurLogLevel::Debug) => 5,
158        Some(OurLogLevel::Info) => 9,
159        Some(OurLogLevel::Warn) => 13,
160        Some(OurLogLevel::Error) => 17,
161        Some(OurLogLevel::Fatal) => 21,
162        // 0 is the default value.
163        // https://github.com/open-telemetry/opentelemetry-collector-contrib/blob/68e1d6cd94bfca9bdf725327d4221f97ce0e0564/pkg/stanza/docs/types/severity.md
164        _ => 0,
165    }
166}
