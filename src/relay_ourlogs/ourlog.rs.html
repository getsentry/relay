12fn otel_value_to_log_attribute(value: OtelValue) -> Option<Attribute> {
13    match value {
14        OtelValue::BoolValue(v) => Some(Attribute::new(AttributeType::Boolean, Value::Bool(v))),
15        OtelValue::DoubleValue(v) => Some(Attribute::new(AttributeType::Double, Value::F64(v))),
16        OtelValue::IntValue(v) => Some(Attribute::new(AttributeType::Integer, Value::I64(v))),
17        OtelValue::StringValue(v) => Some(Attribute::new(AttributeType::String, Value::String(v))),
18        OtelValue::BytesValue(v) => String::from_utf8(v).map_or(None, |str| {
19            Some(Attribute::new(AttributeType::String, Value::String(str)))
20        }),
21        OtelValue::ArrayValue(_) => None,
22        OtelValue::KvlistValue(_) => None,
23    }
24}

26/// Transform an OtelLog to a Sentry log.
27pub fn otel_to_sentry_log(otel_log: OtelLog, received_at: DateTime<Utc>) -> Result<OurLog, Error> {
28    let OtelLog {
29        severity_number,
30        severity_text,
31        body,
32        attributes,
33        trace_id,
34        span_id,
35        time_unix_nano,
36        ..
37    } = otel_log;
38    let span_id = SpanId::try_from(span_id.as_slice())?;
39    let trace_id = TraceId::try_from(trace_id.as_slice())?;
40    let timestamp = Utc.timestamp_nanos(time_unix_nano as i64);
41    let body = body
42        .and_then(|v| v.value)
43        .and_then(|v| match v {
44            OtelValue::StringValue(s) => Some(s),
45            _ => None,
46        })
47        .unwrap_or_else(String::new);
48
49    let received_at_nanos = received_at
50        .timestamp_nanos_opt()
51        .unwrap_or_else(|| UnixTimestamp::now().as_nanos() as i64);
52
53    let mut attribute_data = Object::new();
54
55    attribute_data.insert(
56        "sentry.severity_text".to_owned(),
57        Annotated::new(Attribute::new(
58            AttributeType::String,
59            Value::String(severity_text.clone()),
60        )),
61    );
62    attribute_data.insert(
63        "sentry.severity_number".to_owned(),
64        Annotated::new(Attribute::new(
65            AttributeType::Integer,
66            Value::I64(severity_number as i64),
67        )),
68    );
69    attribute_data.insert(
70        "sentry.timestamp_nanos".to_owned(),
71        Annotated::new(Attribute::new(
72            AttributeType::String,
73            Value::String(time_unix_nano.to_string()),
74        )),
75    );
76    attribute_data.insert(
77        "sentry.timestamp_precise".to_owned(),
78        Annotated::new(Attribute::new(
79            AttributeType::Integer,
80            Value::I64(time_unix_nano as i64),
81        )),
82    );
83    attribute_data.insert(
84        "sentry.observed_timestamp_nanos".to_owned(),
85        Annotated::new(Attribute::new(
86            AttributeType::String,
87            Value::String(received_at_nanos.to_string()),
88        )),
89    );
90    attribute_data.insert(
91        "sentry.trace_flags".to_owned(),
92        Annotated::new(Attribute::new(AttributeType::Integer, Value::I64(0))),
93    );
94    attribute_data.insert(
95        "sentry.body".to_owned(),
96        Annotated::new(Attribute::new(
97            AttributeType::String,
98            Value::String(body.clone()),
99        )),
100    );
101    attribute_data.insert(
102        "sentry.span_id".to_owned(),
103        Annotated::new(Attribute::new(
104            AttributeType::String,
105            Value::String(span_id.to_string()),
106        )),
107    );
108
109    for attribute in attributes.into_iter() {
110        if let Some(value) = attribute.value.and_then(|v| v.value) {
111            let key = attribute.key;
112            if let Some(v) = otel_value_to_log_attribute(value) {
113                attribute_data.insert(key, Annotated::new(v));
114            }
115        }
116    }
117
118    // Map severity_number to OurLogLevel, falling back to severity_text if it's not a number.
119    // Finally default to Info if severity_number is not in range and severity_text is not a valid
120    // log level.
121    let level = match severity_number {
122        1..=4 => OurLogLevel::Trace,
123        5..=8 => OurLogLevel::Debug,
124        9..=12 => OurLogLevel::Info,
125        13..=16 => OurLogLevel::Warn,
126        17..=20 => OurLogLevel::Error,
127        21..=24 => OurLogLevel::Fatal,
128        _ => match severity_text.as_str() {
129            "trace" => OurLogLevel::Trace,
130            "debug" => OurLogLevel::Debug,
131            "info" => OurLogLevel::Info,
132            "warn" => OurLogLevel::Warn,
133            "error" => OurLogLevel::Error,
134            "fatal" => OurLogLevel::Fatal,
135            _ => OurLogLevel::Info,
136        },
137    };
138    let ourlog = OurLog {
139        timestamp: Annotated::new(Timestamp(timestamp)),
140        trace_id: Annotated::new(trace_id),
141        span_id: Annotated::new(span_id),
142        level: Annotated::new(level),
143        attributes: Annotated::new(attribute_data),
144        body: Annotated::new(body),
145        other: Object::default(),
146    };
147
148    Ok(ourlog)
149}

151/// This fills attributes with OTel specific fields to be compatible with the otel schema.
152pub fn ourlog_merge_otel(ourlog: &mut Annotated<OurLog>, received_at: DateTime<Utc>) {
153    let Some(ourlog_value) = ourlog.value_mut() else {
154        return;
155    };
156    let attributes = ourlog_value.attributes.value_mut().get_or_insert_default();
157    // We can only extract microseconds as the conversion from float to Timestamp
158    // messes up with the precision and nanoseconds are never preserved.
159    let timestamp_nanos = ourlog_value
160        .timestamp
161        .value()
162        .map(|timestamp| {
163            ((datetime_to_timestamp(timestamp.into_inner()) * 1e6).round() as i64) * 1000
164        })
165        .unwrap_or_default();
166
167    let received_at_nanos = received_at
168        .timestamp_nanos_opt()
169        .unwrap_or_else(|| UnixTimestamp::now().as_nanos() as i64);
170
171    attributes.insert(
172        "sentry.severity_text".to_owned(),
173        Annotated::new(Attribute::new(
174            AttributeType::String,
175            Value::String(
176                ourlog_value
177                    .level
178                    .value()
179                    .map(|level| level.to_string())
180                    .unwrap_or_else(|| "info".to_owned()),
181            ),
182        )),
183    );
184    attributes.insert(
185        "sentry.severity_number".to_owned(),
186        Annotated::new(Attribute::new(
187            AttributeType::Integer,
188            Value::I64(level_to_otel_severity_number(
189                ourlog_value.level.value().cloned(),
190            )),
191        )),
192    );
193    attributes.insert(
194        "sentry.timestamp_nanos".to_owned(),
195        Annotated::new(Attribute::new(
196            AttributeType::String,
197            Value::String(timestamp_nanos.to_string()),
198        )),
199    );
200    attributes.insert(
201        "sentry.timestamp_precise".to_owned(),
202        Annotated::new(Attribute::new(
203            AttributeType::Integer,
204            Value::I64(timestamp_nanos),
205        )),
206    );
207    attributes.insert(
208        "sentry.observed_timestamp_nanos".to_owned(),
209        Annotated::new(Attribute::new(
210            AttributeType::String,
211            Value::String(received_at_nanos.to_string()),
212        )),
213    );
214    attributes.insert(
215        "sentry.trace_flags".to_owned(),
216        Annotated::new(Attribute::new(AttributeType::Integer, Value::I64(0))),
217    );
218    attributes.insert(
219        "sentry.body".to_owned(),
220        Annotated::new(Attribute::new(
221            AttributeType::String,
222            Value::String(ourlog_value.body.value().cloned().unwrap_or_default()),
223        )),
224    );
225
226    if let Some(span_id) = ourlog_value.span_id.value() {
227        attributes.insert(
228            "sentry.span_id".to_owned(),
229            Annotated::new(Attribute::new(
230                AttributeType::String,
231                Value::String(span_id.to_string()),
232            )),
233        );
234    }
235}

237fn level_to_otel_severity_number(level: Option<OurLogLevel>) -> i64 {
238    match level {
239        Some(OurLogLevel::Trace) => 1,
240        Some(OurLogLevel::Debug) => 5,
241        Some(OurLogLevel::Info) => 9,
242        Some(OurLogLevel::Warn) => 13,
243        Some(OurLogLevel::Error) => 17,
244        Some(OurLogLevel::Fatal) => 21,
245        // 0 is the default value.
246        // https://github.com/open-telemetry/opentelemetry-collector-contrib/blob/68e1d6cd94bfca9bdf725327d4221f97ce0e0564/pkg/stanza/docs/types/severity.md
247        _ => 0,
248    }
249}
