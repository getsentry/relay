11fn otel_value_to_log_attribute(value: OtelValue) -> Option<Attribute> {
12    match value {
13        OtelValue::BoolValue(v) => Some(Attribute::new(AttributeType::Boolean, Value::Bool(v))),
14        OtelValue::DoubleValue(v) => Some(Attribute::new(AttributeType::Double, Value::F64(v))),
15        OtelValue::IntValue(v) => Some(Attribute::new(AttributeType::Integer, Value::I64(v))),
16        OtelValue::StringValue(v) => Some(Attribute::new(AttributeType::String, Value::String(v))),
17        OtelValue::BytesValue(v) => String::from_utf8(v).map_or(None, |str| {
18            Some(Attribute::new(AttributeType::String, Value::String(str)))
19        }),
20        OtelValue::ArrayValue(_) => None,
21        OtelValue::KvlistValue(_) => None,
22    }
23}

25/// Transform an OtelLog to a Sentry log.
26pub fn otel_to_sentry_log(otel_log: OtelLog) -> Result<OurLog, Error> {
27    let OtelLog {
28        severity_number,
29        severity_text,
30        body,
31        attributes,
32        trace_id,
33        span_id,
34        time_unix_nano,
35        ..
36    } = otel_log;
37    let span_id = SpanId::try_from(span_id.as_slice())?;
38    let trace_id = TraceId::try_from(trace_id.as_slice())?;
39    let timestamp = Utc.timestamp_nanos(time_unix_nano as i64);
40    let body = body
41        .and_then(|v| v.value)
42        .and_then(|v| match v {
43            OtelValue::StringValue(s) => Some(s),
44            _ => None,
45        })
46        .unwrap_or_else(String::new);
47
48    let mut attribute_data = Attributes::default();
49    attribute_data.insert("sentry.severity_number".to_owned(), severity_number as i64);
50
51    for attribute in attributes.into_iter() {
52        if let Some(value) = attribute.value.and_then(|v| v.value) {
53            let key = attribute.key;
54            if let Some(v) = otel_value_to_log_attribute(value) {
55                attribute_data.insert_raw(key, Annotated::new(v));
56            }
57        }
58    }
59
60    // Map severity_number to OurLogLevel, falling back to severity_text if it's not a number.
61    // Finally default to Info if severity_number is not in range and severity_text is not a valid
62    // log level.
63    let level = match severity_number {
64        1..=4 => OurLogLevel::Trace,
65        5..=8 => OurLogLevel::Debug,
66        9..=12 => OurLogLevel::Info,
67        13..=16 => OurLogLevel::Warn,
68        17..=20 => OurLogLevel::Error,
69        21..=24 => OurLogLevel::Fatal,
70        _ => match severity_text.as_str() {
71            "trace" => OurLogLevel::Trace,
72            "debug" => OurLogLevel::Debug,
73            "info" => OurLogLevel::Info,
74            "warn" => OurLogLevel::Warn,
75            "error" => OurLogLevel::Error,
76            "fatal" => OurLogLevel::Fatal,
77            _ => OurLogLevel::Info,
78        },
79    };
80    let ourlog = OurLog {
81        timestamp: Annotated::new(Timestamp(timestamp)),
82        trace_id: Annotated::new(trace_id),
83        span_id: Annotated::new(span_id),
84        level: Annotated::new(level),
85        attributes: Annotated::new(attribute_data),
86        body: Annotated::new(body),
87        other: Object::default(),
88    };
89
90    Ok(ourlog)
91}

99    fn parse_otel_log() {

102            "timeUnixNano": "1544712660300000000",
103            "observedTimeUnixNano": "1544712660300000000",
104            "severityNumber": 10,
105            "severityText": "Information",
106            "traceId": "5B8EFFF798038103D269B633813FC60C",
107            "spanId": "EEE19B7EC3C1B174",
108            "body": {
109                "stringValue": "Example log record"
110            },
111            "attributes": [
112                {
113                    "key": "string.attribute",
114                    "value": {
115                        "stringValue": "some string"
116                    }
117                },
118                {
119                    "key": "boolean.attribute",
120                    "value": {
121                        "boolValue": true
122                    }
123                },
124                {
125                    "key": "int.attribute",
126                    "value": {
127                        "intValue": "10"
128                    }
129                },
130                {
131                    "key": "double.attribute",
132                    "value": {
133                        "doubleValue": 637.704
134                    }
135                },
136                {
137                    "key": "array.attribute",
138                    "value": {
139                        "arrayValue": {
140                            "values": [
141                                {
142                                    "stringValue": "many"
143                                },
144                                {
145                                    "stringValue": "values"
146                                }
147                            ]
148                        }
149                    }
150                },
151                {
152                    "key": "map.attribute",
153                    "value": {
154                        "kvlistValue": {
155                            "values": [
156                                {
157                                    "key": "some.map.key",
158                                    "value": {
159                                        "stringValue": "some value"
160                                    }
161                                }
162                            ]
163                        }
164                    }
165                }
166            ]
167        }"#;
168
169        let otel_log: OtelLog = serde_json::from_str(json).unwrap();
170        let our_log: OurLog = otel_to_sentry_log(otel_log).unwrap();
171        let annotated_log: Annotated<OurLog> = Annotated::new(our_log);
172        assert_eq!(
173            get_path!(annotated_log.body),
174            Some(&Annotated::new("Example log record".into()))
175        );
176    }
