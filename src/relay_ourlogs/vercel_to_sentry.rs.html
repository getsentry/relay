11/// Vercel log structure matching their [schema](https://vercel.com/docs/drains/reference/logs).
12#[derive(Debug, Clone, Deserialize)]
13#[serde(rename_all = "camelCase")]
14pub struct VercelLog {
15    /// Unique identifier for the log entry.
16    pub id: String,
17    /// Identifier for the Vercel deployment.
18    pub deployment_id: String,
19    /// Origin of the log.
20    pub source: String,
21    /// Deployment unique URL hostname.
22    pub host: String,
23    /// Unix timestamp in milliseconds.
24    pub timestamp: i64,
25    /// Identifier for the Vercel project.
26    pub project_id: String,
27    /// Log severity level.
28    pub level: VercelLogLevel,
29    /// Log message content (may be truncated if over 256 KB).
30    pub message: Option<String>,
31    /// Identifier for the Vercel build. Only present on build logs.
32    pub build_id: Option<String>,
33    /// Entrypoint for the request.
34    pub entrypoint: Option<String>,
35    /// Origin of the external content. Only on external logs.
36    pub destination: Option<String>,
37    /// Function or dynamic path of the request.
38    pub path: Option<String>,
39    /// Log output type.
40    #[serde(rename = "type")]
41    pub ty: Option<String>,
42    /// HTTP status code of the request.
43    pub status_code: Option<i64>,
44    /// Identifier of the request.
45    pub request_id: Option<String>,
46    /// Deployment environment. One of production or preview.
47    pub environment: Option<String>,
48    /// Git branch name.
49    pub branch: Option<String>,
50    /// JA3 fingerprint digest.
51    pub ja3_digest: Option<String>,
52    /// JA4 fingerprint digest.
53    pub ja4_digest: Option<String>,
54    /// Type of edge runtime. One of edge-function or middleware
55    pub edge_type: Option<String>,
56    /// Name of the Vercel project.
57    pub project_name: Option<String>,
58    /// Region where the request is executed.
59    pub execution_region: Option<String>,
60    /// Trace identifier for distributed tracing.
61    #[serde(deserialize_with = "deserialize_trace_id", flatten)]
62    pub trace_id: Option<String>,
63    /// Span identifier for distributed tracing.
64    #[serde(deserialize_with = "deserialize_span_id", flatten)]
65    pub span_id: Option<String>,
66    /// Proxy information for requests. (Optional)
67    pub proxy: Option<VercelProxy>,
68}

70/// Macro to create a deserializer that accepts two field name variants, preferring the first.
71macro_rules! define_fallback_field_deserializer {
72    ($fn_name:ident, $primary:literal, $fallback:literal) => {
73        fn $fn_name<'d, D: Deserializer<'d>>(d: D) -> Result<Option<String>, D::Error> {
74            #[derive(Deserialize)]
75            struct Helper {
76                #[serde(rename = $primary)]
77                primary_field: Option<String>,
78                #[serde(rename = $fallback)]
79                fallback_field: Option<String>,
80            }
81
82            let helper = Helper::deserialize(d)?;
83            Ok(helper.primary_field.or(helper.fallback_field))
84        }
85    };
86}

88define_fallback_field_deserializer!(deserialize_trace_id, "traceId", "trace.id");
89define_fallback_field_deserializer!(deserialize_span_id, "spanId", "span.id");

91/// Log Level for Vercel Logs
92#[derive(Debug, Clone, Deserialize)]
93#[serde(rename_all = "lowercase")]
94pub enum VercelLogLevel {
95    /// Info level logs.
96    Info,
97    /// Warning level logs.
98    Warning,
99    /// Error level logs.
100    Error,
101    /// Fatal level logs.
102    Fatal,
103    /// Unknown status, for forward compatibility.
104    Unknown(String),
105}

107/// Vercel proxy information for requests.
108#[derive(Debug, Clone, Deserialize)]
109#[serde(rename_all = "camelCase")]
110pub struct VercelProxy {
111    /// Unix timestamp when the proxy request was made.
112    pub timestamp: i64,
113    /// HTTP method of the request.
114    pub method: String,
115    /// Hostname of the request.
116    pub host: String,
117    /// Request path with query parameters.
118    pub path: String,
119    /// User agent strings of the request.
120    pub user_agent: Vec<String>,
121    /// Region where the request is processed.
122    pub region: String,
123    /// Referer of the request.
124    pub referer: Option<String>,
125    /// HTTP status code of the proxy request.
126    pub status_code: Option<i64>,
127    /// Client IP address.
128    pub client_ip: Option<String>,
129    /// Protocol of the request.
130    pub scheme: Option<String>,
131    /// Size of the response in bytes.
132    pub response_byte_size: Option<i64>,
133    /// Original request ID when request is served from cache.
134    pub cache_id: Option<String>,
135    /// How the request was served based on its path and project configuration.
136    pub path_type: Option<String>,
137    /// Variant of the path type.
138    pub path_type_variant: Option<String>,
139    /// Vercel-specific identifier.
140    pub vercel_id: Option<String>,
141    /// Cache status sent to the browser.
142    pub vercel_cache: Option<String>,
143    /// Region where lambda function executed.
144    pub lambda_region: Option<String>,
145    /// Action taken by firewall rules.
146    pub waf_action: Option<String>,
147    /// ID of the firewall rule that matched.
148    pub waf_rule_id: Option<String>,
149}

151/// Maps Vercel log level to Sentry log level.
152fn map_vercel_level_to_sentry(level: VercelLogLevel) -> OurLogLevel {
153    match level {
154        VercelLogLevel::Info => OurLogLevel::Info,
155        VercelLogLevel::Warning => OurLogLevel::Warn,
156        VercelLogLevel::Error => OurLogLevel::Error,
157        VercelLogLevel::Fatal => OurLogLevel::Fatal,
158        VercelLogLevel::Unknown(_) => OurLogLevel::Info,
159    }
160}

162fn get_trace_id(trace_id: Option<&String>, request_id: Option<&String>) -> Annotated<TraceId> {
163    let id_str = trace_id.or(request_id).map_or("", |v| v);
164    TraceId::try_from_str_or_random(id_str)
165}
166
167fn get_span_id(span_id: Option<String>) -> Annotated<SpanId> {
168    match span_id {
169        Some(s) if !s.is_empty() => {
170            SpanId::from_str(&s).map_or_else(|err| Annotated::from_error(err, None), Annotated::new)
171        }
172        _ => Annotated::empty(),
173    }
174}

176/// Transforms a Vercel log record to a Sentry log.
177pub fn vercel_log_to_sentry_log(vercel_log: VercelLog) -> OurLog {
178    let VercelLog {
179        id,
180        deployment_id,
181        source,
182        host,
183        timestamp,
184        project_id,
185        level,
186        message,
187        build_id,
188        entrypoint,
189        destination,
190        path,
191        ty,
192        status_code,
193        request_id,
194        environment,
195        branch,
196        ja3_digest,
197        ja4_digest,
198        edge_type,
199        project_name,
200        execution_region,
201        trace_id,
202        span_id,
203        proxy,
204    } = vercel_log;
205    let trace_id = get_trace_id(trace_id.as_ref(), request_id.as_ref());
206
207    let mut attributes: Attributes = Attributes::default();
208
209    macro_rules! add_optional_attribute {
210        ($name:expr, $value:expr) => {{
211            if let Some(value) = $value {
212                attributes.insert($name.to_owned(), value);
213            }
214        }};
215    }
216
217    macro_rules! add_attribute {
218        ($name:expr, $value:expr) => {{
219            let val = $value;
220            attributes.insert($name.to_owned(), val);
221        }};
222    }
223
224    add_attribute!(ORIGIN, "auto.log_drain.vercel".to_owned());
225    add_attribute!("vercel.id", id);
226    add_attribute!("vercel.deployment_id", deployment_id);
227    add_attribute!("vercel.source", source);
228    add_attribute!(SERVER_ADDRESS, host);
229    add_attribute!("vercel.project_id", project_id);
230
231    add_optional_attribute!("vercel.build_id", build_id);
232    add_optional_attribute!("vercel.entrypoint", entrypoint);
233    add_optional_attribute!("vercel.destination", destination);
234    add_optional_attribute!("vercel.path", path);
235    add_optional_attribute!("vercel.log_type", ty);
236    add_optional_attribute!("vercel.status_code", status_code);
237    add_optional_attribute!("vercel.request_id", request_id);
238    add_optional_attribute!(ENVIRONMENT, environment);
239    add_optional_attribute!("vercel.branch", branch);
240    add_optional_attribute!("vercel.ja3_digest", ja3_digest);
241    add_optional_attribute!("vercel.ja4_digest", ja4_digest);
242    add_optional_attribute!("vercel.edge_type", edge_type);
243    add_optional_attribute!("vercel.project_name", project_name);
244    add_optional_attribute!("vercel.execution_region", execution_region);
245
246    if let Some(proxy) = proxy {
247        let VercelProxy {
248            timestamp,
249            method,
250            host,
251            path,
252            user_agent,
253            referer,
254            region,
255            status_code,
256            client_ip,
257            scheme,
258            response_byte_size,
259            cache_id,
260            path_type,
261            path_type_variant,
262            vercel_id,
263            vercel_cache,
264            lambda_region,
265            waf_action,
266            waf_rule_id,
267        } = proxy;
268
269        add_attribute!("vercel.proxy.timestamp", timestamp);
270        add_attribute!("vercel.proxy.method", method);
271        add_attribute!("vercel.proxy.host", host);
272        add_attribute!("vercel.proxy.path", path);
273        add_attribute!("vercel.proxy.region", region);
274
275        if let Ok(user_agent_string) = serde_json::to_string(&user_agent) {
276            attributes.insert("vercel.proxy.user_agent", user_agent_string);
277        }
278
279        add_optional_attribute!("vercel.proxy.referer", referer);
280        add_optional_attribute!("vercel.proxy.status_code", status_code);
281        add_optional_attribute!("vercel.proxy.client_ip", client_ip);
282        add_optional_attribute!("vercel.proxy.scheme", scheme);
283        add_optional_attribute!("vercel.proxy.response_byte_size", response_byte_size);
284        add_optional_attribute!("vercel.proxy.cache_id", cache_id);
285        add_optional_attribute!("vercel.proxy.path_type", path_type);
286        add_optional_attribute!("vercel.proxy.path_type_variant", path_type_variant);
287        add_optional_attribute!("vercel.proxy.vercel_id", vercel_id);
288        add_optional_attribute!("vercel.proxy.vercel_cache", vercel_cache);
289        add_optional_attribute!("vercel.proxy.lambda_region", lambda_region);
290        add_optional_attribute!("vercel.proxy.waf_action", waf_action);
291        add_optional_attribute!("vercel.proxy.waf_rule_id", waf_rule_id);
292    }
293
294    let timestamp = match Utc.timestamp_millis_opt(timestamp).single() {
295        Some(datetime) => Annotated::new(Timestamp(datetime)),
296        None => {
297            let mut meta = Meta::default();
298            meta.add_remark(Remark::new(RemarkType::Substituted, "timestamp.invalid"));
299            Annotated(Some(Timestamp(Utc::now())), meta)
300        }
301    };
302
303    OurLog {
304        timestamp,
305        trace_id,
306        span_id: get_span_id(span_id),
307        level: Annotated::new(map_vercel_level_to_sentry(level)),
308        body: Annotated::new(message.unwrap_or_default()),
309        attributes: Annotated::new(attributes),
310        ..Default::default()
311    }
312}

692        // Test that camelCase fields take precedence over dotted aliases
693        let json_with_both = r#"{
694            "id": "test-123",
695            "deploymentId": "dpl_test",
696            "source": "lambda",
697            "host": "test.vercel.app",
698            "timestamp": 1573817250283,
699            "projectId": "prj_test",
700            "level": "info",
701            "message": "Test message",
702            "traceId": "1b02cd14bb8642fd092bc23f54c7ffcd",
703            "trace.id": "922dda2462ea4ac2b6a4b339bee90863",
704            "spanId": "f24e8631bd11faa7",
705            "span.id": "a1e13f3f06239d69"
706        }"#;
707
708        let vercel_log: VercelLog =
709            serde_json::from_str(json_with_both).expect("Failed to parse JSON with both formats");
710        let our_log = vercel_log_to_sentry_log(vercel_log);
711        assert_eq!(
712            our_log.trace_id.value().map(|t| t.to_string()),
713            Some("1b02cd14bb8642fd092bc23f54c7ffcd".to_owned())
714        );
715        assert_eq!(
716            our_log.span_id.value().map(|s| s.to_string()),
717            Some("f24e8631bd11faa7".to_owned())
718        );
719    }
