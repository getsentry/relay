11/// Vercel log structure matching their [schema](https://vercel.com/docs/drains/reference/logs).
12#[derive(Debug, Clone, Deserialize)]
13#[serde(rename_all = "camelCase")]
14pub struct VercelLog {
15    /// Unique identifier for the log entry.
16    pub id: String,
17    /// Identifier for the Vercel deployment.
18    pub deployment_id: String,
19    /// Origin of the log.
20    pub source: String,
21    /// Deployment unique URL hostname.
22    pub host: String,
23    /// Unix timestamp in milliseconds.
24    pub timestamp: i64,
25    /// Identifier for the Vercel project.
26    pub project_id: String,
27    /// Log severity level.
28    pub level: VercelLogLevel,
29    /// Log message content (may be truncated if over 256 KB).
30    pub message: Option<String>,
31    /// Identifier for the Vercel build. Only present on build logs.
32    pub build_id: Option<String>,
33    /// Entrypoint for the request.
34    pub entrypoint: Option<String>,
35    /// Origin of the external content. Only on external logs.
36    pub destination: Option<String>,
37    /// Function or dynamic path of the request.
38    pub path: Option<String>,
39    /// Log output type.
40    #[serde(rename = "type")]
41    pub ty: Option<String>,
42    /// HTTP status code of the request.
43    pub status_code: Option<i64>,
44    /// Identifier of the request.
45    pub request_id: Option<String>,
46    /// Deployment environment. One of production or preview.
47    pub environment: Option<String>,
48    /// Git branch name.
49    pub branch: Option<String>,
50    /// JA3 fingerprint digest.
51    pub ja3_digest: Option<String>,
52    /// JA4 fingerprint digest.
53    pub ja4_digest: Option<String>,
54    /// Type of edge runtime. One of edge-function or middleware
55    pub edge_type: Option<String>,
56    /// Name of the Vercel project.
57    pub project_name: Option<String>,
58    /// Region where the request is executed.
59    pub execution_region: Option<String>,
60    /// Trace identifier for distributed tracing.
61    #[serde(deserialize_with = "deserialize_trace_id", flatten)]
62    pub trace_id: Option<String>,
63    /// Span identifier for distributed tracing.
64    #[serde(deserialize_with = "deserialize_span_id", flatten)]
65    pub span_id: Option<String>,
66    /// Proxy information for requests. (Optional)
67    pub proxy: Option<VercelProxy>,
68}

70/// Macro to create a deserializer that accepts two field name variants, preferring the first.
71macro_rules! define_fallback_field_deserializer {
72    ($fn_name:ident, $primary:literal, $fallback:literal) => {
73        fn $fn_name<'d, D: Deserializer<'d>>(d: D) -> Result<Option<String>, D::Error> {
74            #[derive(Deserialize)]
75            struct Helper {
76                #[serde(rename = $primary)]
77                primary_field: Option<String>,
78                #[serde(rename = $fallback)]
79                fallback_field: Option<String>,
80            }
81
82            let helper = Helper::deserialize(d)?;
83            Ok(helper.primary_field.or(helper.fallback_field))
84        }
85    };
86}

88define_fallback_field_deserializer!(deserialize_trace_id, "traceId", "trace.id");
89define_fallback_field_deserializer!(deserialize_span_id, "spanId", "span.id");

91/// Log Level for Vercel Logs
92#[derive(Debug, Clone, Deserialize)]
93#[serde(rename_all = "lowercase")]
94pub enum VercelLogLevel {
95    /// Info level logs.
96    Info,
97    /// Warning level logs.
98    Warning,
99    /// Error level logs.
100    Error,
101    /// Fatal level logs.
102    Fatal,
103    /// Unknown status, for forward compatibility.
104    Unknown(String),
105}

107/// Vercel proxy information for requests.
108#[derive(Debug, Clone, Deserialize)]
109#[serde(rename_all = "camelCase")]
110pub struct VercelProxy {
111    /// Unix timestamp when the proxy request was made.
112    pub timestamp: i64,
113    /// HTTP method of the request.
114    pub method: String,
115    /// Hostname of the request.
116    pub host: String,
117    /// Request path with query parameters.
118    pub path: String,
119    /// User agent strings of the request.
120    pub user_agent: Vec<String>,
121    /// Referer of the request.
122    pub referer: String,
123    /// Region where the request is processed.
124    pub region: String,
125    /// HTTP status code of the proxy request.
126    pub status_code: Option<i64>,
127    /// Client IP address.
128    pub client_ip: Option<String>,
129    /// Protocol of the request.
130    pub scheme: Option<String>,
131    /// Size of the response in bytes.
132    pub response_byte_size: Option<i64>,
133    /// Original request ID when request is served from cache.
134    pub cache_id: Option<String>,
135    /// How the request was served based on its path and project configuration.
136    pub path_type: Option<String>,
137    /// Variant of the path type.
138    pub path_type_variant: Option<String>,
139    /// Vercel-specific identifier.
140    pub vercel_id: Option<String>,
141    /// Cache status sent to the browser.
142    pub vercel_cache: Option<String>,
143    /// Region where lambda function executed.
144    pub lambda_region: Option<String>,
145    /// Action taken by firewall rules.
146    pub waf_action: Option<String>,
147    /// ID of the firewall rule that matched.
148    pub waf_rule_id: Option<String>,
149}

151/// Maps Vercel log level to Sentry log level.
152fn map_vercel_level_to_sentry(level: VercelLogLevel) -> OurLogLevel {
153    match level {
154        VercelLogLevel::Info => OurLogLevel::Info,
155        VercelLogLevel::Warning => OurLogLevel::Warn,
156        VercelLogLevel::Error => OurLogLevel::Error,
157        VercelLogLevel::Fatal => OurLogLevel::Fatal,
158        VercelLogLevel::Unknown(_) => OurLogLevel::Info,
159    }
160}

162fn get_trace_id(trace_id: Option<String>) -> Annotated<TraceId> {
163    match trace_id {
164        Some(s) if !s.is_empty() => match TraceId::from_str(&s) {
165            Ok(id) => Annotated::new(id),
166            Err(_) => {
167                let mut meta = Meta::default();
168                meta.add_remark(Remark::new(RemarkType::Substituted, "trace_id.invalid"));
169                Annotated(Some(TraceId::random()), meta)
170            }
171        },
172        _ => {
173            let mut meta = Meta::default();
174            meta.add_remark(Remark::new(RemarkType::Substituted, "trace_id.missing"));
175            Annotated(Some(TraceId::random()), meta)
176        }
177    }
178}
179
180fn get_span_id(span_id: Option<String>) -> Annotated<SpanId> {
181    match span_id {
182        Some(s) if !s.is_empty() => {
183            SpanId::from_str(&s).map_or_else(|err| Annotated::from_error(err, None), Annotated::new)
184        }
185        _ => Annotated::empty(),
186    }
187}

189/// Transforms a Vercel log record to a Sentry log.
190pub fn vercel_log_to_sentry_log(vercel_log: VercelLog) -> OurLog {
191    let VercelLog {
192        id,
193        deployment_id,
194        source,
195        host,
196        timestamp,
197        project_id,
198        level,
199        message,
200        build_id,
201        entrypoint,
202        destination,
203        path,
204        ty,
205        status_code,
206        request_id,
207        environment,
208        branch,
209        ja3_digest,
210        ja4_digest,
211        edge_type,
212        project_name,
213        execution_region,
214        trace_id,
215        span_id,
216        proxy,
217    } = vercel_log;
218
219    let mut attributes: Attributes = Attributes::default();
220
221    macro_rules! add_optional_attribute {
222        ($name:expr, $value:expr) => {{
223            if let Some(value) = $value {
224                attributes.insert($name.to_owned(), value);
225            }
226        }};
227    }
228
229    macro_rules! add_attribute {
230        ($name:expr, $value:expr) => {{
231            let val = $value;
232            attributes.insert($name.to_owned(), val);
233        }};
234    }
235
236    add_attribute!(ORIGIN, "auto.log_drain.vercel".to_owned());
237    add_attribute!("vercel.id", id);
238    add_attribute!("vercel.deployment_id", deployment_id);
239    add_attribute!("vercel.source", source);
240    add_attribute!(SERVER_ADDRESS, host);
241    add_attribute!("vercel.project_id", project_id);
242
243    add_optional_attribute!("vercel.build_id", build_id);
244    add_optional_attribute!("vercel.entrypoint", entrypoint);
245    add_optional_attribute!("vercel.destination", destination);
246    add_optional_attribute!(URL_PATH, path);
247    add_optional_attribute!("vercel.log_type", ty);
248    add_optional_attribute!("vercel.status_code", status_code);
249    add_optional_attribute!("vercel.request_id", request_id);
250    add_optional_attribute!(ENVIRONMENT, environment);
251    add_optional_attribute!("vercel.branch", branch);
252    add_optional_attribute!("vercel.ja3_digest", ja3_digest);
253    add_optional_attribute!("vercel.ja4_digest", ja4_digest);
254    add_optional_attribute!("vercel.edge_type", edge_type);
255    add_optional_attribute!("vercel.project_name", project_name);
256    add_optional_attribute!("vercel.execution_region", execution_region);
257
258    if let Some(proxy) = proxy {
259        let VercelProxy {
260            timestamp,
261            method,
262            host,
263            path,
264            user_agent,
265            referer,
266            region,
267            status_code,
268            client_ip,
269            scheme,
270            response_byte_size,
271            cache_id,
272            path_type,
273            path_type_variant,
274            vercel_id,
275            vercel_cache,
276            lambda_region,
277            waf_action,
278            waf_rule_id,
279        } = proxy;
280
281        add_attribute!("vercel.proxy.timestamp", timestamp);
282        add_attribute!("vercel.proxy.method", method);
283        add_attribute!("vercel.proxy.host", host);
284        add_attribute!("vercel.proxy.path", path);
285        add_attribute!("vercel.proxy.referer", referer);
286        add_attribute!("vercel.proxy.region", region);
287
288        if let Ok(user_agent_string) = serde_json::to_string(&user_agent) {
289            attributes.insert("vercel.proxy.user_agent", user_agent_string);
290        }
291
292        add_optional_attribute!("vercel.proxy.status_code", status_code);
293        add_optional_attribute!("vercel.proxy.client_ip", client_ip);
294        add_optional_attribute!("vercel.proxy.scheme", scheme);
295        add_optional_attribute!("vercel.proxy.response_byte_size", response_byte_size);
296        add_optional_attribute!("vercel.proxy.cache_id", cache_id);
297        add_optional_attribute!("vercel.proxy.path_type", path_type);
298        add_optional_attribute!("vercel.proxy.path_type_variant", path_type_variant);
299        add_optional_attribute!("vercel.proxy.vercel_id", vercel_id);
300        add_optional_attribute!("vercel.proxy.vercel_cache", vercel_cache);
301        add_optional_attribute!("vercel.proxy.lambda_region", lambda_region);
302        add_optional_attribute!("vercel.proxy.waf_action", waf_action);
303        add_optional_attribute!("vercel.proxy.waf_rule_id", waf_rule_id);
304    }
305
306    let timestamp = match Utc.timestamp_millis_opt(timestamp).single() {
307        Some(datetime) => Annotated::new(Timestamp(datetime)),
308        None => {
309            let mut meta = Meta::default();
310            meta.add_remark(Remark::new(RemarkType::Substituted, "timestamp.invalid"));
311            Annotated(Some(Timestamp(Utc::now())), meta)
312        }
313    };
314
315    OurLog {
316        timestamp,
317        trace_id: get_trace_id(trace_id),
318        span_id: get_span_id(span_id),
319        level: Annotated::new(map_vercel_level_to_sentry(level)),
320        body: Annotated::new(message.unwrap_or_default()),
321        attributes: Annotated::new(attributes),
322        ..Default::default()
323    }
324}

717            "span.id": "a1e13f3f06239d69"
718        }"#;
719
720        let vercel_log: VercelLog =
721            serde_json::from_str(json_with_both).expect("Failed to parse JSON with both formats");

724            our_log.trace_id.value().map(|t| t.to_string()),

728            our_log.span_id.value().map(|s| s.to_string()),
