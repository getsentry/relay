19/// Keeps track of the maximum depth of an SQL expression that the [`NormalizeVisitor`] encounters.
20///
21/// This is used to prevent the serialization of the AST from crashing.
22/// Note that the expression depth does not fully cover the depth of complex SQL statements,
23/// because not everything in SQL is an expression.
24const MAX_EXPRESSION_DEPTH: usize = 64;
25
26/// Derive the SQL dialect from `db_system` (the value obtained from `span.data.system`)
27/// and try to parse the query into an AST.
28pub fn parse_query(
29    db_system: Option<&str>,
30    query: &str,
31) -> Result<Vec<Statement>, sqlparser::parser::ParserError> {
32    relay_log::with_scope(
33        |scope| {
34            scope.set_tag("db_system", db_system.unwrap_or_default());
35            scope.set_extra("query", query.into());
36        },
37        || match std::panic::catch_unwind(|| parse_query_inner(db_system, query)) {
38            Ok(res) => res,
39            Err(_) => Err(sqlparser::parser::ParserError::ParserError(
40                "panicked".to_owned(),
41            )),
42        },
43    )
44}
45
46fn parse_query_inner(
47    db_system: Option<&str>,
48    query: &str,
49) -> Result<Vec<Statement>, sqlparser::parser::ParserError> {
50    // See https://github.com/open-telemetry/opentelemetry-specification/blob/main/specification/trace/semantic_conventions/database.md#notes-and-well-known-identifiers-for-dbsystem
51    //     https://docs.rs/sqlparser/latest/sqlparser/dialect/fn.dialect_from_str.html
52    let dialect = db_system
53        .and_then(sqlparser::dialect::dialect_from_str)
54        .unwrap_or_else(|| Box::new(GenericDialect {}));
55    let dialect = DialectWithParameters(dialect);
56
57    sqlparser::parser::Parser::parse_sql(&dialect, query)
58}

60/// Tries to parse a series of SQL queries into an AST and normalize it.
61pub fn normalize_parsed_queries(
62    db_system: Option<&str>,
63    string: &str,
64) -> Result<(String, Vec<Statement>), ()> {
65    let mut parsed = parse_query(db_system, string).map_err(|_| ())?;
66    let _ = parsed.visit(&mut NormalizeVisitor);
67    let _ = parsed.visit(&mut MaxDepthVisitor::new());
68
69    let concatenated = parsed
70        .iter()
71        .map(|statement| statement.to_string())
72        .join("; ");
73
74    // Insert placeholders that the SQL serializer cannot provide.
75    let replaced = concatenated.replace("___UPDATE_LHS___ = NULL", "..");
76
77    Ok((replaced, parsed))
78}

80/// A visitor that normalizes the SQL AST in-place.
81///
82/// Used for span description normalization.
83struct NormalizeVisitor;
84
85impl NormalizeVisitor {
86    /// Placeholder for string and numerical literals.
87    fn placeholder() -> ValueWithSpan {
88        Value::Placeholder("%s".to_owned()).into()
89    }
90
91    /// Placeholder for a list of items that has been scrubbed.
92    fn ellipsis() -> Ident {
93        Ident::new("..")
94    }
95
96    /// Check if a selected item can be erased into `..`.
97    /// We only collapse simple values (e.g. NULL) and columns (`col1` or `col1 AS foo`).
98    fn is_collapsible(item: &SelectItem) -> bool {
99        match item {
100            SelectItem::UnnamedExpr(expr) | SelectItem::ExprWithAlias { expr, .. } => {
101                matches!(
102                    expr,
103                    Expr::Value(_) | Expr::Identifier(_) | Expr::CompoundIdentifier(_)
104                )
105            }
106            _ => false,
107        }
108    }
109
110    /// Helper function to collapse a sequence of selected items into `..`.
111    fn collapse_items(collapse: &mut Vec<SelectItem>, output: &mut Vec<SelectItem>) {
112        match collapse.len() {
113            0 => {}
114            1 => {
115                output.append(collapse);
116            }
117            _ => {
118                output.push(SelectItem::UnnamedExpr(Expr::Identifier(Self::ellipsis())));
119            }
120        }
121    }
122
123    /// Normalizes `SELECT ...` queries.
124    fn transform_query(&mut self, query: &mut Query) {
125        if let SetExpr::Select(select) = &mut *query.body {
126            self.transform_select(&mut *select);
127        }
128    }
129
130    fn transform_select(&mut self, select: &mut Select) {
131        // Track collapsible selected items (e.g. `SELECT col1, col2`).
132        let mut collapse = vec![];
133
134        // Iterate over selected item.
135        for item in std::mem::take(&mut select.projection) {
136            // Normalize aliases.
137            let item = match item {
138                SelectItem::ExprWithAlias { expr, .. } => SelectItem::UnnamedExpr(expr),
139                SelectItem::QualifiedWildcard(_, options) => SelectItem::Wildcard(options),
140                _ => item,
141            };
142            if Self::is_collapsible(&item) {
143                collapse.push(item);
144            } else {
145                Self::collapse_items(&mut collapse, &mut select.projection);
146                collapse.clear();
147                select.projection.push(item);
148            }
149        }
150        Self::collapse_items(&mut collapse, &mut select.projection);
151    }
152
153    fn simplify_table_alias(alias: &mut Option<TableAlias>) {
154        if let Some(TableAlias { name, columns }) = alias {
155            Self::scrub_name(name);
156            for TableAliasColumnDef { name, data_type: _ } in columns {
157                Self::scrub_name(name);
158            }
159        }
160    }
161
162    fn simplify_compound_identifier(parts: &mut Vec<Ident>) {
163        if let Some(mut last) = parts.pop() {
164            Self::scrub_name(&mut last);
165            *parts = vec![last];
166        }
167    }
168
169    fn simplify_object_name(parts: &mut Vec<ObjectNamePart>) {
170        if let Some(mut last) = parts.pop() {
171            Self::scrub_object_name_part(&mut last);
172            *parts = vec![last];
173        }
174    }
175
176    fn scrub_name(name: &mut Ident) {
177        name.quote_style = None;
178        if let Cow::Owned(s) = TABLE_NAME_REGEX.replace_all(&name.value, "{%s}") {
179            name.value = s
180        };
181    }
182
183    fn scrub_object_name_part(part: &mut ObjectNamePart) {
184        match part {
185            ObjectNamePart::Identifier(ident) => Self::scrub_name(ident),
186            ObjectNamePart::Function(ObjectNamePartFunction { name, args }) => {
187                Self::scrub_name(name);
188                Self::scrub_function_args(args);
189            }
190        }
191    }
192
193    fn scrub_function_args(args: &mut Vec<FunctionArg>) {
194        for arg in args {
195            if let FunctionArg::Named { name, .. } = arg {
196                Self::scrub_name(name);
197            }
198        }
199    }
200
201    fn erase_name(name: &mut Ident) {
202        name.quote_style = None;
203        name.value = "%s".into()
204    }
205
206    fn scrub_statement_filter(filter: &mut Option<ShowStatementFilter>) {
207        if let Some(s) = filter {
208            match s {
209                ShowStatementFilter::Like(s)
210                | ShowStatementFilter::ILike(s)
211                | ShowStatementFilter::NoKeyword(s) => "%s".clone_into(s),
212                ShowStatementFilter::Where(_expr) => {}
213            }
214        }
215    }
216}

218impl VisitorMut for NormalizeVisitor {
219    type Break = ();
220
221    fn pre_visit_relation(&mut self, relation: &mut ObjectName) -> ControlFlow<Self::Break> {
222        Self::simplify_object_name(&mut relation.0);
223        ControlFlow::Continue(())
224    }
225
226    fn pre_visit_value(&mut self, value: &mut Value) -> ControlFlow<Self::Break> {
227        *value = Value::Placeholder("%s".into());
228        ControlFlow::Continue(())
229    }
230
231    fn pre_visit_table_factor(
232        &mut self,
233        table_factor: &mut TableFactor,
234    ) -> ControlFlow<Self::Break> {
235        match table_factor {
236            TableFactor::Table { alias, .. } => {
237                Self::simplify_table_alias(alias);
238            }
239            TableFactor::Derived {
240                subquery, alias, ..
241            } => {
242                self.transform_query(subquery);
243                Self::simplify_table_alias(alias);
244            }
245            TableFactor::TableFunction { alias, .. } => {
246                Self::simplify_table_alias(alias);
247            }
248            TableFactor::UNNEST {
249                alias,
250                with_offset_alias,
251                ..
252            } => {
253                Self::simplify_table_alias(alias);
254                if let Some(ident) = with_offset_alias {
255                    Self::scrub_name(ident);
256                }
257            }
258            TableFactor::NestedJoin { alias, .. } => {
259                Self::simplify_table_alias(alias);
260            }
261            TableFactor::Pivot { alias, .. } => {
262                Self::simplify_table_alias(alias);
263            }
264            TableFactor::Function {
265                name, args, alias, ..
266            } => {
267                Self::simplify_object_name(&mut name.0);
268                Self::scrub_function_args(args);
269                Self::simplify_table_alias(alias);
270            }
271            TableFactor::JsonTable { columns, alias, .. } => {
272                for column in columns {
273                    match column {
274                        JsonTableColumn::Named(JsonTableNamedColumn { name, .. }) => {
275                            Self::scrub_name(name);
276                        }
277                        JsonTableColumn::ForOrdinality(ident) => Self::scrub_name(ident),
278                        JsonTableColumn::Nested(JsonTableNestedColumn { path: _, columns }) => {
279                            columns.clear(); // bit aggressive, might revisit later.
280                        }
281                    }
282                }
283                Self::simplify_table_alias(alias);
284            }
285            TableFactor::Unpivot {
286                name,
287                columns,
288                alias,
289                ..
290            } => {
291                Self::scrub_name(name);
292                for ExprWithAlias { expr: _, alias } in columns {
293                    if let Some(alias) = alias {
294                        Self::scrub_name(alias);
295                    }
296                }
297                Self::simplify_table_alias(alias);
298            }
299            // Not altering the following directly, might revisit later:
300            TableFactor::OpenJsonTable { .. } => {}
301            TableFactor::MatchRecognize { .. } => {}
302            TableFactor::XmlTable { .. } => {}
303            TableFactor::SemanticView { .. } => {}
304        }
305        ControlFlow::Continue(())
306    }
307
308    fn pre_visit_expr(&mut self, expr: &mut Expr) -> ControlFlow<Self::Break> {
309        match expr {
310            // Simple values like numbers and strings are replaced by a placeholder:
311            Expr::Value(x) => *x = Self::placeholder(),
312            // `IN (val1, val2, val3)` is replaced by `IN (%s)`.
313            Expr::InList { list, .. } => *list = vec![Expr::Value(Self::placeholder())],
314            // `"table"."col"` is replaced by `col`.
315            Expr::CompoundIdentifier(parts) => {
316                Self::simplify_compound_identifier(parts);
317            }
318            // `"col"` is replaced by `col`.
319            Expr::Identifier(ident) => {
320                Self::scrub_name(ident);
321            }
322            // Recurse into subqueries.
323            Expr::Subquery(query) => self.transform_query(query),
324            // Remove negative sign, e.g. `-100` becomes `%s`.
325            Expr::UnaryOp {
326                op: UnaryOperator::Minus,
327                expr: inner,
328            } => {
329                if let Expr::Value(_) = **inner {
330                    *expr = Expr::Value(Self::placeholder())
331                }
332            }
333            // Simplify `CASE WHEN..` expressions.
334            Expr::Case {
335                operand,
336                conditions,
337                else_result,
338                case_token: _,
339                end_token: _,
340            } => {
341                operand.take();
342                *conditions = vec![CaseWhen {
343                    condition: Expr::Identifier(Self::ellipsis()),
344                    result: Expr::Identifier(Self::ellipsis()),
345                }];
346                else_result.take();
347            }
348            _ => {}
349        }
350        ControlFlow::Continue(())
351    }
352
353    fn post_visit_expr(&mut self, expr: &mut Expr) -> ControlFlow<Self::Break> {
354        // Casts are omitted for simplification. Because we replace the entire expression,
355        // the replacement has to occur *after* visiting its children.
356        match expr {
357            Expr::Cast { expr: inner, .. } => {
358                *expr = take_expr(inner);
359            }
360            Expr::BinaryOp {
361                left,
362                op: op @ (BinaryOperator::Or | BinaryOperator::And),
363                right,
364            } => {
365                remove_redundant_parentheses(op, left);
366                remove_redundant_parentheses(op, right);
367                if left == right {
368                    //     /\
369                    //    /  \
370                    //   /\   B
371                    //  /  \
372                    // A    A
373                    *expr = take_expr(left);
374                } else {
375                    //     /\
376                    //    /  \
377                    //   /\   B
378                    //  /  \
379                    // A    B
380                    if let Expr::BinaryOp {
381                        left: left_left,
382                        op: left_op,
383                        right: left_right,
384                    } = left.as_mut()
385                        && left_op == op
386                        && left_right == right
387                    {
388                        *left = Box::new(take_expr(left_left));
389                    }
390                }
391            }
392            Expr::Nested(inner) if matches!(inner.as_ref(), &Expr::Nested(_)) => {
393                // Remove multiple levels of parentheses.
394                // These can occur because of the binary op reduction above.
395                *expr = take_expr(inner);
396            }
397            _ => (),
398        }
399
400        ControlFlow::Continue(())
401    }
402
403    fn post_visit_statement(&mut self, statement: &mut Statement) -> ControlFlow<Self::Break> {
404        match statement {
405            Statement::Query(query) => {
406                self.transform_query(query);
407            }
408            Statement::Insert(Insert {
409                columns, source, ..
410            }) => {
411                *columns = vec![Self::ellipsis()];
412                if let Some(source) = source.as_mut()
413                    && let SetExpr::Values(v) = &mut *source.body
414                {
415                    v.rows = vec![vec![Expr::Value(Self::placeholder())]]
416                }
417            }
418            Statement::Update { assignments, .. } => {
419                if assignments.len() > 1
420                    && assignments
421                        .iter()
422                        .all(|a| matches!(a.value, Expr::Value(_)))
423                {
424                    *assignments = vec![Assignment {
425                        target: AssignmentTarget::ColumnName(ObjectName(vec![
426                            ObjectNamePart::Identifier(Ident::new("___UPDATE_LHS___")),
427                        ])),
428                        value: Expr::Value(Value::Null.into()),
429                    }]
430                } else {
431                    for assignment in assignments.iter_mut() {
432                        if let AssignmentTarget::ColumnName(target) = &mut assignment.target {
433                            Self::simplify_object_name(&mut target.0);
434                        }
435                    }
436                }
437            }
438            Statement::Savepoint { name } => Self::erase_name(name),
439            Statement::ReleaseSavepoint { name } => Self::erase_name(name),
440            Statement::Declare { stmts } => {
441                for Declare {
442                    names, for_query, ..
443                } in stmts
444                {
445                    for name in names {
446                        Self::erase_name(name);
447                    }
448                    if let Some(for_query) = for_query {
449                        self.transform_query(for_query.as_mut());
450                    }
451                }
452            }
453            Statement::Fetch { name, into, .. } => {
454                Self::erase_name(name);
455                if let Some(into) = into {
456                    into.0 = vec![ObjectNamePart::Identifier(Ident {
457                        value: "%s".into(),
458                        quote_style: None,
459                        span: Span::empty(),
460                    })];
461                }
462            }
463            Statement::Close { cursor } => match cursor {
464                CloseCursor::All => {}
465                CloseCursor::Specific { name } => Self::erase_name(name),
466            },
467            Statement::AlterTable {
468                name, operations, ..
469            } => {
470                Self::simplify_object_name(&mut name.0);
471                for operation in operations {
472                    match operation {
473                        AlterTableOperation::AddConstraint {
474                            constraint,
475                            not_valid: _,
476                        } => match constraint {
477                            TableConstraint::Unique { name, .. } => {
478                                if let Some(name) = name {
479                                    Self::scrub_name(name);
480                                }
481                            }
482                            TableConstraint::ForeignKey {
483                                name,
484                                columns,
485                                referred_columns,
486                                ..
487                            } => {
488                                if let Some(name) = name {
489                                    Self::scrub_name(name);
490                                }
491                                for column in columns {
492                                    Self::scrub_name(column);
493                                }
494                                for column in referred_columns {
495                                    Self::scrub_name(column);
496                                }
497                            }
498                            TableConstraint::Check { name, .. } => {
499                                if let Some(name) = name {
500                                    Self::scrub_name(name);
501                                }
502                            }
503                            TableConstraint::Index { name, .. } => {
504                                if let Some(name) = name {
505                                    Self::scrub_name(name);
506                                }
507                            }
508                            TableConstraint::FulltextOrSpatial { opt_index_name, .. } => {
509                                if let Some(name) = opt_index_name {
510                                    Self::scrub_name(name);
511                                }
512                            }
513                            TableConstraint::PrimaryKey {
514                                name, index_name, ..
515                            } => {
516                                if let Some(name) = name {
517                                    Self::scrub_name(name);
518                                }
519                                if let Some(name) = index_name {
520                                    Self::scrub_name(name);
521                                }
522                            }
523                        },
524                        AlterTableOperation::AddColumn { column_def, .. } => {
525                            let ColumnDef { name, .. } = column_def;
526                            Self::scrub_name(name);
527                        }
528                        AlterTableOperation::DropConstraint { name, .. } => Self::scrub_name(name),
529                        AlterTableOperation::DropColumn { column_names, .. } => {
530                            for column_name in column_names {
531                                Self::scrub_name(column_name);
532                            }
533                        }
534                        AlterTableOperation::RenameColumn {
535                            old_column_name,
536                            new_column_name,
537                        } => {
538                            Self::scrub_name(old_column_name);
539                            Self::scrub_name(new_column_name);
540                        }
541                        AlterTableOperation::ChangeColumn {
542                            old_name, new_name, ..
543                        } => {
544                            Self::scrub_name(old_name);
545                            Self::scrub_name(new_name);
546                        }
547                        AlterTableOperation::RenameConstraint { old_name, new_name } => {
548                            Self::scrub_name(old_name);
549                            Self::scrub_name(new_name);
550                        }
551                        AlterTableOperation::AlterColumn { column_name, .. } => {
552                            Self::scrub_name(column_name);
553                        }
554                        _ => {}
555                    }
556                }
557            }
558            Statement::Analyze { columns, .. } => {
559                Self::simplify_compound_identifier(columns);
560            }
561            Statement::Truncate { .. } => {}
562            Statement::Msck { .. } => {}
563            Statement::Directory { path, .. } => {
564                "%s".clone_into(path);
565            }
566            Statement::Call(_) => {}
567            Statement::Copy { source, values, .. } => {
568                if let CopySource::Table { columns, .. } = source {
569                    Self::simplify_compound_identifier(columns);
570                }
571                *values = vec![Some("..".into())];
572            }
573            Statement::CopyIntoSnowflake {
574                files,
575                pattern,
576                validation_mode,
577                from_obj,
578                ..
579            } => {
580                if let Some(from_obj) = from_obj {
581                    Self::simplify_object_name(&mut from_obj.0);
582                }
583                *files = None;
584                *pattern = None;
585                *validation_mode = None;
586            }
587            Statement::Delete { .. } => {}
588            Statement::CreateView {
589                columns,
590                cluster_by,
591                ..
592            } => {
593                for column in columns {
594                    Self::scrub_name(&mut column.name);
595                }
596                Self::simplify_compound_identifier(cluster_by);
597            }
598            Statement::CreateTable { .. } => {}
599            Statement::CreateVirtualTable {
600                module_name,
601                module_args,
602                ..
603            } => {
604                Self::scrub_name(module_name);
605                Self::simplify_compound_identifier(module_args);
606            }
607            Statement::CreateIndex(CreateIndex { name, include, .. }) => {
608                // `table_name` is visited by `visit_relation`, but `name` is not.
609                if let Some(name) = name {
610                    Self::simplify_object_name(&mut name.0);
611                }
612                Self::simplify_compound_identifier(include);
613            }
614            Statement::CreateRole { .. } => {}
615            Statement::AlterIndex { name, operation } => {
616                // Names here are not visited by `visit_relation`.
617                Self::simplify_object_name(&mut name.0);
618                match operation {
619                    sqlparser::ast::AlterIndexOperation::RenameIndex { index_name } => {
620                        Self::simplify_object_name(&mut index_name.0);
621                    }
622                }
623            }
624            Statement::AlterView { .. } => {}
625            Statement::AlterRole { name, .. } => {
626                Self::scrub_name(name);
627            }
628            Statement::AttachDatabase { schema_name, .. } => Self::scrub_name(schema_name),
629            Statement::Drop { .. } => {}
630            Statement::DropFunction { .. } => {}
631            Statement::CreateExtension { name, .. } => Self::scrub_name(name),
632            Statement::Flush { channel, .. } => *channel = None,
633            Statement::Discard { .. } => {}
634            Statement::ShowFunctions { filter } => Self::scrub_statement_filter(filter),
635            Statement::ShowVariable { variable } => Self::simplify_compound_identifier(variable),
636            Statement::ShowVariables { filter, .. } => Self::scrub_statement_filter(filter),
637            Statement::ShowCreate { .. } => {}
638            Statement::ShowColumns { .. } => {}
639            Statement::ShowTables { .. } => {}
640            Statement::ShowCollation { filter } => Self::scrub_statement_filter(filter),
641            Statement::Use(
642                Use::Database(db_name)
643                | Use::Catalog(db_name)
644                | Use::Schema(db_name)
645                | Use::Role(db_name)
646                | Use::Warehouse(db_name),
647            ) => Self::simplify_object_name(&mut db_name.0),
648            Statement::Use(_) => {} // gave up at this point
649            Statement::StartTransaction { .. } => {}
650            Statement::Comment { comment, .. } => *comment = None,
651            Statement::Commit { .. } => {}
652            Statement::Rollback {
653                savepoint: Some(savepoint),
654                ..
655            } => {
656                Self::erase_name(savepoint);
657            }
658            Statement::CreateSchema { .. } => {}
659            Statement::CreateDatabase {
660                location,
661                managed_location,
662                ..
663            } => {
664                *location = None;
665                *managed_location = None;
666            }
667            Statement::CreateFunction { .. } => {}
668            Statement::CreateProcedure { .. } => {}
669            Statement::CreateMacro { .. } => {}
670            Statement::CreateStage { comment, .. } => *comment = None,
671            Statement::Assert { .. } => {}
672            Statement::Grant {
673                grantees,
674                granted_by,
675                ..
676            } => {
677                for grantee in grantees {
678                    if let Some(GranteeName::ObjectName(name)) = &mut grantee.name {
679                        Self::simplify_object_name(&mut name.0);
680                    }
681                }
682
683                *granted_by = None;
684            }
685            Statement::Revoke {
686                grantees,
687                granted_by,
688                ..
689            } => {
690                for grantee in grantees {
691                    if let Some(GranteeName::ObjectName(name)) = &mut grantee.name {
692                        Self::simplify_object_name(&mut name.0);
693                    }
694                }
695                *granted_by = None;
696            }
697            Statement::Deallocate { name, .. } => {
698                Self::scrub_name(name);
699            }
700            Statement::Execute {
701                name: Some(name), ..
702            } => {
703                Self::simplify_object_name(&mut name.0);
704            }
705            Statement::Prepare { name, .. } => Self::scrub_name(name),
706            Statement::Kill { id, .. } => *id = 0,
707            Statement::ExplainTable { .. } => {}
708            Statement::Explain { .. } => {}
709            Statement::Merge { .. } => {}
710            Statement::Cache { .. } => {}
711            Statement::UNCache { .. } => {}
712            Statement::CreateSequence { .. } => {}
713            Statement::CreateType { .. } => {}
714            Statement::Pragma { .. } => {}
715            Statement::LockTables { tables } => {
716                for table in tables {
717                    let LockTable {
718                        table,
719                        alias,
720                        lock_type: _,
721                    } = table;
722                    Self::scrub_name(table);
723                    if let Some(alias) = alias {
724                        Self::scrub_name(alias);
725                    }
726                }
727            }
728            Statement::UnlockTables => {}
729            Statement::Install { extension_name } | Statement::Load { extension_name } => {
730                Self::scrub_name(extension_name)
731            }
732            Statement::ShowStatus {
733                filter: _,
734                global: _,
735                session: _,
736            } => {}
737            Statement::Unload { to, .. } => {
738                Self::scrub_name(to);
739            }
740            _ => {}
741        }
742
743        ControlFlow::Continue(())
744    }
745}

747/// Get ownership of an `Expr` and leave a NULL value in its place.
748///
749/// We cannot use [`std::mem::take`], because [`Expr`] does not implement [`Default`].
750fn take_expr(expr: &mut Expr) -> Expr {
751    let mut swapped = Expr::Value(Value::Null.into());
752    std::mem::swap(&mut swapped, expr);
753    swapped
754}

756/// Removes parentheses for equal operators, e.g. `(a OR b) OR c`.
757///
758/// Only use this function on operations which have the
759/// [associative property](https://en.wikipedia.org/wiki/Associative_property).
760fn remove_redundant_parentheses(outer_op: &BinaryOperator, expr: &mut Expr) {
761    if let Expr::Nested(inner) = expr
762        && let Expr::BinaryOp { op, .. } = inner.as_ref()
763        && op == outer_op
764    {
765        *expr = take_expr(inner.as_mut());
766    }
767}

769/// Limits the maximum expression depth of an SQL statement.
770///
771/// This prevents stack overflows when serializing the query back to a string.
772struct MaxDepthVisitor {
773    /// The current depth of an expression.
774    current_expr_depth: usize,
775}
776
777impl MaxDepthVisitor {
778    pub fn new() -> Self {
779        Self {
780            current_expr_depth: 0,
781        }
782    }
783}
784
785impl VisitorMut for MaxDepthVisitor {
786    type Break = ();
787
788    fn pre_visit_expr(&mut self, expr: &mut Expr) -> ControlFlow<Self::Break> {
789        if self.current_expr_depth > MAX_EXPRESSION_DEPTH {
790            *expr = Expr::Value(Value::Placeholder("..".to_owned()).into());
791            return ControlFlow::Continue(());
792        }
793        self.current_expr_depth += 1;
794        ControlFlow::Continue(())
795    }
796
797    fn post_visit_expr(&mut self, _expr: &mut Expr) -> ControlFlow<Self::Break> {
798        self.current_expr_depth = self.current_expr_depth.saturating_sub(1);
799        ControlFlow::Continue(())
800    }
801}
