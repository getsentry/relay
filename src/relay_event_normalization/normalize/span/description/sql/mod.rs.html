12/// Removes SQL comments starting with "--" or "#".
13static COMMENTS: LazyLock<Regex> =
14    LazyLock::new(|| Regex::new(r"(?:--|#).*(?P<newline>\n|$)").unwrap());
15
16/// Removes MySQL inline comments.
17static INLINE_COMMENTS: LazyLock<Regex> =
18    LazyLock::new(|| Regex::new(r"/\*(?:.|\n)*?\*/").unwrap());
19
20/// Regex with multiple capture groups for SQL tokens we should scrub.
21///
22/// Slightly modified from
23/// <https://github.com/getsentry/sentry/blob/65fb6fdaa0080b824ab71559ce025a9ec6818b3e/src/sentry/spans/grouping/strategy/base.py#L170>
24/// <https://github.com/getsentry/sentry/blob/17af7efe869007f85c5322e48aa9f80a8515bde4/src/sentry/spans/grouping/strategy/base.py#L163>
25static NORMALIZER_REGEX: LazyLock<Regex> = LazyLock::new(|| {
26    Regex::new(
27        r#"(?xi)
28        # Capture `SAVEPOINT` savepoints.
29        ((?-x)(?P<pre>SAVEPOINT )(?P<savepoint>(?:(?:"[^"]+")|(?:'[^']+')|(?:`[^`]+`)|(?:[a-z]\w+)))) |
30        # Capture single-quoted strings, including the remaining substring if `\'` is found.
31        ((?-x)(?P<single_quoted_strs>N?'(?:\\'|[^'])*(?:'|$)(::\w+(\[\]?)?)?)) |
32        # Capture placeholders.
33        (   (?P<placeholder> (?:\?+|\$\d+|%(?:\(\w+\))?s|:\w+) (::\w+(\[\]?)?)? )   ) |
34        # Capture ODBC escape sequence.
35        ((?-x)(?P<odbc_escape_sequence>\{(?:ts?|d)\s+'.+'\})) |
36        # UUIDs
37        ([0-9a-f]{8}.?[0-9a-f]{4}.?[0-9a-f]{4}.?[0-9a-f]{4}.?[0-9a-f]{12}) |
38        # Capture numbers.
39        ((?-x)(?P<number>(-?\b(?:[0-9]+\.)?[0-9]+(?:[eE][+-]?[0-9]+)?\b)(::\w+(\[\]?)?)?)) |
40        # Hex constants
41        ((?-x)(?P<hex>(\b0x[0-9a-f]+\b)(::\w+(\[\]?)?)?)) |
42        # Capture booleans (as full tokens, not as substrings of other tokens).
43        ((?-x)(?P<bool>(\b(?:true|false)\b)))
44        "#,
45    )
46    .unwrap()
47});
48
49/// For MySQL, also look for double quoted strings.
50static DOUBLE_QUOTED_STRING_REGEX: LazyLock<Regex> = LazyLock::new(|| {
51    Regex::new(
52        r#"(?xi)
53        # Capture double-quoted strings, including the remaining substring if `\'` is found.
54        ((?-x)(?P<single_quoted_strs>N?"(?:\\"|[^"])*(?:"|$)(::\w+(\[\]?)?)?))
55        "#,
56    )
57    .unwrap()
58});
59
60/// Removes extra whitespace and newlines.
61static WHITESPACE: LazyLock<Regex> = LazyLock::new(|| Regex::new(r"(\s*\n\s*)|(\s\s+)").unwrap());
62
63/// Removes whitespace around parentheses.
64static PARENS: LazyLock<Regex> =
65    LazyLock::new(|| Regex::new(r"((?P<pre>\()\s+)|(\s+(?P<post>\)))").unwrap());
66
67static STRIP_QUOTES: LazyLock<Regex> =
68    LazyLock::new(|| Regex::new(r#"["`](?P<entity_name>\w+)($|["`])"#).unwrap());
69
70/// Regex to shorten table or column references, e.g. `"table1"."col1"` -> `col1`.
71static COLLAPSE_ENTITIES: LazyLock<Regex> =
72    LazyLock::new(|| Regex::new(r"(?:\w+\.)+(?P<entity_name>\w+)").unwrap());
73
74/// Regex to make multiple placeholders collapse into one.
75/// This can be used as a second pass after [`NORMALIZER_REGEX`].
76static COLLAPSE_PLACEHOLDERS: LazyLock<Regex> = LazyLock::new(|| {
77    Regex::new(
78        r"(?xi)
79        (?P<pre>(?:VALUES|IN) \s+\() (?P<values> ( %s ( \)\s*,\s*\(\s*%s | \s*,\s*%s )* )) (?P<post>\s*\)?)
80        ",
81    )
82    .unwrap()
83});
84
85/// Collapse simple lists of column names.
86/// For example:
87///   SELECT a, b, count(*) FROM x -> SELECT .., count(*) FROM x
88///   SELECT "a.b" AS a__b, "a.c" AS a__c FROM x -> SELECT .. FROM x
89///
90/// Assumes that column names have already been normalized.
91static COLLAPSE_COLUMNS: LazyLock<Regex> = LazyLock::new(|| {
92    let col = r"\w+(\s+AS\s+\w+)?";
93    Regex::new(
94        format!(
95            r"(?ix)
96        (?P<pre>(SELECT(\s+(DISTINCT|ALL))?\s|,|\())
97        \s*
98        (?P<columns>({col}(?:\s*,\s*{col})+))
99        (?P<post>\s*(,|\s+|\)))"
100        )
101        .as_str(),
102    )
103    .unwrap()
104});
105
106/// Regex to identify SQL queries that are already normalized.
107///
108/// Looks for `?`, `$1` or `%s` identifiers, commonly used identifiers in
109/// Python, Ruby on Rails and PHP platforms.
110static ALREADY_NORMALIZED_REGEX: LazyLock<Regex> =
111    LazyLock::new(|| Regex::new(r"/\?|\$1|%s").unwrap());
112
113/// Normalizes the given SQL-query-like string.
114pub fn scrub_queries(db_system: Option<&str>, string: &str) -> (Option<String>, Mode) {
115    let t = Instant::now();
116    let (res, mode) = scrub_queries_inner(db_system, string);
117    relay_statsd::metric!(
118        timer(Timers::SpanDescriptionNormalizeSQL) = t.elapsed(),
119        mode = mode.as_str(),
120    );
121    (res, mode)
122}
123
124/// The scrubbing mode that was applied to an SQL string.
125#[derive(Debug)]
126pub enum Mode {
127    /// The SQL parser was able to parse & sanitize the string.
128    Parsed(Vec<sqlparser::ast::Statement>),
129    /// SQL parsing failed and the scrubber fell back to a sequence of regexes.
130    Regex,
131}
132
133impl Mode {
134    fn as_str(&self) -> &str {
135        match self {
136            Mode::Parsed(_) => "parser",
137            Mode::Regex => "regex",
138        }
139    }
140}
141
142fn scrub_queries_inner(db_system: Option<&str>, string: &str) -> (Option<String>, Mode) {
143    if let Ok((queries, ast)) = normalize_parsed_queries(db_system, string) {
144        return (Some(queries), Mode::Parsed(ast));
145    }
146
147    let mark_as_scrubbed = ALREADY_NORMALIZED_REGEX.is_match(string);
148
149    let mut string = Cow::from(string.trim());
150
151    if let Cow::Owned(s) = COMMENTS.replace_all(&string, "\n") {
152        string = Cow::Owned(s);
153    }
154    if let Cow::Owned(s) = INLINE_COMMENTS.replace_all(&string, "") {
155        string = Cow::Owned(s);
156    }
157    if let Cow::Owned(s) = NORMALIZER_REGEX.replace_all(&string, "$pre%s") {
158        string = Cow::Owned(s);
159    }
160    if db_system == Some("mysql")
161        && let Cow::Owned(s) = DOUBLE_QUOTED_STRING_REGEX.replace_all(&string, "%s")
162    {
163        string = Cow::Owned(s);
164    }
165    if let Cow::Owned(s) = WHITESPACE.replace_all(&string, " ") {
166        string = Cow::Owned(s);
167    }
168    if let Cow::Owned(s) = PARENS.replace_all(&string, "$pre$post") {
169        string = Cow::Owned(s);
170    }
171    if let Cow::Owned(s) = COLLAPSE_PLACEHOLDERS.replace_all(&string, "$pre%s$post") {
172        string = Cow::Owned(s);
173    }
174    if let Cow::Owned(s) = STRIP_QUOTES.replace_all(&string, "$entity_name") {
175        string = Cow::Owned(s);
176    }
177    if let Cow::Owned(s) = COLLAPSE_ENTITIES.replace_all(&string, "$entity_name") {
178        string = Cow::Owned(s);
179    }
180    if let Cow::Owned(s) = COLLAPSE_COLUMNS.replace_all(&string, "$pre..$post") {
181        string = Cow::Owned(s);
182    }
183
184    let result = match string {
185        Cow::Owned(scrubbed) => Some(scrubbed.trim().to_owned()),
186        Cow::Borrowed(s) if mark_as_scrubbed => Some(s.trim().to_owned()),
187        Cow::Borrowed(_) => None,
188    };
189    (result, Mode::Regex)
190}
