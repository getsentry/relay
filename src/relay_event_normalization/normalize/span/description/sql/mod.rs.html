12/// Removes SQL comments starting with "--" or "#".
13static COMMENTS: Lazy<Regex> = Lazy::new(|| Regex::new(r"(?:--|#).*(?P<newline>\n|$)").unwrap());
14
15/// Removes MySQL inline comments.
16static INLINE_COMMENTS: Lazy<Regex> = Lazy::new(|| Regex::new(r"/\*(?:.|\n)*?\*/").unwrap());
17
18/// Regex with multiple capture groups for SQL tokens we should scrub.
19///
20/// Slightly modified from
21/// <https://github.com/getsentry/sentry/blob/65fb6fdaa0080b824ab71559ce025a9ec6818b3e/src/sentry/spans/grouping/strategy/base.py#L170>
22/// <https://github.com/getsentry/sentry/blob/17af7efe869007f85c5322e48aa9f80a8515bde4/src/sentry/spans/grouping/strategy/base.py#L163>
23static NORMALIZER_REGEX: Lazy<Regex> = Lazy::new(|| {
24    Regex::new(
25        r#"(?xi)
26        # Capture `SAVEPOINT` savepoints.
27        ((?-x)(?P<pre>SAVEPOINT )(?P<savepoint>(?:(?:"[^"]+")|(?:'[^']+')|(?:`[^`]+`)|(?:[a-z]\w+)))) |
28        # Capture single-quoted strings, including the remaining substring if `\'` is found.
29        ((?-x)(?P<single_quoted_strs>N?'(?:\\'|[^'])*(?:'|$)(::\w+(\[\]?)?)?)) |
30        # Capture placeholders.
31        (   (?P<placeholder> (?:\?+|\$\d+|%(?:\(\w+\))?s|:\w+) (::\w+(\[\]?)?)? )   ) |
32        # Capture ODBC escape sequence.
33        ((?-x)(?P<odbc_escape_sequence>\{(?:ts?|d)\s+'.+'\})) |
34        # UUIDs
35        ([0-9a-f]{8}.?[0-9a-f]{4}.?[0-9a-f]{4}.?[0-9a-f]{4}.?[0-9a-f]{12}) |
36        # Capture numbers.
37        ((?-x)(?P<number>(-?\b(?:[0-9]+\.)?[0-9]+(?:[eE][+-]?[0-9]+)?\b)(::\w+(\[\]?)?)?)) |
38        # Hex constants
39        ((?-x)(?P<hex>(\b0x[0-9a-f]+\b)(::\w+(\[\]?)?)?)) |
40        # Capture booleans (as full tokens, not as substrings of other tokens).
41        ((?-x)(?P<bool>(\b(?:true|false)\b)))
42        "#,
43    )
44    .unwrap()
45});
46
47/// For MySQL, also look for double quoted strings.
48static DOUBLE_QUOTED_STRING_REGEX: Lazy<Regex> = Lazy::new(|| {
49    Regex::new(
50        r#"(?xi)
51        # Capture double-quoted strings, including the remaining substring if `\'` is found.
52        ((?-x)(?P<single_quoted_strs>N?"(?:\\"|[^"])*(?:"|$)(::\w+(\[\]?)?)?))
53        "#,
54    )
55    .unwrap()
56});
57
58/// Removes extra whitespace and newlines.
59static WHITESPACE: Lazy<Regex> = Lazy::new(|| Regex::new(r"(\s*\n\s*)|(\s\s+)").unwrap());
60
61/// Removes whitespace around parentheses.
62static PARENS: Lazy<Regex> =
63    Lazy::new(|| Regex::new(r"((?P<pre>\()\s+)|(\s+(?P<post>\)))").unwrap());
64
65static STRIP_QUOTES: Lazy<Regex> =
66    Lazy::new(|| Regex::new(r#"["`](?P<entity_name>\w+)($|["`])"#).unwrap());
67
68/// Regex to shorten table or column references, e.g. `"table1"."col1"` -> `col1`.
69static COLLAPSE_ENTITIES: Lazy<Regex> =
70    Lazy::new(|| Regex::new(r"(?:\w+\.)+(?P<entity_name>\w+)").unwrap());
71
72/// Regex to make multiple placeholders collapse into one.
73/// This can be used as a second pass after [`NORMALIZER_REGEX`].
74static COLLAPSE_PLACEHOLDERS: Lazy<Regex> = Lazy::new(|| {
75    Regex::new(
76        r"(?xi)
77        (?P<pre>(?:VALUES|IN) \s+\() (?P<values> ( %s ( \)\s*,\s*\(\s*%s | \s*,\s*%s )* )) (?P<post>\s*\)?)
78        ",
79    )
80    .unwrap()
81});
82
83/// Collapse simple lists of column names.
84/// For example:
85///   SELECT a, b, count(*) FROM x -> SELECT .., count(*) FROM x
86///   SELECT "a.b" AS a__b, "a.c" AS a__c FROM x -> SELECT .. FROM x
87///
88/// Assumes that column names have already been normalized.
89static COLLAPSE_COLUMNS: Lazy<Regex> = Lazy::new(|| {
90    let col = r"\w+(\s+AS\s+\w+)?";
91    Regex::new(
92        format!(
93            r"(?ix)
94        (?P<pre>(SELECT(\s+(DISTINCT|ALL))?\s|,|\())
95        \s*
96        (?P<columns>({col}(?:\s*,\s*{col})+))
97        (?P<post>\s*(,|\s+|\)))"
98        )
99        .as_str(),
100    )
101    .unwrap()
102});
103
104/// Regex to identify SQL queries that are already normalized.
105///
106/// Looks for `?`, `$1` or `%s` identifiers, commonly used identifiers in
107/// Python, Ruby on Rails and PHP platforms.
108static ALREADY_NORMALIZED_REGEX: Lazy<Regex> = Lazy::new(|| Regex::new(r"/\?|\$1|%s").unwrap());
109
110/// Normalizes the given SQL-query-like string.
111pub fn scrub_queries(db_system: Option<&str>, string: &str) -> (Option<String>, Mode) {
112    let t = Instant::now();
113    let (res, mode) = scrub_queries_inner(db_system, string);
114    relay_statsd::metric!(
115        timer(Timers::SpanDescriptionNormalizeSQL) = t.elapsed(),
116        mode = mode.as_str(),
117    );
118    (res, mode)
119}
120
121/// The scrubbing mode that was applied to an SQL string.
122#[derive(Debug)]
123pub enum Mode {
124    /// The SQL parser was able to parse & sanitize the string.
125    Parsed(Vec<sqlparser::ast::Statement>),
126    /// SQL parsing failed and the scrubber fell back to a sequence of regexes.
127    Regex,
128}
129
130impl Mode {
131    fn as_str(&self) -> &str {
132        match self {
133            Mode::Parsed(_) => "parser",
134            Mode::Regex => "regex",
135        }
136    }
137}
138
139fn scrub_queries_inner(db_system: Option<&str>, string: &str) -> (Option<String>, Mode) {
140    if let Ok((queries, ast)) = normalize_parsed_queries(db_system, string) {
141        return (Some(queries), Mode::Parsed(ast));
142    }
143
144    let mark_as_scrubbed = ALREADY_NORMALIZED_REGEX.is_match(string);
145
146    let mut string = Cow::from(string.trim());
147
148    if let Cow::Owned(s) = COMMENTS.replace_all(&string, "\n") {
149        string = Cow::Owned(s);
150    }
151    if let Cow::Owned(s) = INLINE_COMMENTS.replace_all(&string, "") {
152        string = Cow::Owned(s);
153    }
154    if let Cow::Owned(s) = NORMALIZER_REGEX.replace_all(&string, "$pre%s") {
155        string = Cow::Owned(s);
156    }
157    if db_system == Some("mysql")
158        && let Cow::Owned(s) = DOUBLE_QUOTED_STRING_REGEX.replace_all(&string, "%s")
159    {
160        string = Cow::Owned(s);
161    }
162    if let Cow::Owned(s) = WHITESPACE.replace_all(&string, " ") {
163        string = Cow::Owned(s);
164    }
165    if let Cow::Owned(s) = PARENS.replace_all(&string, "$pre$post") {
166        string = Cow::Owned(s);
167    }
168    if let Cow::Owned(s) = COLLAPSE_PLACEHOLDERS.replace_all(&string, "$pre%s$post") {
169        string = Cow::Owned(s);
170    }
171    if let Cow::Owned(s) = STRIP_QUOTES.replace_all(&string, "$entity_name") {
172        string = Cow::Owned(s);
173    }
174    if let Cow::Owned(s) = COLLAPSE_ENTITIES.replace_all(&string, "$entity_name") {
175        string = Cow::Owned(s);
176    }
177    if let Cow::Owned(s) = COLLAPSE_COLUMNS.replace_all(&string, "$pre..$post") {
178        string = Cow::Owned(s);
179    }
180
181    let result = match string {
182        Cow::Owned(scrubbed) => Some(scrubbed.trim().to_owned()),
183        Cow::Borrowed(s) if mark_as_scrubbed => Some(s.trim().to_owned()),
184        Cow::Borrowed(_) => None,
185    };
186    (result, Mode::Regex)
187}
