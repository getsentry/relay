1use std::mem;
2
3use relay_event_schema::processor::{ProcessingResult, ProcessingState, Processor};
4use relay_event_schema::protocol::{DebugId, DebugImage, NativeDebugImage};
5use relay_protocol::{Annotated, Meta};
6
7/// Converts legacy data structures to current format.
8pub struct LegacyProcessor;
9
10impl Processor for LegacyProcessor {
11    fn process_debug_image(
12        &mut self,
13        image: &mut DebugImage,
14        _meta: &mut Meta,
15        _state: &ProcessingState<'_>,
16    ) -> ProcessingResult {
17        if let DebugImage::Apple(apple) = image {
18            let native = NativeDebugImage {
19                code_id: Annotated::empty(),
20                code_file: mem::replace(&mut apple.name, Annotated::empty()).map_value(From::from),
21                debug_id: mem::replace(&mut apple.uuid, Annotated::empty())
22                    .map_value(DebugId::from),
23                debug_file: Annotated::empty(),
24                debug_checksum: Annotated::empty(),
25                arch: mem::replace(&mut apple.arch, Annotated::empty()),
26                image_addr: mem::replace(&mut apple.image_addr, Annotated::empty()),
27                image_size: mem::replace(&mut apple.image_size, Annotated::empty()),
28                image_vmaddr: mem::replace(&mut apple.image_vmaddr, Annotated::empty()),
29                other: mem::take(&mut apple.other),
30            };
31
32            *image = DebugImage::MachO(Box::new(native));
33        }
34
35        Ok(())
36    }
37}
