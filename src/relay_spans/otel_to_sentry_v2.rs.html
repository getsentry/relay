20/// Transform an OTEL span to a Sentry span V2.
21///
22/// This uses attributes in the OTEL span to populate various fields in the Sentry span.
23/// * The Sentry span's `name` field may be set based on `db` or `http` attributes
24///   if the OTEL span's `name` is empty.
25/// * The Sentry span's `sentry.description` attribute may be set based on `db` or `http` attributes
26///   if the OTEL span's `sentry.description` attribute is empty.
27///
28/// All other attributes are carried over from the OTEL span to the Sentry span.
29pub fn otel_to_sentry_span(
30    otel_span: OtelSpan,
31    resource: Option<&Resource>,
32    scope: Option<&InstrumentationScope>,
33) -> SentrySpanV2 {
34    let OtelSpan {
35        trace_id,
36        span_id,
37        parent_span_id,
38        flags,
39        name,
40        kind,
41        attributes,
42        status,
43        links,
44        start_time_unix_nano,
45        end_time_unix_nano,
46        trace_state: _,
47        dropped_attributes_count: _,
48        events: _,
49        dropped_events_count: _,
50        dropped_links_count: _,
51    } = otel_span;
52
53    let start_timestamp = Utc.timestamp_nanos(start_time_unix_nano as i64);
54    let end_timestamp = Utc.timestamp_nanos(end_time_unix_nano as i64);
55
56    let span_id = SpanId::try_from(span_id.as_slice()).into();
57    let trace_id = TraceId::try_from_slice_or_random(trace_id.as_slice());
58
59    let parent_span_id = match parent_span_id.as_slice() {
60        &[] => Annotated::empty(),
61        bytes => SpanId::try_from(bytes).into(),
62    };
63
64    let mut sentry_attributes = Attributes::new();
65
66    relay_otel::otel_scope_into_attributes(&mut sentry_attributes, resource, scope);
67
68    sentry_attributes.insert(ORIGIN, "auto.otlp.spans".to_owned());
69
70    let mut name = if name.is_empty() { None } else { Some(name) };
71    for (key, value) in attributes.into_iter().flat_map(|attribute| {
72        let value = attribute.value?.value?;
73        Some((attribute.key, value))
74    }) {
75        match key.as_str() {
76            key if key.starts_with("db") => {
77                name = name.or(Some("db".to_owned()));
78            }
79            "http.method" | "http.request.method" => {
80                let http_op = match kind {
81                    2 => "http.server",
82                    3 => "http.client",
83                    _ => "http",
84                };
85                name = name.or(Some(http_op.to_owned()));
86            }
87            _ => (),
88        }
89
90        if let Some(v) = otel_value_to_attribute(value) {
91            sentry_attributes.0.insert(key, Annotated::new(v));
92        }
93    }
94
95    let sentry_links: Vec<Annotated<SpanV2Link>> = links
96        .into_iter()
97        .map(|link| otel_to_sentry_link(link).into())
98        .collect();
99
100    if let Some(status_message) = status.clone().map(|status| status.message) {
101        sentry_attributes.insert(STATUS_MESSAGE.to_owned(), status_message);
102    }
103
104    SentrySpanV2 {
105        name: name.into(),
106        trace_id,
107        span_id,
108        parent_span_id,
109        is_remote: otel_flags_is_remote(flags).unwrap_or(false).into(),
110        start_timestamp: Timestamp(start_timestamp).into(),
111        end_timestamp: Timestamp(end_timestamp).into(),
112        status: status
113            .map(|status| otel_to_sentry_status(status.code))
114            .unwrap_or(SpanV2Status::Ok)
115            .into(),
116        kind: otel_to_sentry_kind(kind),
117        links: sentry_links.into(),
118        attributes: Annotated::new(sentry_attributes),
119        ..Default::default()
120    }
121}

123fn otel_flags_is_remote(value: u32) -> Option<bool> {
124    if value & OtelSpanFlags::ContextHasIsRemoteMask as u32 == 0 {
125        None
126    } else {
127        Some(value & OtelSpanFlags::ContextIsRemoteMask as u32 != 0)
128    }
129}
130
131fn otel_to_sentry_kind(kind: i32) -> Annotated<SpanKind> {
132    match kind {
133        kind if kind == OtelSpanKind::Unspecified as i32 => Annotated::empty(),
134        kind if kind == OtelSpanKind::Internal as i32 => Annotated::new(SpanKind::Internal),
135        kind if kind == OtelSpanKind::Server as i32 => Annotated::new(SpanKind::Server),
136        kind if kind == OtelSpanKind::Client as i32 => Annotated::new(SpanKind::Client),
137        kind if kind == OtelSpanKind::Producer as i32 => Annotated::new(SpanKind::Producer),
138        kind if kind == OtelSpanKind::Consumer as i32 => Annotated::new(SpanKind::Consumer),
139        _ => Annotated::from_error(ErrorKind::InvalidData, Some(Value::I64(kind as i64))),
140    }
141}

143fn otel_to_sentry_status(status_code: i32) -> SpanV2Status {
144    if status_code == OtelStatusCode::Unset as i32 || status_code == OtelStatusCode::Ok as i32 {
145        SpanV2Status::Ok
146    } else {
147        SpanV2Status::Error
148    }
149}

153fn otel_to_sentry_link(otel_link: OtelLink) -> Result<SpanV2Link, Error> {
154    // See the W3C trace context specification:
155    // <https://www.w3.org/TR/trace-context-2/#sampled-flag>
156    const W3C_TRACE_CONTEXT_SAMPLED: u32 = 1 << 0;
157
158    let attributes = Attributes::from_iter(otel_link.attributes.into_iter().filter_map(|kv| {
159        let value = kv.value?.value?;
160        let attr_value = otel_value_to_attribute(value)?;
161        Some((kv.key, Annotated::new(attr_value)))
162    }));
163
164    let span_link = SpanV2Link {
165        trace_id: Annotated::new(hex::encode(otel_link.trace_id).parse()?),
166        span_id: SpanId::try_from(otel_link.span_id.as_slice())?.into(),
167        sampled: (otel_link.flags & W3C_TRACE_CONTEXT_SAMPLED != 0).into(),
168        attributes: Annotated::new(attributes),
169        other: Default::default(),
170    };
171
172    Ok(span_link)
173}
