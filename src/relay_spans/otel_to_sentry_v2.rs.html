16/// Transform an OTEL span to a Sentry span V2.
17///
18/// This uses attributes in the OTEL span to populate various fields in the Sentry span.
19/// * The Sentry span's `name` field may be set based on `db` or `http` attributes
20///   if the OTEL span's `name` is empty.
21/// * The Sentry span's `sentry.description` attribute may be set based on `db` or `http` attributes
22///   if the OTEL span's `sentry.description` attribute is empty.
23///
24/// All other attributes are carried over from the OTEL span to the Sentry span.
25pub fn otel_to_sentry_span(otel_span: OtelSpan) -> Result<SentrySpanV2, Error> {
26    let OtelSpan {
27        trace_id,
28        span_id,
29        parent_span_id,
30        flags,
31        name,
32        kind,
33        attributes,
34        status,
35        links,
36        start_time_unix_nano,
37        end_time_unix_nano,
38        trace_state: _,
39        dropped_attributes_count: _,
40        events: _,
41        dropped_events_count: _,
42        dropped_links_count: _,
43    } = otel_span;
44
45    let start_timestamp = Utc.timestamp_nanos(start_time_unix_nano as i64);
46    let end_timestamp = Utc.timestamp_nanos(end_time_unix_nano as i64);
47
48    let span_id = SpanId::try_from(span_id.as_slice())?;
49    let trace_id = TraceId::try_from(trace_id.as_slice())?;
50    let parent_span_id = match parent_span_id.as_slice() {
51        &[] => None,
52        bytes => Some(SpanId::try_from(bytes)?),
53    };
54
55    let mut sentry_attributes = Attributes::new();
56    let mut name = if name.is_empty() { None } else { Some(name) };
57    let mut description = None;
58    let mut http_method = None;
59    let mut http_route = None;
60
61    for (key, value) in attributes.into_iter().flat_map(|attribute| {
62        let value = attribute.value?.value?;
63        Some((attribute.key, value))
64    }) {
65        match key.as_str() {
66            "sentry.description" => {
67                description = otel_value_to_string(value.clone());
68            }
69            key if key.starts_with("db") => {
70                name = name.or(Some("db".to_owned()));
71                if key == "db.statement" {
72                    description = description.or_else(|| otel_value_to_string(value.clone()));
73                }
74            }
75            "http.method" | "http.request.method" => {
76                let http_op = match kind {
77                    2 => "http.server",
78                    3 => "http.client",
79                    _ => "http",
80                };
81                http_method = otel_value_to_string(value.clone());
82                name = name.or(Some(http_op.to_owned()));
83            }
84            "http.route" | "url.path" => {
85                http_route = otel_value_to_string(value.clone());
86            }
87            _ => (),
88        }
89
90        if let Some(v) = otel_value_to_attr(value) {
91            sentry_attributes.insert_raw(key, Annotated::new(v));
92        }
93    }
94
95    if let (Some(http_method), Some(http_route)) = (http_method, http_route) {
96        description = description.or_else(|| Some(format!("{http_method} {http_route}")));
97    }
98
99    // Put the fixed up description back into the attributes
100    if let Some(description) = description {
101        sentry_attributes.insert("sentry.description".into(), description);
102    }
103
104    let sentry_links: Vec<Annotated<SpanV2Link>> = links
105        .into_iter()
106        .map(|link| otel_to_sentry_link(link).map(Into::into))
107        .collect::<Result<_, _>>()?;
108
109    let event_span = SentrySpanV2 {
110        name: name.into(),
111        parent_span_id: parent_span_id.into(),
112        span_id: span_id.into(),
113        is_remote: Annotated::from(otel_flags_is_remote(flags)),
114        start_timestamp: Timestamp(start_timestamp).into(),
115        end_timestamp: Timestamp(end_timestamp).into(),
116        status: status
117            .map(|status| otel_to_sentry_status(status.code))
118            .into(),
119        trace_id: Annotated::new(trace_id),
120        kind: otel_to_sentry_kind(kind),
121        links: sentry_links.into(),
122        attributes: Annotated::new(sentry_attributes),
123        ..Default::default()
124    };
125
126    Ok(event_span)
127}

129fn otel_value_to_string(value: OtelValue) -> Option<String> {
130    match value {
131        OtelValue::StringValue(v) => Some(v),
132        OtelValue::BoolValue(v) => Some(v.to_string()),
133        OtelValue::IntValue(v) => Some(v.to_string()),
134        OtelValue::DoubleValue(v) => Some(v.to_string()),
135        OtelValue::BytesValue(v) => String::from_utf8(v).ok(),
136        _ => None,
137    }
138}
139
140fn otel_flags_is_remote(value: u32) -> Option<bool> {
141    if value & OtelSpanFlags::ContextHasIsRemoteMask as u32 == 0 {
142        None
143    } else {
144        Some(value & OtelSpanFlags::ContextIsRemoteMask as u32 != 0)
145    }
146}
147
148fn otel_to_sentry_kind(kind: i32) -> Annotated<SpanV2Kind> {
149    match kind {
150        kind if kind == OtelSpanKind::Unspecified as i32 => Annotated::empty(),
151        kind if kind == OtelSpanKind::Internal as i32 => Annotated::new(SpanV2Kind::Internal),
152        kind if kind == OtelSpanKind::Server as i32 => Annotated::new(SpanV2Kind::Server),
153        kind if kind == OtelSpanKind::Client as i32 => Annotated::new(SpanV2Kind::Client),
154        kind if kind == OtelSpanKind::Producer as i32 => Annotated::new(SpanV2Kind::Producer),
155        kind if kind == OtelSpanKind::Consumer as i32 => Annotated::new(SpanV2Kind::Consumer),
156        _ => Annotated::from_error(ErrorKind::InvalidData, Some(Value::I64(kind as i64))),
157    }
158}
159
160fn otel_to_sentry_status(status_code: i32) -> SpanV2Status {
161    if status_code == OtelStatusCode::Unset as i32 || status_code == OtelStatusCode::Ok as i32 {
162        SpanV2Status::Ok
163    } else {
164        SpanV2Status::Error
165    }
166}

168fn otel_value_to_attr(otel_value: OtelValue) -> Option<Attribute> {
169    let (ty, value) = match otel_value {
170        OtelValue::StringValue(s) => (AttributeType::String, Value::String(s)),
171        OtelValue::BoolValue(b) => (AttributeType::Boolean, Value::Bool(b)),
172        OtelValue::IntValue(i) => (AttributeType::Integer, Value::I64(i)),
173        OtelValue::DoubleValue(d) => (AttributeType::Double, Value::F64(d)),
174        OtelValue::BytesValue(bytes) => {
175            let s = String::from_utf8(bytes).ok()?;
176            (AttributeType::String, Value::String(s))
177        }
178        OtelValue::ArrayValue(_) | OtelValue::KvlistValue(_) => return None,
179    };
180
181    Some(Attribute::new(ty, value))
182}

184fn otel_to_sentry_link(otel_link: OtelLink) -> Result<SpanV2Link, Error> {
185    // See the W3C trace context specification:
186    // <https://www.w3.org/TR/trace-context-2/#sampled-flag>
187    const W3C_TRACE_CONTEXT_SAMPLED: u32 = 1 << 0;
188
189    let attributes = Attributes::from_iter(otel_link.attributes.into_iter().filter_map(|kv| {
190        let value = kv.value?.value?;
191        let attr_value = otel_value_to_attr(value)?;
192        Some((kv.key, Annotated::new(attr_value)))
193    }));
194
195    let span_link = SpanV2Link {
196        trace_id: Annotated::new(hex::encode(otel_link.trace_id).parse()?),
197        span_id: SpanId::try_from(otel_link.span_id.as_slice())?.into(),
198        sampled: (otel_link.flags & W3C_TRACE_CONTEXT_SAMPLED != 0).into(),
199        attributes: Annotated::new(attributes),
200        other: Default::default(),
201    };
202
203    Ok(span_link)
204}
