14impl Database {
15    pub fn new(budget: &MemoryBudget) -> Self {
16        Self {
17            inner: Default::default(),
18            budget: budget.clone(),
19        }
20    }

22    /// Removes and merges intern information contained in the `packet` into the database.
23    pub fn intern(&mut self, packet: &mut proto::TracePacket) -> Result<&Tables, BudgetExceeded> {
24        let data = packet.interned_data.take();
25        let seq_id = SequenceId::new(packet);
26
27        // Technically this is off by-one, but significantly easier to handle.
28        if self.inner.len() >= consts::MAX_SEQUENCE_IDS {
29            return Err(BudgetExceeded);
30        }
31
32        if utils::has_incremental_state_cleared(packet) {
33            self.inner.remove(&seq_id);
34        }
35
36        let tables = self.inner.entry(seq_id).or_insert_with(|| Tables {
37            function_names: self.budget.map(),
38            mapping_paths: self.budget.map(),
39            build_ids: self.budget.map(),
40            frames: self.budget.map(),
41            callstacks: self.budget.map(),
42            mappings: self.budget.map(),
43        });
44
45        if let Some(data) = data {
46            tables.merge(data)?;
47        };
48        Ok(tables)
49    }
50}

52#[derive(Debug)]
53pub struct Tables {
54    function_names: BudgetMap<u64, String>,
55    mapping_paths: BudgetMap<u64, String>,
56    build_ids: BudgetMap<u64, Vec<u8>>,
57    frames: BudgetMap<u64, proto::Frame>,
58    callstacks: BudgetMap<u64, proto::Callstack>,
59    mappings: BudgetMap<u64, proto::Mapping>,
60}
61
62impl Tables {
63    pub fn resolve_callstack(&self, id: u64) -> Option<&proto::Callstack> {
64        self.callstacks.get(&id)
65    }
66
67    pub fn resolve_frame(&self, id: u64) -> Option<&proto::Frame> {
68        self.frames.get(&id)
69    }
70
71    pub fn resolve_function_name(&self, id: Option<u64>) -> Option<&str> {
72        id.and_then(|id| self.function_names.get(&id))
73            .map(String::as_str)
74    }
75
76    pub fn resolve_mapping(&self, id: Option<u64>) -> Option<&proto::Mapping> {
77        id.and_then(|id| self.mappings.get(&id))
78    }
79
80    pub fn resolve_mapping_path(&self, id: u64) -> Option<&str> {
81        self.mapping_paths.get(&id).map(String::as_str)
82    }
83
84    pub fn resolve_build_id(&self, id: Option<u64>) -> Option<&[u8]> {
85        id.and_then(|id| self.build_ids.get(&id)).map(Vec::as_slice)
86    }
87
88    fn merge(&mut self, data: proto::InternedData) -> Result<(), BudgetExceeded> {
89        let proto::InternedData {
90            function_names,
91            mapping_paths,
92            build_ids,
93            frames,
94            callstacks,
95            mappings,
96        } = data;
97
98        self.function_names
99            .extend(interned_strings(function_names))?;
100        self.mapping_paths.extend(interned_strings(mapping_paths))?;
101        self.build_ids.extend(
102            build_ids
103                .into_iter()
104                .filter_map(|is| Some((is.iid?, is.r#str?))),
105        )?;
106        self.frames
107            .extend(frames.into_iter().filter_map(|f| Some((f.iid?, f))))?;
108        self.callstacks
109            .extend(callstacks.into_iter().filter_map(|c| Some((c.iid?, c))))?;
110        self.mappings
111            .extend(mappings.into_iter().filter_map(|m| Some((m.iid?, m))))?;
112
113        Ok(())
114    }
115}
116
117fn interned_strings(strings: Vec<proto::InternedString>) -> impl Iterator<Item = (u64, String)> {
118    strings
119        .into_iter()
120        .filter_map(|is| Some((is.iid?, String::from_utf8(is.r#str?).ok()?)))
121}
