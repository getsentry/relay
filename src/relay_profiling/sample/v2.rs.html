1//! Sample Format V2
2//!
3//! This version of the sample format expects a collection of samples to be sent with no reference
4//! to the events collected while the profiler was running.
5//!
6//! We collect a profiler ID, meaning to be a random identifier for this specific instance of the
7//! profiler and not a persistent ID. It only needs to be valid from the start of the profiler to
8//! when it stops and will be useful to then group samples on the backend.
9//!
10//! Spans are expected to carry the profiler ID to know which samples are associated with them.
11//!

26const MAX_PROFILE_CHUNK_DURATION_SECS: f64 = MAX_PROFILE_CHUNK_DURATION.as_secs_f64();
27#[derive(Debug, Serialize, Deserialize)]
28pub struct ProfileMetadata {
29    /// Random UUID identifying a chunk
30    pub chunk_id: EventId,
31    /// Random UUID for each profiler session
32    pub profiler_id: EventId,
33
34    #[serde(default, skip_serializing_if = "DebugMeta::is_empty")]
35    pub debug_meta: DebugMeta,
36
37    #[serde(skip_serializing_if = "Option::is_none")]
38    pub environment: Option<String>,
39    pub platform: String,
40    pub release: Option<String>,
41
42    #[serde(skip_serializing_if = "Option::is_none")]
43    pub client_sdk: Option<ClientSdk>,
44
45    /// Hard-coded string containing "2" to indicate the format version.
46    pub version: Version,
47}

49#[derive(Debug, Serialize, Deserialize)]
50pub struct Sample {
51    /// Unix timestamp in seconds with millisecond precision when the sample
52    /// was captured.
53    pub timestamp: FiniteF64,
54    /// Index of the stack in the `stacks` field of the profile.
55    pub stack_id: usize,
56    /// Thread or queue identifier
57    pub thread_id: String,
58}

60#[derive(Debug, Serialize, Deserialize)]
61pub struct ProfileChunk {
62    // `measurements` contains CPU/memory measurements we do during the capture of the chunk.
63    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
64    pub measurements: BTreeMap<String, ChunkMeasurement>,
65    /// This struct contains all the metadata related to the chunk but all fields are expected to
66    /// be at the top-level of the object.
67    #[serde(flatten)]
68    pub metadata: ProfileMetadata,
69    pub profile: ProfileData,
70}

72impl ProfileChunk {
73    pub fn normalize(&mut self) -> Result<(), ProfileError> {
74        let platform = self.metadata.platform.as_str();
75
76        if self.metadata.client_sdk.is_none() {
77            self.metadata.client_sdk = default_client_sdk(platform);
78        }
79
80        self.profile.normalize(platform)
81    }
82}

84#[derive(Debug, Default, Serialize, Deserialize)]
85pub struct ProfileData {
86    /// `samples` contains the list of samples referencing a stack and thread identifier.
87    /// If 2 stack of frames captured at 2 different timestamps are identical, you're expected to
88    /// reference the same `stack_id`.
89    pub samples: Vec<Sample>,
90    /// `stacks` contains a list of stacks indicating the index of the frame in the `frames` field.
91    /// We do this to not have to repeat frames in different stacks.
92    pub stacks: Vec<Vec<usize>>,
93    /// `frames` contains a list of unique frames found in the profile.
94    pub frames: Vec<Frame>,
95
96    /// `thread_metadata` contains information about the thread or the queue. The identifier is a
97    /// string and can be any unique identifier for the thread or stack (an integer or an address
98    /// for example).
99    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
100    pub thread_metadata: BTreeMap<String, ThreadMetadata>,
101}
102
103impl ProfileData {
104    fn is_above_max_duration(&self) -> bool {
105        if self.samples.is_empty() {
106            return false;
107        }
108        let mut min = self.samples[0].timestamp;
109        let mut max = self.samples[0].timestamp;
110
111        for sample in self.samples.iter().skip(1) {
112            if sample.timestamp < min {
113                min = sample.timestamp
114            } else if sample.timestamp > max {
115                max = sample.timestamp
116            }
117        }
118
119        let duration = max.saturating_sub(min);
120        duration.to_f64() > MAX_PROFILE_CHUNK_DURATION_SECS
121    }
122    /// Ensures valid profile chunk or returns an error.
123    ///
124    /// Mutates the profile chunk. Removes invalid samples and threads.
125    /// Throws an error if the profile chunk is malformed.
126    /// Removes extra metadata that are not referenced in the samples.
127    pub fn normalize(&mut self, platform: &str) -> Result<(), ProfileError> {
128        self.remove_single_samples_per_thread();
129
130        if self.samples.is_empty() {
131            return Err(ProfileError::NotEnoughSamples);
132        }
133
134        if !self.all_stacks_referenced_by_samples_exist() {
135            return Err(ProfileError::MalformedSamples);
136        }
137
138        if !self.all_frames_referenced_by_stacks_exist() {
139            return Err(ProfileError::MalformedStacks);
140        }
141
142        self.samples.sort_by_key(|s| s.timestamp);
143
144        if self.is_above_max_duration() {
145            return Err(ProfileError::DurationIsTooLong);
146        }
147
148        self.strip_pointer_authentication_code(platform);
149        self.remove_unreferenced_threads();
150
151        Ok(())
152    }
153
154    fn strip_pointer_authentication_code(&mut self, platform: &str) {
155        let addr = match platform {
156            // https://github.com/microsoft/plcrashreporter/blob/748087386cfc517936315c107f722b146b0ad1ab/Source/PLCrashAsyncThread_arm.c#L84
157            "cocoa" => 0x0000000FFFFFFFFF,
158            _ => return,
159        };
160        for frame in &mut self.frames {
161            frame.strip_pointer_authentication_code(addr);
162        }
163    }
164
165    /// Checks that all stacks referenced by the samples exist in the stacks.
166    fn all_stacks_referenced_by_samples_exist(&self) -> bool {
167        self.samples
168            .iter()
169            .all(|sample| self.stacks.get(sample.stack_id).is_some())
170    }
171
172    /// Checks that all frames referenced by the stacks exist in the frames.
173    fn all_frames_referenced_by_stacks_exist(&self) -> bool {
174        self.stacks.iter().all(|stack| {
175            stack
176                .iter()
177                .all(|frame_id| self.frames.get(*frame_id).is_some())
178        })
179    }
180
181    fn remove_unreferenced_threads(&mut self) {
182        let thread_ids = self
183            .samples
184            .iter()
185            .map(|sample| sample.thread_id.clone())
186            .collect::<HashSet<_>>();
187        self.thread_metadata
188            .retain(|thread_id, _| thread_ids.contains(thread_id));
189    }
190
191    /// Removes a sample when it's the only non-idle sample on its thread
192    fn remove_single_samples_per_thread(&mut self) {
193        let mut sample_count_by_thread_id: hashbrown::HashMap<String, u32> = HashMap::new();
194
195        for s in &self.samples {
196            if let Some(stack) = self.stacks.get(s.stack_id) {
197                // We only count non-idle samples
198                if stack.is_empty() {
199                    continue;
200                }
201            } else {
202                continue;
203            }
204            *sample_count_by_thread_id
205                .entry(s.thread_id.to_owned())
206                .or_default() += 1;
207        }
208
209        sample_count_by_thread_id.retain(|_, count| *count > 1);
210        self.samples
211            .retain(|sample| sample_count_by_thread_id.contains_key(&sample.thread_id));
212    }
213}

215pub fn parse(payload: &[u8]) -> Result<ProfileChunk, ProfileError> {
216    let d = &mut serde_json::Deserializer::from_slice(payload);
217    serde_path_to_error::deserialize(d).map_err(ProfileError::InvalidJson)
218}

231        let second_payload = serde_json::to_vec(&first_parse.unwrap()).unwrap();
232        let second_parse = parse(&second_payload[..]);
