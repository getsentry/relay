1//! Profiling protocol and processing for Sentry.
2//!
3//! Profiles are captured during the life of a transaction on Sentry clients and sent to Relay for
4//! ingestion as JSON objects with some metadata. They are sent alongside a transaction (usually in
5//! the same envelope) and are usually big objects (average size around 300KB and it's not unusual
6//! to see several MB). Their size is linked to the amount of code executed during the transaction
7//! (so the duration of the transaction influences the size of the profile).
8//!
9//! It's preferrable to have a profile with a transaction. If a transaction is dropped, the profile
10//! should be dropped as well.
11//!
12//! # Envelope
13//!
14//! ## Transaction Profiling
15//!
16//! To send a profile of a transaction to Relay, the profile is enclosed in an item of type
17//! `profile`:
18//! ```json
19//! {"type": "profile", "size": ...}
20//! { ... }
21//! ```
22//! ## Continuous Profiling
23//!
24//! For continuous profiling, we expect to receive chunks of profile in an item of type
25//! `profile_chunk`:
26//! ```json
27//! {"type": "profile_chunk"}
28//! { ... }
29//! ```
30//!
31//! # Protocol
32//!
33//! Each item type expects a different format.
34//!
35//! For `Profile` item type, we expect the Sample format v1 or Android format.
36//! For `ProfileChunk` item type, we expect the Sample format v2.
37//!
38//! # Ingestion
39//!
40//! Relay will forward those profiles encoded with `msgpack` after unpacking them if needed and push a message on Kafka.

72const MAX_PROFILE_DURATION: Duration = Duration::from_secs(30);
73/// For continuous profiles, each chunk can be at most 1 minute.
74/// In certain circumstances (e.g. high cpu load) the profiler
75/// the profiler may be stopped slightly after 60, hence here we
76/// give it a bit more room to handle such cases (66 instead of 60)

79/// Unique identifier for a profile.

81/// Same format as event IDs.

84/// Determines the type/use of a [`ProfileChunk`].
85#[derive(Debug, Clone, Copy, PartialEq, Eq)]
86pub enum ProfileType {
87    /// A backend profile.
88    Backend,
89    /// A UI profile.
90    Ui,
91}
92
93impl ProfileType {
94    /// Converts a platform to a [`ProfileType`].
95    ///
96    /// The profile type is currently determined based on the contained profile
97    /// platform. It determines the data category this profile chunk belongs to.
98    ///
99    /// This needs to be synchronized with the implementation in Sentry:
100    /// <https://github.com/getsentry/sentry/blob/ed2e1c8bcd0d633e6f828fcfbeefbbdd98ef3dba/src/sentry/profiles/task.py#L995>
101    pub fn from_platform(platform: &str) -> Self {
102        match platform {
103            "cocoa" | "android" | "javascript" => Self::Ui,
104            _ => Self::Backend,
105        }
106    }
107}

109#[derive(Debug, Deserialize)]
110struct MinimalProfile {
111    #[serde(alias = "profile_id", alias = "chunk_id")]
112    event_id: ProfileId,
113    platform: String,
114    release: Option<String>,
115    #[serde(default)]
116    version: sample::Version,
117}
118
119impl Filterable for MinimalProfile {
120    fn release(&self) -> Option<&str> {
121        self.release.as_deref()
122    }
123}
124
125impl Getter for MinimalProfile {
126    fn get_value(&self, path: &str) -> Option<Val<'_>> {
127        match path.strip_prefix("event.")? {
128            "release" => self.release.as_deref().map(|release| release.into()),
129            "platform" => Some(self.platform.as_str().into()),
130            _ => None,
131        }
132    }
133}

158pub fn parse_metadata(payload: &[u8]) -> Result<ProfileMetadata, ProfileError> {
159    let profile = match minimal_profile_from_json(payload) {
160        Ok(profile) => profile,
161        Err(err) => {
162            relay_log::debug!(
163                error = &err as &dyn Error,
164                from = "minimal",
165                "invalid profile"
166            );
167            return Err(ProfileError::InvalidJson(err));
168        }
169    };
170    match profile.version {
171        sample::Version::V1 => {
172            let d = &mut Deserializer::from_slice(payload);
173            let _: sample::v1::ProfileMetadata = match serde_path_to_error::deserialize(d) {
174                Ok(profile) => profile,
175                Err(err) => {
176                    relay_log::debug!(
177                        error = &err as &dyn Error,
178                        from = "metadata",
179                        platform = profile.platform,
180                        "invalid profile",
181                    );
182                    return Err(ProfileError::InvalidJson(err));
183                }
184            };
185        }
186        _ => match profile.platform.as_str() {
187            "android" => {
188                let d = &mut Deserializer::from_slice(payload);
189                let _: android::legacy::ProfileMetadata = match serde_path_to_error::deserialize(d)
190                {
191                    Ok(profile) => profile,
192                    Err(err) => {
193                        relay_log::debug!(
194                            error = &err as &dyn Error,
195                            from = "metadata",
196                            platform = "android",
197                            "invalid profile",
198                        );
199                        return Err(ProfileError::InvalidJson(err));
200                    }
201                };
202            }
203            _ => return Err(ProfileError::PlatformNotSupported),
204        },
205    };
206
207    Ok(ProfileMetadata {
208        id: profile.event_id,
209        platform: profile.platform,
210    })
211}

213pub fn expand_profile(
214    payload: &[u8],
215    event: &Event,
216    client_ip: Option<IpAddr>,
217    filter_settings: &ProjectFiltersConfig,
218    global_config: &GlobalConfig,
219) -> Result<(ProfileId, Vec<u8>), ProfileError> {
220    let profile = match minimal_profile_from_json(payload) {
221        Ok(profile) => profile,
222        Err(err) => {
223            relay_log::debug!(
224                error = &err as &dyn Error,
225                from = "minimal",
226                platform = event.platform.as_str(),
227                project_id = event.project.value().unwrap_or(&0),
228                sdk_name = event.sdk_name(),
229                sdk_version = event.sdk_version(),
230                transaction_id = ?event.id.value(),
231                "invalid profile",
232            );
233            return Err(ProfileError::InvalidJson(err));
234        }
235    };
236
237    if let Err(filter_stat_key) = relay_filter::should_filter(
238        &profile,
239        client_ip,
240        filter_settings,
241        global_config.filters(),
242    ) {
243        return Err(ProfileError::Filtered(filter_stat_key));
244    }
245
246    let transaction_metadata = extract_transaction_metadata(event);
247    let transaction_tags = extract_transaction_tags(event);
248    let processed_payload = match (profile.platform.as_str(), profile.version) {
249        (_, sample::Version::V1) => {
250            sample::v1::parse_sample_profile(payload, transaction_metadata, transaction_tags)
251        }
252        ("android", _) => {
253            android::legacy::parse_android_profile(payload, transaction_metadata, transaction_tags)
254        }
255        (_, _) => return Err(ProfileError::PlatformNotSupported),
256    };
257    match processed_payload {
258        Ok(payload) => Ok((profile.event_id, payload)),
259        Err(err) => match err {
260            ProfileError::InvalidJson(err) => {
261                relay_log::debug!(
262                    error = &err as &dyn Error,
263                    from = "parsing",
264                    platform = profile.platform,
265                    project_id = event.project.value().unwrap_or(&0),
266                    sdk_name = event.sdk_name(),
267                    sdk_version = event.sdk_version(),
268                    transaction_id = ?event.id.value(),
269                    "invalid profile",
270                );
271                Err(ProfileError::InvalidJson(err))
272            }
273            _ => {
274                relay_log::debug!(
275                    error = &err as &dyn Error,
276                    from = "parsing",
277                    platform = profile.platform,
278                    project_id = event.project.value().unwrap_or(&0),
279                    sdk_name = event.sdk_name(),
280                    sdk_version = event.sdk_version(),
281                    transaction_id = ?event.id.value(),
282                    "invalid profile",
283                );
284                Err(err)
285            }
286        },
287    }
288}

290/// Intermediate type for all processing on a profile chunk.
291pub struct ProfileChunk {
292    profile: MinimalProfile,
293    payload: Bytes,
294}
295
296impl ProfileChunk {
297    /// Parses a new [`Self`] from raw bytes.
298    pub fn new(payload: Bytes) -> Result<Self, ProfileError> {
299        match minimal_profile_from_json(&payload) {
300            Ok(profile) => Ok(Self { profile, payload }),
301            Err(err) => {
302                relay_log::debug!(
303                    error = &err as &dyn Error,
304                    from = "minimal",
305                    "invalid profile chunk",
306                );
307                Err(ProfileError::InvalidJson(err))
308            }
309        }
310    }
311
312    /// Returns the platform of the profile chunk.
313    pub fn platform(&self) -> &str {
314        &self.profile.platform
315    }
316
317    /// Returns the [`ProfileType`] this chunk belongs to.
318    ///
319    /// This is currently determined from the platform via [`ProfileType::from_platform`].
320    pub fn profile_type(&self) -> ProfileType {
321        ProfileType::from_platform(&self.profile.platform)
322    }
323
324    /// Applies inbound filters to the profile chunk.
325    ///
326    /// The profile needs to be filtered (rejected) when this returns an error.
327    pub fn filter(
328        &self,
329        client_ip: Option<IpAddr>,
330        filter_settings: &ProjectFiltersConfig,
331        global_config: &GlobalConfig,
332    ) -> Result<(), ProfileError> {
333        relay_filter::should_filter(
334            &self.profile,
335            client_ip,
336            filter_settings,
337            global_config.filters(),
338        )
339        .map_err(ProfileError::Filtered)
340    }
341
342    /// Normalizes and 'expands' the profile chunk into its normalized form Sentry expects.
343    pub fn expand(&self) -> Result<Vec<u8>, ProfileError> {
344        match (self.profile.platform.as_str(), self.profile.version) {
345            ("android", _) => android::chunk::parse(&self.payload),
346            (_, sample::Version::V2) => {
347                let mut profile = sample::v2::parse(&self.payload)?;
348                profile.normalize()?;
349                Ok(serde_json::to_vec(&profile)
350                    .map_err(|_| ProfileError::CannotSerializePayload)?)
351            }
352            (_, _) => Err(ProfileError::PlatformNotSupported),
353        }
354    }
355}

357/// The result of expanding a binary Perfetto trace via [`expand_perfetto`].
358///
359/// Carries the serialized Sample v2 JSON payload together with the profile
360/// metadata needed downstream (platform, profile type, inbound filtering) so
361/// that callers do **not** need to deserialize the payload a second time.
362#[derive(Debug)]
363pub struct ExpandedPerfettoChunk {
364    /// Serialized Sample v2 JSON payload, ready for ingestion.
365    pub payload: Vec<u8>,
366    /// Platform string extracted from the metadata (e.g. `"android"`).
367    pub platform: String,
368    /// Release string from the metadata, used for inbound filtering.
369    pub release: Option<String>,
370}
371
372impl ExpandedPerfettoChunk {
373    /// Returns the [`ProfileType`] derived from the platform.
374    pub fn profile_type(&self) -> ProfileType {
375        ProfileType::from_platform(&self.platform)
376    }
377
378    /// Applies inbound filters to the profile chunk.
379    pub fn filter(
380        &self,
381        client_ip: Option<IpAddr>,
382        filter_settings: &ProjectFiltersConfig,
383        global_config: &GlobalConfig,
384    ) -> Result<(), ProfileError> {
385        relay_filter::should_filter(self, client_ip, filter_settings, global_config.filters())
386            .map_err(ProfileError::Filtered)
387    }
388}
389
390impl Filterable for ExpandedPerfettoChunk {
391    fn release(&self) -> Option<&str> {
392        self.release.as_deref()
393    }
394}
395
396impl Getter for ExpandedPerfettoChunk {
397    fn get_value(&self, path: &str) -> Option<Val<'_>> {
398        match path.strip_prefix("event.")? {
399            "release" => self.release.as_deref().map(|r| r.into()),
400            "platform" => Some(self.platform.as_str().into()),
401            _ => None,
402        }
403    }
404}

406/// Expands a binary Perfetto trace into a Sample v2 profile chunk.
407///
408/// Returns an [`ExpandedPerfettoChunk`] with the serialized JSON payload plus
409/// the profile metadata needed for downstream processing (platform, profile type,
410/// inbound filtering) — avoiding a second JSON deserialization pass in callers.
411pub fn expand_perfetto(
412    perfetto_payload: &[u8],
413    json_payload: &[u8],
414) -> Result<ExpandedPerfettoChunk, ProfileError> {
415    let d = &mut Deserializer::from_slice(json_payload);
416    let mut chunk: sample::v2::ProfileChunk =
417        serde_path_to_error::deserialize(d).map_err(ProfileError::InvalidJson)?;
418
419    let platform = chunk.metadata.platform.clone();
420    let release = chunk.metadata.release.clone();
421
422    let (profile_data, debug_images) = perfetto::convert(perfetto_payload)?;
423    chunk.profile = profile_data;
424    chunk.metadata.debug_meta.images.extend(debug_images);
425    chunk.normalize()?;
426
427    let payload = serde_json::to_vec(&chunk).map_err(|_| ProfileError::CannotSerializePayload)?;
428    Ok(ExpandedPerfettoChunk {
429        payload,
430        platform,
431        release,
432    })
433}

525        assert!(!output.profile.frames.is_empty());
